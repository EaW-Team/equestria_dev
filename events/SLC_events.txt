country_event = {
	id = SLC_commie_collapse_tree.3
	title = SLC_commie_collapse_tree.3.t
	desc = SLC_commie_collapse_tree.3.d
	is_triggered_only = yes
	fire_only_once = yes

	option = {
		name = SLC_commie_collapse_tree.3.a
		custom_effect_tooltip = SLC_tt_rav_fucked_up_3
		hidden_effect = {
			if = {
				limit = {
					is_faction_leader = yes
				}
				dismantle_faction = yes
			}
			else_if = {
				limit = {
					is_in_faction = yes
				}
				leave_faction = yes
			}
			create_faction = SLC_revolt_resistance_faction_name
			every_country = {
    			limit = {
        	        has_war_with = SLC
        	    }
        	    white_peace = SLC
    		}
    		set_country_flag = SLC_rav_collapse_flag

			get_sorted_scored_countries_temp = {
				scorer = SLC_nb_of_cores_scorer
				array = temp_occupied_countries
				scores = trash
			}
			if = {
				# Sometimes, you can occupy yourself
				limit = {
					is_in_array = {
						array = temp_occupied_countries
						value = SLC
					}
				}

				remove_from_temp_array = {
					array = temp_occupied_countries
					value = SLC
				}
			}

			for_each_scope_loop = {
			    array = core_states
			    add_to_temp_array = { temp_core_states = THIS }
			}

			set_temp_variable = { root_capital = ROOT.capital:id }
			set_temp_variable = { temp_leader = temp_occupied_countries^0 }
			set_temp_variable = { temp_leader_distance = temp_leader:capital:distance_to@var:root_capital }
			for_each_scope_loop = {
			    array = temp_occupied_countries

				log = "[THIS.GetName]"

				# Release the country
			    ROOT = {
			        release = PREV
					if = {
						limit = { country_exists = PREV }
						declare_war_on = {
							target = PREV
							type = annex_everything
						}
					}
			    }

				if = {
					limit = { exists = yes } # Sometimes there are no more core to free

					# Which country is the farthest from ROOT ?
					set_temp_variable = { temp_curr_distance = THIS.capital:distance_to@var:root_capital }
					if = {
						limit = { check_variable = { temp_leader_distance < temp_curr_distance } }

						set_temp_variable = { temp_leader_distance = temp_curr_distance }
						set_temp_variable = { temp_leader = THIS }
					}

					# Focus tree loading
					if = {
						limit = { capital_scope = { is_on_continent = africa }}

						load_focus_tree = generic_zebra_focus
					} else = {
						load_focus_tree = generic_griffon_focus
					}
					# Check that the country has the technology for the template
					if = {
						limit = {
							NOT = { has_tech = interwar_antiair }
						}

						set_technology = {
							interwar_antiair = 1
							popup = no
						}
					}
					if = {
						limit = {
							NOT = { has_tech = tech_support }
						}

						set_technology = {
							tech_support = 1
							popup = no
						}
					}
					# Spawn the rebels
					division_template = {
						name = "Rebels"
						regiments = {
							infantry = { x = 0 y = 0 }
							infantry = { x = 0 y = 1 }
							infantry = { x = 1 y = 0 }
							infantry = { x = 1 y = 1 }
							infantry = { x = 1 y = 2 }
							infantry = { x = 2 y = 0 }
							infantry = { x = 2 y = 1 }
						}
						support = {
							engineer = { x = 0 y = 0 }
							anti_air = { x = 0 y = 1 }
						}
					}
					every_controlled_state = {
						limit = { impassable = no }

						set_temp_variable = { num_units_to_create = building_level@arms_factory }
						add_to_temp_variable = { num_units_to_create = building_level@industrial_complex }
						divide_temp_variable = { num_units_to_create = 3 }
						add_to_temp_variable = { num_units_to_create = 2 }
						round_temp_variable = num_units_to_create
						clamp_temp_variable = { var = num_units_to_create min = 3 max = 100 }
						for_loop_effect = {
							end = num_units_to_create
							create_unit = {
								division = "division_template = \"Rebels\" start_experience_factor = 1 start_equipment_factor = 1"
								owner = PREV
							}
						}
					}
				}
			}
			for_each_scope_loop = {
				array = temp_core_states
			    if = {
			        limit = { NOT = { is_core_of = ROOT } }

			        add_core_of = ROOT
			    }
			}

			# Add all the newly released country to the faction
			var:temp_leader = {
				hidden_effect = { create_faction = SLC_revolt_faction_name }
			}
			for_each_scope_loop = {
				array = temp_occupied_countries

				if = {
					limit = {
						exists = yes
						NOT = { tag = var:temp_leader }
					}

					var:temp_leader = {
						add_to_faction = PREV
					}
				}
			}

			# Handle the exclave, states with no core
			every_owned_state = {
				limit = {
					NOT = { is_core_of = ROOT }
				}

				add_to_temp_array = { temp_coreless_states = THIS.id }
				log = "Adding coreless state : [?THIS.id] [THIS.GetName]"
			}
			set_temp_variable = { assigned_state = 1 }
			while_loop_effect = {
				# Loop until all cores are assigned, or no more can be assigned
				limit = {
					check_variable = { assigned_state > 0 }
					check_variable = { temp_coreless_states^num > 0 }
				}

				set_temp_variable = { assigned_state = 0 }
				for_each_loop = {
					array = temp_coreless_states
					break = break_for
					index = index
					value = state_id

					var:state_id = {
						random_neighbor_state = {
							limit = {
								owner = {
									NOT = { tag = ROOT }
									has_war_with = ROOT
								}
							}

							PREV = {
								transfer_state_to = var:PREV.owner
							}

							# Remove the state
							set_temp_variable = { assigned_state = 1 }
							remove_from_temp_array = {
								array = temp_coreless_states
								index = index
							}
							set_temp_variable = { break_for = 1 }
						}
					}
				}
			}

			# Check that all states has been attributed, else assigned them to the biggest enemy country
			if = {
				limit = { check_variable = { temp_coreless_states^num > 0 } }

				for_each_loop = {
					array = temp_coreless_states
					value = state_id

					if = {
						limit = { check_variable = { temp_occupied_countries^num > 0 } }

						var:state_id = {
							transfer_state_to = var:temp_occupied_countries^0
						}
					}
				}
			}
		}
	}
}
