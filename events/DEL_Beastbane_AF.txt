country_event = {
	id = DEL_AF_event_prep.1
	hidden = yes

	is_triggered_only = yes
	immediate = {
		log = "[GetDateText]: [Root.GetName]: event DEL_AF_event_prep.1" 
		hidden_effect = {
			DEL = {
				if = {
					limit = {
						has_war = yes
					}
					set_country_flag = { flag = bypass_on_peaceconference_ended days = 1 value = 1 }
					every_enemy_country = {
						set_country_flag = { flag = bypass_on_peaceconference_ended days = 1 value = 1 }
						white_peace = DEL
					}
				}
				if = {
					limit = {
						is_subject = yes
					}
					set_autonomy = {
						target = DEL
						autonomous_state = autonomy_free
					}
				}
				if = {
					limit = {
						exists = no
					}
					every_country = {
						limit = {
							any_owned_state = { is_core_of = DEL }
						}
						release = DEL
					}
				}
				if = {
					limit = {
						NOT = {
							owns_state = 1139
						}
					}
					for_each_scope_loop = {
						array = original_cores
						if = {
							limit = {
								NOT = { is_core_of = DEL }
							}
							add_core_of = DEL
						}
						transfer_state_to = PREV
					}
				}
			}
		}
	}
	option = {
		name = DEL_AF_event_prep.1.a
		if = {
			limit = {
				TBK = {
					is_ai = no
				}
			}
			TBK = {
				country_event = {
					id = DEL_AF_event_prep.2
				}
			}
		}
		else = {
			DEL = {
				country_event = {
					id = DEL_AF_event_start.1
				}
			}
		}
	}
}
