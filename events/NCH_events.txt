country_event = { #FBK Positive
		 id = greneclyf.3
         title = greneclyf.3.t
         desc = greneclyf.3.d
         picture = GFX_event_gytha

         is_triggered_only = yes

         option = {      
            name = greneclyf.3.a
            add_equipment_to_stockpile = {
                type = convoy_1
                amount = 1 
             }       
          	
			FBK = {

				if = {
					limit = {
						NCH = {
							has_country_flag = NCH_CHN_Submarine_Flag
						}
					}
					add_ideas = NCH_trading_with_NCH_fascist
					set_country_flag = NCH_trading_with_flag
					
				}
				else = {
					add_ideas = NCH_trading_with_NCH_harmonic
					set_country_flag = NCH_trading_with_flag
				}
				set_country_flag = NCH_FBK_accepted_us
				
			}
			
			NCH = {
				if = {
					limit = {
						check_variable = { NCH_trade_partners_iter = 0 }
					}
					add_ideas = NCH_trade_1
				}
				if = {
					limit = {
						check_variable = { NCH_trade_partners_iter = 1 }
					}
					swap_ideas = {add_idea = NCH_trade_2 remove_idea = NCH_trade_1}
				}
				if = {
					limit = {
						check_variable = { NCH_trade_partners_iter = 2 }
					}
					swap_ideas = {add_idea = NCH_trade_3 remove_idea = NCH_trade_2}
				}
				if = {
					limit = {
						check_variable = { NCH_trade_partners_iter = 3 }
					}
					swap_ideas = {add_idea = NCH_trade_4 remove_idea = NCH_trade_3}
				}
				
				add_to_variable = {
					var = NCH_trade_partners_iter
					value = 1
				}
			}
		}
}
