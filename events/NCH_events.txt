	id = greneclyf.108

	title = greneclyf.108.t
	desc = greneclyf.108.d
	picture = GFX_event_gytha

	is_triggered_only = yes

	option = { #lets try EQS!
		name = greneclyf.108.a
		trigger = {
			has_country_flag = NCH_CHN_denied_us_flag
		}
		event_target:equestria = {
			country_event = {id = greneclyf.106 days = 4}
		}
	}
	option = { #lets try CHN!
		name = greneclyf.108.b
		trigger = {
			has_country_flag = NCH_EQS_denied_us_flag
			OR = { country_exists = CHN country_exists = CCN }
		}
		if = {
			limit = { country_exists = CHN }
			CHN = { country_event = { days = 4 id = greneclyf.107 } }
		}
		else_if = {
			limit = { country_exists = CCN }
			CCN = { country_event = { days = 4 id = greneclyf.107 } }
		}
	}
	option = { #we give up
		name = greneclyf.108.c
		trigger = {
			NOT = { has_country_flag = NCH_EQS_trade_accepted_flag }
			NOT = { has_country_flag = NCH_CHN_trade_accepted_flag }
		}
		add_to_variable = {
			gytha_sanity = 4
		}
	}
}
