country_event = { #Hidden Event for smooth transition
	id = north_war.3
	immediate = {log = "[GetDateText]: [Root.GetName]: event north_war.3"}
	hidden = yes

	is_triggered_only = yes

	option = {
		name = north_war.3.a
		HIP = {
			set_country_flag = { flag = bypass_on_peaceconference_ended days = 1 value = 1 }
			white_peace = BAT
			white_peace = CTH
		}
		every_other_country = {
			limit = {
				OR = {
					is_subject_of = BAT
					is_in_faction_with = BAT
				}
			}
			set_country_flag = { flag = bypass_on_peaceconference_ended days = 1 value = 1 }
			white_peace = HIP
		}
		every_other_country = {
			limit = {
				OR = {
					is_subject_of = CTH
					is_in_faction_with = CTH
				}
			}
			set_country_flag = { flag = bypass_on_peaceconference_ended days = 1 value = 1 }
			white_peace = HIP
		}
		every_other_country = {
			limit = {
				OR = {
					is_subject_of = ROOT
					is_in_faction_with = ROOT
				}
			}
			set_country_flag = { flag = bypass_on_peaceconference_ended days = 1 value = 1 }
			white_peace = BAT
			white_peace = CTH
		}
		BAT = {
			if = {
				limit = {
					OR = {
						AND = {
							country_exists = WAR
							WAR = {
								has_war_with = BAT
								has_war_with = CTH
							}
						}
						AND = {
							country_exists = WAR
							OR = {
								is_subject_of = HIP
								is_in_faction_with = HIP
							}
						}
					}
				}
				transfer_state = 688
				transfer_state = 1206
				transfer_state = 743
				transfer_state = 744
			}
			else_if = {
				limit = {
					NOT = {
						688 = { owner = { tag = BAT }}
						1206 = { owner = { tag = BAT }}
						743 = { owner = { tag = BAT }}
						744 = { owner = { tag = BAT }}
					}
				}
				transfer_state = 688
				transfer_state = 1206
				transfer_state = 743
				transfer_state = 744
			}
			transfer_state = 710
			transfer_state = 1199
			country_event = { 
				id = north_war.2
			}
		}
		CTH = {
			transfer_state = 711
			transfer_state = 698
			transfer_state = 697
			country_event = { 
				id = north_war.2
			}
		}
		every_state = {
			limit = {
				is_core_of = ZUM
			}
			remove_core_of = HIP
		}
	}
}
