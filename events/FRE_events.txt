### Buy Fighters ###
country_event = {
	id = freeponies.2
	immediate = {log = "[Root.GetName]: event freeponies.2"}#Auto-logging
	title = freeponies.2.t
	desc = freeponies.2.d
	picture = GFX_report_event_fighters

	is_triggered_only = yes


	option = { #Yes
		name = freeponies.2.a
		ai_chance = {
			factor = 10
		}
		FROM = { country_event = freeponies.3 }
		effect_tooltip = {
			FROM = {
				if = {
					limit = {
						not = {
							has_tech = early_fighter
						}
					}
					set_technology = { early_fighter = 1 }
				}
				add_equipment_to_stockpile = { type = fighter_equipment_0 amount = 25 producer = NIM }
			}
		}
	}

	option = { #No
		name = freeponies.2.b
		ai_chance = {
			factor = 0
		}
		FROM = { country_event = freeponies.4 }

	}
}

		original_tag = FRE
		has_completed_focus = FRE_welth_of_the_mountains

			set_country_flag = FRE_breezies_are_angry

		}
	}
}

	id = freeponies.10
	immediate = {log = "[Root.GetName]: event freeponies.10"}#Auto-logging
	title = freeponies.10.t
	desc = freeponies.10.d
	#picture = GFX_report_event_generic_bombers

	trigger = {
		original_tag = FRE
		has_completed_focus = FRE_begin_rapid_industrialisation
		has_stability < 0.5
		NOT = { has_country_flag = FRE_longhouse_anger }
		NOT = {
			has_idea = FRE_slowed_industralisation
		}
	}

	mean_time_to_happen = {
		months = 10
	}

	option = { ## Give in to their demands - -Construction Speed + Consumer Goods 1100 Days
		name = freeponies.10.a
		add_timed_idea = { idea = FRE_slowed_industralisation days = 180 }
		set_country_flag = { flag = FRE_longhouse_anger days = 200 value = 1 }
	}

	option = { ## They will obay - pp -stability
		name = freeponies.10.b
		add_stability = -0.05
		add_political_power = -50
		set_country_flag = { flag = FRE_longhouse_anger days = 180 value = 1 }
	}

### Civil War! ###
country_event = {
	id = freeponies.14
	immediate = {log = "[Root.GetName]: event freeponies.14"}#Auto-logging
	title = freeponies.14.t
	desc = freeponies.14.d
	#picture = GFX_report_event_generic_bombers

	fire_only_once = yes

	trigger = {
		original_tag = FRE
		has_completed_focus = FRE_begin_rapid_industrialisation
		has_stability < 0.01
		has_country_flag = FRE_rebellion
	}

	mean_time_to_happen = {
		months = 2
	}

	option = { ## Darn them!
		name = freeponies.14.a
		if = {
			limit = {
				NOT = { has_government = neutrality }
			}
			start_civil_war = {
				ideology = neutrality
				size = 0.6
			}
		}
		else = {
			start_civil_war = {
				ideology = democratic
				size = 0.6
			}
		}
	}
}
