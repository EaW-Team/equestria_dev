country_event = { # Yes peace (Macawia side)
	id = peace_with_macawia.2
	immediate = {log = "[GetDateText]: [Root.GetName]: event peace_with_macawia.2"}
	title = peace_with_macawia.2.t
	desc = peace_with_macawia.2.d
	picture = GFX_event_minitrue

	is_triggered_only = yes
	
	option = { # Yes, let's end this
		name = peace_with_macawia.2.a

		# End all subject relationships in Griffonia
		every_subject_country = {
			limit = {
				is_griffon_continent_country_tag = yes
			}
			MCW = {
				end_puppet = PREV
			}
		}

		# Kick them from the faction
		if = {
			limit = {
				MCW = {
					is_faction_leader = yes
				}
			}
			every_allied_country = {
				limit = {
					is_griffon_continent_country_tag = yes
				}
				MCW = {
					remove_from_faction = PREV
				}
			}
		}
		else = {
			if = {
				limit = {
					is_in_faction = yes
				}
				every_allied_country = {
					limit = {
						is_griffon_continent_country_tag = no
					}
					var:faction_leader = {
						remove_from_faction = PREV
					}
				}
				var:faction_leader = {
					remove_from_faction = MCW
				}
			}
		}

		# Make the white peace
		set_country_flag = { flag = bypass_on_peaceconference_ended days = 1 value = 1 }
		white_peace = FROM
		every_allied_country = {
			limit = {
				is_griffon_continent_country_tag = no
			}
			set_country_flag = { flag = bypass_on_peaceconference_ended days = 1 value = 1 }
			white_peace = FROM
		}

		FROM = {
			country_event = peace_with_macawia.3
		}

		ai_chance = { base = 1 }
	}
	option = { # No way
		name = peace_with_macawia.2.b

		FROM = {
			country_event = peace_with_macawia.4
		}

		ai_chance = { base = 0 }
	}
}
