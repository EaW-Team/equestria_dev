# Nuclear Attack on Griffonstone
news_event = {
	id = nuke_dropped.101
	title = nuke_dropped.101.t
	desc = nuke_dropped.101.d
	picture = GFX_news_event_nuke
	
	major = yes

	is_triggered_only = yes

	option = {
		name = nuke_dropped.101.a
		trigger = {
			NOT = { 
				owns_state = 389 
				tag = event_target:griffonstone_nuker
			}
		}
		hidden_effect = {
			event_target:griffonstone_nuker = {
				PREV = {
					add_ai_strategy = {
						type = antagonize
						id = PREV
						value = 500
					}
					add_ai_strategy = {
						type = contain
						id = PREV
						value = 500
					}
					add_ai_strategy = {
						type = befriend
						id = PREV
						value = -500
					}
					add_ai_strategy = {
						type = alliance
						id = PREV
						value = -500
					}
				}
			}
		}
		add_opinion_modifier = {
			target = event_target:griffonstone_nuker
			modifier = nuked_griffonstone
		}
		add_opinion_modifier = { 
			target = event_target:griffonstone_nuker 
			modifier = CHN_embargo 
		}
		every_other_country = {
			limit = {
				tag = event_target:griffonstone_nuker
			}
			add_opinion_modifier = { 
				target = ROOT 
				modifier = CHN_embargo 
			}
		}
	}
	
	option = {
		name = nuke_dropped.101.b
		trigger = {
			owns_state = 389
		}
		add_ideas = BOI_rage_on_griffonstone_nuker
		every_other_country = {
			limit = {
				NOT = {
					is_in_faction_with = event_target:griffonstone_nuker
					tag = event_target:griffonstone_nuker
				}
			}
			add_ai_strategy = {
				type = befriend
				id = ROOT
				value = 500
			}
			add_ai_strategy = {
				type = alliance
				id = ROOT
				value = 500
			}
			add_ai_strategy = {
				type = send_volunteers_desire
				id = ROOT
				value = 500
			}
			add_ai_strategy = {
				type = support
				id = ROOT
				value = 500
			}	
		}
	}
	option = {
		name = nuke_dropped.101.c
		trigger = {
			tag = event_target:griffonstone_nuker
		}
		custom_effect_tooltip = despicable_action
		add_political_power = -150
		add_stability = -0.10
		add_war_support = -0.10
		effect_tooltip = {
			FROM.FROM.OWNER = {
				add_ideas = BOI_rage_on_griffonstone_nuker
			}
		}
	}
}

# Nuclear Attack on Greifenheim
news_event = {
	id = nuke_dropped.102
	immediate = { log = "[GetDateText]: [Root.GetName]: event nuke_dropped.102" }
	title = nuke_dropped.102.t
	desc = {
		trigger = { 
		    From.From.Owner = {
		        tag = GRI
		    }
			NOT = { 
				has_global_flag = GRI_grover_vi_dead
				From.From.Owner = {
					has_country_leader = {
						name = "Emperor Grover VI"
						ruling_only = yes
					}
					has_focus_tree = imperial_grover_focus
				}
			}  
		}
		text = nuke_dropped.102.d.grover_alive
	}
	desc = {
		trigger = { 
		    OR = {
		        NOT = {
		            From.From.Owner = {
		                tag = GRI
		            }
		        }
		        AND = {
		            has_global_flag = GRI_grover_vi_dead }
                    From.From.Owner = {
		                tag = GRI
		            }
		        }
		    }
		text = nuke_dropped.102.d.grover_ded
	}
	desc = {
		trigger = { 
			From.From.Owner = {
			    tag = GRI
				OR = {
					has_country_leader = {
						name = "Emperor Grover VI"

					has_focus_tree = imperial_grover_focus
				} 
			}
		}
		text = nuke_dropped.102.d.grover_emperor
	}
	picture = GFX_news_event_nuke
	
	major = yes

	is_triggered_only = yes

	option = {
		name = nuke_dropped.102.a
