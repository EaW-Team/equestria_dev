	}

		has_variable = global.GRF_demilitarized_states^0

			var:v = {
				owner = {
					OR = {
						NOT = { tag = GRF }
						is_subject = yes
					}
				}
			}
		}
	}
	immediate = {
		set_temp_variable = { start_idx = global.GRF_demilitarized_states^num }
		subtract_from_temp_variable = { start_idx = 1 }
		for_loop_effect = {
			start = start_idx
			end = 0
			add = -1
			compare = greater_than_or_equals
			var:global.GRF_demilitarized_states^v = {
				if = {
					limit = {
							owner = {
								OR = {
									NOT = { tag = GRF }
									is_subject = yes
								}
							}
					}
					set_demilitarized_zone = no
					remove_from_array = { 
						array = global.GRF_demilitarized_states
						index = v
					}
				}
			}
		}
	}
}
