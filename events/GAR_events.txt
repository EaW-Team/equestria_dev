country_event = {
    id = GAR_initial_event_final.0
    title = GAR_initial_event_final.0.t
    desc =  GAR_initial_event_final.0.d

    is_triggered_only = yes

    option = {
        name = GAR_initial_event_final.0.a
        hidden_effect = {
            set_temp_variable = { temp_popularity_coup = GAR_clans_influence_array^NIG }
            add_to_temp_variable = { temp_popularity_coup = GAR_clans_influence_array^BLA }
            set_temp_variable = { temp_popularity_golian = GAR_clans_influence_array^WYV }
            add_to_temp_variable = { temp_popularity_golian = GAR_clans_influence_array^STO }
            if = {
                limit = {
                    check_variable = {
				        var = temp_popularity_golian
				        value = 60
				        compare = greater_than_or_equals
			        }
                }
                country_event = {
                    id = GAR_initial_event_final.1
                    days = 1
                }
            }
            else_if = {
                limit = {
                    has_completed_focus = GAR_dragars_reform_of_the_army
                }
                 country_event = {
                    id = GAR_initial_event_final.4
                    days = 1
                }
            }
            else_if = { # Demonia/Garik got enough influence
                limit = {
                    check_variable = {
				        var = temp_popularity_coup
				        value = 60
				        compare = greater_than_or_equals
			        }
                }
                country_event = {
                    id = GAR_initial_event_final.2
                    days = 1
                }
            }
            else = { # Noone got enough influence
                country_event = {
                    id = GAR_initial_event_final.3
                    days = 1
                }
            }
        }
    }
}
