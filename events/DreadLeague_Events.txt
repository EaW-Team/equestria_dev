country_event = {
	id = deadempire.29
	hidden = yes

	trigger = {
		country_exists = DED
		NOT = { has_war_with = DED }
		has_country_flag = capitulated_DED
	}

	mean_time_to_happen = { days = 1 }
	fire_only_once = yes

	immediate = {
		annex_country = { target = DED transfer_troops = no }
		every_state = {
			limit = { is_core_of = DED }
			remove_core_of = DED
		}
	}
	option = {}
}

country_event = {
    id = deadempire.30
  	hidden = yes

		is_triggered_only = yes
		immediate = {
			set_country_flag = DED_conversion_on
		}

    option = {
			if = {
				limit = { has_war = no }
				every_owned_state = {
					limit = {
						NOT = { is_core_of = ROOT }
					}
					decrease_population_necromancy = yes
				}
				else = {
					every_state = {
						limit = {
							NOT = { is_core_of = ROOT }
							OR = {
								is_owned_by = ROOT
								is_controlled_by = ROOT
							}
						}
						decrease_population_necromancy = yes
					}
				}
			}
			country_event = { id = deadempire.31 days = 30 }
		}

}

		}

		trigger = {
			tag = DED
			has_completed_focus = DED_Black_Crusade
			any_neighbor_country = {
				NOT = { has_war_with = ROOT }
			}
		}

    option = {
			every_neighbor_country = {

				ROOT = { declare_war_on = { target = PREV type = annex_everything } }
			}
		}

}

country_event = {
    id = deadempire.33
		title = deadempire.33.t
    desc = deadempire.33.d
		picture = GFX_badlands

		mean_time_to_happen = {
			days = 7
		}

		trigger = {
			NOT = { tag = DED }
			country_exists = DED
			OR = {
				is_major = yes
				is_faction_leader = yes
			}
			NOT = {
				has_annex_war_goal = DED
				has_war_with = DED
				has_country_flag = DED_wargoal_flag
			}
			DED = {
				has_completed_focus = DED_Black_Crusade
				has_country_flag = { flag = DED_killed_nations value > 9 }
			}
		}

    option = {
			declare_war_on = { target = DED type = annex_everything }
			hidden_effect = { set_country_flag = DED_wargoal_flag }
			ai_chance = { factor = 100 }
		}
		option = {
			add_political_power = -100
			hidden_effect = { set_country_flag = { flag = DED_wargoal_flag days = 180 } }
			ai_chance = { factor = 0 }
		}
}
