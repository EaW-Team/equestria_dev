### "Battle On The Ice" Focus Event Chain

country_event = {
	id = dread_start.8

	title = dread_start.8.t
	desc = {
		text = dread_start.8.descA
		trigger = {
			NOT = { has_country_flag = DED_Event_altered_plan }
			NOT = { has_country_flag = DED_Event_angered_corpus }
		}
	}
	desc = {
		text = dread_start.8.descB
		trigger = {
			NOT = { has_country_flag = DED_Event_altered_plan }
			has_country_flag = DED_Event_angered_corpus
		}
	}
	desc = {
		text = dread_start.8.descC
		trigger = {
			has_country_flag = DED_Event_altered_plan
		}
	}

	#picture = GFX_event_magehold

	is_triggered_only = yes

	option = {
		name = dread_start.8.a
		if = {
			limit = {
				NOT = { has_country_flag = DED_Event_altered_plan }
				NOT = { has_country_flag = DED_Event_angered_corpus }
			}
			hidden_effect = { country_event = { id = dread_start.801 days = 3 } }
		}
		else_if = {
			limit = {
				NOT = { has_country_flag = DED_Event_altered_plan }
				has_country_flag = DED_Event_angered_corpus
			}
			hidden_effect = { country_event = { id = dread_start.803 days = 3 } }
		}
		else = {
			hidden_effect = { country_event = { id = dread_start.805 days = 3 } }
		}
	}
	option = {
		name = dread_start.8.b
		if = {
			limit = {
				NOT = { has_country_flag = DED_Event_altered_plan }
				NOT = { has_country_flag = DED_Event_angered_corpus }
			}
			hidden_effect = { country_event = { id = dread_start.802 days = 3 } }
		}
		else_if = {
			limit = {
				NOT = { has_country_flag = DED_Event_altered_plan }
				has_country_flag = DED_Event_angered_corpus
			}
			hidden_effect = { country_event = { id = dread_start.804 days = 3 } }
		}
		else = {
			hidden_effect = { country_event = { id = dread_start.806 days = 3 } }
		}
	}
}
