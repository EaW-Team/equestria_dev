country_event = {
	id = bakawar.1
	immediate = {log = "[Root.GetName]: event bakawar.1"}#Auto-logging
	title = bakawar.1.t
	desc = bakawar.1.d
	picture = 

	trigger = {
		tag = BAK
		OR = {
			has_completed_focus = BAK_big_landair_four
			has_completed_focus = BAK_bbbc_navdoc
			has_completed_focus = BAK_shbb_stab
			has_completed_focus = BAK_bbbc_specops
			has_completed_focus = BAK_bbbc_carrierres
		}
		FBK = {
			NOT = {
				OR = {
					is_in_faction_with = BAK
					is_guaranteed_by = BAK
					has_non_aggression_pact_with = BAK
					is_puppet = yes
				}
			}
			has_war = yes
			surrender_progress > 0.7
		}
		OR = {
			BAK = {
				is_neighbor_of = FBK
			}
			any_country = {
				is_puppet_of = BAK
				is_neighbor_of = FBK
			}
		}
		NOT = {
			any_country = {
				OR = {
					AND = {
						is_in_faction_with = BAK
						has_war_with = FBK
					}
					AND = {
						is_puppet_of = BAK
						has_war_with = FBK
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 2
	}
	
	option = {
		name = bakawar.1.a
		FBK = {
			country_event = { days = 1 id = bakawar.2 }
		}
		ai_chance = {
			factor = 75
			modifier = {
				factor = 0
				OR = {
					has_war = yes
					manpower_per_military_factory < 1000
					num_divisions < 5
					has_war_support < 0.5
					has_stability < 0.65
					FBK = {
						alliance_strength_ratio > 0.5
					}
					any_country = {
						has_war_with = FBK
						alliance_strength_ratio > 0.65
					}
				}
			}
		}
	}
	option = {
		name = bakawar.1.b
		ai_chance = { factor = 25 }
	}
}
