	title = lsp_old.29.t
	desc = lsp_old.29.d

	is_triggered_only = yes

	option = {##Oh no!
		name = lsp_old.29.a
		if = {
			limit = {
				has_idea = LSP_fragile_food_supply
			}
			swap_ideas = {
				remove_idea = LSP_fragile_food_supply
				add_idea = LSP_famine
			}
		}
		else_if = {
			limit = {
				has_idea = LSP_famine
			}
			swap_ideas = {
				remove_idea = LSP_famine
				add_idea = LSP_severe_famine
			}
		}
		else = {
			add_ideas = LSP_fragile_food_supply
		}
	}

}
