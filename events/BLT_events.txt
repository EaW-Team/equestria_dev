country_event = {
	id = boltria.bad_dream.2
	title = boltria.bad_dream.t
	desc = boltria.bad_dream.2.d
	is_triggered_only = yes

	option = {#stay silent
		name = boltria.bad_dream.2.a
		hidden_effect = { country_event = boltria.bad_dream.3 }
	}
	option = {#fight back
		name = boltria.bad_dream.2.b
		set_temp_variable = {
			success_chance = 50
			tooltip = BLT_base_success_chance_tt
		}
		if = {
			limit = { BLT_boltrus_is_hardened = yes }
			add_to_temp_variable = {
				success_chance = 35
				tooltip = BLT_extra_success_chance_hardened_tt
			}
		}
		set_temp_variable = { fail_chance = 100 }
		subtract_from_temp_variable = { fail_chance = var:success_chance }
		random_list = {
			var:success_chance = {
				custom_effect_tooltip = boltria_roll_success_generic
				hidden_effect = { country_event = boltria.bad_dream.4 }
			}
			var:fail_chance = {
				custom_effect_tooltip = boltria_roll_failure_generic
				hidden_effect = { country_event = boltria.bad_dream.5 }
			}
		}
	}
	option = {#stay steady, remember the plan
		name = boltria.bad_dream.2.c
		trigger = { BLT_boltrus_is_conniving = yes }
		custom_effect_tooltip = BLT_boltrus_is_conniving_OP
		hidden_effect = { country_event = boltria.bad_dream.6 }
	}
}
