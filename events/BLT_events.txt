country_event = {#SLC win
	id = boltria.5
	title = boltria.5.t
	desc = boltria.5.d
	is_triggered_only = yes

	option = {
		name = boltria.5.a
		hidden_effect = {
			random = {
				chance = var:BLT_bevak_loyalty_value
				set_country_flag = BLT_bevak_loyalty_roll_success
			}
			if = {
				limit = { has_country_flag = BLT_bevak_loyalty_roll_success }
				country_event = { id = boltria.9 days = 3 }
			}
			else = {
				country_event = { id = boltria.bevak_loyalty_failure.1 days = 3 }
			}
			clr_country_flag = BLT_bevak_loyalty_roll_success
		}
	}
}
