add_namespace = cityfall
# The Fall of Vaverfront (Changelings)
news_event = {
	id = cityfall.1
	title = cityfall.1.t
	desc = cityfall.1.d
	picture = GFX_news_event_vanhoover_surrenders
	
	major = yes
	
	trigger = {
		69 = { is_controlled_by = CHN }
		69 = { is_owned_by = OLE }
		OLE = { has_war_with = CHN }
		NOT = { has_global_flag = fall_of_vaverfront_CHN }
	}
	
	immediate = {
		set_global_flag = fall_of_vaverfront_CHN
	}
	
	mean_time_to_happen = {
		days = 2
	}
	
	option = {
		name = cityfall.1.a
		trigger = {
			NOT = {
				TAG = CHN
				TAG = OLE
			}
		}
	}
	option = {
		name = cityfall.1.b
		trigger = { TAG = CHN }
	}
	option = {
		name = cityfall.1.c
		trigger = { TAG = OLE }
	}
}
