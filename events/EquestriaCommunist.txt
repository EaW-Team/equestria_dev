country_event = { #### Communist Civil War
	id = eqscom.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event eqscom.1"}
	title = eqscom.1.t
	desc = eqscom.1.d
	picture = GFX_event_eqs_communists

	trigger = {
		tag = EQS
		NOT = { has_government = communism }
		communism > 0.5
		has_country_flag = stalliongradian_influence
		has_defensive_war = no
		NOT = { has_global_flag = the_war_ended }
	}
	is_triggered_only = yes
	#mean_time_to_happen = { days = 1 }
	fire_only_once = yes

	option = {
		name = eqscom.1.a
		trigger = { is_ai = no }
		hidden_effect = {
			set_capital = { 
				state = 75
				remember_old_capital = no
			}
			start_civil_war = {
				ruling_party = communism
				ideology = democratic
				size = 0.5
				capital = 3
			}
			random_other_country = {
				limit = {
					original_tag = EQS
					has_government = democratic
				}
				#load_oob = "EQS_DEMCOM_Civil_War"
				set_stability = 0.6
				add_ideas = communist_civil_war
				add_popularity = {
					ideology = democratic
					popularity = 0.35
				}
			}
			set_global_flag = equestrian_com_civil_war

			#load_oob = "EQSCOM_Civil_War"
			set_stability = 0.6
			add_ideas = communist_civil_war
			add_popularity = {
				ideology = communism
				popularity = 0.5
			}
			add_ideas = celestial_resistance

			#country_event = { id = spain.2 days = 7 }
		}
		custom_effect_tooltip = communists_chosen
	}
	option = {
		name = eqscom.1.b
		trigger = { is_ai = no }
		hidden_effect = {
			set_capital = { 
				state = 3 
				remember_old_capital = no
			}
			start_civil_war = {
				ruling_party = democratic
				ideology = communism
				size = 0.5
				capital = 75
			}
			random_other_country = {
				limit = {
					original_tag = EQS
					has_government = communism
				}
				add_ideas = communist_civil_war
				#load_oob = "EQSCOM_Civil_War"
				set_equipment_fraction = 0.5
				set_stability = 0.6
				add_popularity = {
					ideology = communism
					popularity = 0.4
				}
			}
			set_global_flag = equestrian_com_civil_war
			add_ideas = communist_civil_war
			#load_oob = "EQS_DEMCOM_Civil_War"
			set_equipment_fraction = 0.5
			set_stability = 0.6
			add_popularity = {
				ideology = democratic
				popularity = 0.4
			}

			#country_event = { id = spain.2 days = 7 }
		}
		custom_effect_tooltip = celestia_chosen
	}

}
