country_event = {
	id = quaggatai_diplo.1
	immediate = {log = "[Root.GetName]: event quaggatai_diplo.1"}
	title = quaggatai_diplo.1.t
	desc = quaggatai_diplo.1.d
	picture = GFX_event_zebrica_6
	
	is_triggered_only = yes
	
	option = { # Accept
		name = quaggatai_diplo.1.a
		
		add_political_power = 25
		
		CGI = {
			country_event = {
				id = quaggatai_diplo.2
				hours = 3
			}
		}
		
		ai_chance = {
			base = 10
			
			modifier = {
				factor = 0
				has_war_with = CGI
			}
			modifier = {
				factor = 5
				has_opinion = {
					target = CGI
					value > 0
				}
			}
		}
	}
	
	option = { # Refuse
		name = quaggatai_diplo.1.b
		
		CGI = {
			country_event = {
				id = quaggatai_diplo.3
				hours = 3
			}
		}
		
		ai_chance = {
			base = 5
		}
	}
}

# Trade Deal yay
country_event = {
	id = quaggatai_diplo.2
	immediate = {log = "[Root.GetName]: event quaggatai_diplo.2"}
	title = quaggatai_diplo.2.t
	desc = quaggatai_diplo.2.d
	picture = GFX_event_zebrica_6
	
	is_triggered_only = yes
	
	option = { # Good
		name = quaggatai_diplo.2.a
		
		if = {
			limit = {
				FROM = {
					tag = CVA
				}
			}
			add_to_variable = {
				var = CGI_diamond_road_strength
				value = 3
			}
			custom_effect_tooltip = {
				localization_key = CGI_diamond_road_add
				VALUE = "3"
				POINT_POINTS = CGI_points
			}
			CGI_update_diamond_road = yes
			set_country_flag = CGI_diamond_road_CVA_trade_deal
			custom_effect_tooltip = CGI_disable_mission_to_colthage
		}
		else = {
			add_to_variable = {
				var = CGI_diamond_road_strength
				value = 4
			}
			custom_effect_tooltip = {
				localization_key = CGI_diamond_road_add
				VALUE = "4"
				POINT_POINTS = CGI_points
			}
			CGI_update_diamond_road = yes
			set_country_flag = CGI_diamond_road_CTH_trade_deal
			custom_effect_tooltip = CGI_disable_contacts_with_coltvan_resistance
		}
	}
}
