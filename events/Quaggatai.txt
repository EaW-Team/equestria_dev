country_event = {
	id = quaggatai_camelia.10
	immediate = {log = "[Root.GetName]: event quaggatai_camelia.10"}
	title = quaggatai_camelia.10.t
	desc = quaggatai_camelia.10.d
	picture = GFX_event_zebrica_6
	
	is_triggered_only = yes
	fire_only_once = yes
	
	option = {
		name = quaggatai_camelia.10.a
		
		# TODO: Add effect for more situations
		
		set_temp_variable = { diamond_road_buff = 7 }
		set_temp_variable = { pp_gain = 50 }
		set_temp_variable = { army_xp_gain = 0 }
		set_temp_variable = { manpower_gain = 0 }
		
		if = {
			limit = {
				has_country_flag = CGI_camelia_has_send_military_mission
			}
			add_to_temp_variable = { diamond_road_buff = 2 }
			add_to_temp_variable = { pp_gain = 25 }
			add_to_temp_variable = { army_xp_gain = 25 }
		}
		if = {
			limit = {
				has_country_flag = CGI_camelia_has_send_volunteer_batallion
			}
			add_to_temp_variable = { diamond_road_buff = 1 }
			add_to_temp_variable = { army_xp_gain = 15 }
			add_to_temp_variable = { manpower_gain = 2500 } # Half the send menpower
		}
		if = {
			limit = {
				has_country_flag = CGI_camelia_has_send_guns
			}
			add_to_temp_variable = { diamond_road_buff = 1 }
		}
		
		add_political_power = pp_gain
		if = {
			limit = {
				check_temp_variable = { army_xp_gain > 0 }
			}
			army_experience = army_xp_gain
		}
		if = {
			limit = {
				check_temp_variable = { manpower_gain > 0 }
			}
			add_manpower = manpower_gain
		}
		
		add_to_variable = {
			var = CGI_diamond_road_strength
			value = diamond_road_buff
		}
		custom_effect_tooltip = {
			localization_key = CGI_diamond_road_add
			VALUE = "[?diamond_road_buff]"
			POINT_POINTS = CGI_points
		}
		CGI_update_diamond_road = yes
	}
}
