country_event = {
	id = bronzehill_council.1
	immediate = {log = "[Root.GetName]: event bronzehill_council.1"}#Auto-logging
	title = bronzehill_council.1.t
	
	desc = {
		text = bronzehill_council.1.d
		trigger = {
			has_country_flag = BRZ_softfur_in_the_council
		}
	}
	desc = {
		text = bronzehill_council.1.d_no_lux
	}
	
	picture = GFX_event_griffon_back
	
	is_triggered_only = yes
	fire_only_once = yes
	
	immediate = {
		hidden_effect = {
			country_event = {
				id = bronzehill_council.4
				days = 30
			}
			country_event = {
				id = bronzehill_council.5
				days = 65
			}
			country_event = {
				id = bronzehill_council.6
				days = 100
			}
			country_event = {
				id = bronzehill_council.7
				days = 140
			}
			country_event = {
				id = bronzehill_council.8
				days = 180
			}
			
			add_ideas = BRZ_powerstruggle_in_the_council
			
			activate_mission = BRZ_council_timeout_mission
			
			activate_mission = BRZ_copperhill_influence_mission
			activate_mission = BRZ_loudbark_influence_mission
			if = {
				limit = {
					has_country_flag = BRZ_softfur_in_the_council
				}
				activate_mission = BRZ_softfur_influence_mission
			} else = {
				activate_mission = BRZ_steelpaw_influence_mission
			}
			
			# Initialize the variables for powerstruggle
			set_variable = {
				BRZ_council_powerstruggle_influence_copperhill = 20
			}
			set_variable = {
				BRZ_council_powerstruggle_influence_loudbark = 20
			}
			if = {
				limit = {
					has_country_flag = BRZ_softfur_in_the_council
				}
				set_variable = {
					BRZ_council_powerstruggle_influence_softfur = 3
				}
			} else = {
				set_variable = {
					BRZ_council_powerstruggle_influence_softfur = 0
				}
			}
			if = {
				limit = {
					has_country_flag = BRZ_softfur_in_the_council
				}
				set_variable = {
					BRZ_council_powerstruggle_radicalization = 0
				}
			} else = {
				set_variable = {
					BRZ_council_powerstruggle_radicalization = 20
				}
			}
		}
	}
	
	option = {
		name = bronzehill_council.1.a
		
		effect_tooltip = {
			add_ideas = BRZ_powerstruggle_in_the_council
			
			activate_mission = BRZ_council_timeout_mission
			
			activate_mission = BRZ_copperhill_influence_mission
			activate_mission = BRZ_loudbark_influence_mission
			if = {
				limit = {
					has_country_flag = BRZ_softfur_in_the_council
				}
				activate_mission = BRZ_softfur_influence_mission
			} else = {
				activate_mission = BRZ_steelpaw_influence_mission
			}
		}
	}
}

					var = temp_copperhill_loudbark_comparison

					var = temp_copperhill_softfur_comparison

			}

# Hidden poll drift (weekly)
country_event = {
	id = bronzehill_barnaby.6
	immediate = {log = "[Root.GetName]: event bronzehill_barnaby.6"}#Auto-logging
	title = bronzehill_barnaby.6.t
	desc = bronzehill_barnaby.6.d
	picture = GFX_event_griffon_back
	
	hidden = yes
	
	is_triggered_only = yes
	
	option = {
		name = bronzehill_barnaby.6.a
		
		every_core_state = {
			
			if = {
				limit = {
					has_dynamic_modifier = {
						modifier = BRZ_local_party_support
						scope = THIS
					}
				}
				random_list = {
					3 = {
						# Nothing happens
					}
					5 = {
						# Reduce regionalists, if present
						if = {
							limit = {
								check_variable = { BRZ_local_party_support^6 > 0.05 }
							}
							set_temp_variable = { BRZ_party_boost_buffer = -0.02 }
							set_temp_variable = { BRZ_party_boost_selector = 6 }
							BRZ_add_to_party_absolute = yes # reductions don't really work otherwise
						}
						
					}
					2 = { # Random drift
						random_list = {
							var:BRZ_state_is_rural = {
								set_temp_variable = { BRZ_party_boost_buffer = 0.02 }
								set_temp_variable = { BRZ_party_boost_selector = 1 } # Agrarian Conservatives
								BRZ_add_to_party = yes
							}
							var:BRZ_state_is_rural = {
								set_temp_variable = { BRZ_party_boost_buffer = 0.02 }
								set_temp_variable = { BRZ_party_boost_selector = 2 } # Agrarian Socialists
								BRZ_add_to_party = yes
							}
							2 = {
								set_temp_variable = { BRZ_party_boost_buffer = 0.02 }
								set_temp_variable = { BRZ_party_boost_selector = 0 } # Conservatives
								BRZ_add_to_party = yes
							}
							2 = {
								set_temp_variable = { BRZ_party_boost_buffer = 0.02 }
								set_temp_variable = { BRZ_party_boost_selector = 4 } # Socialist
								BRZ_add_to_party = yes
							}
							1 = {
								set_temp_variable = { BRZ_party_boost_buffer = 0.02 }
								set_temp_variable = { BRZ_party_boost_selector = 3 } # Liberals
								BRZ_add_to_party = yes
							}
						}
					}
				}
			}
			else = {
				BRZ_add_initial_party_support = yes
				
				add_dynamic_modifier = {
					modifier = BRZ_local_party_support
					scope = THIS
				}
			}
			
			# Update local seats
			set_variable = { BRZ_local_state_seats = state_population_k }
			divide_variable = { BRZ_local_state_seats = 100 }
			round_variable = BRZ_local_state_seats
		}
		
		country_event = {
			id = bronzehill_barnaby.6
			days = 7
		}
	}
}

country_event = {
	id = bronzehill_barnaby.28
	immediate = {log = "[Root.GetName]: event bronzehill_barnaby.28"}#Auto-logging
	title = bronzehill_barnaby.28.t
	desc = bronzehill_barnaby.28.d
	picture = GFX_event_griffon_back
	
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = {
				has_country_flag = BRZ_conservative_cabinet
			}
			
			hidden_effect = {
				# recruit dawnfeather
				if = {
					limit = {
						NOT = { has_country_flag = BRZ_dawnfeather_advisor }
					}
					every_possible_country = {
						limit = {
							has_character = PYT_archibald_dawnfeather
						}
						PYT_archibald_dawnfeather = {
							set_nationality = ROOT
							add_advisor_role = {
								advisor = {
									slot = political_advisor
									cost = 150
									idea_token = BRZ_archibald_dawnfeather
									traits = { BRZ_friendship_promoter }
								}
							}
						}
					}
					set_country_flag = BRZ_dawnfeather_advisor
				}
				# recruit attaboy
				if = {
					limit = {
						NOT = { has_country_flag = BRZ_attaboy_advisor }
					}
					every_possible_country = {
						limit = {
							has_character = BRZ_clement_attaboy
						}
						BRZ_clement_attaboy = {
							set_nationality = ROOT
							add_advisor_role = {
								advisor = {
									slot = political_advisor
									cost = 150
									idea_token = BRZ_clement_attaboy
									traits = { welfare_reformer }
								}
							}
						}
					}
					set_country_flag = BRZ_attaboy_advisor
				}
			}
		}
		else_if = {	
			limit = {
				has_country_flag = BRZ_socialist_cabinet
			}
			hidden_effect = {
				# recruit schüler
				if = {
					limit = {
						NOT = { has_country_flag = BRZ_schuler_advisor }
					}
					every_possible_country = {
						limit = {
							has_character = YAL_georgina_fredrika_schuler
						}
						YAL_georgina_fredrika_schuler = {
							set_nationality = ROOT
							add_advisor_role = {
								advisor = {
									slot = political_advisor
									cost = 150
									idea_token = BRZ_georgina_fredrika_schuler
									traits = { YAL_Social_Democrat_chick_leader }
								}
							}
						}
					}
					set_country_flag = BRZ_schuler_advisor
				}
				# make softfur an advisor
				if = {
					limit = {
						NOT = { has_country_flag = BRZ_softfur_advisor }
					}
					every_possible_country = {
						limit = {
							has_character = BRZ_rosey_softfur
						}
						BRZ_rosey_softfur = {
							set_nationality = ROOT
							add_advisor_role = {
								advisor = {
									slot = political_advisor
									cost = 150
									idea_token = BRZ_rosey_softfur_advisor
									traits = { BRZ_socialist_theorist }
								}
							}
						}
					}
					set_country_flag = BRZ_softfur_advisor
				}
			}
		}
	}
	
	option = {
		name = bronzehill_barnaby.28.a
		
		if = {
			limit = {
				has_idea = BRZ_winston_copperhill
			}
			
			custom_effect_tooltip = available_political_advisor
			show_ideas_tooltip = BRZ_clement_attaboy
			
			custom_effect_tooltip = available_political_advisor
			show_ideas_tooltip = BRZ_archibald_dawnfeather
		}
		else_if = {	
			limit = {
				OR = {
					has_idea = BRZ_genevieve_guildedwing
					has_idea = BRZ_georgina_fredrika_schuler
				}
			}
		
			custom_effect_tooltip = available_political_advisor
			show_ideas_tooltip = BRZ_georgina_fredrika_schuler
			
			custom_effect_tooltip = available_political_advisor
			show_ideas_tooltip = BRZ_rosey_softfur_advisor
		}
	}
}
