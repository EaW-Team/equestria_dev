country_event = {
	id = bronzehill_council.1
	immediate = {log = "[Root.GetName]: event bronzehill_council.1"}
	title = bronzehill_council.1.t
	
	desc = {
		text = bronzehill_council.1.d
		trigger = {
			has_country_flag = BRZ_softfur_in_the_council
		}
	}
	desc = {
		text = bronzehill_council.1.d_no_lux
	}
	
	picture = GFX_event_griffon_back
	
	is_triggered_only = yes
	fire_only_once = yes
	
	immediate = {
		hidden_effect = {
			country_event = {
				id = bronzehill_council.4
				days = 30
			}
			country_event = {
				id = bronzehill_council.5
				days = 65
			}
			country_event = {
				id = bronzehill_council.6
				days = 100
			}
			country_event = {
				id = bronzehill_council.7
				days = 140
			}
			country_event = {
				id = bronzehill_council.8
				days = 180
			}
			
			add_ideas = BRZ_powerstruggle_in_the_council
			
			activate_mission = BRZ_council_timeout_mission
			
			activate_mission = BRZ_copperhill_influence_mission
			activate_mission = BRZ_loudbark_influence_mission
			if = {
				limit = {
					has_country_flag = BRZ_softfur_in_the_council
				}
				activate_mission = BRZ_softfur_influence_mission
			} else = {
				activate_mission = BRZ_steelpaw_influence_mission
			}
			
			# Initialize the variables for powerstruggle
			set_variable = {
				BRZ_council_powerstruggle_influence_copperhill = 20
			}
			set_variable = {
				BRZ_council_powerstruggle_influence_loudbark = 20
			}
			if = {
				limit = {
					has_country_flag = BRZ_softfur_in_the_council
				}
				set_variable = {
					BRZ_council_powerstruggle_influence_softfur = 3
				}
			} else = {
				set_variable = {
					BRZ_council_powerstruggle_influence_softfur = 0
				}
			}
			if = {
				limit = {
					has_country_flag = BRZ_softfur_in_the_council
				}
				set_variable = {
					BRZ_council_powerstruggle_radicalization = 0
				}
			} else = {
				set_variable = {
					BRZ_council_powerstruggle_radicalization = 20
				}
			}
		}
	}
	
	option = {
		name = bronzehill_council.1.a
		
		effect_tooltip = {
			add_ideas = BRZ_powerstruggle_in_the_council
			
			activate_mission = BRZ_council_timeout_mission
			
			activate_mission = BRZ_copperhill_influence_mission
			activate_mission = BRZ_loudbark_influence_mission
			if = {
				limit = {
					has_country_flag = BRZ_softfur_in_the_council
				}
				activate_mission = BRZ_softfur_influence_mission
			} else = {
				activate_mission = BRZ_steelpaw_influence_mission
			}
		}
	}
}

	picture = GFX_event_griffon_back
	
	is_triggered_only = yes
	
	fire_only_once = yes
	
	immediate = { # Calculate relative scores
		hidden_effect = {
			BRZ = {
				set_variable = {
					var = temp_copperhill_loudbark_comparison 
					value = BRZ_council_powerstruggle_influence_copperhill
				}
				subtract_from_variable = {
					var = temp_copperhill_loudbark_comparison
					value = BRZ_council_powerstruggle_influence_loudbark
				}
				set_variable = {
					var = temp_copperhill_softfur_comparison
					value = BRZ_council_powerstruggle_influence_copperhill
				}
				subtract_from_variable = {
					var = temp_copperhill_softfur_comparison
					value = BRZ_council_powerstruggle_influence_softfur
				}
				set_variable = {
					var = temp_loudbark_softfur_comparison
					value = BRZ_council_powerstruggle_influence_loudbark
				}
				subtract_from_variable = {
					var = temp_loudbark_softfur_comparison
					value = BRZ_council_powerstruggle_influence_softfur
				}
			}
			
			if = {
				limit = {
					is_subject_of = GRI
				}
				GRI = {
					end_puppet = ROOT
				}
			}
			if = {

				}

			}

				type = befriend

			}

			}

				id = bronzehill_news.1
				days = 2
			}
		}
	}

			check_variable = { temp_copperhill_loudbark_comparison > 0 } # Copperhill wins over Loudbark
			check_variable = { temp_copperhill_softfur_comparison > 0 } # Copperhill wins over Softfur
		}
		
		name = bronzehill_council.2.a
		
		# Remove any active council struggle missions
		if = {
			limit = {
				has_active_mission = BRZ_council_timeout_mission
			}
			remove_mission = BRZ_council_timeout_mission
		}
		if = {
			limit = {
				has_active_mission = BRZ_copperhill_influence_mission
			}
			remove_mission = BRZ_copperhill_influence_mission
		}
		if = {
			limit = {
				has_active_mission = BRZ_loudbark_influence_mission
			}
			remove_mission = BRZ_loudbark_influence_mission
		}
		if = {
			limit = {
				has_active_mission = BRZ_softfur_influence_mission
			}
			remove_mission = BRZ_softfur_influence_mission
		}
		if = {
			limit = {
				has_active_mission = BRZ_steelpaw_influence_mission
			}
			remove_mission = BRZ_steelpaw_influence_mission
		}

# Hidden poll drift + public trust mechanic (weekly)
country_event = {
	id = bronzehill_barnaby.6
	immediate = {log = "[Root.GetName]: event bronzehill_barnaby.6"}
	title = bronzehill_barnaby.6.t
	desc = bronzehill_barnaby.6.d
	picture = GFX_event_griffon_back
	
	hidden = yes
	
	is_triggered_only = yes
	
	option = {
		name = bronzehill_barnaby.6.a
		
		every_core_state = {
			
			if = {
				limit = {
					has_dynamic_modifier = {
						modifier = BRZ_local_party_support
						scope = THIS
					}
				}
				random_list = {
					3 = {
						# Nothing happens
					}
					5 = {
						# Reduce regionalists, if present
						if = {
							limit = {
								check_variable = { BRZ_local_party_support^6 > 0.05 }
							}
							set_temp_variable = { BRZ_party_boost_buffer = -0.02 }
							set_temp_variable = { BRZ_party_boost_selector = 6 }
							BRZ_add_to_party_absolute = yes # reductions don't really work otherwise
						}
						
					}
					2 = { # Random drift
						random_list = {
							var:BRZ_state_is_rural = {
								set_temp_variable = { BRZ_party_boost_buffer = 0.02 }
								set_temp_variable = { BRZ_party_boost_selector = 1 } # Agrarian Conservatives
								BRZ_add_to_party = yes
							}
							var:BRZ_state_is_rural = {
								set_temp_variable = { BRZ_party_boost_buffer = 0.02 }
								set_temp_variable = { BRZ_party_boost_selector = 2 } # Agrarian Socialists
								BRZ_add_to_party = yes
							}
							2 = {
								set_temp_variable = { BRZ_party_boost_buffer = 0.02 }
								set_temp_variable = { BRZ_party_boost_selector = 0 } # Conservatives
								BRZ_add_to_party = yes
							}
							2 = {
								set_temp_variable = { BRZ_party_boost_buffer = 0.02 }
								set_temp_variable = { BRZ_party_boost_selector = 4 } # Socialist
								BRZ_add_to_party = yes
							}
							1 = {
								set_temp_variable = { BRZ_party_boost_buffer = 0.02 }
								set_temp_variable = { BRZ_party_boost_selector = 3 } # Liberals
								BRZ_add_to_party = yes
							}
						}
					}
				}
			}
			else = {
				BRZ_add_initial_party_support = yes
				
				add_dynamic_modifier = {
					modifier = BRZ_local_party_support
					scope = THIS
				}
			}
			
			# Update local seats
			set_variable = { BRZ_local_state_seats = state_population_k }
			divide_variable = { BRZ_local_state_seats = 100 }
			round_variable = BRZ_local_state_seats
		}
		
		if = {
			limit = {
				has_country_flag = BRZ_barnaby_public_trust_active
			}
			subtract_from_variable = {
				BRZ_barnaby_public_trust = BRZ_barnaby_public_trust_decay
			}
			if = {
				limit = { 
					check_variable = { BRZ_barnaby_public_trust < 0 }
				}
				country_event = bronzehill_barnaby.35
				set_variable = { BRZ_barnaby_public_trust = 0 }
			}
			if = {
				limit = {
					check_variable = {
						BRZ_promote_constitutionalism_cost > 50
					}
				}
				subtract_from_variable = {
					BRZ_promote_constitutionalism_cost = 1
				}
			}
		}
		
		country_event = {
			id = bronzehill_barnaby.6
			days = 7
		}
	}
}
