#############################################################
country_event = {
	id = lakecityflavor.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event lakecityflavor.1"}
	title = lakecityflavor.1.t
	desc = lakecityflavor.1.d
	picture = GFX_event_lake_city_city3
	is_triggered_only = yes

	# the holiday is upon us!
	option = {
		name = lakecityflavor.1.a
		trigger = {
			has_war = no
		}
		hidden_effect = {
			country_event = {
				id = lakecityflavor.2
				days = 20
			}
		}
	}

	# "This event is a waste of time" #never fires again#
	option = {
		name = lakecityflavor.1.b
		trigger = {
			has_war = no
		}
		country_event = {
			id = lakecityflavor.3
			days = 2
		}
	}

	# Can't fire the event during war.
	option = {
		name = lakecityflavor.1.c
		trigger = {
			has_war = yes
		}
		country_event = {
			id = lakecityflavor.1
			days = 380
		}
	}
}

	desc = lakecityflavor.4.d
	picture = GFX_event_boat_in_ruined_dock
	fire_only_once = yes

	trigger = {
		tag = LCT
		has_stability < 0.3
		has_war = no
	}

	mean_time_to_happen = {
		days = 60
	}

	# investigate
	option = {
		name = lakecityflavor.4.a

		hidden_effect = {
			country_event = {
				id = lakecityflavor.5
				days = 30
				random = 720 #days to hours applied
			}
		}
	}

	# ignore
	option = {
		name = lakecityflavor.4.b
		add_stability = -0.05
	}
}

# prince's birthday trigger event
country_event = {
	id = lakecityflavor.12
	immediate = {log = "[GetDateText]: [Root.GetName]: event lakecityflavor.12"}

	hidden = yes
	is_triggered_only = yes

	option = {
		trigger = {
			has_country_leader = {
				character = LCT_heavenly_snow
				ruling_only = yes
			}
		}
		if = {
			limit = { NOT = { has_country_flag = hide_birthdays } }
			country_event = lakecityflavor.13
		}
		else = {
			add_political_power = 5
			add_stability = 0.01
		}
	}

	option = {
		trigger = {
			NOT = {
				has_country_leader = {
					character = LCT_heavenly_snow
					ruling_only = yes
				}
			}
		}
		#yes, do nothing, this is intended
	}
}
