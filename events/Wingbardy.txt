				has_idea = WNG_garibald_talonuel_iii
			}

			add_ideas = WNG_gumberto_ii
		}

		if = {
			limit = {
				has_completed_focus = WNG_absolute_rule
			}
			hidden_effect = {
				country_event = wingbardy_gumberto.29
