# Class Conflicts Divide [Root.GetName]
country_event = {
	id = political.4
	immediate = {log = "[GetDateText]: [Root.GetName]: event political.4"}
	title = political.4.t
	desc = political.4.d
	picture = GFX_event_eqs_communists

	trigger = {
		has_idea_with_trait = communist_revolutionary
		NOT = { has_government = communism }
		communism > 0.2
		check_has_focus_tree_to_switch_to_communism = yes
	}

	mean_time_to_happen = {
		days = 365
		modifier = {
			factor = 0.5
			has_country_flag = communism_popular_movement
		}
	}

	option = {
		name = political.4.a
		add_popularity = {
			ideology = communism
			popularity = 0.05
		}
		add_stability = -0.05
	}
}
