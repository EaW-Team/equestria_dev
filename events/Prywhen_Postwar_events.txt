		has_country_flag = accepted_prywhen_communists

		days = 250
		modifier = {
			factor = 3
			communism < 0.2
		}
		modifier = {
			factor = 2
			communism < 0.3
		}
		modifier = {
			factor = 0.6
			communism < 0.45
		}
		modifier = {
			factor = 0.45
			communism < 0.5
		}
		modifier = {
			factor = 0.35

		}

	option = {
		name = griffonwill.3.a
		remove_ideas = GRW_prywhen_communism_support

			limit = { original_tag = GRY }

		}
		else_if = {
			limit = { has_focus_tree = generic_griffon_focus }
			complete_national_focus = griffon_political_effort
			complete_national_focus = griffon_dreams_of_a_republic
			complete_national_focus = griffon_gather_support
			complete_national_focus = griffon_popular_revolution
		}
		start_civil_war = { ruling_party = communism ideology = ROOT size = 0.5 }
		if = {
			limit = { original_tag = LUS }
			load_focus_tree = Lushi_peasants_focus
		}
		add_popularity = { ideology = communism popularity = 0.5 }
		set_country_flag = civil_war_happened
		add_opinion_modifier = {
			modifier = common_cause target = GRW
		}
		GRW = {
			add_opinion_modifier = {
				modifier = common_cause target = ROOT

		}

			STG = { create_production_license = { target = GRW cost_factor = 0 equipment = { type = light_tank_equipment_1 } } }
		}
		else = {
			STG = { create_production_license = { target = GRW cost_factor = 0 equipment = { type = gw_tank_equipment } } }
		}
		if = {
			limit = {
				NOT = {
					has_tech = gwtank
					has_tech = gwtank_chassis
				}
			}
			if = {
				limit = {
					has_dlc = "No Step Back"
				}
				set_technology = {
					gwtank_chassis = 1
				}
			}
			else = {			
				set_technology = {
					gwtank = 1
				}
			}
			add_tech_bonus = {
				name = armor_bonus
				bonus = 1
				uses = 1
				category = armor
			}
		}
		else = {
			add_tech_bonus = {
				name = armor_bonus
				bonus = 2
				uses = 1
				category = armor
			}
		}
		add_equipment_to_stockpile = { type = light_tank_chassis amount = 1 producer = STG }
