  hidden = yes

	trigger = {
		original_tag = FAW
		NOT = { has_government = fascism }

			original_tag = FAT
			exists = yes
			NOT = { has_war_with = FAW }
			has_government = ROOT
		}
	}

	mean_time_to_happen = { days = 2 }

  option = {
		random_other_country = {
			limit = {
				original_tag = FAT
				exists = yes
				NOT = { has_war_with = FAW }
				has_government = ROOT
			}
			if = {
				limit = { is_ai = yes }
				change_tag_from = ROOT
			}
			annex_country = { target = ROOT transfer_troops = yes }
		}
  }
}
