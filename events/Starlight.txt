country_event = { ## Who becomes new leader?
	id = starlight.10
	immediate = {log = "[GetDateText]: [Root.GetName]: event starlight.10"}
	title = starlight.10.t
	desc = starlight.10.d
	picture = GFX_event_our_town

	is_triggered_only = yes
	#I'll set up the triggers and code later, this is for testing
	option = { # Feather Bangs wins
		name = starlight.10.a
		set_politics = {
			ruling_party = neutrality
			elections_allowed = no
		}
		set_cosmetic_tag = STL_MY
		load_focus_tree = stl_feather_bangs
	}
	option = { # Party Favor wins
		name = starlight.10.b

		create_country_leader = {
			name = "Party Favor"

			ideology = leninism
			traits = {
				STL_stg_simp
			}
		}
		set_cosmetic_tag = STL_OUR

			ideology = communism
			name = STL_stallionist_party
			long_name = STL_stallionist_party_long
		}
		load_focus_tree = stl_party_favor
	}
	option = { # +Double Diamond wins
		name = starlight.10.c
		load_focus_tree = stl_double_diamond
	}
	option = { # Night Glider wins

			}
		}
