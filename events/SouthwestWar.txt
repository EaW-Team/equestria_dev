add_namespace = southwest

country_event = {
	id = southwest.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event southwest.1"}
	title = southwest.1.t
	desc = southwest.1.d

	is_triggered_only = yes

	option = {
		name = southwest.1.a
		if = {
			limit = {
				tag = LSP
				is_ai = no
				SCS = {
					is_ai = yes
				}
			}
			create_faction = SCS_LSP_faction
			add_to_faction = SCS
		}
		if = {
			limit = {
				tag = SCS
			}
			if = {
				limit = {
					OR = {
						LSP = {
							is_ai = yes
						}
						AND = {
							LSP = {
								is_ai = no
							}
							SCS = {
								is_ai = no
							}
						}
					}
				}
				create_faction = SCS_LSP_faction
				add_to_faction = LSP
			}
		}
	}
}
