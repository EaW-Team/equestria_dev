news_event = { #River cup 1007
	id = river_cup.1
	immediate = {log = "[Root.GetName]: event river_cup.1"}#Auto-logging
	title = river_cup.1.t
	desc = river_cup.1.d
	picture = GFX_news_event_friendship_games

	is_triggered_only = yes

	option = {
		name = river_cup.1.a
		trigger = {
			owns_state = event_target:rc_winner
		}
		add_political_power = 10
	}

	option = {
		name = river_cup.1.b
		trigger = {
			owns_state = event_target:rc_runner_up
		}
		add_political_power = 5
	}

	option = {
		name = river_cup.1.c
		trigger = {
			is_river_fed_member_or_RivFed = yes
			NOT = {
				owns_state = event_target:rc_winner
				owns_state = event_target:rc_runner_up
			}
		}
	}

	option = {
		name = river_cup.1.e
		trigger = {
			NOT = {
				is_river_fed_member_or_RivFed = yes
			}
		}
	}

}
