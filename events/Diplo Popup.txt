# Faction invite
country_event = {
	id = diplopopup.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event diplopopup.1"}
	hidden = yes
	is_triggered_only = yes
	trigger = {
		FROM = { is_in_faction_with = FROM.FROM }
		country_exists = FROM
		country_exists = FROM.FROM
	}
	option = {
		every_country = {
			limit = {
				is_ai = no
				NOT = {
					tag = FROM
					tag = FROM.FROM
				}
			}
			show_faction_invited_popup = yes
		}
	}
}

# Event to automatically make popup go away in three days
country_event = {
	id = diplopopup.99
	immediate = {
		log = "[GetDateText]: [ROOT.GetName]: event diplopopup.99"
		for_each_loop = {
			array = eaw_diplo_popup_unique_id
			value = v
			break = b
			if = {
				limit = {
					event_target:unique_id = {
						state = v
					}
				}
				set_temp_variable = { unique_id_found = 1 }
				set_temp_variable = { current_index = i }
				set_temp_variable = { b = 1 }
			}
		}
		if = {
			limit = {
				check_variable = { unique_id_found = 1 }
			}
			set_temp_variable = { close_index = current_index }
			eaw_diplo_popup_close_popup = yes
		}
	}
	is_triggered_only = yes
	hidden = yes
}
