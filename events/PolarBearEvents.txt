country_event = { ## Loyalty Switch Hunter bad half
	id = polar.1001
	hidden = yes
	trigger = {
		tag = PLB
		check_variable = { hunter_loyalty < 40 }
		check_variable = { hunter_loyalty > 21 }
		NOT = { has_idea = PLB_LOY_Hunter_bad_half }
	}
	is_triggered_only = yes
	option = {
		name = polar.1001.a
		add_ideas = PLB_LOY_Hunter_bad_half
		if ={
			limit = { has_idea = PLB_LOY_Hunter_good }
			remove_ideas = { PLB_LOY_Hunter_good }
		}
		
		if ={
			limit = { has_idea = PLB_LOY_Hunter_bad }
			remove_ideas = { PLB_LOY_Hunter_bad }
		}
	}
}
