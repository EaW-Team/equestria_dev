					limit = {
						OR = {
							region = 283

							state = 1065
						}
					}
					add_to_array = { ROOT.kiria_states = THIS }
				}

					limit = {
						OR = {
							region = 94
							region = 85
						}
					}
					add_to_array = { ROOT.zaikiria_states = THIS }
				}
				else_if = {
					limit = {
						OR = {
							region = 125
							region = 131
						}
					}
					add_to_array = { ROOT.zongo_states = THIS }
				}
				else_if = {
					limit = {
						NOT = {
							region = 134

							region = 133

						}
					}
					add_to_array = { ROOT.other_states = THIS }
				}
			}
		}
	}

	option = {
		name = maglibing_postkiria.1.a
