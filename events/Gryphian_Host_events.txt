						rule = GRY_ai_behavior
						option = HARMONIST_HISTORICAL
					}
				}
			}
		}

		add_stability = -0.02
		set_country_flag = GRY_raid_failed

			limit = {
				has_country_flag = GRY_raiding_brodfeld
