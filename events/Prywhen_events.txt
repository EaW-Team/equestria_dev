	id = prywhen.42
	immediate = {log = "[GetDateText]: [Root.GetName]: event prywhen.42"}
	title = prywhen.42.t
	desc = prywhen.42.d
	picture = GFX_event_medieval_village

	trigger = {
		tag = BRF
		has_idea = GRW_risk_of_famine
		NOT = { has_country_flag = GRW_famine_start_flag }
		date > 1007.4.4
	}
	mean_time_to_happen = {
		days = 300
		modifier = { #GLA got sabbotaged by monarchists, famine risk increased
			has_country_flag = BRF_sabotage
			factor = 0.5
		}
		modifier = { #Potato focus during cw as GLA decrease risk
			has_completed_focus = GRW_adopt_potatos
			factor = 1.5
		}
		modifier = { #GLA ricefields focus during cw decreases risk
			has_completed_focus = GRW_ricefield_training
			factor = 1.5
		}
		modifier = { #Controlling Brodfeld agroregion decreases risk
			589 = { is_controlled_by = ROOT }
			factor = 1.5
		}
		modifier = { #Brodfeld focus Tighten the Belt decreases risk
			has_completed_focus = BRF_temporal_carrot
			factor = 2
		}
		modifier = {
			OR = {
				has_completed_focus = GRW_victory_in_civil_war
				has_completed_focus = BRF_free_freedom
			}
			factor = 1.5
		}
		modifier = {
			is_ai = yes
			factor = 2
		}
	}
	option = { ## yes
		name = prywhen.42.a
		add_ideas = GRW_famine_start
		set_country_flag = GRW_famine_start_flag
		save_current_manpower = yes
		every_owned_state = {
			add_manpower = -10000
		}
		restore_previous_manpower = yes
	}
}
