add_namespace = utility


country_event = { ### Set rule to prevent joining factions
	id = utility.1
	hidden = yes
	mean_time_to_happen = { days = 2 }
	trigger = {
		is_ai = yes
		NOT = { has_country_flag = join_factions_set }
	}
	
	option = {
		set_rule = { can_join_factions = no }
		set_country_flag = join_factions_set
	}
}
