	desc = YANinsurgency.1.d
	#picture = 
	is_triggered_only = yes
	trigger = {
		has_country_flag = YAN_insurgency_occured
		any_owned_state = {
			resistance > 39
			NOT = {
				is_demilitarized_zone = yes 
			}
			
		}
	}

			add_resistance = 5 
		}

		add_to_variable = {
			variable = YAN_insurgency_strength
			value = 100
		}

	}
}
