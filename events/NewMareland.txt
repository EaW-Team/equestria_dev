	picture = GFX_event_equestria_fail
	is_triggered_only = yes
	option = {
		ai_chance = {
			factor = 10
		}
		name = mareland.2.a
		add_political_power = -50
		FROM = {
			if = {
				limit = { is_subject = yes }
				end_puppet_this = yes
			}
			if = {
				limit = { is_in_faction = yes }
				leave_faction = yes
			}
		}
		set_country_flag = { flag = bypass_on_peaceconference_ended days = 1 value = 1 }
		white_peace = FROM
		hidden_effect = {
			every_other_country = {
				limit = {
					OR = {
						AND = { ROOT = { is_faction_leader = yes is_in_faction_with = ROOT } }
						is_subject_of = ROOT
					}
					has_war_with = FROM
				}
				set_country_flag = { flag = bypass_on_peaceconference_ended days = 1 value = 1 }
				white_peace = FROM
			}
		}
	}
	option = {
		name = mareland.2.b
		ai_chance = {
			factor = 2
		}
		FROM = {
			country_event = { id = mareland.3 hours = 2 }
		}
	}
}
