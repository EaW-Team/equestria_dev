country_event = { ### Win the raid
	id = hostraidlakecity.1
	title = hostraidlakecity.1.t
	desc = hostraidlakecity.1.d
	picture = GFX_event_griffon_soldier
	is_triggered_only = yes

	option = {
		name = hostraidlakecity.1.a

		if = {
			limit = {
				check_variable = {
					GRY_lakecity_raided < 1
				}
			}

			random_list = {

				65 = {
					add_to_variable = {
						var = GRY_lakecity_raided
						value = 1
					}
					add_tech_bonus = {
						name = RIV_arms_development
						bonus = 0.50
						uses = 1
						category = infantry_weapons
					}
				}
				#
				35 = {
					if = {
						limit = {
							LCT = { has_tech = advanced_infantry_weapons }
						}

						add_equipment_to_stockpile = {
							type = infantry_equipment_3
							amount = 1000
							producer = LCT
						}
					}
					else_if = {
						limit = {
							LCT = { has_tech = improved_infantry_weapons }
						}
						add_equipment_to_stockpile = {

					}

							days = 180
						}
					}

					65 = {
						if = {
							limit = {
								LCT = { has_tech = advanced_infantry_weapons }
							}

							add_equipment_to_stockpile = {
								type = infantry_equipment_3
								amount = 1000
								producer = LCT
							}
						}
						else_if = {
							limit = {
								LCT = { has_tech = improved_infantry_weapons }
							}
							add_equipment_to_stockpile = {
								type = infantry_equipment_2
								amount = 1000
								producer = LCT
							}
						}
						else_if = {
							limit = {
								LCT = { has_tech = infantry_weapons1 }

							}

							add_equipment_to_stockpile = {
								type = infantry_equipment_1
								amount = 1000
