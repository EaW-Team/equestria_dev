# Triggered by on_startup if ED gamerule is active
# Triggers for Chiropterra (BAT)

	id = evildiarchy_newmareland.1

	desc = evildiarchy_newmareland.1.d
	picture = GFX_event_night_riders

	fire_only_once = yes
	is_triggered_only = yes

		name = evildiarchy_newmareland.1.a
		
		hidden_effect = {
			country_event = {
				id = evildiarchy_newmareland.2
				days = 4
			}
		}
	}
}

# The Princesses Visit New Mareland
# Triggers for everyone except Equestria
country_event = { 
	id = evildiarchy_newmareland.4
	immediate = {log = "[GetDateText]: [Root.GetName]: event evildiarchy_newmareland.3"}
	title = evildiarchy_newmareland.4.t
	desc = evildiarchy_newmareland.4.d
	picture = GFX_event_night_riders

	fire_only_once = yes
	is_triggered_only = yes

	option = { # (generic, switching to EQC)
		trigger = {
			NOT = {
				tag = EQC
				tag = BAT
			}
		}
		name = evildiarchy_newmareland.4.a
		
		EQC = {
			change_tag_from = PREV
		}
	}
	option = { # (generic, switching to EQS)
		name = evildiarchy_newmareland.4.b
		
		EQS = {
			change_tag_from = PREV
		}		
	}
	option = { # (generic, not switching)
		trigger = {
			NOT = { tag = BAT }
		}
		name = evildiarchy_newmareland.4.c
	}
	option = { # (EQC)
		trigger = {
			tag = EQC
		}
		name = evildiarchy_newmareland.4.e
	}
	option = { # (BAT, switching to EQC)
		trigger = {
			tag = BAT
		}
		name = evildiarchy_newmareland.4.f
		
		EQC = {
			change_tag_from = PREV
		}
	}
	option = { # (BAT, not switching)
		trigger = {
			tag = BAT
		}
		name = evildiarchy_newmareland.4.g
	}
}
