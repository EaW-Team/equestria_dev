# New Society Development Trigger
#
# EaW_society_development_var will store a number indicating the level of development of society.
# Example:
#	check_variable = {
#		var = EaW_society_development_var
#		value = 5
#		compare = greater_than_or_equals
#	}
# It means that the country has either outdated_industry, or a more advanced level of society (including graduations). This will remove a lot of checks and reduce the probability of errors in the code during jumps in development.
#
# Defined values:
# modern_society = 6
#	outdated_industry2 = 5.5
# outdated_industry = 5
#	pre_industrial_society3 = 4.75
#	pre_industrial_society2 = 4.5
#	pre_industrial_society1 = 4.25
# pre_industrial_society = 4
#	agrarian_society6 = 3.83
#	agrarian_society5 = 3.67
#	agrarian_society4 = 3.5
#	agrarian_society3 = 3.33
#	agrarian_society2 = 3.17

# communal_society = 2
#	tribal_society4 = 1.75
