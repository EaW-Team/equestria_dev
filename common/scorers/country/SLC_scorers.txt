# This scorer is used to sort the country occupied by SLC with their number of cores
SLC_nb_of_cores_scorer = {
	targets = {
		targets_dynamic = no
		target_non_existing = yes
		target_array = SLC.occupied_countries

		score = {
			base = 1
			# Add tension added by country
			modifier = {
				always = yes
				add = core_states^num
			}
		}
	}
}
