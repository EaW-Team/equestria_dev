						}
					}
					country_event = { rila_extremist.1 hours = 12 }
				}
				else_if = {
					limit = {
						AND = {
							check_variable = { JEB_moderate_revolutionary_clubs > JEB_michel }
							check_variable = { JEB_moderate_revolutionary_clubs > JEB_progressive_clubs }
							check_variable = { JEB_moderate_revolutionary_clubs > JEB_extremist_revolutionary_clubs }
						}
					}
					country_event = { rila_moderate.1 hours = 12 }
				}
				else_if = {
					limit = {
						AND = {
							check_variable = { JEB_progressive_clubs > JEB_michel }
							check_variable = { JEB_progressive_clubs > JEB_extremist_revolutionary_clubs }
							check_variable = { JEB_progressive_clubs > JEB_moderate_revolutionary_clubs }
						}
					}
					country_event = { rila_progressive.1 hours = 12 }
				}
				else = { #in the event two clubs have the same influence points
					country_event = { rila_moderate.1 hours = 12 } #moderate socialists are the most compromising ones anyway
				}
			}
		}
	}
