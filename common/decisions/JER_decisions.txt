JER_revolution_category = {
	JER_revolution_mission = {
		icon = generic_ignite_civil_war
		allowed = {
			OR = {
				original_tag = JER
				original_tag = JEB
				original_tag = JRR
				original_tag = VIN
				original_tag = JFT
			}
		}
		available = {
			if = {
				limit = {
					original_tag = JER
					has_completed_focus = JER_state_of_aquileia
				}
				custom_trigger_tooltip = {
					tooltip = JER_revolution_crushed_tp_JEB
					JEB = { check_variable = { monarchist_support > revolutionary_support } }
				}
				custom_trigger_tooltip = {
					tooltip = JER_revolution_crushed_tp_JRR
					JRR = { check_variable = { monarchist_support > revolutionary_support } }
				}
				custom_trigger_tooltip = {
					tooltip = JER_revolution_crushed_tp_VIN
					VIN = { check_variable = { monarchist_support > revolutionary_support } }
				}
				custom_trigger_tooltip = {
					tooltip = JER_revolution_crushed_tp_JFT
					JFT = { check_variable = { monarchist_support > revolutionary_support } }
				}
			}
			else_if = {
				limit = {
				    original_tag = JER
					has_completed_focus = JER_fight_for_freedom
				}
				custom_trigger_tooltip = {
					tooltip = JER_revolution_success_tp_JEB
					JEB = { check_variable = { monarchist_support < revolutionary_support } }
				}
				custom_trigger_tooltip = {
					tooltip = JER_revolution_success_tp_JRR
					JRR = { check_variable = { monarchist_support < revolutionary_support } }
				}
				custom_trigger_tooltip = {
					tooltip = JER_revolution_success_tp_VIN
					VIN = { check_variable = { monarchist_support < revolutionary_support } }
				}
				custom_trigger_tooltip = {
					tooltip = JER_revolution_success_tp_JFT
					JFT = { check_variable = { monarchist_support < revolutionary_support } }
				}
			}
			else = { hidden_trigger = { always = no } }
		}
		activation = {
			JER = { 
			    OR = {
			        has_completed_focus = JER_state_of_aquileia
                    has_completed_focus = JER_fight_for_freedom					
				}
			}
		}
		is_good = no
		days_mission_timeout = 365
		fire_only_once = yes
		timeout_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision timeout JER_revolution_mission"
			if = {
				limit = { original_tag = JEB }
				JEB = {
					if = {
						limit = { check_variable = { monarchist_support > revolutionary_support } }
						country_event = { id = aquileia.9 days = 1 }
						set_country_flag = REVOLUTION_is_a_monarchist
					}
					else = {
						country_event = { id = aquileia.10 days = 1 }
						set_country_flag = REVOLUTION_is_a_revolutionary
					}
				}
			}
			else_if = {
				limit = { original_tag = JRR }
				JRR = {
					if = {
						limit = { check_variable = { monarchist_support > revolutionary_support } }
						country_event = { id = aquileia.11 days = 1 }
						set_country_flag = REVOLUTION_is_a_monarchist
					}
					else = {
						country_event = { id = aquileia.12 days = 1 }
						set_country_flag = REVOLUTION_is_a_revolutionary
					}
				}
			}
			else_if = {
				limit = { original_tag = VIN }
				VIN = {
					if = {
						limit = { check_variable = { monarchist_support > revolutionary_support } }
						country_event = { id = aquileia.13 days = 1 }
						set_country_flag = REVOLUTION_is_a_monarchist
					}
					else = {
						country_event = { id = aquileia.14 days = 1 }
						set_country_flag = REVOLUTION_is_a_revolutionary
					}
				}
			}
			else_if = {
				limit = { original_tag = JFT }
				JFT = {
					if = {
						limit = { check_variable = { monarchist_support > revolutionary_support } }
						country_event = { id = aquileia.15 days = 1 }
						set_country_flag = REVOLUTION_is_a_monarchist
					}
					else = {
						country_event = { id = aquileia.45 days = 1 }
						set_country_flag = REVOLUTION_is_a_revolutionary
					}
				}
			}
			else_if = {
				limit = { 
				    original_tag = JER 
					has_completed_focus = JER_fight_for_freedom
				}
				JER = {
					country_event = { id = aquileia.16 }
					set_country_flag = REVOLUTION_is_a_revolutionary
				}
			}
			else_if = {
				limit = { 
				    original_tag = JER 
					has_completed_focus = JER_state_of_aquileia
				}
				JER = {
					country_event = { id = aquileia.46 days = 4 }
					set_country_flag = REVOLUTION_is_a_monarchist
				}
			}
			hidden_effect = {
				set_global_flag = JER_revolution_begun
			}
		}
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision JER_revolution_mission"
			if = {
				limit = { 
				    original_tag = JER 
					has_completed_focus = JER_state_of_aquileia
				}
				JER = {
					country_event = { id = aquileia.47 days = 1 }
				}
			}
			else_if = {
				limit = { 
				    original_tag = JER 
					has_completed_focus = JER_fight_for_freedom
				}
				JER = {
					country_event = { id = aquileia.17 days = 1 }
				}
			}
		}
	}
