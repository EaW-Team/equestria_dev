				AND = {
				    original_tag = JER
					has_completed_focus = JER_fight_for_freedom
				    custom_trigger_tooltip = {
					    tooltip = JER_revolution_success_tp
					    JEB = { check_variable = { revolutionary_support > monarchist_support } }
					    JRR = { check_variable = { revolutionary_support > monarchist_support } }
					    VIN = { check_variable = { revolutionary_support > monarchist_support } }

			}
		}
		visible = {
			NOT = {
				has_global_flag = JER_revolution_begun
				has_global_flag = JER_revolution_ended
				has_global_flag = JER_revolution_ended_2
			}
		}
		activation = {
			JER = { 
			    OR = {
			        has_completed_focus = JER_state_of_aquileia
                    has_completed_focus = JER_fight_for_freedom					
				}
			}
		}
		is_good = no
		days_mission_timeout = 365
		fire_only_once = yes
		timeout_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision timeout JER_revolution_mission"
			if = {
				limit = { original_tag = JEB }
				JEB = {
					if = {
						limit = { check_variable = { monarchist_support > revolutionary_support } }
						country_event = { id = aquileia.9 days = 1 }
					}
					else = {
						country_event = { id = aquileia.10 days = 1 }
					}
				}
			}
			if = {
				limit = { original_tag = JRR }
				JRR = {
					if = {
						limit = { check_variable = { monarchist_support > revolutionary_support } }
						country_event = { id = aquileia.11 days = 1 }
					}
					else = {
						country_event = { id = aquileia.12 days = 1 }
					}
				}
			}
			if = {
				limit = { original_tag = VIN }
				VIN = {
					if = {
						limit = { check_variable = { monarchist_support > revolutionary_support } }
						country_event = { id = aquileia.13 days = 1 }
					}
					else = {
						country_event = { id = aquileia.14 days = 1 }
					}
				}
			}
			if = {
				limit = { original_tag = JFT }
				JFT = {
					if = {
						limit = { check_variable = { monarchist_support > revolutionary_support } }
						country_event = { id = aquileia.15 days = 1 }
					}
					else = {
						country_event = { id = aquileia.45 days = 1 }
					}
				}
			}
			if = {
				limit = { 
				    original_tag = JER 
					has_completed_focus = JER_fight_for_freedom
				}
				JER = {
					country_event = { id = aquileia.16 }
				}
			}
			if = {
				limit = { 
				    original_tag = JER 
					has_completed_focus = JER_state_of_aquileia
				}
				JER = {
					country_event = { id = aquileia.46 days = 4 }
				}
			}
			hidden_effect = {
				set_global_flag = JER_revolution_begun
			}
		}
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision JER_revolution_mission"
			if = {
				limit = { 
				    original_tag = JER 
					has_completed_focus = JER_state_of_aquileia
				}
