BAR_make_zombies = {
		icon = griffon

		visible = {
			original_tag = BAR
		}
		available = {
			check_variable = { BAR_test_subjects > 1 }
		}
		days_remove = 30
		
		remove_effect = {
		if = { 
		limit = { has_country_flag = BAR_pufferbombs }
		add_manpower = 20000
		else = {
		add_manpower = 15000
		    }
		}
		hidden_effect = {
		subtract_from_variable = {
        var = BAR_test_subjects 
        value = 2
                }
			}
		}
	ai_will_do = {
		factor = 100
		}
	}
