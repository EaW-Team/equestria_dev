coring_decisions = {
	generic_core_country = {

		icon = infiltrate_state

		allowed = {
            always = no
        }

		target_trigger = {
			always = no
		}
        
        visible = {
            FROM = {
				check_variable = {
					core_states^num > 0
				}
				any_uncored_cores_owned_by_ROOT_or_subject = yes
			}
        }

        cost = freedec?0

        days_remove = 30
        
		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove generic_core_country target: [From.GetName]"
            every_state = {
                limit = {
					owner = {
						OR = {
							tag = ROOT
							is_subject_of = ROOT
						}
					}
					is_core_of = FROM
					NOT = { is_core_of = ROOT }
                }
                add_core_of = ROOT
            }
        }
        
        ai_will_do = {
            factor = 5000
        }
	}

	generic_core_country_original_cores = {

		icon = infiltrate_state

		allowed = {
            always = no
        }

		target_trigger = {
			always = no
		}
        
        visible = {
            FROM = {
				check_variable = {
					original_cores^num > 0
				}
				any_uncored_cores_owned_by_ROOT_or_subject = yes
			}
        }

        cost = freedec?0

        days_remove = 30
        
		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove generic_core_country target: [From.GetName]"
            every_state = {
                limit = {
					owner = {
						OR = {
							tag = ROOT
							is_subject_of = ROOT
						}
					}
					is_in_array = {
						FROM.original_cores = THIS
					}
					NOT = { is_core_of = ROOT }
                }
                add_core_of = ROOT
            }
        }
        
        ai_will_do = {
            factor = 5000
        }
	}
}
