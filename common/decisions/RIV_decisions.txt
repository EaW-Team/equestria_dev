	RIV_invite_country_decision = {
		icon = rivfed
		targets = { FIR FBK WAT }
		targets_dynamic = yes
		target_trigger = {
			FROM = {
				OR = {
					original_tag = FIR
					original_tag = FBK
					original_tag = WAT
					#original_tag = NCH
				}
			}
		}
		target_root_trigger = {
			is_subject = no
		}
		visible = {
			is_subject = no
			FROM = {
				OR = {
					original_tag = FIR
					original_tag = FBK
					original_tag = WAT
				}
				exists = yes
				is_subject = no
				has_war = no
				is_river_fed_member = no
				NOT = { has_country_flag = intived_to_RIV }
			}
		}

		available = {
			FROM = {
				if = {
					limit = { original_tag = FIR }
					has_completed_focus = FIR_riv_relations
				}
				else_if = {
					limit = { original_tag = FBK }
					has_completed_focus = FBK_apply_for_membership
				}
			}
		}

		ai_will_do = {
			factor = 100
			modifier = {
				factor = 0
				FROM = {
					original_tag = FIR
					OR = {
						has_government = fascism
						has_government = communism
					}
				}
			}
			modifier = {
				factor = 0
				FROM = {
					is_ai = yes
					is_in_faction = yes
				}
			}
			modifier = {
				factor = 0.1
				country_exists = RIV
				NOT = { FROM = { has_opinion = { target = RIV value > 25 } } }
			}
			modifier = {
				factor = 0
				country_exists = RIV
				NOT = { FROM = { has_opinion = { target = RIV value > -50 } } }
			}
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision RIV_invite_country_decision target: [From.GetName]"
			FROM = {
				set_country_flag = { flag = intived_to_RIV value = 1 days = 360 }
				country_event = { id = riverfed.12 hours = 1 }
			}
		}
	}
