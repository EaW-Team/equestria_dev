SIC_army_reform_category = {

	SIC_army_reform = {

		icon = generic_prepare_civil_war

		available = {
			hidden_trigger = { has_army_experience > 50 }
		}

		custom_cost_trigger = {
			has_army_experience > 50
		}

		custom_cost_text = decision_cost_army_xp_50

		ai_will_do = {
			factor = 10
		}
		days_remove = 90

		visible = {
			has_completed_focus = SIC_examine_militia
			OR = {
				has_idea = SIC_militia_dependence
				has_idea = SIC_militia_dependence1
				has_idea = SIC_militia_dependence2
				has_idea = SIC_militia_dependence3
				has_idea = SIC_militia_dependence4
			}
		}
		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove SIC_army_reform"
			if = {
				limit = { has_idea = SIC_militia_dependence }
				swap_ideas = {
					remove_idea = SIC_militia_dependence
					add_idea = SIC_militia_dependence1
				}
			}
			else_if = {
				limit = { has_idea = SIC_militia_dependence1  }
				swap_ideas = {
					remove_idea = SIC_militia_dependence1
					add_idea = SIC_militia_dependence2
				}
			}
			else_if = {
				limit = { has_idea = SIC_militia_dependence2 }
				swap_ideas = {
					remove_idea = SIC_militia_dependence2
					add_idea = SIC_militia_dependence3
				}
			}
			else_if = {
				limit = { has_idea = SIC_militia_dependence3 }
				swap_ideas = {
					remove_idea = SIC_militia_dependence3
					add_idea = SIC_militia_dependence4
				}
			}
			else_if = {
				limit = { has_idea = SIC_militia_dependence4 }
				remove_ideas = SIC_militia_dependence4
			}
		}
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision SIC_army_reform"
			hidden_effect = { army_experience = -50 }
		}
	}
}
