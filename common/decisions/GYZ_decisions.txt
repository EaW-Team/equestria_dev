GYZ_the_hayzeb_issue = {
	GYZ_ticking_timer = {
		days_mission_timeout = 65
		is_good = no
		activation = {
			always = no
		}
		available = {
			hidden_trigger = {
				always = no
			}
		}
		cancel_trigger = {
			has_global_flag = SK_formed
		}
		timeout_effect = {
			if = {
				limit = {
					has_idea = GYZ_hayzeb_resistance_1
				}
				activate_mission = GYZ_ticking_timer
				swap_ideas = {
					remove_idea = GYZ_hayzeb_resistance_1
					add_idea = GYZ_hayzeb_resistance_2
				}
			}
			else_if = {
				limit = {
					has_idea = GYZ_hayzeb_resistance_2
				}
				activate_mission = GYZ_ticking_timer
				swap_ideas = {
					remove_idea = GYZ_hayzeb_resistance_2
					add_idea = GYZ_hayzeb_resistance_3
				}
			}
			else_if = {
				limit = {
					has_idea = GYZ_hayzeb_resistance_3
				}
				activate_mission = GYZ_ticking_timer
				swap_ideas = {
					remove_idea = GYZ_hayzeb_resistance_3
					add_idea = GYZ_hayzeb_resistance_4
				}
			}
			else = {
				country_event = GYZ_storm_civil_war.2
			}
		}
	}
	
	GYZ_propagandize_hayzeb = {
		days_remove = 30
		available = {
			has_completed_focus = GYZ_handle_the_hayzeb_problem
		}
		cost = GYZ_propagandize_hayzeb_cost
		complete_effect = {

			multiply_variable = { GYZ_propagandize_hayzeb_cost = 2 }
			add_days_mission_timeout = {
				mission = GYZ_ticking_timer
				days = 30
			}
			custom_effect_tooltip = GYZ_double_cost_tt
		}
	}

		days_remove = 30
		
		available = {
			has_completed_focus = GYZ_handle_the_hayzeb_problem
		}

		custom_cost_trigger = {
			command_power > GYZ_infiltrate_hayzeb_cost
		}
		custom_cost_text = decision_cost_CP_hayzeb
		complete_effect = {
			multiply_variable = { GYZ_infiltrate_hayzeb_cost = -1 }
			hidden_effect = {add_command_power = GYZ_infiltrate_hayzeb_cost }
			multiply_variable = { GYZ_infiltrate_hayzeb_cost = -2 } # offsetting above
			add_days_mission_timeout = {
				mission = GYZ_ticking_timer
