FEZ_black_market_category = {
	FEZ_black_market_griefwald = {
		icon = generic_form_nation

		visible = {
			FROM = {
				exists = yes
				NOT = { has_war_with = ROOT }
			}
		}

		fire_only_once = no
		cost = 50
		days_remove = 30
		target_root_trigger = {
			OR = {
				has_country_flag = black_market
				has_country_flag = black_market2
				has_country_flag = black_market3
				has_country_flag = black_market4
			}
		}
		targets = { AET FLO AVI HAU WNG VED D01 D02 D03 D04 D05 D06 D07 D08 D09 D10 D11 D12 D13 D14 D15 }
		target_trigger = {
			FROM = {
				OR = {
					AND = {
						ROOT = { has_country_flag = black_market }
						OR = {
							original_tag = AET
							original_tag = FLO
							original_tag = AVI
						}
					}
					AND = {
						ROOT = { has_country_flag = black_market2 }
						original_tag = WNG
					}
					AND = {
						ROOT = { has_country_flag = black_market3 }
						original_tag = HAU
					}
					AND = {
						ROOT = { has_country_flag = black_market4 }
						original_tag = VED
					}
				}
			}
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision FEZ_black_market"
			FROM = {
				if = {
					limit = { original_tag = AET }
					country_event = fezera.2
				}
				else_if = {
					limit = { original_tag = FLO }
					country_event = fezera.3
				}
				else_if = {
					limit = { original_tag = AVI }
					country_event = fezera.4
				}
				else_if = {
					limit = { original_tag = HAU }
					country_event = fezera.15
				}
				else_if = {
					limit = { original_tag = WNG }
					country_event = fezera.16
				}
				else_if = {
					limit = { original_tag = VED }
					country_event = fezera.16
				}
			}
		}
	}
}
