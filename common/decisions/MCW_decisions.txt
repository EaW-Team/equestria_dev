foreign_politics = {
	MCW_peace_with_griffonia = { # Fires for Macawia's enemies
		available = {
			is_griffon_continent_country_tag = yes
		}

		activation = {
			NOT = { has_country_flag = disable_peace_with_macawia_mission }
			is_subject = no
			OR = {
				is_in_faction = no
				is_faction_leader = yes
			}
			has_war_with = MCW
			MCW = {
				is_major = yes
				has_capitulated = no
				any_allied_country = { # Has at least 1
					is_griffon_continent_country_tag = yes
				}
				all_allied_country = {
					OR = {
						is_griffon_continent_country_tag = no
						has_capitulated = yes
					}
				}
			}
		}

		available = {
			always = no
		}

		cancel_trigger = {
			OR = {
				NOT = { has_war_with = MCW }
				MCW = { has_capitulated = yes }
				hidden_trigger = {
					custom_trigger_tooltip = {
						tooltip = MCW_allies_in_griffonia
						MCW = {
							all_allied_country = {
								NOT = { is_griffon_continent_country_tag = yes }
								has_capitulated = no
							}
						}
					}
				}		
			}
		}

		days_mission_timeout = 5 # 1 year

		timeout_effect = {
			country_event = peace_with_macawia.1
		}
	}
}
