    BAK_election_mission = {
		icon = generic_ignite_civil_war
		available = {
			always = no
		}
		visible = {
			has_idea = BAK_elections
		}
		activation = {
			has_idea = BAK_elections
		}
		is_good = no
		days_mission_timeout = 330
		fire_only_once = yes
		timeout_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision timeout BAK_election_mission"
			if = { #Fascists Win
				limit = { 
				    AND = {
				        OR = {
				            has_country_flag = BAK_AI_FASCIST
				            has_game_rule = {
				                rule = BAK_ai_behavior
				                option = FASCIST
			                }
				        }
				        is_ai = yes
					}
				}
				country_event = { 
				    id = bakara.1
				}
			}
			else_if = { #Communists Win
				limit = { 
				    AND = {
				        OR = {
				            has_country_flag = BAK_AI_COMMUNIST
				            has_game_rule = {
				                rule = BAK_ai_behavior
				                option = COMMUNIST
			                }
				        }
				        is_ai = yes
					}
				}
				country_event = { 
				    id = bakara.2
				}
			}
			else_if = { #Neutrality Win
				limit = { 
				    AND = {
				        OR = {
				            has_country_flag = BAK_AI_NEUTRALITY
				            has_game_rule = {
				                rule = BAK_ai_behavior
				                option = NEUTRALITY
			                }
				        }
				        is_ai = yes
					}
				}
				country_event = { 
				    id = bakara.3
				}
			}
			else_if = { #Harmonists Win
				limit = { 
				    AND = {
				        OR = {
				            has_country_flag = BAK_AI_HISTORICAL
				            has_game_rule = {
				                rule = BAK_ai_behavior
				                option = HARMONY_HISTORICAL
			                }
			                AND = {
				                is_historical_focus_on = yes
				                has_game_rule = {
					                rule = BAK_ai_behavior
					                option = DEFAULT
				                }
			                }
				        }
				        is_ai = yes
					}
				}
				country_event = { 
				    id = bakara.4
				}
			}
			else_if = { #Fascists Win
				limit = { 
				    AND = {
					    check_variable = { BAK_fascist_support_v > BAK_communist_support_v }
						check_variable = { BAK_fascist_support_v > BAK_neutrality_support_v }
						check_variable = { BAK_fascist_support_v > BAK_harmonist_support_v }
					}
				}
				country_event = { 
				    id = bakara.1
				}
			}
			else_if = { #Communists Win
				limit = { 
				    AND = {
					    check_variable = { BAK_communist_support_v > BAK_fascist_support_v }
						check_variable = { BAK_communist_support_v > BAK_neutrality_support_v }
						check_variable = { BAK_communist_support_v > BAK_harmonist_support_v }
					}
				}
				country_event = { 
				    id = bakara.2
				}
			}
			else_if = { #Neutrality Win
				limit = { 
				    AND = {
					    check_variable = { BAK_neutrality_support_v > BAK_fascist_support_v }
						check_variable = { BAK_neutrality_support_v > BAK_communist_support_v }
						check_variable = { BAK_neutrality_support_v > BAK_harmonist_support_v }
					}
				}
				country_event = { 
				    id = bakara.3
				}
			}
			else = { #Harmonists Win
			    country_event = { 
				    id = bakara.4
				}
			}
		}
	}
