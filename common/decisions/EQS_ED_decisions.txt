	EQS_ED_targeted_excavate_artefacts = {
		
		state_target = yes 
		target_array = global.states_with_artefacts 
		target_trigger = {
			FROM = { has_variable = artefact_in_state }
		}
		visible = {
			FROM = {
				
				if = {
					limit = {
						is_on_continent = asia 
					}
					ROOT = { has_completed_focus = EQS_ED_contact_the_griffons }
				}
			}
		}
		
		available = {
			FROM = {
				NOT = { is_controlled_by = EQC }
			}
			custom_trigger_tooltip = {
				tooltip = has_enough_excavations_available
				ROOT = {
					check_variable = { current_excavation_projects < max_excavation_projects }
				}
			}
		}
		
		cancel_trigger = {
			FROM = { is_fully_controlled_by = EQC }
		}
		
		cancel_effect = {
			add_to_variable = { current_excavation_projects = -1 }
		}
		
		days_remove = 60
		days_re_enable = EQS_snitch_artefacts_day_reeneable
		fixed_random_seed = no 
		
		complete_effect = {
			add_to_variable = { current_excavation_projects = 1 }
			if = {
				limit = {
					FROM = { is_on_continent = asia }
				}
				EQS_AF_calculate_snitch_chance = yes 
				round_temp_variable = tmp_check_var 
				set_variable = { FROM.successful_excavation_chance = tmp_check_var }
				hidden_effect = {
					set_temp_variable = { pp_to_remove = current_pp_toinvest }
					set_temp_variable = { cp_to_remove = current_cp_toinvest }
					set_temp_variable = { eq_to_remove = current_eq_toinvest }
					
					multiply_temp_variable = { pp_to_remove = -1 }
					multiply_temp_variable = { cp_to_remove = -1 }
					multiply_temp_variable = { eq_to_remove = -1 }
					
					add_political_power = pp_to_remove
					add_command_power = cp_to_remove
					add_equipment_to_stockpile = { type = support_equipment amount = eq_to_remove }
					
					set_variable = { current_pp_toinvest = 0 }
					set_variable = { current_eq_toinvest = 0 }
					set_variable = { current_cp_toinvest = 0 }
					EQS_AF_display_invest_chance = yes 
				}
			}
			else = {
				set_variable = { FROM.successful_excavation_chance = 100 }
			}
		}
		
		remove_effect = {
			set_temp_variable = { check_var = 100 }
			subtract_from_temp_variable = { check_var = FROM.successful_excavation_chance }
			clamp_temp_variable = { var = check_var min = 0 }
			random_list = {
				FROM.successful_excavation_chance = {
					hidden_effect = {
						add_ideas = FROM.artefact_in_state
					}
					FROM = { clear_variable = artefact_in_state }
				}
				check_var = {
				
				}
			}
			add_to_variable = { current_excavation_projects = -1 }
		}
	}
