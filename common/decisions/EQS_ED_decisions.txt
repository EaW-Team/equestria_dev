	EQS_weather_spell_darkness_conditions_target = {
	#	visible = { has_completed_focus = EQS_ED_weather_magic }
		custom_cost_text = decision_cost_350_wmp
		custom_cost_trigger = {
			check_variable = { var = weather_pb_var value = 350 compare = greater_than_or_equals }
		}
		
		fixed_random_seed = no 
		 
		complete_effect = { 
			add_to_variable = { weather_pb_var = -350 }
			hidden_effect = {
				thaumaturgy_roll = yes
				if = {
					limit = {
						check_variable = { roll_result > 4 }
					}
					FROM.owner = {
						add_ideas = EQS_WM_darkness 
					}
				}
				else = {
					set_temp_variable = { check_rl = 5 }
					subtract_from_temp_variable = { check_rl = roll_result }
					random_list = {
						roll_result = {
							FROM.owner = { add_ideas = EQS_WM_darkness }
						}
						check_rl = {
							
						}
					}
				}
			}
			custom_effect_tooltip = weather_magic_darkness
		}
		
		state_target = yes 
		
		target_trigger = {
			FROM.owner = { 
				has_war_with = ROOT 
				is_neighbor_of = ROOT 
			}
			FROM = { is_capital = yes }
			
		}
		
		days_re_enable = 45
		days_remove = 45
		
		remove_effect = {
			hidden_effect = { FROM.owner = { remove_ideas = EQS_WM_darkness } }
		}	
	}
