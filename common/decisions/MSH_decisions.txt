MSH_subsidiary_economics_category = {
    MSH_change_payment_type = {
		icon = GFX_decision_ger_mefo_bills

		cost = 0

		ai_will_do = {
			factor = 0
		}

		complete_effect = {
			log = "[GetDateText] [Root.GetName]: decision MSH_change_payment_type"
			if = {
				limit = { 
					has_country_flag = MSH_paying_with_political_power 
				}
				custom_effect_tooltip = MSH_change_to_paying_with_gold_tp
				clr_country_flag = MSH_paying_with_political_power
				set_country_flag = MSH_paying_with_gold
			}
			else_if = {
				limit = { 
					has_country_flag = MSH_paying_with_gold 
				}
				custom_effect_tooltip = MSH_change_to_paying_with_political_power_tp
				clr_country_flag = MSH_paying_with_gold
				set_country_flag = MSH_paying_with_political_power
			}
		}
	}
}
