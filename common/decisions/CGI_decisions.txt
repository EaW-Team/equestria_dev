	CGI_make_contacts_with_the_coltvan_resistance = {
		icon = GFX_decision_gre_investment_decisions
		
		fire_only_once = yes 
		
		visible = {
			has_completed_focus = CGI_opportunities_in_zonicia
			
			NOT = { has_country_flag = CGI_diamond_road_CTH_trade_deal }
		}
		
		available = {
			# Colthage has not already cored Coltva
			NOT = {
				any_country_with_original_tag = {
					original_tag_to_check = CTH
					any_core_state = {
						is_core_of = CVA
					}
				}
			}
			if = {
				limit = {
					country_exists = CVA
				}
				NOT = { has_war_with = CVA }
			}
		}
		
		cost = 30
		
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete CGI_make_contacts_with_the_coltvan_resistance"
						
			if = {
				limit = {
					country_exists = CVA
				}
				CVA = {
					country_event = quaggatai_diplo.1
				}
				
				custom_effect_tooltip = CGI_if_they_accept
				custom_effect_tooltip = {
					localization_key = CGI_diamond_road_add
					VALUE = "3"
					POINT_POINTS = CGI_points
				}
				custom_effect_tooltip = CGI_disable_mission_to_colthage
				custom_effect_tooltip = CGI_if_they_refuse
				custom_effect_tooltip = CGI_get_back_invested_pp
			}
			else = {
				CTH = {
					every_owned_state = {
						limit = {
							is_core_of = CVA
						}
						add_resistance = 5
					}
				}
				hidden_effect = {
					CVA = {
						add_ideas = CGI_diamond_road_trade_deal_with_quaggatai_CVA
					}
				}
				
				add_to_variable = {
					var = CGI_diamond_road_strength
					value = 3
				}
				custom_effect_tooltip = {
					localization_key = CGI_diamond_road_add
					VALUE = "3"
					POINT_POINTS = CGI_points
				}
				CGI_update_diamond_road = yes
				
				set_country_flag = CGI_diamond_road_CVA_trade_deal
				custom_effect_tooltip = CGI_disable_mission_to_colthage
			}
		}
	}
