foreign_politics = {
	CGI_attack_breakaways = {
		allowed = {
			original_tag = CGI
		}

		visible = {
			NOT = { has_country_flag = CGI_coltdar_lost }
		}
		
		target_root_trigger = {
			has_country_flag = CGI_attacked_colthage
		}

		targets = { CTH CVA TRO HIE }
		targets_dynamic = yes

		target_trigger = {
			FROM = { # Needed because of targets_dynamic
				OR = {
					original_tag = CTH
					original_tag = CVA
					original_tag = TRO
					original_tag = HIE
				}
			}
			FROM = { exists = yes }
			NOT = { has_war_with = FROM }
		}

		ai_will_do = {
			factor = 100
		}

		war_with_target_on_remove = yes

		days_remove = 15
		cancel_trigger = {
			has_country_flag = CGI_coltdar_lost
		}

		cancel_effect = {
			eaw_clear_alert_imminent_declaration_of_war_FROM = yes
		}

		complete_effect = {
			eaw_alert_send_imminent_declaration_of_war_FROM = yes
		}
		
		remove_effect = {
			eaw_clear_alert_imminent_declaration_of_war_FROM = yes
			declare_war_on = {
				target = FROM
				type = annex_everything
			}
		}
	}
}
