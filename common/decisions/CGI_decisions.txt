CGI_the_diamond_road = {
	CGI_invest_into_the_diamond_road = {
		icon = GFX_decision_gre_investment_decisions
		
		available = {
			num_of_available_civilian_factories > 1
		}
		
		visible = {
			has_completed_focus = CGI_strengthen_the_diamond_road
		}
		
		cost = 35

		days_remove = 45

		modifier = {
			civilian_factory_use = 2
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove CGI_invest_into_the_diamond_road"
			
			add_to_variable = {
				var = CGI_diamond_road_strength
				value = 4
			}
			custom_effect_tooltip = {
				localization_key = CGI_diamond_road_add
				VALUE = "4"
				POINT_POINTS = "points"
			}
			
			CGI_update_diamond_road = yes
		}
	}
	
	CGI_buy_up_modern_tools = {
		icon = GFX_decision_gre_investment_decisions
		
		available = {
			num_of_available_civilian_factories > 1
		}
		
		visible = {
			has_completed_focus = CGI_embracing_modernity
			NOT = { check_variable = { CGI_diamond_road_increase_society_counter > 2 } }
		}
		
		cost = 25

		days_remove = 50

		modifier = {
			civilian_factory_use = 2
		}
		
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete CGI_buy_up_modern_tools"
			
			add_to_variable = { CGI_diamond_road_increase_society_counter = 1 }
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove CGI_buy_up_modern_tools"
			
			
			add_to_variable = {
				var = CGI_diamond_road_strength
				value = -5
			}
			custom_effect_tooltip = {
				localization_key = CGI_diamond_road_remove
				VALUE = "5"
				POINT_POINTS = "points"
			}
			
			CGI_update_diamond_road = yes
			
			increase_society_development_gradual = yes
			
			custom_effect_tooltip = CGI_diamond_road_increase_society_counter_tooltip
		}
	}

		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete CGI_buy_up_modern_tools"
			
			
		}

	CGI_make_contacts_with_the_coltvan_resistance = {
		icon = GFX_decision_gre_investment_decisions
				
		visible = {
			has_completed_focus = CGI_opportunities_in_zonicia
			
			NOT = { has_country_flag = CGI_diamond_road_CTH_trade_deal }
		}
		
		available = {
			# Has not already cored Coltva
			NOT = {
				any_country_with_original_tag = {
					original_tag_to_check = CTH
					any_core_state = {
						is_core_of = CVA
					}
				}
			}
		}
		
		cost = 30
		
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete CGI_buy_up_modern_tools"
			
			set_country_flag = CGI_diamond_road_CVA_trade_deal
			
			CTH = {
				every_owned_state = {
					limit = {
						is_core_of = CVA
					}
					add_resistance = 5
				}
			}
		}
	}
