on_actions = {
	on_startup = {
		effect = {
			STG = {
				set_variable = { stg_max_hunger_lose = 5 }
				set_variable = { stg_max_hunger_exp = 6 }
				add_dynamic_modifier = {
					modifier = STG_hunger_modifier
				}
			}
		}
	}
	on_weekly_STG = {
		effect = {
			if = {	
				limit = {
					has_country_flag = famine_in_stg_flag
				}
				randomize_temp_variable = {
					var = stg_hunger_lower
					min = 1
					max = 10
					distribution = uniform
				}
				set_variable = { stg_food_lose = stg_hunger_lower }
				randomize_temp_variable = {
					var = stg_hunger_exp
					min = 1
					max = stg_max_hunger_exp
					distribution = uniform
				}
				set_variable = { stg_food_gain = stg_hunger_exp }
				set_variable = { stg_food_change = 0 }
				#add_to_variable = { stg_food_change = stg_food_gain }
				subtract_from_variable = { stg_food_change = 10 }
				#round_variable = stg_food_change
				add_to_variable = { hunger_level_100 = stg_food_change }
				clamp_variable = {
					var = hunger_level_100
					min = 0
					max = 100 
				}
				
				calculate_stg_bonus_penalties = yes
				
			}
		}
	}
