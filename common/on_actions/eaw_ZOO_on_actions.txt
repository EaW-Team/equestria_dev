on_actions = {

	on_weekly_ZOO = {
		effect = {
			# Outside and constantly calculated so as to make the variable available to display prior to the first 'real' weekly pulse.
			set_variable = { ZOO_map_gui_next_construction = global.date }
			add_to_variable = { ZOO_map_gui_next_construction = 0.00168 }
			if = {
				limit = {
					has_country_flag = ZOO_grand_railroad_project_enabled
					has_country_flag = ZOO_railroad_construction
					has_dynamic_modifier = {
						modifier = ZOO_grand_railroad_project
					}
				}
				add_to_variable = { ZOO_railway_tiles_stored = modifier@ZOO_weekly_railway_tiles }
				ZOO_weekly_railroad_construction = yes
			}
			add_to_variable = { ZOO_choo_choo_dirty = 0.00001 }
		}
	}
}
