on_actions = {
	on_state_control_changed = {
		effect = {
			# Check if the state has unrest or not, and removes the modifier if necessary
			if = {
				limit = {
					FROM = {
						original_tag = MGY
					}
				}
				MGY_Unrest_onStateOwnershipChange_effect = yes
			}
		}
	}
	on_unit_leader_created = {
		# Effect to assign the "Maredjay Officer" trait to new commanders
		effect = {
			character = {
				if = {
					limit = {
						is_army_leader = yes
					}
					if = {
						limit = {
							FROM = {
								has_idea = MGY_MaredjayArmy
							}
						}
						ROOT = {
							add_unit_leader_trait = maredjay_officer
						}
					}
				}
			}
		}
	}
	on_startup = {
		effect = {
			MGY = {
				country_event = {
					id = MGY_Lore.1
				}
			}
		}
	}
	on_weekly_MGY = {
		effect = {
			if = {
				#Storm King Dynamic Modifier decay
				limit = {
					has_dynamic_modifier = {
						modifier = MGY_MODIFIER_StormKing_Dynmod
					}
				}
				log = "[GetDateText]: ROOT: [Root.GetName] on_weekly_MGY: MGY_Misc_DealWithStorm_effect"
				MGY_Misc_DealWithStorm_effect = yes
			}
			if = {
				#To update how much food is gained per week
				limit = {
					has_completed_focus = MGY_AgrarianCensus
				}
				MGY_Granary_OnWeek_effect = yes
			}
		}
	}
}
