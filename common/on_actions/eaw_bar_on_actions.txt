on_actions = {
	on_startup = {
		effect = {
			BAR = {
				bar_expedition_init = yes
				bar_activities_init = yes
				if = {
					limit = {
						BAR = {
							is_ai = no
						}
					}
					activate_mission = BAR_ambrosius_boredom
				}
			}
		}
	}
	on_monthly_BAR = {
		effect = {
			log = "[GetDateText]: ROOT: [Root.GetName] on_monthly_BAR"
			if = {
				limit = {
					original_tag = BAR
				}
				if = {
					limit = { 
						country_exists = BAR
						BAR = {
							has_country_flag = barrad_early_game
							has_country_flag = barrad_silver_star_alive
						}
					}
					if = {
						limit = {
							OR = {
								AND = {
									has_country_flag = barrad_asinti_alive
									check_variable = { BAR_relations_asinti_silver < 70 }
								}
								AND = {
									has_country_flag = barrad_leopold_alive
									check_variable = { BAR_relations_silver_leopold < 70 }
								}
								AND = {
									has_country_flag = barrad_viira_alive
									check_variable = { BAR_relations_silver_viira < 70 }
								}
							}
						}
						if = {
							limit = {
								has_country_flag = BAR_silver_sad
							}
							country_event = { days = 1 random_days = 14 id = bar_silver_star_nightmares.11 }
						}
						else = {
							country_event = { days = 1 random_days = 14 id = bar_silver_star_nightmares.1 }
						}
					} else = {
						BAR = {
							clr_country_flag = BAR_silver_sad
						}
					}
				}
				if = {
					limit = {
						country_exists = BAR
						BAR = {
							has_variable = viira_mana_overload
						}
					}
					BAR = {
						bar_viira_spells_overload_monthly_tickdown = yes
					}	
				}
				if = {
					limit = {
						BAR = {
							has_war = yes
							has_country_flag = BAR_asinti_kill_count
						}
					}
					BAR = {
						set_variable = { inflicted_casualties_last_month = inflicted_casualties }
						count_inflicted_casualties_on_current_enemies = yes
						subtract_from_variable = { inflicted_casualties_last_month = inflicted_casualties }
						multiply_variable = { inflicted_casualties_last_month = -1 }
						if = {
							limit = {
								check_variable = { BAR_asinti_blood_resource < 2000000 }
							}
							add_to_variable = { BAR_asinti_blood_resource = inflicted_casualties_last_month }
						}
						else = {
							set_variable = { BAR_asinti_blood_resource = 2000000 }
						}
					}	
				}
				if = {
					limit = {
						BAR = {
							has_country_flag = bar_viira_investigation_step_12
							NOT = { has_country_flag = bar_viira_investigation_step_13 }
							nuclear_reactor > 1
						}
					}
					set_country_flag = bar_viira_investigation_step_13
				}
				if = {
					limit = {
						has_power_balance = { id = bar_sane_ambrosius_balance }
						has_war = yes
					}
					if = {
						limit = {
							has_idea = extensive_conscription
						}
						BAR = {	
							add_power_balance_value = {
								id = bar_sane_ambrosius_balance
								value = -0.01
							}
						}
					}
					if = {
						limit = {
							has_idea = service_by_requirement
						}
						BAR = {	
							add_power_balance_value = {
								id = bar_sane_ambrosius_balance
								value = -0.02
							}
						}
					}
					if = {
						limit = {
							has_idea = all_adults_serve
						}
						BAR = {	
							add_power_balance_value = {
								id = bar_sane_ambrosius_balance
								value = -0.03
							}
						}
					}
					if = {
						limit = {
							has_idea = scraping_the_barrel
						}
						BAR = {	
							add_power_balance_value = {
								id = bar_sane_ambrosius_balance
								value = -0.04
							}
						}
					}
					if = {
						limit = {
							has_idea = tot_economic_mobilisation
						}
						BAR = {	
							add_power_balance_value = {
								id = bar_sane_ambrosius_balance
								value = -0.01
							}
						}
					}
				}
				if = {
					limit = {
						has_country_flag = BAR_undead_convert
					}
					set_temp_variable = {
						bar_ss_temp_factories_count = 0
					}
					every_controlled_state = {
						limit = {
							is_core_of = BAR
						}
						add_to_temp_variable = {
							bar_ss_temp_factories_count = arms_factory_level
						} 
						add_to_temp_variable = {
							bar_ss_temp_factories_count = industrial_complex_level
						} 
					}
					multiply_temp_variable = {
						bar_ss_temp_factories_count = 50
					}
					set_variable = {
						bar_ss_required_factory_pop = bar_ss_temp_factories_count
					}
					if = {
						limit = {
							check_variable = {
								bar_ss_pop_quantity > bar_ss_required_factory_pop
							}
						}
						set_variable = {
							bar_ss_workforce_factor = 100
						}	
					} else = {
						set_temp_variable = {
							bar_ss_employment_percentage = bar_ss_pop_quantity
						}	
						divide_temp_variable = {
							bar_ss_employment_percentage = bar_ss_required_factory_pop
						}
						set_variable = {
							bar_ss_workforce_factor = bar_ss_employment_percentage
						}
					}
				}
			}	
		}
	}
	
	on_declare_war = {
		effect = {
			if = {
				limit = {
					ROOT = { tag = BAR }
					has_power_balance = { id = bar_sane_ambrosius_balance }
				}
				BAR = {	
					add_power_balance_value = {
						id = bar_sane_ambrosius_balance
						value = -0.05
					}
				}
			}
		}
	}

	on_liberate  = {
		effect = {
			if = {
				limit = {
					FROM  = { tag = BAR }
					has_power_balance = { id = bar_sane_ambrosius_balance }
				}
				BAR = {	
					add_power_balance_value = {
						id = bar_sane_ambrosius_balance
						value = 0.02
					}
				}
			}
		}
	}
	
	on_peace  = {
		effect = {
			if = {
				limit = {
					THIS = { tag = BAR }
					has_power_balance = { id = bar_sane_ambrosius_balance }
				}
				BAR = {	
					add_power_balance_value = {
						id = bar_sane_ambrosius_balance
						value = 0.02
					}
				}
			}
		}
	}
	
	on_nuke_drop = {
		effect = {
			if = {
				limit = {
					ROOT = { tag = BAR }
					has_power_balance = { id = bar_sane_ambrosius_balance }
				}
				BAR = {	
					add_power_balance_value = {
						id = bar_sane_ambrosius_balance
						value = -0.03
					}
				}
			}
		}
	}
}
