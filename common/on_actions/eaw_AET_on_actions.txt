	on_annex = {
		effect = {
			if = {
				limit = {
					AND = {
						ROOT = { has_country_flag = AET_style_meter_unlocked }
					}
				}
				set_variable = {
					AET_from_states_annexed = FROM.num_owned_states
				}
				set_variable = {
					AET_state_count_loop = 0
				}
				set_variable = {
					AET_state_loop_subtractor = 1
				}
				while_loop_effect = { #did you know you can approximate square roots to the nearest integer rounded-down via repeated subtraction?
					limit = {
						check_variable = {
							AET_from_states_annexed > 1
						}
					}
					subtract_from_variable = {
						AET_from_states_annexed = AET_state_loop_subtractor
					}
					add_to_variable = {
						AET_state_count_loop = 1
					}
					add_to_variable = {
						AET_state_loop_subtractor = 2
					}
					set_variable = {
						AET_style_score_add = AET_state_count_loop
					}
				}
				AET_style_meter_add = yes
			}
		}
	}

	on_daily_AET = {
		effect = {
			if = {
				limit = {
					has_country_flag = AET_HLDV_active
				}
	
				# cbb with date variables
				if = {
					limit = {
						AET_is_in_maar_mode = yes
						has_country_flag = AET_maar_sacrifice
					}
					subtract_from_variable = { AET_HLDV_days_remaining = 1 }
					set_variable = { AET_HLDV_cooldown_progress = 120 }
					subtract_from_variable = { AET_HLDV_cooldown_progress = AET_HLDV_days_remaining }
				}
				else_if = {
					limit = {
						AET_is_in_trimmel_mode = yes
						any_of = {
							array = AET_HLDV_trimmel_timer_array_status
							value = check_v
							check_variable = { check_v = 1 }
						}
					}
					AET_HLDV_trimmel_daily_pulse = yes
				}

				add_to_variable = { AET_HLDV_dirty = 0.00001 }
			}
		}
