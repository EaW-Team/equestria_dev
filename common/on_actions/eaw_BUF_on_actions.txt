on_actions = {
	on_startup = {
		effect = {
			BUF = {
				add_ideas = poor_science_base
				adjust_research_modifier = yes
			}
		}
	}

	on_puppet = {
		effect = {
			if = {
				limit = {
					original_tag = BUF
					FROM = {
						original_tag = EQS
						has_government = democratic
					}
				}
				country_event = bufstate.36
			}
		}
	}

	on_release_as_puppet = {
		effect = {
			if = {
				limit = {
					original_tag = BUF
					FROM = {
						original_tag = EQS
						has_government = democratic
					}
				}
				country_event = bufstate.36
			}
		}
	}
	on_daily_BUF = {
		effect = {
			if = {
				limit = {
					has_focus_tree = BUF_civil_war_tree
					NOT = {
						has_war_with = LSP
						has_war_with = SCS
						country_exists = SCS
						country_exists = LSP
					}

				}
				country_event = cw_strongheart.2
			}
		}
	}
	
	on_weekly_BUF = {
		#BUFFALO BOP
		effect = {
			#####################
			### THUNDERHOOVES ###
			#####################
			if = { #THUNDERHOOVES - GOOD = No Fash & No Commie |||| BAD = Some Fash OR Commie // Losing Cores
				limit = {
					has_country_leader = {
						ruling_only = yes
						character = BUF_chief_thunderhooves
					}
				}				
				#### BAD ####

				#CONCERNING SUPPORT FOR EXTREMISTS
				if = {
					limit = {
						OR = {
							communism > 0.09
							fascism > 0.09
						}
					}
					#Add modifier (only once)
					if = {
						limit = {
							NOT = {
								has_power_balance_modifier = {
									id = BUF_power_balance
									modifier = BUF_bop_modifier_concerning_support_for_extremists
								}
							}
						}
						add_power_balance_modifier = {
							id = BUF_power_balance
							modifier = BUF_bop_modifier_concerning_support_for_extremists
						}
					}
					#Remove opposite modifier if necessary
					if = {
						limit = {
							has_power_balance_modifier = {
								id = BUF_power_balance
								modifier = BUF_bop_modifier_eradicate_extremists
							}
						}
						remove_power_balance_modifier = {
							id = BUF_power_balance
							modifier = BUF_bop_modifier_eradicate_extremists
						}
					}
				}
				else = {
					if = {
						limit = {
							has_power_balance_modifier = {
								id = BUF_power_balance
								modifier = BUF_bop_modifier_concerning_support_for_extremists
							}
						}
						remove_power_balance_modifier = {
							id = BUF_power_balance
							modifier = BUF_bop_modifier_concerning_support_for_extremists
						}
					}
				}
				#### GOOD ####
				#ERADICATE EXTREMISTS
				if = {
					limit = {
						fascism < 0.01
						communism < 0.01
					}
					if = {
						limit = {
							NOT = {
								has_power_balance_modifier = {
									id = BUF_power_balance
									modifier = BUF_bop_modifier_eradicate_extremists
								}
							}
						}
						add_power_balance_modifier = {
							id = BUF_power_balance
							modifier = BUF_bop_modifier_eradicate_extremists
						}
					}
					if = {
						limit = {
							has_power_balance_modifier = {
								id = BUF_power_balance
								modifier = BUF_bop_modifier_concerning_support_for_extremists
							}
						}
						remove_power_balance_modifier = {
							id = BUF_power_balance
							modifier = BUF_bop_modifier_concerning_support_for_extremists
						}
					}
				}
				else = {
					if = {
						limit = {
							has_power_balance_modifier = {
								id = BUF_power_balance
								modifier = BUF_bop_modifier_eradicate_extremists
							}
						}
						remove_power_balance_modifier = {
							id = BUF_power_balance
							modifier = BUF_bop_modifier_eradicate_extremists
						}
					}
				}
			}
			###################
			### STRONGHEART ###
			###################
			else_if = { #STRONGHEART - GOOD = Winning war against Fascists |||| BAD = Offensive war against Harmonists
				limit = {
					has_country_leader = {
						ruling_only = yes
						character = BUF_chieftess_strongheart
					}
				}
				#### BAD ####

				#TRAITORS TO HARMONY
				if = {
					limit = {
						any_enemy_country = {
							has_government = democratic
							has_defensive_war_with = BUF
						}
					}
					#Add modifier (only once)
					if = {
						limit = {
							NOT = {
								has_power_balance_modifier = {
									id = BUF_power_balance
									modifier = BUF_bop_modifier_traitors_to_harmony
								}
							}
						}
						add_power_balance_modifier = {
							id = BUF_power_balance
							modifier = BUF_bop_modifier_traitors_to_harmony
						}
					}
				}
				else = {
					if = {
						limit = {
							has_power_balance_modifier = {
								id = BUF_power_balance
								modifier = BUF_bop_modifier_traitors_to_harmony
							}
						}
						remove_power_balance_modifier = {
							id = BUF_power_balance
							modifier = BUF_bop_modifier_traitors_to_harmony
						}
					}
				}
				#### GOOD ####
				#ANTI-SUPREMACIST CRUSADE
				if = {
					limit = {
						has_war = yes
						any_enemy_country = {
							has_government = fascism
						}
						NOT = {
							any_war_score < 10
						}
					}
					if = {
						limit = {
							NOT = {
								has_power_balance_modifier = {
									id = BUF_power_balance
									modifier = BUF_bop_modifier_anti_supremacist_crusade
								}
							}
						}
						add_power_balance_modifier = {
							id = BUF_power_balance
							modifier = BUF_bop_modifier_anti_supremacist_crusade
						}
					}
				}
				else = {
					if = {
						limit = {
							has_power_balance_modifier = {
								id = BUF_power_balance
								modifier = BUF_bop_modifier_anti_supremacist_crusade
							}
						}
						remove_power_balance_modifier = {
							id = BUF_power_balance
							modifier = BUF_bop_modifier_anti_supremacist_crusade
						}
					}
				}
			}

			######################
			### ELDER GREYHORN ###
			######################
			else_if = { #GREYHORN - GOOD = Controlling non-starting Southwest state |||| BAD = Losing control of core state
				limit = {
					OR = {
						has_country_leader = {
							ruling_only = yes
							character = BUF_the_elder
						}
						has_country_leader = {
							ruling_only = yes
							character = BUF_elders_grandniece
						}
					}
				}
				#### BAD ####

				#FAILED PROMISES
				if = {
					limit = {
						any_core_state = {
							is_owned_by = ROOT
							NOT = { is_controlled_by = ROOT }
						}
					}
					if = {
						limit = {
							NOT = {
								has_power_balance_modifier = {
									id = BUF_power_balance
									modifier = BUF_bop_modifier_failed_promises
								}
							}
						}
						add_power_balance_modifier = {
							id = BUF_power_balance
							modifier = BUF_bop_modifier_failed_promises
						}
					}
				}
				else_if = {
					limit = {
						has_power_balance_modifier = {
							id = BUF_power_balance
							modifier = BUF_bop_modifier_failed_promises
						}
					}
					remove_power_balance_modifier = {
						id = BUF_power_balance
						modifier = BUF_bop_modifier_failed_promises
					}
				}
				#### GOOD ####
				#REVANCHIST HAWK
				if = {
					limit = {
						any_controlled_state = {
							BUF_state_is_southwestern_territory = yes
						}
					}
					if = {
						limit = {
							NOT = {
								has_power_balance_modifier = {
									id = BUF_power_balance
									modifier = BUF_bop_modifier_revanchist_hawk
								}
							}
						}
						add_power_balance_modifier = {
							id = BUF_power_balance
							modifier = BUF_bop_modifier_revanchist_hawk
						}
					}
				}
				else = {
					if = {
						limit = {
							has_power_balance_modifier = {
								id = BUF_power_balance
								modifier = BUF_bop_modifier_revanchist_hawk
							}
						}
						remove_power_balance_modifier = {
							id = BUF_power_balance
							modifier = BUF_bop_modifier_revanchist_hawk
						}
					}
				}
			}
			# UNKNOWN LEADER -> KILL BOP
			else = {
				remove_power_balance = {
					id = BUF_power_balance
				}
			}

			#### CLEAN ANY OBSOLETE BOP MODIFIERS (based on country leader)
			BUF_clean_all_obsolete_bop_modifiers_effect = yes
		}
	}
	
	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					FROM.FROM = {
						has_dynamic_modifier = { modifier = BUF_partisan_tactics_modifier }
					}
					FROM = { tag = BUF }
				}
				FROM.FROM = {
					remove_dynamic_modifier = { modifier = BUF_partisan_tactics_modifier }
				}
				#Cost decrease for Partisan decisions in Buffalo
				hidden_effect = {
					FROM = {
						divide_variable = { BUF_partisan_decision_cost = BUF_partisan_decision_cost_increase }
						set_variable = { BUF_partisan_decision_cost_for_trigger = BUF_partisan_decision_cost } #For trigger purposes
						subtract_from_variable = { BUF_partisan_decision_cost_for_trigger = 1 } 
					}
				}
			}
		}
	}
}
