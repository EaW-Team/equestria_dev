on_actions = {
	on_daily_WNG = {
		effect = {
			if = {
				limit = {
					has_government = neutrality
					has_completed_focus = WNG_the_royal_intervention
				}
				set_temp_variable = { WNG_neutral_support = WNG.party_popularity@neutrality }
				multiply_temp_variable = { WNG_neutral_support = 100 }
				set_variable = {
					var = WNG_popular_support_max
					value = WNG_neutral_support
				}
				if = {
					limit = {
						check_variable = {
							var = WNG_popular_support_max
							value = WNG_popular_support_min
							compare = less_than
						}
					}
					set_variable = {
						var = WNG_popular_support_max
						value = WNG_popular_support_min
					}
				}
				clamp_variable = {
					var = WNG_popular_support
					min = WNG_popular_support_min
					max = WNG_popular_support_max
				}
			}
		}
	}
	on_daily_ERI = {
        effect = {
				if = {
					limit = {
						has_country_flag = ERI_prelude_war
							OR = {
								AND = {
									date > 1007.5.5
									date < 1007.5.10
								}
								ERI = {
									casualties_inflicted_by = {
										opponent = EYR
										thousands > 15
									}
								}
								EYR = {
									surrender_progress > 0.29
								}
							}
					}
                country_event = erie.3
				}
		}
    }
	on_daily_EYR = {
        effect = {
				if = {
					limit = {
						has_country_flag = EYR_prelude_war
							OR = {
								AND = {
									date > 1007.5.10
									date < 1007.5.25
								}
								EYR = {
									casualties_inflicted_by = {
										opponent = ERI
										thousands > 18
									}
								}
								ERI = {
									surrender_progress > 0.39
								}
							}
					}
                country_event = eyrie.3
				}
		}
    }
	on_daily_BAL = {
		effect = {
			if = {
				limit = {
					has_focus_tree = baltimare_focus
					NOT = {
						has_war_with = JUN
						has_war_with = CES
						has_war_with = NMS
						count_triggers = {
							amount = 2
							country_exists = CES
							country_exists = JUN
							country_exists = NMS
						}
					}
				}
				country_event = baltimare.20
			}
		}
	}
	on_daily_JUN = {
		effect = {
			if = {
				limit = {
					has_focus_tree = jungle_focus
					NOT = {
						has_war_with = BAL
						has_war_with = CES
						has_war_with = NMS
						count_triggers = {
							amount = 2 #at least 2 triggers are true
							country_exists = CES
							country_exists = BAL
							country_exists = NMS
						}
					}
					
				}
				country_event = jungle.8
			}
		}
	}
	on_daily_NMS = {
		effect = {
			if = {
				limit = {
					has_focus_tree = nightmare_focus
					NOT = {
						has_war_with = BAL
						has_war_with = JUN
						has_war_with = CES
						count_triggers = {
							amount = 2 #at least 2 triggers are true
							country_exists = CES
							country_exists = BAL
							country_exists = JUN
						}
					}
				}
				country_event = supporters.9
			}
		}
	}
}
