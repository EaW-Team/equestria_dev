	on_daily_AET = {
		effect = {
			if = {
				limit = {
					has_completed_focus = AET_style_meter
				}
				count_inflicted_casualties_on_current_enemies = yes

				##find today's kills
				set_variable = { AET_kills_total_count = inflicted_casualties }
				set_variable = { AET_kills_today = AET_kills_total_count }
				subtract_from_variable = { AET_kills_today = AET_kills_yesterday_count }
				set_variable = { AET_kills_total_count = AET_kills_yesterday_count }

				##set up the week's kills
				set_variable = { AET_kills_day_7 = AET_kills_day_6 }
				set_variable = { AET_kills_day_6 = AET_kills_day_5 }
				set_variable = { AET_kills_day_5 = AET_kills_day_4 }
				set_variable = { AET_kills_day_4 = AET_kills_day_3 }
				set_variable = { AET_kills_day_3 = AET_kills_day_2 }
				set_variable = { AET_kills_day_2 = AET_kills_day_1 }
				set_variable = { AET_kills_day_1 = AET_kills_today }

				##create moving average
				set_variable = { AET_moving_average = AET_kills_day_1 }
				add_to_variable = { AET_moving_average = AET_kills_day_2 }
				add_to_variable = { AET_moving_average = AET_kills_day_3 }
				add_to_variable = { AET_moving_average = AET_kills_day_4 }
				add_to_variable = { AET_moving_average = AET_kills_day_5 }
				add_to_variable = { AET_moving_average = AET_kills_day_6 }
				add_to_variable = { AET_moving_average = AET_kills_day_7 }
				divide_variable = { AET_moving_average = 7 }

				##find style
				divide_variable = { AET_kills_today = AET_moving_average }
				divide_variable = { AET_kills_today = 4 }
				round_variable = AET_kills_today
	
				##find casualty stank
				if = {
					limit = {
						check_variable = { AET_casualty_stank > 0 }
					}
					subtract_from_variable = { AET_kills_today = AET_casualty_stank }
					clamp_variable = { AET_kills_today = 0 }
				}
				set_country_flag = AET_casualty_stank_stunk
				add_to_variable = { AET_style_meter_number = AET_kills_today }
			}
		}
	}
