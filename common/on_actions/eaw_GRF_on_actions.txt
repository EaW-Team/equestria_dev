on_actions = {
	on_puppet = {
		effect = {
			GRF_remove_has_equipment_market_trade_desire_ai_strategy = yes
		}
	}

	on_release_as_puppet = {
		effect = {
			GRF_remove_has_equipment_market_trade_desire_ai_strategy = yes
		}
	}

	on_release_as_free = {
		effect = {
			GRF_remove_has_equipment_market_trade_desire_ai_strategy = yes
		}
	}

	on_government_change = {
		effect = {
			GRF_remove_has_equipment_market_trade_desire_ai_strategy = yes
		}
	}

	on_ruling_party_change = {
		effect = {
			GRF_remove_has_equipment_market_trade_desire_ai_strategy = yes
		}
	}

	on_capitulation = {
		effect = {
			if = {
				limit = { 
					tag = GRA
					country_exists = GRB
					FROM = {
						OR = {
							tag = STG
							tag = YAK
						}
					}
				}
				GRB = {
					country_event = { id = griffons.142 days = 5 }
				}
			}
		}
	}

	on_annex = {
		effect = {
			if = {
				limit = {
					tag = GRF
					FROM = { tag = GFF }
					NOT = { has_global_flag = GRF_civil_war_begun }
				}
				GRF = {
					country_event = {
						id = griffons.168
						days = 14
					}
				}
			}
		}
	}

	on_civil_war_end = {
		effect = {
			if = {		#GRF disband the militia after the civil war
				limit = {
					original_tag = GRF
					has_country_flag = GRF_civil_war_flag
				}
				country_event = { id = griffons.142 days = 5 }
			}
		}
	}

	on_peace = {
		effect = {
			if = {		#GRF disband the militia after the civil war
				limit = {
					original_tag = GRF
					has_country_flag = GRF_civil_war_flag
				}
				country_event = { id = griffons.142 days = 5 }
			}
		}
	}
}
