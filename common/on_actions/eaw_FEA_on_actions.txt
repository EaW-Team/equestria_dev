on_actions = {
	on_startup = {
		effect = {
			FEA = {
				country_event = { id = feat.15 days = 60 random_hours = 2880 }#days to hours applied
				country_event = { id = feat.20 days = 72 random_hours = 144 }#days to hours applied
				country_event = { id = feat.21 days = 105 random_hours = 144 }#days to hours applied
			}
		}
	}

	on_subject_annexed = {
		effect = {
			if = {
				limit = {
					OR = {
						tag = STW
						tag = FEA
					}
					OR = {
						has_country_flag = FEA_STW_marriage
						FROM = { has_country_flag = FEA_STW_marriage }
					}
					FROM = {
						OR = {
							tag = STW
							tag = FEA
						}
						OR = {
							has_country_leader = { ruling_only = yes character = STW_gabriela_eagleclaw }
							has_idea = FEA_beloved_monarch
							has_idea = FEA_beloved_monarch_2
							has_idea = FEA_beloved_monarch_3
						}
					}
					OR = {
						has_country_leader = { ruling_only = yes character = STW_gabriela_eagleclaw }
						has_idea = FEA_beloved_monarch
						has_idea = FEA_beloved_monarch_2
						has_idea = FEA_beloved_monarch_3
					}
				}
				FROM = {
					set_country_flag = FEA_STW_union
					set_cosmetic_tag = FEA_STW_union
				}
			}
		}
	}

	on_puppet = {
		effect = {
			if = {
				limit = {
					original_tag = FEA
					OR = {
						has_cosmetic_tag = FEA_PvdA_old_flag
						has_cosmetic_tag = FEA_PvdA_old_flag_GRI_Puppet
					}
					FROM = { has_government = communism }
				}
				drop_cosmetic_tag = yes
			}
		}
	}

	on_release_as_puppet = {
		effect = {
			if = {
				limit = {
					FROM = { has_country_flag = FEA_STW_union }
					OR = {
						tag = STW
						tag = FEA
					}
				}
				FROM = {
					set_autonomy = {
						target = ROOT
						autonomy_state = autonomy_personal_union_eaw
					}
					clr_country_flag = FEA_STW_union
					drop_cosmetic_tag = yes
					reapply_secondary_cosmetic_tags = yes
				}
			}
		}
	}

	on_release_as_free = {
		effect = {
			if = {
				limit = {
					FROM = { has_country_flag = FEA_STW_union }
					OR = {
						tag = STW
						tag = FEA
					}
				}
				FROM = {
					clr_country_flag = FEA_STW_union
					drop_cosmetic_tag = yes
					reapply_secondary_cosmetic_tags = yes
				}
			}
		}
	}

	on_government_change = {
		effect = {
			if = {
				limit = {
					original_tag = FEA
					OR = {
						has_cosmetic_tag = FEA_PvdA_old_flag
						has_cosmetic_tag = FEA_PvdA_old_flag_GRI_Puppet
					}
					OR = {
						NOT = { has_government = communism }
						AND = {
							is_subject = yes
							overlord = { has_government = communism }
						}
					}
				}
				drop_cosmetic_tag = yes
			}
		}
	}
}
