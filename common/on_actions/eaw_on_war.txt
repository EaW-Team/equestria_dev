on_actions = {
	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					OR = { # AF26 - This idea prevents getting into wars
						ROOT = {
							has_idea = EQS_ED_the_shield_of_harmony
						}
						FROM = {
							has_idea = EQS_ED_the_shield_of_harmony
						}
					}
					NOT = { ROOT = { tag = KTZ } }
					NOT = { FROM = { tag = KTZ } }
				}
				white_peace = FROM
				remove_wargoal = {
					type = all
					target = FROM
				}
				FROM = {
					every_allied_country = {
						ROOT = {
							remove_wargoal = {
								type = all
								target = PREV
							}
						}
					}
				}
			}
		}
	}	

	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					OR = { # AF26 - This prevents EQC and GRW getting into war too early
						AND = {
							ROOT = {
								tag = EQC
							}
							FROM = {
								tag = GRW
							}
						}
						AND = {
							ROOT = {
								tag = GRW
							}
							FROM = {
								tag = EQC
							}
						}
					}
					EQC = {
						has_war_with = GRW
						NOT = {
							has_completed_focus = EQC_ED_destroy_heretical_sites
						}
					}
				}
				white_peace = FROM
				remove_wargoal = {
					type = all
					target = FROM
				}
			}
		}
	}
}
