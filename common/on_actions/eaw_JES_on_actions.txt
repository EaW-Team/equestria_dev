on_actions = {
	on_declare_war = {
		effect = {
			if = {
				limit = {
					tag = FRN
					FROM = {
						tag = JES
					}
					NOT = {
						has_country_flag = FRN_JES_war_lost
					}
				}
				FRN = {
					activate_targeted_decision = {
						target = JES
						decision = FRN_JES_war_timer
					}
				}
				JES = {
					country_event = {
						id = tarrin.1
						days = 1
					}
				}
			}
		}
	}
}
