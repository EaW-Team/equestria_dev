		effect = {
			ERI = {
				add_opinion_modifier = { target = EYR modifier = mortal_enemies }
			}
		}
	}

	on_declare_war = {
		effect = {
			if = {
				limit = {
					original_tag = ERI
					FROM = { original_tag = EYR }
				}
				set_country_flag = { flag = block_white_peace value = 1 days = 21 }
				FROM = { set_country_flag = { flag = block_white_peace value = 1 days = 21 } }
			}
		}
	}
}
