on_actions = {
	on_startup = {
		effect = {
			ZAI = {
				set_sand_mechanics = yes
			}
		}
	}

	on_weekly_ZAI = {
		effect = {
			if = {
				limit = {
					OR = {
						has_cosmetic_tag = Terathopia
						has_cosmetic_tag = Terathopia_2
						has_cosmetic_tag = Terathopia_3
					}
				}
				if = {
					limit = {
						has_idea = ZAI_special_designer
					}
					add_equipment_to_stockpile = {
						type = monster_equipment
						amount = 25
					}
				}
				if = {
					limit = {
						NOT = {
							has_country_flag = ZAI_event_triggered
						}
						has_war = yes
						casualties > 99
					}
					country_event = {
						id = eaw_debug.25
					}
				}
				else_if = {
					limit = {
						NOT = {
							has_country_flag = ZAI_event_2_triggered
						}
						controls_state = 697
						has_navy_size = {
							size > 29
						}
					}
					country_event = {
						id = cityfall.1010
					}
				}
				else_if = {
					limit = {
						NOT = {
							has_country_flag = ZAI_doctrinal_conflict
						}
						has_war = no
						has_completed_focus = NE_attack_east
						has_completed_focus = NE_attack_north
						has_completed_focus = NE_attack_northwest
						has_completed_focus = NE_attack_southwest
					}
					country_event = {
						id = eaw_debug.20
					}
				}
			}
		}
	}

	on_monthly_ZAI = {
		effect = {
			if = {
				limit = {
					OR = {
						has_cosmetic_tag = Terathopia
						has_cosmetic_tag = Terathopia_2
						has_cosmetic_tag = Terathopia_3
					}
				}
				for_each_scope_loop = {
					array = global.countries
					if = {
						limit = {
							NOT = {
								original_tag = ZAI
								original_tag = ZZZ
							}
						}
						add_opinion_modifier = { target = ZAI modifier = mortal_enemies }
						add_opinion_modifier = { target = ZAI modifier = embargo }
						add_opinion_modifier = { target = ZAI modifier = CHN_Embargo }
						reverse_add_opinion_modifier = { target = ZAI modifier = mortal_enemies }
						reverse_add_opinion_modifier = { target = ZAI modifier = embargo }
						reverse_add_opinion_modifier = { target = ZAI modifier = CHN_Embargo }
					}
				}
			}
		}
	}
}
