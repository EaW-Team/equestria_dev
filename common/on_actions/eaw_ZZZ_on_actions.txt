on_actions = {
	on_startup = {
		effect = {
			ZZZ = {
				if = {
					limit = {
						is_ai = no
					}
					country_event = utility.20
				}
			}
		}
	}

	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					OR = {
						tag = ZZZ
						FROM = { tag = zzz }
					}
				}
				white_peace = FROM
			}
		}
	}

	on_state_control_changed = {
		effect = {
			#This state needs to be impossible for any country but ZZZ to control.
			if = {
				limit = {
					FROM = { tag = ZZZ }
					FROM.FROM = { state = 1209 }
				}
				log = "State 1209 was taken from ZZZ. This should never happen!"
				ZZZ = { transfer_state = 1209 }
			}
		}
	}

	on_guarantee = {
		effect = {
			#ZZZ safety check
			if = {
				limit = {
					FROM = {
						tag = ZZZ
					}
				}
				diplomatic_relation = {
					country = ZZZ
					relation = guarantee
					active = no
				}
			}
		}
	}

	on_join_faction = {
		effect = {
			#ZZZ safety check
			if = {
				limit = { tag = ZZZ }
				leave_faction = yes
			}
		}
	}

	on_send_volunteers = {
		effect = {
			if = {
				limit = {
					tag = ZZZ
				}
				recall_volunteers_from = FROM
			}
		}
	}

	on_subject_autonomy_level_change = {
		effect = {
			#ZZZ safety check
			if = {
				limit = { tag = ZZZ }
				FROM = {
					set_autonomy = {
						target = ROOT
						autonomous_state = autonomy_free
						end_wars = yes
						end_civil_wars = yes
					}
				}
			}
		}
	}

	on_generate_wargoal = {
		effect = {
			#ZZZ safety check
			if = {
				limit = {
					FROM = { tag = ZZZ }
				}
				remove_wargoal = {
					type = all
					target = FROM
				}
			}
		}
	}

	on_create_faction = {
		effect = {
			#This keeps ZZZ's hidden faction at the back of the list
			ZZZ = {
				country_event = {
					id = utility.19
					hours = 1
				}
			}
		}
	}
}
