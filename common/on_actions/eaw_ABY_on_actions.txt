on_actions = {
	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					OR = {
						AND = {
							original_tag = WNG
							FROM = {
								original_tag = ABY
							}
						}
						AND = {
							original_tag = ABY
							FROM = {
								original_tag = WNG
							}
						}
					}
				}
				ABY = {
					if = {
						limit = {
							has_idea = ABY_risk_of_famine
						}
						country_event = {
							id = abyssinia.44
							days = 1
						}
					}
					if = {
						limit = {
							has_government = fascism
							NOT = {
								has_country_flag = ABY_unleashed_potential
							}
						}
						country_event = {
							id = abyssinia.144
							days = 3
						}
					}
					else_if = {
						limit = {
							has_government = communism
							NOT = {
								has_country_flag = ABY_conscripts_raised
							}
						}
						country_event = {
							id = abyssinia.175
							days = 3
						}
					}
				}
			}
		}
	}

	on_army_leader_lost_combat = {
		effect = {
			if = {
				limit = {
					FROM = {
						original_tag = ABY
						NOT = { has_country_flag = ABY_general_killed }
					}
				}
				if = {
					limit = {
						has_character_flag = ABY_doomed_to_die
						set_temp_variable = { chance = 0.50 }
						check_variable = { random < chance }
						NOT = { has_trait = wounded }
					}
					unit_leader_event = { id = abyssinia.33 }
				}
			}
		}
	}

	on_capitulation = {
		effect = {
			if = {
				limit = {
					original_tag = ABY
					has_country_flag = ABY_limited_MDL_war_flag
				}
				clr_country_flag = ABY_limited_MDL_war_flag
			}
			if = {
				limit = {
					original_tag = ABY
					ABY = { is_subject = no }
					FROM = { original_tag = WNG }
				}
				FROM = {
					set_country_flag = { flag = bypass_annex_compliance days = 1 value = 1 }
					annex_country = { target = ROOT }
				}
				if = {
					limit = { has_cosmetic_tag = ABY_patriotic_revolt }
					drop_cosmetic_tag = yes
					set_cosmetic_tag = ABY_default_cosmetic
				}
			}
			if = {
				limit = {
					original_tag = ABY
					FROM = { has_country_flag = SON_ABY_war_white_peace_flag }
				}
				ABY = {
					overlord = { country_event = zarhay.152 }
				}
			}
			if = {
				limit = {
					OR = {
						original_tag = EQS
						original_tag = EQC
					}
					has_government = democratic
					ABY = {
						exists = yes
						has_capitulated = no
						NOT = {
							any_owned_state = {
								OR = {
									is_controlled_by = FROM
									CONTROLLER = { is_ally_with = FROM }
								}
							}
						}
						has_war_with = FROM
					}
				}
				ABY = {
					if = {
						limit = { is_in_faction_with = EQS }
						leave_faction = yes
					}
					else_if = {
						limit = { is_in_faction_with = EQC }
						leave_faction = yes
					}
					if = {
						limit = { has_war_with = FROM }
						set_country_flag = { flag = bypass_on_peaceconference_ended days = 1 value = 1 }
						white_peace = FROM
						remove_wargoal = {
							type = all
							target = FROM
						}
					}
				}
			}
			else_if = {
				limit = {
					original_tag = ABY
					has_war_with = FROM
					OR = {
						is_in_faction_with = EQS
						is_in_faction_with = EQC
					}
					FROM = { is_zebrican_continent_country = yes }
				}
				leave_faction = yes
				if = {
					limit = {
						EQS = { has_war_with = FROM }
					}
					EQS = {
						set_country_flag = { flag = bypass_on_peaceconference_ended days = 1 value = 1 }
						white_peace = FROM
						every_other_country = {
							limit = {
								has_war_with = FROM
								is_in_faction_with = EQS
							}
							set_country_flag = { flag = bypass_on_peaceconference_ended days = 1 value = 1 }
							white_peace = FROM
						}
					}
				}
				else_if = {
					limit = {
						EQC = { has_war_with = FROM }
					}
					EQC = {
						set_country_flag = { flag = bypass_on_peaceconference_ended days = 1 value = 1 }
						white_peace = FROM
						every_other_country = {
							limit = {
								has_war_with = FROM
								is_in_faction_with = EQC
							}
							set_country_flag = { flag = bypass_on_peaceconference_ended days = 1 value = 1 }
							white_peace = FROM
						}
					}
				}
			}
		}
	}

	on_capitulation_immediate = {
		effect = {
			if = {
				limit = {
					original_tag = MDL
					FROM = { original_tag = ABY }
				}
				FROM = { country_event = abyssinia.49 }
			}
			if = {
				limit = {
					OR = {
						any_occupied_country = { original_tag = ABY }
						has_war_with = ABY
						has_war_with = ABP
					}
					NOT = { is_zebrican_continent_country = yes }
					NOT = { has_global_flag = katztime } #Not during Katzen AF
				}
				if = {
					limit = {
						ABP = { exists = yes }
					}
					every_owned_state = {
						limit = {
							OR = {
								is_core_of = ABY
								is_claimed_by = ABY
							}
						}
						transfer_state_to = ABP
					}
				}
				else = {
					every_owned_state = {
						limit = {
							OR = {
								is_core_of = ABY
								is_claimed_by = ABY
							}
						}
						transfer_state_to = ABY 
					}
				}
				ABY = {
					if = {
						limit = { has_cosmetic_tag = ABY_patriotic_revolt }
						drop_cosmetic_tag = yes
						set_cosmetic_tag = ABY_default_cosmetic
					}
					set_capital = {
						state = 775
						remember_old_capital = no
					}
				}
			}
		}
	}

	on_ruling_party_change = {
		effect = {
			if = {
				limit = { original_tag = ABY }
				if = {
					limit = {
						NOT = {
							has_government = neutrality
							has_government = democratic
						}
						has_power_balance = { id = ABY_neutral_harmony }
					}
					remove_power_balance = { id = ABY_neutral_harmony }
				}
			}
		}
	}

	on_puppet = {
		effect = {
			if = {
				limit = {
					ROOT = { original_tag = MGY }
					FROM = { original_tag = ABY }
				}
				generate_character = {
					token_base = ABY_MGY_ankha
					name = ABY_MGY_ankha
				}
				if = {
					limit = { has_character = ABY_MGY_ankha }
					set_portraits = {
						character = ABY_MGY_ankha 
						army = { 
							large = "gfx/leaders/ABY/oh_no.tga"
						}
					}
				}
				add_country_leader_role = {
					character = ABY_MGY_ankha
					country_leader = {
						expire = "1965.1.1"
						ideology = conservatism
						desc = ABY_MGY_ankha_desc
					}
					promote_leader = yes
				}
				add_country_leader_role = {
					character = ABY_MGY_ankha
					country_leader = {
						expire = "1965.1.1"
						ideology = despotism
						desc = ABY_MGY_ankha_desc
					}
					promote_leader = yes
				}
				add_country_leader_role = {
					character = ABY_MGY_ankha
					country_leader = {
						expire = "1965.1.1"
						ideology = nazism
						desc = ABY_MGY_ankha_desc
					}
					promote_leader = yes
				}
			}
			if = {
				limit = {
					ROOT = { original_tag = ABY }
				}
				if = {
					limit = {
						OR = {
							has_cosmetic_tag = ABY_patriotic_revolt
							has_cosmetic_tag = ABY_bitchdom
						}
					}
					drop_cosmetic_tag = yes
				}
			}
		}
	}

	on_civil_war_end = {
		effect = {
			if = {
				limit = { original_tag = ABY }
				if = {
					limit = {
						FROM = { has_government = communism }
					}
					country_event = abyssinia.129
				}
				else_if = {
					limit = {
						has_government = communism
						NOT = { has_country_flag = ABY_first_civil_war_end }
					}
					country_event = abyssinia.159
				}
				else_if = {
					limit = {
						has_government = communism
						has_focus_tree = ABY_cummunism_real_civil_war_tree
						OR = {
							AND = {
								FROM = { original_tag = ABY }
								var:global.donkey_revolt_var = { exists = no }
							}
							AND = {
								FROM = { original_tag = ASS }
								var:global.cat_revolt_var = { exists = no }
							}
						}
					}
					country_event = abyssinia.183
				}
				if = {
					limit = { has_cosmetic_tag = ABY_patriotic_revolt }
					drop_cosmetic_tag = yes
					set_cosmetic_tag = ABY_default_cosmetic
				}
			}
			else_if = {
				limit = {
					original_tag = ASS
					has_cosmetic_tag = ABY_new_donkeystan
					FROM = {
						original_tag = ABY
						has_government = communism
					}
				}
				if = {
					limit = {
						ASS = {
							exists = yes
							has_government = fascism
							NOT = { has_war_with = ROOT }
						}
					}
					ASS = { country_event = abyssinia.184 }
				}
			}
		}
	}
