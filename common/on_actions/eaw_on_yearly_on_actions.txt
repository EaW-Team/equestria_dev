on_actions = {
	# Months counter initialization
	on_startup = {
		effect = {
			set_variable = { global.month = 1 }
		}
	}

	# ROOT is country, this is ran for every country
	on_monthly = {
		effect = {
			# Months counter
			if = {
				limit = {
					NOT = { has_global_flag = on_monthly_done }
				}
				log = "[GetDateText]: global on_monthly"
				set_AI_strategies_on_every_possible_country = yes
				add_to_variable = { global.month = 1 }
				if = {
					limit = {
						check_variable = { global.month > 12 }
					}
					set_variable = { global.month = 1 }
				}
				set_global_flag = { flag = on_monthly_done value = 1 days = 1 }
			}
			if = {
				limit = {
					date > 1010.1.1
					date < 1010.3.1
					779 = {
						owner = {
							tag = ROOT
						}
						state_population > 0
					}
				}
				779 = {
					owner = {
						country_event = { id = abyssinia.121 days = 8 }
					}
				}
			}
			# Disbanding land units, trigger inside
			country_event = { id = utility.14 days = 1 random = 480 }

			build_air_base = yes
			if = {
				limit = {
					has_war = yes
					num_of_civilian_factories_available_for_projects > 29
				}
				country_event = { id = utility.18 days = 9 random = 72 }
				if = {
					limit = {
						num_of_civilian_factories_available_for_projects > 44
					}
					country_event = { id = utility.18 days = 19 random = 72 }
				}
			}

			# ON YEARLY
			if = {
				#don't touch!
				limit = {
					NOT = {
						check_variable = { global.last_year = global.year }
						has_global_flag = on_yearly_done
					}
				}
				#this is done only ONCE per year, even though on_daily fires for every country
				#will also be done on startup
				log = "[GetDateText]: global on_yearly"

				#your code here
				SRS = { country_event = socrep_grif.76 }

				if = { # on_yearly_TBK
					limit = {
						TBK = {
							exists = yes
						}
					}
					TBK = { #place effects for tbk in here to simulate on_yearly_tbk
						if = {
							limit = {
								NOT = {
									check_variable = { global.year = 1007 }
								}
								OR = {
									has_completed_focus = TBK_house_of_trade
									has_completed_focus = TBK_criminal_ministry
								}
							}
							add_to_variable = {
								var = TBK_dirty_money
								value = TBK_yearly_money
							}
						}
					}
				}

				#don't touch!
				set_variable = { global.last_year = global.year }
				set_global_flag = { flag = on_yearly_done value = 1 days = 1 }
			}

			if = {
				limit = {
					check_variable = { occupied_countries^num > 0 }
				}
				core_all_occupied_through_compliance = yes
			}
		}
	}

	# ROOT is country, this is ran for every country
	on_weekly = {
		effect = {
			handle_cosmetic_tag_and_puppet_change_weekly = yes
			miscellaneous_on_weekly_effects = yes
			miniaturized_nukes_increased_production_effect = yes
			if = { ## Children of Concord
				limit = {
					original_tag = KIC
					has_country_flag = KIC_children_of_concord_flag
				}
				if = {
					limit = {
						check_variable = {
							var = COC_power_struggle_DR_var
							value = 2
							compare = greater_than_or_equals
						}
					}
					add_ideas = COC_PS_DR_hidden_buffs_1
				}
				if = {
					limit = {
						check_variable = {
							var = COC_power_struggle_DR_var
							value = 4
							compare = greater_than_or_equals
						}
					}
					add_ideas = COC_PS_DR_hidden_buffs_2
					if = {
						limit = {
							EQS = { exists = yes is_subject = no }
						}
						country_event = { id = kirin_council.320 hours = 1 }
					}
				}
				if = {
					limit = {
						check_variable = {
							var = COC_power_struggle_KF_var
							value = 2
							compare = greater_than_or_equals
						}
					}
					add_ideas = COC_PS_KF_hidden_buffs_1
				}
				if = {
					limit = {
						check_variable = {
							var = COC_power_struggle_KF_var
							value = 4
							compare = greater_than_or_equals
						}
					}
					add_ideas = COC_PS_KF_hidden_buffs_2
					if = {
						limit = {
							OR = {
								STG = { exists = yes is_subject = no }
								SRS = { exists = yes is_subject = no }
								GRW = { exists = yes is_subject = no }
							}
						}
						country_event = { id = kirin_council.319 hours = 1 }
					}
				}
			}

			if = { ## CFR
				limit = {
					country_exists = CFR
					tag = CFR
				}
				CFR = {
					if = {
						limit = {
							has_country_flag = CFR_imago_parastatal_available
						}
						for_each_scope_loop = {
							array = CFR.CFR_infiltrated_countries_array
							if = {
								limit = {
									num_of_military_factories > 9
								}
								set_variable = { CFR_fraudulent_arms_contract_AGENTS_REQ = num_of_military_factories }
								divide_variable = { CFR_fraudulent_arms_contract_AGENTS_REQ = 10 }
								multiply_variable = { CFR_fraudulent_arms_contract_AGENTS_REQ = 5 }
								round_variable = CFR_fraudulent_arms_contract_AGENTS_REQ
							}
						}
					}
					if = {
						limit = {
							has_country_flag = CFR_just_fuck_my_shit_up_darling_available
						}
						for_each_scope_loop = {
							array = CFR.CFR_infiltrated_countries_array
							if = {
								limit = {
									num_divisions > 9
								}
								set_variable = { CFR_provoke_army_defections_AGENTS_REQ = num_divisions }
								divide_variable = { CFR_provoke_army_defections_AGENTS_REQ = 10 }
								round_variable = CFR_provoke_army_defections_AGENTS_REQ
								set_variable = { CFR_provoke_army_defections_DIVISIONS = CFR_provoke_army_defections_AGENTS_REQ }
								multiply_variable = { CFR_provoke_army_defections_AGENTS_REQ = 15 }
							}
						}
					}
					if = {
						limit = {
							has_country_flag = CFR_weekly_five_infiltrators_flag
						}
						add_to_variable = { infiltrators_var = 5 } ## Why does it add 770???
					}
				}
			}
		}
	}

	on_monthly = {
		effect = {
			if = {
				limit = {
					has_navy_size = {
    					size > 1
    					type = capital_ship
					}
				}
				set_temp_variable = { capital_ships_temp_var = num_ships_with_type@capital }
				add_to_temp_variable = { capital_ships_temp_var = num_ships_with_type@carrier }
				#log = "[GetDateText]: [Root.GetName]: Capital ships Num [?capital_ships_temp_var|R]"
				set_temp_variable = { screening_ships_temp_var = num_ships_with_type@screen }
				#log = "[GetDateText]: [Root.GetName]: Screening ships Num [?screening_ships_temp_var|R]"
				subtract_from_temp_variable = { capital_ships_temp_var = capital_ships_var }
				subtract_from_temp_variable = { screening_ships_temp_var = screening_ships_var }
				divide_temp_variable = { screening_ships_temp_var = capital_ships_temp_var }
				#log = "[GetDateText]: [Root.GetName]: Screens/Capitals Ratio: [?screening_ships_temp_var|R]"
				if = {
					limit = {
						OR = {
							check_variable = {
								screening_ships_temp_var < 4 
							}
						}
					}
					#log = "[GetDateText]: [Root.GetName]: Build screens"
					set_country_flag = AI_build_more_screens
				}
				else = {
					#log = "[GetDateText]: [Root.GetName]: Build capital"
					clr_country_flag = AI_build_more_screens
				}
			}
		}
	}
	on_weekly_WAR = {
		effect = {
			if = {
				limit = {
					has_completed_focus = WAR_push_back_darkness
				}
				every_owned_state = {
					limit = {
						has_dynamic_modifier = {
							modifier = WAR_lunar_effect 
						}
					}
					subtract_from_variable = { WAR_lunar_value = modifier@WAR_Weekly_Lunar_Resistance }
				}
			}
   		}
	}
}
