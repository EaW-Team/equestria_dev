on_actions = {
	on_monthly_IST = {
		effect = {
			if = {
				limit = {
					always = no # TODO
				}
				if = {
					limit = {
						YAN = { all_original_cores_owned_by_ROOT_or_subject = yes }
					}
					country_event = maglibing_titles.1
				}
				if = {
					limit = {
						GAR = { all_original_cores_owned_by_ROOT_or_subject = yes }
						CNT = { all_original_cores_owned_by_ROOT_or_subject = yes }
					}
					country_event = maglibing_titles.2
				}
				if = {
					limit = {
						HYE = { all_original_cores_owned_by_ROOT_or_subject = yes }
					}
					country_event = maglibing_titles.3
				}
				if = {
					limit = {
						STM = { all_original_cores_owned_by_ROOT_or_subject = yes }
						GYZ = { all_original_cores_owned_by_ROOT_or_subject = yes }
						SLC = { all_original_cores_owned_by_ROOT_or_subject = yes }
						THU = { all_original_cores_owned_by_ROOT_or_subject = yes }
					}
					country_event = maglibing_titles.4
				}
			}
		}
	}
	on_peace = {
		effect = {
			if = {
				limit = {
					has_global_flag = IST_resistance_civil_war
				}
				if = {
					limit = {
						tag = IST
					}
					country_event = maglibing_civil_war.2
				}
				else_if = {
					limit = {
						original_tag = IST
					}
					country_event = maglibing_civil_war.3
				}
			}
			if = {
				limit = {
					has_completed_focus = IST_focus_war_4
					NOT = { has_country_flag = IST_kiria_victory }
				}
				country_event = maglibing_postkiria.1
				set_country_flag = IST_kiria_victory
			}
			if = {
				limit = {
					has_completed_focus = IST_we_wait_no_longer
					NOT = { has_country_flag = IST_middle_victory }
				}
				country_event = maglibing.8
				set_country_flag = IST_middle_victory
			}
		}
	}
	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					ROOT = {
						original_tag = IST
						NOT = { tag = IST }
					}
				}
				if = {
					limit = {
						FROM = { IST_is_Dimasupil = yes }
						FROM.FROM = { is_in_array = { FROM.project_blazing_sun_states = THIS } }
					}
					FROM.FROM = {
						damage_building = { type = infrastructure damage = 1.5 }
						damage_building = { type = nuclear_reactor damage = 1.5 }
						damage_building = { type = industrial_complex damage = 1.5 }
						damage_building = { type = arms_factory damage = 1.5 }
						damage_building = { type = dockyard damage = 1.5 }
						damage_building = { type = fuel_silo damage = 1.5 }
						damage_building = { type = synthetic_refinery damage = 1.5 }
						damage_building = { type = synthetic_refinery damage = 1.5 }
						damage_building = { type = rail_way damage = 1 }
						damage_building = { type = supply_node damage = 1 }
						remove_from_array = { FROM.project_blazing_sun_states = THIS }
						add_dynamic_modifier = {
							modifier = IST_Blazing_Sun_DM
							days = 60
						}
					}
					FROM = { 
						scoped_sound_effect = "ui_scorched_earth" 
						scoped_sound_effect = "alert_hostile_troops_sound"
					}
				}
				if = {
					limit = {
						any_allied_country = {
							FROM.FROM = { is_core_of = PREV }
							set_temp_variable = { temp_country = THIS }
						}
					}
					transfer_state_to = var:temp_country
				}
			}
		}
	}
}
