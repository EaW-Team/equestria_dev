on_actions = {
	on_startup = {
		effect = {
			every_country = {
				limit = { 
                    tag = PLB
                }
                set_variable = { hunter_loyalty = 50 }
                set_variable = { hunter_loyalty_rate = .2 }
                set_variable = { hunter_loyalty_max = 100 }
				
				set_variable = { warrior_loyalty = 50 }
                set_variable = { warrior_loyalty_rate = .2 }
                set_variable = { warrior_loyalty_max = 100 }
				
				set_variable = { diplo_loyalty = 50 }
                set_variable = { diplo_loyalty_rate = .2 }
                set_variable = { diplo_loyalty_max = 100 }
				
				set_variable = { industry_loyalty = 50 }
                set_variable = { industry_loyalty_rate = .2 }
                set_variable = { industry_loyalty_max = 100 }
            }
		}
	}
	
	on_daily = {
		effect = {
			if = {
				limit = {
					tag = PLB
				}
				
				add_to_variable = { hunter_loyalty = hunter_loyalty_rate }
				add_to_variable = { warrior_loyalty = warrior_loyalty_rate }
				add_to_variable = { diplo_loyalty = diplo_loyalty_rate }
				add_to_variable = { industry_loyalty = industry_loyalty_rate }
				
				clamp_variable = { var = hunter_loyalty min = 0 max = hunter_loyalty_max }
				
				clamp_variable = { var = warrior_loyalty min = 0 max = warrior_loyalty_max }
				
				clamp_variable = { var = diplo_loyalty min = 0 max = diplo_loyalty_max }
				
				clamp_variable = { var = industry_loyalty min = 0 max = industry_loyalty_max }
			}
		}
	}
	
}
