on_actions = {
	on_startup = {
		effect = {
			every_country = {
				limit = { 
                    tag = PLB
                }
                set_variable = { hunter_loyalty = 40 }
                set_variable = { hunter_loyalty_rate = .1 }
                set_variable = { hunter_loyalty_max = 40 }
				
				set_variable = { warrior_loyalty = 40 }
                set_variable = { warrior_loyalty_rate = .1 }
                set_variable = { warrior_loyalty_max = 40 }
				
				set_variable = { diplo_loyalty = 40 }
                set_variable = { diplo_loyalty_rate = .1 }
                set_variable = { diplo_loyalty_max = 40 }
				
				set_variable = { industry_loyalty = 40 }
                set_variable = { industry_loyalty_rate = .1 }
                set_variable = { industry_loyalty_max = 40 }
            }
		}
	}
	
	on_daily = {
		effect = {
			if = {
				limit = {
					tag = PLB
				}
				
				if = {
					limit = {
						check_variable = {
							hunter_loyalty < hunter_loyalty_max
						}
					}
					add_to_variable = { hunter_loyalty = hunter_loyalty_rate }
				}
				if = {
					limit = {
						check_variable = {
							warrior_loyalty < warrior_loyalty_max
						}
					}
					
					add_to_variable = { warrior_loyalty = warrior_loyalty_rate }
				}
				if = {
					limit = {
						check_variable = {
							diplo_loyalty < diplo_loyalty_max
						}
					}
							
					add_to_variable = { diplo_loyalty = diplo_loyalty_rate }
					
				}
							
				if = {
					limit = {
						check_variable = {
							industry_loyalty < industry_loyalty_max
						}
					}
								
					add_to_variable = { industry_loyalty = industry_loyalty_rate }
				}
			}
		}
	}
	
}
