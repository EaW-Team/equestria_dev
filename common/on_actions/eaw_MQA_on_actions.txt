on_actions = {
	on_startup = {
		effect = {
			MQA = {
			#	country_event = { id = sambovato.1 days = 120 }
			}
		}
	}

	on_declare_war = {
		effect = {
			if = {
				limit = {
					OR = {
						AND = {
							tag = ZOO
							FROM = { tag = MQA }
						}
						AND = {
							tag = MQA
							FROM = { tag = ZOO }
						}
					}
				}
				MQA = {
					set_country_flag = ZOO_conquest_MQA #Flag for Zongo's naval content to indicate a prior state of war has happened between Zongo and Sambovato
				}
			}
		}
	}

	on_ruling_party_change = {
		effect = {
			if = {
				limit = { has_character = MQA_ceyda_vatosoa }

				if = {
					limit = { has_government = democratic }
					MQA_ceyda_vatosoa = {
						set_portraits = {
							civilian = {
								large = "gfx/leaders/MQA/Ceyda_Vatosoa_harmony.tga"
							}
							army = {
								large = "gfx/leaders/MQA/Ceyda_Vatosoa_harmony.tga"
							}
						}
					}
				}
				else_if = {
					limit = { has_government = fascism }
					MQA_ceyda_vatosoa = {
						set_portraits = {
							civilian = {
								large = "gfx/leaders/MQA/Ceyda_Vatosoa_Supremacy.tga"
							}
							army = {
								large = "gfx/leaders/MQA/Ceyda_Vatosoa_Supremacy.tga"
							}
						}
					}
				}
				else = {
					MQA_ceyda_vatosoa = {
						set_portraits = {
							civilian = {
								large = "gfx/leaders/MQA/Ceyda_Vatosoa.tga"
							}
							army = {
								large = "gfx/leaders/MQA/Ceyda_Vatosoa.tga"
							}
						}
					}
				}
			}
		}
	}
}
