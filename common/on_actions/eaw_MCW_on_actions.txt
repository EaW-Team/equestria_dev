on_actions = {
	on_send_volunteers = {
		effect = {
			if = {
				limit = {
					ROOT = {
						original_tag = MCW
					}
				}
				FROM = {
					set_country_flag = MCW_sent_volunteers_flag
				}
			}
		}
	}

	on_daily_MCW = {
		effect = {
			if = {
				limit = {
					has_country_flag = MCW_parliament_enable
				}
				f_MCW_reload_parliament = yes
			}
		}
	}
}
