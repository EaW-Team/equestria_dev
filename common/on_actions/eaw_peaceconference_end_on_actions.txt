on_actions = {
	#ROOT is winner #FROM is loser
	on_peaceconference_ended = {
		effect = {
			# PUT EVERYTHING IN THE IF BELOW
			# THE bypass_on_peaceconference_ended FLAG **HAS** TO BE SET BEFORE CALLING THE white_peace COMMAND AND CLEARED AFTERWARDS
			if = {
				limit = {
					NOT = {
						has_country_flag = bypass_on_peaceconference_ended
						FROM = { has_country_flag = bypass_on_peaceconference_ended }
					}
				}
				log = "[GetDateText]: ROOT: [Root.GetName] FROM: [From.GetName] on_peaceconference_ended"
				transfer_impassible_states_back = yes
				if = {
					limit = {
						FROM = { is_subject = yes }
					}
					every_country_with_original_tag = {
						original_tag_to_check = FROM
						limit = {
							NOT = { tag = FROM }
							is_subject = yes
							is_ai = yes
							overlord = { is_ai = yes }
						}
						FROM = {
							annex_country = { target = PREV transfer_troops = yes }
						}
					}
				}
				if = {
					limit = {
						is_in_array = { annex_compliance = FROM }
					}
					remove_from_array = { annex_compliance = FROM }
				}
				if = {
					limit = {
						is_in_array = { annex_compliance_subject = FROM }
					}
					remove_from_array = { annex_compliance_subject = FROM }
				}
				set_AI_strategies_on_every_possible_country_delayed = yes
				FROM = { clr_country_flag = already_capitulated }

				handle_op_paperclip = yes

				# NZW related code start
				if = { # Hippogriff loss
					limit = {
						ROOT = {
							OR = {
								has_country_flag = NZW_war_participant
								AND = {
									original_tag = BAT
									has_global_flag = BAT_won_civil_war # Lunar Civil War
								}
							}
						}
						FROM = {
							original_tag = HIP
							NOT = {
								has_country_flag = BAT_HIP_defeated
							}
						}
					}
					set_country_flag = BAT_HIP_defeated
					if = {
						limit = {
							NOT = { has_global_flag = NZW_ended }
						}
						NZW_on_end_effect = yes
					}
					if = {
						limit = { # Major alliance partner
							OR = {
								tag = NZ1
								tag = NZ2
							}
							has_global_flag = NZW_alliance_signed
						}
						NZ1 = {
							country_event = north_war_alliance.5 # Option to dissolve alliance
						}
						NZ2 = {
							country_event = north_war_alliance.5 # Option to dissolve alliance
						}
					}
				}
				if = { # Hippogriff victory
					limit = {
						ROOT = {
							original_tag = HIP
							NOT = {
								has_country_flag = BAT_HIP_defeated
							}
						}
						FROM = {
							OR = {
								AND = {
									OR = {
										tag = NZ1
										tag = NZ2
										tag = WNG
									}
									has_country_flag = NZW_war_participant
								}
								tag = HIE
							}
						}
						NOT = { # No one remaining
							ROOT = {
								any_enemy_country = {
									is_subject = no
									has_country_flag = NZW_war_participant
									has_capitulated = no
								}
							}
						}
					}
					set_country_flag = HIP_won_north_zebrican_war
					NZW_on_end_effect = yes
					if = {
						limit = {
							has_global_flag = NZW_alliance_signed
						}
						NZ1 = { set_major = no }
						NZ2 = { set_major = no }
					}
					mark_focus_tree_layout_dirty = yes
				}
				else_if = { # Hippogriffs started the war (south)
					limit = {
						ROOT = {
							original_tag = HIP
							has_completed_focus = HIP_strike_first
						}
						FROM = {
							OR = { # Southern tag
								original_tag = TBK
								original_tag = ZAR
								original_tag = BAT
								original_tag = WAR
							}
						}
						OR = {
							FROM = {
								owns_state = 712
							}
							ROOT = {
								owns_state = 712
							}
						}
					}
					set_country_flag = HIP_won_north_zebrican_war
					mark_focus_tree_layout_dirty = yes
				}
				if = { # Chiropterra defeated, remove cored regions
					limit = {
						FROM = {
							original_tag = BAT
						}
						NOT = {
							has_country_flag = BAT_commonwealth_established
							has_country_flag = BAT_national_catastrophe
						}
					}
					BAT = {
						every_core_state = {
							limit = {
								NOT = {
									state = 689
									state = 690
									state = 691
									state = 712
									state = 713
									state = 815
								}
							}
							remove_core_of = BAT
						}
					}
				}
				if = { # Chiropterra as victor in the NZW (not Lunar victory)
					limit = {
						original_tag = BAT
						NOT = {
							has_country_flag = BAT_commonwealth_established
							has_country_flag = BAT_national_catastrophe
						}
						NOT = {
						    has_global_flag = BAT_won_civil_war
						}
						OR = { # Was in North Zebrican War
							has_global_flag = NZW_in_progress
							has_global_flag = NZW_ended
						}
						NOT = { has_country_flag = NZW_BAT_victory_handled }
						FROM = {
						    original_tag = HIP
						}
					}
					set_country_flag = NZW_BAT_victory_handled # fire only once
				}
				if = { # If not in an alliance, you get this now, otherwise after the alliance is either kept or NZ1 defeated
					limit = {
						original_tag = BAT
						NOT = {
							has_global_flag = NZW_alliance_signed
							has_global_flag = BAT_won_civil_war
						}
						FROM = {
						    original_tag = HIP
						}
					}
					if = {
						limit = { # Controls the home region
							NOT = {
								WAR = {
									exists = yes
									is_subject = no
								}
							}
							NOT = {
								ZAR = {
									exists = yes
									is_subject = no
								}
							}
							NOT = {
								TBK = {
									exists = yes
									is_subject = no
								}
							}
						}
						if = {
							limit = {
								NOT = { has_focus_tree = tzintacalia_tree_7 }
							}
							country_event = bat.81 # Chiropterran commonwealth proclaimed
						}
					}
					else = {
						country_event = bat.79 # Declare war on rest of your home region (should never happen)
					}
				}
				if = { # Chiropterra triumphed over HIP (with Lunar victory)
					limit = {
						original_tag = BAT
						NOT = {
							has_country_flag = BAT_commonwealth_established
							has_country_flag = BAT_national_catastrophe
						}
						has_global_flag = BAT_won_civil_war
						FROM = {
						    original_tag = HIP
						}

					}
					if = {
						limit = {
							NOT = {
								WAR = {
									exists = yes
									is_subject = no
								}
								ZAR = {
									exists = yes
									is_subject = no
								}
								TBK = {
									exists = yes
									is_subject = no
								}
							}
						}
						country_event = {
							id = bat.130
						}
					}
					else = {
						country_event = {
							id = bat.79
						}
					}
				}
				if = { # Anti-Aris Axis (now Azghat Pact) defeated by outside power after NZW
					limit = {
						has_global_flag = NZW_alliance_kept_post_war # This flag is only set after the end of the NZW
						NOT = { has_global_flag = NZW_azghat_pact_defeated }
						OR = {
							FROM = { tag = NZ1 }
							FROM = { tag = NZ2 }
						}
					}
					NZ1 = { set_major = no }
					NZ2 = { set_major = no }
					set_global_flag = NZW_azghat_pact_defeated
				} # NZW related code end

				# remove demilitarized flag if previous overlord no longer exists, has become a subject or if country gains a new overlord
				# done to prevent countries from ending up with demilitarized puppets that don't want any				
				FROM = {
					if = {
						limit = {
							FROM = { has_country_flag = demilitarized }
						}
						if = {
							limit = {
								OR = {
									if = {
										limit = { is_subject = yes }
										overlord = {
											NOT = {
												tag = FROM.overlord_before_pc
											}
										}
									}
									is_subject = no
									var:overlord_before_pc = {
										OR = {
											exists = no
											is_subject = yes
										}
									}
								}
							}
							clr_country_flag = demilitarized
							log = "[GetDateText]: ROOT: [Root.GetName] FROM: [From.GetName] demilitarized flag cleared"
							set_temp_variable = { current_overlord = overlord }
							# need to end puppet before puppeting anew or it will take lower possible autonomy instead of default
							end_puppet_this = yes
							var:current_overlord = { puppet = FROM }
						}
						clear_variable = overlord_before_pc
					}
				}
				# if a country is an AI puppet with only one state and surrounded by another country, annex them to reduce bordergore
				FROM = { country_event = { id = utility.10 hours = 1 } }

				update_global_arrays_all = yes
			}
		}
	}
}
