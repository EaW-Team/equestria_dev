on_actions = {

	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					NOT = { has_global_flag = CGI_coltdar_war_happened }
					OR = {
						AND = {
							tag = CGI
							FROM = {
								OR = {
									original_tag = CTH
									original_tag = CVA
									original_tag = HIE
									original_tag = TRO
								}
								owns_state = 875 # Colthage
							}
						}
						AND = {
							OR = {
								original_tag = CTH
								original_tag = CVA
								original_tag = HIE
								original_tag = TRO
							}
							owns_state = 875 # Colthage
							FROM = {
								tag = CGI
							}
						}
					}
				}
				set_global_flag = CGI_coltdar_war_happened

				if = {
					limit = {
						tag = CGI # Quaggatai attacked
					}
					set_variable = {
						CGI_coltdar_war_enemy = FROM
					}
					country_event = quaggatai.21
				}
				else = { # Quaggatai is defender
					CGI = {
						set_variable = {
							CGI_coltdar_war_enemy = ROOT
						}
						country_event = quaggatai.20
					}
				}
			}
		}
	}
	
	on_weekly_CGI = {
		effect = {
			if = {
				limit = {
					has_country_flag = CGI_diamond_road_active
				}
				add_equipment_to_stockpile = {
					type = infantry_equipment
					amount = CGI_diamond_road_guns
					producer = CGI
				}
				add_equipment_to_stockpile = {
					type = artillery_equipment
					amount = CGI_diamond_road_arty
					producer = CGI
				}
				add_equipment_to_stockpile = {
					type = support_equipment
					amount = CGI_diamond_road_support
					producer = CGI
				}
			}
		}
	}

	on_monthly_CGI = {
		effect = {
			if = {
				limit = {
					has_country_flag = CGI_diamond_road_collapse_active
				}
				if = {
					limit = {
						has_dynamic_modifier = {
							modifier = CGI_diamond_road_modifier
						}
					}
					add_to_variable = { CGI_diamond_road_strength = -1 }
					CGI_update_diamond_road = yes

					if = {
						limit = {
							check_variable = { CGI_diamond_road_strength < 10.1 }
						}
						country_event = quaggatai.103 # Total collapse
						clr_country_flag = CGI_diamond_road_collapse_active
					}
				}
				else = {
					clr_country_flag = CGI_diamond_road_collapse_active
				}
			}
			if = {
				limit = {
					has_country_flag = CGI_diamond_road_active
					has_idea = CGI_muraq
					has_war = no
					NOT = { has_country_flag = CGI_diamond_road_collapse_active }
				}
				add_to_variable = { CGI_diamond_road_strength = 1 }
				CGI_update_diamond_road = yes
			}
		}
	}

	on_peaceconference_ended = {
		effect = {
			# PUT EVERYTHING IN THE IF BELOW
			# THE bypass_on_peaceconference_ended FLAG **HAS** TO BE SET BEFORE CALLING THE white_peace COMMAND AND CLEARED AFTERWARDS
			if = {
				limit = {
					NOT = {
						has_country_flag = bypass_on_peaceconference_ended
						FROM = { has_country_flag = bypass_on_peaceconference_ended }
					}
				}
				if = {
					limit = {
						original_tag = CGI
						NOT = { CGI = { has_country_flag = CGI_conquered_colthage } }
						OR = {
							FROM = { original_tag = CTH }
							FROM = { original_tag = TRO }
							FROM = { original_tag = HIE }
							FROM = { original_tag = CVA }
						}
					}
					random_country_with_original_tag = {
						original_tag_to_check = CTH
						limit = {
							is_subject_of = CGI
						}
						CGI = { annex_country = { target = CTH } }
					}
					random_country_with_original_tag = {
						original_tag_to_check = TRO
						limit = {
							is_subject_of = CGI
						}
						CGI = { annex_country = { target = TRO } }
					}
					random_country_with_original_tag = {
						original_tag_to_check = HIE
						limit = {
							is_subject_of = CGI
						}
						CGI = { annex_country = { target = HIE } }
					}
					random_country_with_original_tag = {
						original_tag_to_check = CVA
						limit = {
							is_subject_of = CGI
						}
						CGI = { annex_country = { target = CVA } }
					}
					random_country_with_original_tag = {
						original_tag_to_check = CGI
						limit = {
							CTH = { all_cores_owned_by_ROOT_or_subject = yes }
							#CVA = { all_cores_owned_by_ROOT_or_subject = yes }
							1095 = { is_owned_by = CGI }
							1093 = { is_owned_by = CGI }
							1094 = { is_owned_by = CGI }
							HIE = { all_cores_owned_by_ROOT_or_subject = yes }
							TRO = { all_cores_owned_by_ROOT_or_subject = yes }
						}
						CGI = {
							set_country_flag = CGI_conquered_colthage
							country_event = quaggatai.100
						}
					}
					FROM = { transfer_navy = { target = CGI } }
				}

				# Camelia win/loss
				if = {
					limit = {
						OR = {
							tag = BLT # Boltrian Kingdom
							tag = CME # Free Camel Tribes
						}
						FROM = {
							OR = {
								tag = BLT
								tag = CME
							}
						}
					}

					set_global_flag = camelia_united

					CGI = {
						if = {
							limit = {
								has_country_flag = CGI_diamond_road_active
								has_country_flag = CGI_supported_BLT
								ROOT = { tag = BLT }
							}
							country_event = quaggatai_camelia.8
						}
						else_if = {
							limit = {
								has_country_flag = CGI_diamond_road_active
								has_country_flag = CGI_supported_BLT
								ROOT = { tag = CME }
							}
							country_event = quaggatai_camelia.9
						}
						else_if = {
							limit = {
								has_country_flag = CGI_diamond_road_active
								has_country_flag = CGI_supported_CME
								ROOT = { tag = CME }
							}
							country_event = quaggatai_camelia.10
						}
						else_if = {
							limit = {
								has_country_flag = CGI_diamond_road_active
								has_country_flag = CGI_supported_CME
								ROOT = { tag = BLT }
							}
							country_event = quaggatai_camelia.11
						}
					}
				}
			}
		}
	}
}
