on_actions = {
	on_weekly_MTA = {
		effect = {
			if = {
				limit = {
					has_cosmetic_tag = MTA_eotc
				}
				set_temp_variable = { MTA_amount_of_divisions = num_divisions }
				if = {
					limit = { has_completed_focus = MTA_energy_conservation }
					multiply_temp_variable = { MTA_amount_of_divisions = 0.9 }
				}
				multiply_temp_variable = { MTA_amount_of_divisions = -1 }
				add_to_variable = { MTA_eotc_sparrowber_mana = MTA.MTA_amount_of_divisions }
				if = {
					limit = {
						check_variable = { MTA_eotc_sparrowber_mana < 0 }
					}
					country_event = marcycle.7
				}
				every_neighbor_country = {
					limit = {
						NOT = {
							has_war_with = ROOT
						}
					}
					country_event = marcycle.4
					ROOT = { 
						hidden_effect = {
							create_wargoal = { 
								type = annex_everything
								target = PREV
							}
						}
						declare_war_on = { target = PREV type = annex_everything } 
					}
				}
				every_enemy_country = {
					MTA_eotc_upgrade_resistance = yes
				}
				every_subject_country = {
					ROOT = {
						annex_country = { target = PREV }
					}
				}
			}
		}
	}
}
