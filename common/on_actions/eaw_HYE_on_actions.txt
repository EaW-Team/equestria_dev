on_actions = {
	on_puppet = {
		effect = {
			if = {
				limit = {
					FROM = {
						OR = {
							original_tag = GAR
							original_tag = CNT
						}
						is_in_faction_with = HYE					
					}
					ROOT = {
						OR = {
							original_tag = SLC
							original_tag = THU
							original_tag = GYZ
							original_tag = STM
						}
					}
				}
				HYE = {
					set_country_flag = { flag = bypass_annex_compliance days = 1 value = 1 }
					annex_country = { target = ROOT  }
				}
			}

			if = {
				limit = {
					original_tag = HYE
					has_cosmetic_tag = HYE_fed
					is_subject = yes
					FROM = {
						NOT = { has_government = communism }
					}
				}
				drop_cosmetic_tag = yes
			}
		}
	}

	on_release_as_puppet = {
		effect = {
			if = {
				limit = {
					FROM = {
						OR = {
							original_tag = GAR
							original_tag = CNT
						}
						is_in_faction_with = HYE
					}
					ROOT = {
						OR = {
							original_tag = SLC
							original_tag = THU
							original_tag = GYZ
							original_tag = STM
						}
					}
				}
				HYE = {
					set_country_flag = { flag = bypass_annex_compliance days = 1 value = 1 }
					annex_country = { target = ROOT  }
				}
			}
		}
	}

	on_annex = {
		effect = {
			if = {
				limit = {
					OR = {
						original_tag = GAR
						original_tag = CNT
					}
					is_in_faction_with = HYE
					FROM = {
						OR = {
							original_tag = SLC
							original_tag = THU
							original_tag = GYZ
							original_tag = STM
						}
					}
				}
				every_owned_state = {
					limit = {
						OR = {
							is_core_of = SLC
							is_core_of = THU
							is_core_of = GYZ
							is_core_of = STM
						}
					}
					transfer_state_to = HYE
				}
			}
		}
	}

	on_state_control_changed = {
		effect = {
			# HYE Underworld events
			if = {
				limit = { original_tag = HYE }

				if = { # Underworld event
					limit = {
						FROM.FROM = { state = 1030 }
					}
					country_event = hyenia.23
				}
				else_if = { # Underworld event
					limit = {
						FROM.FROM = { state = 1033 }
					}
					country_event = hyenia.24
				}
				else_if = { # Underworld event
					limit = {
						FROM.FROM = { state = 1048 }
					}
					country_event = hyenia.25
				}
				else_if = { # Underworld event
					limit = {
						FROM.FROM = { state = 1116 }
					}
					country_event = hyenia.26
				}
				else_if = { # Underworld event
					limit = {
						FROM.FROM = { state = 919 }
					}
					country_event = hyenia.27
				}
				else_if = { # Underworld event
					limit = {
						FROM.FROM = { state = 973 }
					}
					country_event = hyenia.28
				}
				else_if = { # Underworld event
					limit = {
						FROM.FROM = { state = 978 }
					}
					country_event = hyenia.29
				}
			}
		}
	}
}
