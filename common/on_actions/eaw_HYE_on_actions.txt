on_actions = {
	on_puppet = {
		effect = {
			if = {
				limit = {
					FROM = {
						OR = {
							original_tag = GAR
							original_tag = CNT
						}
						is_in_faction_with = HYE					
					}
					ROOT = {
						OR = {
							original_tag = SLC
							original_tag = THU
							original_tag = GYZ
							original_tag = STM
						}
					}
				}
				HYE = {
					set_country_flag = { flag = bypass_annex_compliance days = 1 value = 1 }
					annex_country = { target = ROOT  }
				}
			}

			if = {
				limit = {
					original_tag = HYE
					has_cosmetic_tag = HYE_fed
					is_subject = yes
					FROM = {
						NOT = { has_government = communism }
					}
				}
				drop_cosmetic_tag = yes
			}
		}
	}

	on_release_as_puppet = {
		effect = {
			if = {
				limit = {
					FROM = {
						OR = {
							original_tag = GAR
							original_tag = CNT
						}
						is_in_faction_with = HYE
					}
					ROOT = {
						OR = {
							original_tag = SLC
							original_tag = THU
							original_tag = GYZ
							original_tag = STM
						}
					}
				}
				HYE = {
					set_country_flag = { flag = bypass_annex_compliance days = 1 value = 1 }
					annex_country = { target = ROOT  }
				}
			}
		}
	}

	on_capitulation_immediate = {
		effect = {
			if = {
				limit = {
					FROM = {
						OR = {
							original_tag = HYE
							is_in_faction_with = HYE
						}
					}
					HYE = {
						is_subject = no
						has_completed_focus = HYE_ascend_the_malachite_throne
					}
					ROOT = {
						OR = {
							original_tag = STM
							original_tag = SLC
							original_tag = THU
							original_tag = GYZ
						}
					}
				}
				HYE = {
					annex_country = { target = ROOT }				
				}
			}
		}
	}

	on_state_control_changed = {
		effect = {
			# HYE Underworld events
			if = {
				limit = { original_tag = HYE }

				if = { # Underworld event
					limit = {
						FROM.FROM = { state = 1030 }
					}
					country_event = hyenia.23
				}
				else_if = { # Underworld event
					limit = {
						FROM.FROM = { state = 1033 }
					}
					country_event = hyenia.24
				}
				else_if = { # Underworld event
					limit = {
						FROM.FROM = { state = 1048 }
					}
					country_event = hyenia.25
				}
				else_if = { # Underworld event
					limit = {
						FROM.FROM = { state = 1116 }
					}
					country_event = hyenia.26
				}
				else_if = { # Underworld event
					limit = {
						FROM.FROM = { state = 919 }
					}
					country_event = hyenia.27
				}
				else_if = { # Underworld event
					limit = {
						FROM.FROM = { state = 973 }
					}
					country_event = hyenia.28
				}
				else_if = { # Underworld event
					limit = {
						FROM.FROM = { state = 978 }
					}
					country_event = hyenia.29
				}
			}
		}
	}

	on_monthly_HYE = {
		effect = {
			if = {
				limit = {
					OR = {
						has_completed_focus = HYE_reconnect_with_the_underworld
						has_completed_focus = HYE_our_sisters_under_the_soil
						has_completed_focus = HYE_formal_invitation_to_the_underworld
					}
				}

				HYE_underground_population_increase = yes
			}
		}
	}
	on_weekly_HYE = {
		effect = {
			if = {
				limit = {
					OR = {
						has_completed_focus = HYE_reconnect_with_the_underworld
						has_completed_focus = HYE_our_sisters_under_the_soil
						has_completed_focus = HYE_formal_invitation_to_the_underworld
					}
				}

				HYE_underground_conscription_weekly = yes
			}
		}
	}

	on_daily_HYE = {
		effect = {
			if = {
				limit = {
					NOT = {
						has_country_flag = HYE_reclamation_war_won_flag
					}
					has_war = yes
					OR = {
						AND = {
							is_ai = no
							OR = {
								STM = {
									surrender_progress > 0.6
									has_war_with = HYE
								}
								THU = {
									surrender_progress > 0.6
									has_war_with = HYE
								}
								SLC = {
									surrender_progress > 0.6
									has_war_with = HYE
								}
								GYZ = {
									surrender_progress > 0.6
									has_war_with = HYE
								}
							}
						}
						AND = {
							is_ai = yes
							OR = {
								STM = {
									surrender_progress > 0.5
									has_war_with = HYE
								}
								THU = {
									surrender_progress > 0.5
									has_war_with = HYE
								}
								SLC = {
									surrender_progress > 0.5
									has_war_with = HYE
								}
								GYZ = {
									surrender_progress > 0.5
									has_war_with = HYE
								}
							}
						}
					}
				}
				country_event = hyenia.6
			}
			if = {
				limit = {
					AND = {
						has_completed_focus = HYE_a_meeting_with_the_king
						has_completed_focus = HYE_reconnecting_with_the_queen
					}
				}
				HYE_support_gui_reload = yes
			}
		}
	}
}
