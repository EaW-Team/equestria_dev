## This is strictly for the capitulation message gui ##

on_actions = {
	on_capitulation_immediate = {
		effect = {
			set_temp_variable = { temp_winning_country = FROM }
			set_temp_variable = { temp_losing_country = ROOT }

			every_country = {
				limit = { is_ai = no }

				add_to_array = { losing_country = temp_losing_country }
				add_to_array = { winning_country = temp_winning_country }
			}
		}
	}
}
