on_actions = {
	on_startup = {
		effect = {
			EYR = {
				country_lock_all_division_template = yes
				country_event = { id = eyrie.1 hours = 18 }
			}
		}
	}

	on_puppet = {
		effect = {
			if = {
				limit = { 
					FROM = { original_tag = EYR }
					ROOT = {
						OR = {
							original_tag = ERI
							original_tag = AVI
						}
					}
				}
				FROM = {
					set_country_flag = { flag = bypass_annex_compliance days = 1 value = 1 }
					annex_country = { target = ROOT }
				}
			}
		}
	}

	on_unit_leader_created = {
		effect = {
			if = {
				limit = {
					is_army_leader = yes
					var:creator = {
						original_tag = EYR
						has_completed_focus = EYR_guard_professionalisation
					}
				}
				add_unit_leader_trait = EYR_guardsgriff_postwar
			}
		}
	}

	on_capitulation = {
		effect = {
			if = {
				limit = {
					tag = EYR
					FROM = { tag = ERI has_country_flag = ERI_prelude_war }
				}
				FROM = { country_event = { hours = 6 id = erie.105 } }
			}
			if = {
				limit = {
					tag = EYR
					FROM = { tag = ERI has_government = neutrality }
				}
				FROM = { country_event = { days = 3 id = erie.45 } }
			}
		}
	}

	on_declare_war = {
		effect = {
			if = {
				limit = {
					OR = {
						AND = {
							original_tag = EYR
							has_power_balance = {
								id = EYR_noble_power_struggle
							}
						}
						FROM = {
							original_tag = EYR
							has_power_balance = {
								id = EYR_noble_power_struggle
							}
						}
					}
				}
				EYR = {
					country_event = { id = eyrie_roman.22 days = 3 }
				}
			}

			if = {
				limit = {
					original_tag = EYR
					FROM = { original_tag = ERI }
				}
				set_country_flag = { flag = block_white_peace value = 1 days = 3 }
				FROM = { set_country_flag = { flag = block_white_peace value = 1 days = 3 } }
			}
		}
	}

	on_monthly_EYR = {
		effect = {
			if = {
				limit = {
					is_in_faction_with = GRD
					GRD = {
						exists = yes
					}
					has_war = no
				}
				if = {
					limit = {
						has_country_flag = EYR_give_heartlands_to_empire
						any_occupied_country = {
							is_griffon_empire_country_tag = yes
						}
					}
					country_event = eyrie.79
				}
				else_if = {
					limit = {
						NOT = {
							has_country_flag = EYR_BOI_warning
						}
						any_occupied_country = {
							original_tag = BOI
						}
					}
					set_country_flag = EYR_BOI_warning
					GRD = {
						country_event = eyrie.107
					}
					country_event = eyrie.108
				}
				else_if = {
					limit = {
						any_occupied_country = {
							OR = {
								original_tag = JER
								original_tag = FAT
								original_tag = FAW
								original_tag = TRD
							}
						}
					}
					GRD = {
						country_event = eyrie.83
					}
				}
			}
			if = {
				limit = {
					has_character = ERI_knight_commander_cyrille
					surrender_progress > 0.10
					any_enemy_country = { is_major = yes }
				}
				ERI_knight_commander_cyrille = { unit_leader_event = { id = eyrie.106 random = 72 days = 7 } }
			}
		}
	}
	on_weekly_EYR = {
        effect = {
            log = "[GetDateText]: ROOT: [Root.GetName] on_weekly_EYR"
            if = {
                limit = {
					EYR_economist_and_smuggler = {
						is_hired_as_advisor = yes
						OR = {
							has_trait = eyrian_economist
							has_trait = eyrian_economist_eco
							has_trait = eyrian_economist_eco_postwar_mil
						}
					}
				}
			add_equipment_to_stockpile = { 
				type = infantry_equipment 
				amount = 15 
            	}
        	}
			else_if = {
                limit = {
					EYR_economist_and_smuggler = {
						is_hired_as_advisor = yes
						OR = {
							has_trait = eyrian_economist_guns
							has_trait = eyrian_economist_guns_postwar
						}
					}
				}
			add_equipment_to_stockpile = { 
				type = infantry_equipment 
				amount = 25 
            	}
			add_equipment_to_stockpile = { 
				type = artillery_equipment
				amount = 10 
            	}
			add_equipment_to_stockpile = { 
				type = support_equipment
				amount = 5 
            	}
        	}
			else_if = {
                limit = {
					EYR_economist_and_smuggler = {
						is_hired_as_advisor = yes
						has_trait = eyrian_economist_guns_postwar_is
					}
				}
			add_equipment_to_stockpile = { 
				type = infantry_equipment 
				amount = 40 
            	}
			add_equipment_to_stockpile = { 
				type = support_equipment
				amount = 10 
            	}
        	}
			else_if = {
                limit = {
					EYR_economist_and_smuggler = {
						is_hired_as_advisor = yes
						has_trait = eyrian_economist_guns_postwar_aaa
					}
				}
			add_equipment_to_stockpile = { 
				type = artillery_equipment 
				amount = 10
            	}
			add_equipment_to_stockpile = { 
				type = anti_air_equipment
				amount = 5
            	}
			add_equipment_to_stockpile = { 
				type = anti_tank_equipment
				amount = 5 
            	}
        	}
			else_if = {
                limit = {
					EYR_economist_and_smuggler = {
						is_hired_as_advisor = yes
						has_trait = eyrian_economist_guns_postwar_tt
					}
				}
			add_equipment_to_stockpile = { 
				type = motorized_equipment 
				amount = 5 
            	}
				if = {
					limit = { has_dlc = "No Step Back" }
					add_equipment_to_stockpile = { type = light_tank_chassis_0 amount = 3 }
				}
				else = {
					add_equipment_to_stockpile = { type = gw_tank_equipment amount = 3 }
				}
        	}
			if = {
				limit = {
					is_in_faction_with = GRD
					GRD = {
						exists = yes
					}
					is_ai = yes
					NOT = {
						has_global_flag = GRI_empire_rising_flag
					}
					has_war_with_major = yes
				}
				GRD = {
					country_event = eyrie.81
				}
			}
		}
	}

	on_daily_EYR = {
		effect = {
			if = {
				limit = {
					has_country_flag = EYR_prelude_war
					NOT = { has_country_flag = EYR_sweatlord_tryhards }
						AND = {
						date > 1007.5.5
						date < 1007.5.10
					}
				}
				country_event = eyrie.3
			}
		}
	}
}
