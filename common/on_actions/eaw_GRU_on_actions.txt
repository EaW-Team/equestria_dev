on_actions = {
	on_startup = {
		effect = {
			GRU = { country_event = { id = jan_brazowy.1 days = 70 random_hours = 200 } } ## Jan Brązowy event chain ##
		}
	}

	on_puppet = {
		effect = {
			GRU_load_commie_tree = yes
		}
	}

	on_release_as_puppet = {
		effect = {
			GRU_give_released_puppet_special_cosmetic_tags = yes
		}
	}

	on_release_as_free = {
		effect = {
			GRU_load_commie_tree = yes
		}
	}

	on_subject_free = {
		effect = {
			GRU_load_commie_tree = yes
		}
	}

	on_government_change = {
		effect = {
			if = {
				limit = { original_tag = GRU }
				if = {
					limit = { has_government = communism }
					load_focus_tree = gryphus_commie_tree
					else = {
						load_focus_tree = gryphus
					}
				}
			}
		}
	}
}
