on_actions = {
	on_startup = {
		effect = {
			CHN = {
				find_global_platonic_countries = yes
			}
		}
	}
	on_war_relation_added = {
		effect = {
			CHN = {
				find_global_platonic_countries = yes 
			}
			if = {
				limit = {
					ROOT = {
						OR = {
							tag = CHN
							is_in_faction_with = CHN
							is_subject_of = CHN
						}
					}
					FROM = {
						has_idea = YAK_anti_chrysalis_pact_3
					}
				}
				FROM = {
					country_event = newyaks.50
				}
			}
			if = {
				limit = {
					original_tag = CHN
					NOT = {
						has_government = democratic
					}
					PLB = {
						has_defensive_war_with = CHN
						is_subject = no
					}
				}
				if = {
					limit = {
						NOT = {
							has_global_flag = CHN_great_northern_war
						}
					}
					set_global_flag = CHN_great_northern_war
					news_event = {
						id = plb_news.1
						days = 1
					}
				}
			}
		}
	}
	on_peaceconference_ended = {
		effect = {
			CHN = {
				find_global_platonic_countries = yes 
			}
		}
	}
}
