on_actions = {
	on_annex = {
		effect = {
			if = {
				limit = {
					FROM = { original_tag = TRD }
				}
				every_other_country = {
					limit = {
						OR = {
							has_idea = TRD_trading_with_skyfall_1
							has_idea = TRD_trading_with_skyfall_2
						}
					}
					add_opinion_modifier = {
						target = ROOT
						modifier = TRD_disrupted_trade
					}
				}
			}
		}
	}

	on_monthly_TRD = {
		effect = {
			if = {
				limit = {
					has_completed_focus = TRD_robber_baron
					has_government = neutrality
					is_subject = no
				}
				GRF_recalc_mercs = yes
			}
		}
	}
}
