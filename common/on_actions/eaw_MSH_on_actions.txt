on_actions = {
	on_startup = {
		effect = {
			MSH = {
				add_dynamic_modifier = {
					modifier = MSH_spiritual_trend
				}
				MSH_calculate_piety_effects = yes
				MSH_Influences_start = yes
			}
		}
	}

	on_declare_war = {
		effect = {
			if = {
				limit = {
					FROM = {
						is_great_lakes_country = yes
						NOT = { original_tag = MSH }
					}
					ROOT = {
						original_tag = MSH
						has_country_flag = MSH_struggle_country_flag
						NOT = {
							has_idea = MSH_the_great_struggle_idea
						}
					}
				}
				ROOT = {
					add_ideas = MSH_the_great_struggle_idea
				}
			}
			if = {
				limit = {
					ROOT = {
						is_great_lakes_country = yes
						NOT = { original_tag = MSH }
					}
					FROM = {
						original_tag = MSH
						has_country_flag = MSH_struggle_country_flag
						NOT = {
							has_idea = MSH_the_great_struggle_idea
						}
					}
				}
				FROM = {
					add_ideas = MSH_the_great_struggle_idea
				}
			}
		}
	}

	on_capitulation = {
		effect = {
			if = {
				limit = {
					original_tag = MSH
					has_country_flag = MSH_limited_MGY_war_flag
				}
				clr_country_flag = MSH_limited_MGY_war_flag
			}
		}
	}

	on_civil_war_end = {
		effect = {
			if = {
				limit = { original_tag = MSH }
				if = {
					limit = {
						tag = MSL
						NOT = {
							AND = {
								country_exists = MSP
								country_exists = MSR
							}
						}
					}
					country_event = {
						id = macigena_civ_war.4
						days = 1
					}
				}
				else_if = {
					limit = {
						tag = MSP
						NOT = {
							AND = {
								country_exists = MSL
								country_exists = MSR
							}
						}
					}
					country_event = {
						id = macigena_civ_war.5
						days = 1
					}
				}
				else_if = {
					limit = {
						tag = MSR
						NOT = {
							AND = {
								country_exists = MSP
								country_exists = MSL
							}
						}
					}
					country_event = {
						id = macigena_civ_war.6
						days = 1
					}
				}
			}
		}
	}

	on_peaceconference_ended = {
		effect = {
		effect = {
			# PUT EVERYTHING IN THE IF BELOW
			# THE bypass_on_peaceconference_ended FLAG **HAS** TO BE SET BEFORE CALLING THE white_peace COMMAND AND CLEARED AFTERWARDS
			if = {
				limit = {
					NOT = {
						has_country_flag = bypass_on_peaceconference_ended
						FROM = { has_country_flag = bypass_on_peaceconference_ended }
					}
				}
				log = "[GetDateText]: ROOT: [Root.GetName] FROM: [From.GetName] on_peaceconference_ended"
				if = {
					limit = {
						tag = MSH
						FROM = {
							tag = SON
							is_subject = no
						}
					}
					country_event = {
						id = maciegna.6
					}
				}
				if = {
					limit = {
						tag = MSH
						FROM = {
							tag = ZWA
							is_subject = no
						}
					}
					country_event = {
						id = maciegna.12
					}
				}
				if = {
					limit = {
						tag = MSH
						FROM = {
							tag = ZUG
							is_subject = no
						}
					}
					country_event = {
						id = maciegna.13
					}
				}
				if = {
					limit = {
						tag = MSH
						FROM = {
							tag = ZEL
							is_subject = no
						}
					}
					country_event = {
						id = maciegna.19
					}
				}
				if = {
					limit = {
						tag = MSH
						FROM = {
							tag = HAY
							is_subject = no
						}
					}
					country_event = {
						id = maciegna.20
					}
				}
				if = {
					limit = {
						tag = MSH
						FROM = {
							tag = TIM
							is_subject = no
						}
					}
					country_event = {
						id = maciegna.24
					}
				}
				if = {
					limit = {
						tag = MSH
						FROM = {
							tag = ALO
							is_subject = no
						}
					}
					country_event = {
						id = maciegna.25
					}
				}
				if = {
					limit = {
						tag = MSH
						FROM = {
						    OR = {
								tag = MAZ
								tag = event_target:fz_army
								tag = event_target:bre_army
								tag = event_target:rest_army
							}
							is_subject = no
						}
					}
					country_event = {
						id = maciegna.34
					}
				}
				if = {
					limit = {
						original_tag = MSH
						has_country_flag = MSH_spiritual_trend_unlocked
					}
					MSH_Influences_re_render = yes
				}
			}
		}
	}

	on_weekly_MSH = {
		effect = {
			log = "[GetDateText]: ROOT: [Root.GetName] on_weekly_MSH"
			if = {
				limit = {
					has_idea = MSH_from_timbucktu_with_love_idea
				}
				add_equipment_to_stockpile = {
					type = infantry_equipment
					amount = 50
				}
			}
			if = {
			    limit = {
				    has_country_flag = MSH_spiritual_trend_unlocked
				}
				set_temp_variable = { manually_re_render_gui = 1 }
				for_each_scope_loop = {
					array = MSH_Influences_States
					if = {
						limit = {
							is_owned_by = MSH
						}
						## GLOBAL

						set_temp_variable = { t_change = MSH.modifier@MSH_weekly_legalist_support_global }
						add_to_temp_variable = { t_change = modifier@MSH_weekly_state_legalist_support_global }
						set_temp_variable = { t_index = 0 }
						MSH_change_support = yes

						set_temp_variable = { t_change = MSH.modifier@MSH_weekly_purist_support_global }
						add_to_temp_variable = { t_change = modifier@MSH_weekly_state_purist_support_global }
						set_temp_variable = { t_index = 1 }
						MSH_change_support = yes

						set_temp_variable = { t_change = MSH.modifier@MSH_weekly_reformist_support_global }
						add_to_temp_variable = { t_change = modifier@MSH_weekly_state_reformist_support_global }
						set_temp_variable = { t_index = 2 }
						MSH_change_support = yes
					}
				}
				MSH_Influences_re_render = yes
			}
			if = {
				limit = {
					country_exists = MSP
					has_civil_war = yes
					has_completed_focus = MSH_uncertain_purist_9
				}
				if = {
					limit = {
						country_exists = MSL
						MSP = {
							has_idea = MSH_civ_war_purist_steal_msl_spirit
						}
					}
					MSL = {
						send_equipment_fraction = {
							value = 0.05
							target = MSP
						}
					}
				}
				if = {
					limit = {
						country_exists = MSR
						MSP = {
							has_idea = MSH_civ_war_purist_steal_msr_spirit
						}
					}
					MSR = {
						send_equipment_fraction = {
							value = 0.05
							target = MSP
						}
					}
				}
			}
		}
	}

	on_daily_MSH = {
		effect = {
			if = {
				limit = {
					has_country_flag = MSH_limited_MGY_war_flag
					has_war = yes
				}
				if = {
					limit = {
						any_enemy_country = {
							OR = {
								owns_state = 770
								owns_state = 766
								owns_state = 763
							}
							surrender_progress > 0.29
						}
					}
					every_enemy_country = { 
						limit = {
							OR = {
								owns_state = 770
								owns_state = 766
								owns_state = 763
							}
						}
						country_event = maciegna_unification_diplomacy.33
					}
					MGY = {
						country_event = maciegna_unification_diplomacy.34
					}
					clr_country_flag = MSH_limited_MGY_war_flag
				}
				else_if = {
					limit = {
						surrender_progress > 0.29
					}
					random_enemy_country = { 
						limit = {
							OR = {
								owns_state = 741
								owns_state = 738
								owns_state = 740
							}
						}
						country_event = maciegna_unification_diplomacy.35
					}
					MGY = {
						country_event = maciegna_unification_diplomacy.36
					}
					clr_country_flag = MSH_limited_MGY_war_flag
				}
			}
		}
	}
}
