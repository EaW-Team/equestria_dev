on_actions = {
	on_capitulation = {
		effect = {
			if = {
				limit = {
					tag = NMS
					FROM = {
						OR = {
							original_tag = CES
							original_tag = EQS
						}
					}
					has_war_with = CES
					has_war_with = EQS
				}
				CES = {
					set_country_flag = { flag = bypass_annex_compliance days = 1 value = 1 }
					annex_country = { target = ROOT }
				}
			}

			if = {
				limit = {
					OR = {
						original_tag = EQS
						original_tag = NLR
					}
					exists = yes
					has_government = fascism
					is_subject = no
					CES = {
						has_country_flag = CES_shadow_path
						has_capitulated = no
						is_subject = no
						has_war_with = PREV
					}
				}
				if = {
					limit = { original_tag = EQS }
					CES = {
						country_event = {
							id = daring_shadow.18
							days = 1
						}
					}
				}
				else = {
					CES = {
						country_event = {
							id = daring_shadow.20
							days = 1
						}
					}
				}
			}
		}
	}

	on_annex = {
		effect = {
			if = {
				limit = {
					original_tag = CES
					FROM = { tag = NMS }
					count_triggers = {
						amount = 2 #at least 2 triggers are true
						country_exists = CES
						country_exists = JUN
						country_exists = BAL
					}
				}
				country_event = { id = daring.3 }
			}
			if = {
				limit = {
					FROM = { tag = JUN }
					country_exists = CES
				}
				CES = {
					country_event = {
						id = daring.15
						days = 7
					}
				}	
			}
		}
	}

	on_monthly_CES = {
		effect = {
			if = {
				limit = {
					original_tag = CES
				}
				if = {
					limit = {
						CES = {
							exists = yes
							has_war = yes
							has_country_flag = CES_shadow_path
							has_country_flag = CES_shadow_enable_manpower_casualty_mechanic
						}
					}
					CES = {
						set_variable = { inflicted_casualties_last_month = inflicted_casualties }
						count_inflicted_casualties_on_current_enemies = yes
						subtract_from_variable = { inflicted_casualties_last_month = inflicted_casualties }
						divide_variable = { inflicted_casualties_last_month = -10 }
						add_manpower = inflicted_casualties_last_month
						set_variable = { CES_manpower_gained_last_month = inflicted_casualties_last_month }
					}	
				}
			}	
		}
	}
}
