on_actions = {

	on_startup = {
		
		effect = {
			
		}
	}

	on_army_leader_daily = {
		effect = {
			if = {
				limit = {
					FROM = {
						original_tag = EQC
						NOT = { has_country_flag = EQC_ED_hold_nmm_protest }
						is_ai = no
					}
					has_character_flag = is_EQC_ED_nightmare_moon
					is_leading_army = yes
				}
				FROM = {
					set_country_flag = EQC_ED_hold_nmm_protest
				}
				unit_leader_event = evildiarchy_newmareland.nmm_reset_1
				unit_leader_event = {
					id = evildiarchy_newmareland.nmm_reset_2
					hours = 1
				}
			}
			else_if = {
				limit = {
					FROM = {
						original_tag = EQC
						NOT = { has_country_flag = EQC_ED_hold_db_protest }
						is_ai = no
					}
					has_character_flag = is_EQC_ED_daybreaker
					is_leading_army = yes
				}
				FROM = {
					set_country_flag = EQC_ED_hold_db_protest
				}
				unit_leader_event = evildiarchy_newmareland.db_reset_1
				unit_leader_event = {
					id = evildiarchy_newmareland.db_reset_2
					hours = 1
				}
			}
		}
	}
	
	on_daily_EQS = {
		effect = {
			if = { 
				limit = {
					has_completed_focus = EQS_ED_weather_magic
				}
				add_to_variable = { weather_pb_var = daily_weather_magic_gain }
				clamp_variable = { var = weather_pb_var max = 300 }
			}
			if = {
				limit = {
					has_completed_focus = EQS_ED_slowing_them_down
				}
				EQS_AF_calculate_snitch_chance = yes
			}
			if = {
				limit = {
					has_completed_focus = EQS_ED_our_most_valuable_asset
				}
				set_temp_variable = { tmp_jspdrift = EQS_AF_daily_suspicion_drift }
				EQS_ED_calculate_jetset_icon_pos = yes 
			}
		}
	}

	on_peaceconference_ended = {
		effect = {
			if = {
				limit = {
					has_game_rule = { # This means April Fools is activated
						rule = eqc_april_fool
						option = EQC_APRIL_FOOL
					}
					ROOT = {
						tag = EQS
					}
					FROM = {
						tag = EQC
					}
					country_event = evildiarchy_equestria.200
				}
			}
		}
	}
}
