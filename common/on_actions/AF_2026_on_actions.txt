on_actions = {

	on_startup = {
		
		effect = {
			
			2 = { set_variable = { artefact_in_state = token:EQS_star_swirls_hat_artefact } }
			add_to_array = { global.states_with_artefacts = 2 }
			3 = { set_variable = { artefact_in_state = token:EQS_knuckerbockers_shell_artefact } }
			add_to_array = { global.states_with_artefacts = 3 }
			665 = {  set_variable = { artefact_in_state = token:EQC_elixir_of_life_artefact } } 
			add_to_array = { global.states_with_artefacts = 665 }
			382 = { set_variable = { artefact_in_state = token:EQC_griffichronicon_artefact } } 
			add_to_array = { global.states_with_artefacts = 382 }
		}
	}

	on_army_leader_daily = {
		effect = {
			if = {
				limit = {
					FROM = {
						original_tag = EQC
						NOT = { has_country_flag = EQC_ED_hold_nmm_protest }
					}
					has_character_flag = is_EQC_ED_nightmare_moon
					is_leading_army = yes
				}
				FROM = {
					set_country_flag = EQC_ED_hold_nmm_protest
				}
				unit_leader_event = evildiarchy_newmareland.nmm_reset_1
				unit_leader_event = {
					id = evildiarchy_newmareland.nmm_reset_2
					hours = 1
				}
			}
			else_if = {
				limit = {
					FROM = {
						original_tag = EQC
						NOT = { has_country_flag = EQC_ED_hold_db_protest }
					}
					has_character_flag = is_EQC_ED_daybreaker
					is_leading_army = yes
				}
				FROM = {
					set_country_flag = EQC_ED_hold_db_protest
				}
				unit_leader_event = evildiarchy_newmareland.db_reset_1
				unit_leader_event = {
					id = evildiarchy_newmareland.db_reset_2
					hours = 1
				}
			}
		}
	}
}
