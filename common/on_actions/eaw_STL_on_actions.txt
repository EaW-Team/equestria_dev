on_actions = {
	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					original_tag = STL
					NOT = { # to account for NG thing
						AND = {
							is_in_faction_with = EQS
							has_completed_focus = STL_help_equestria_ng
						}
					}
					STG = {
						has_defensive_war_with = STL
					}
				}
				STL_set_up_STG_war_offensive = yes
			}
		}
	}

	on_monthly_STL = {
		effect = {
			if = {
				limit = {
					has_country_leader = {
						character = STL_sugar_belle
						ruling_only = yes
					}
				}
				# CHECKING STG
				if = {
					limit = {
						NOT = {
							has_country_flag = STL_friendly_STG
							has_country_flag = STL_not_friendly_STG
						}
						OR = {
							STG = {
								has_country_leader = {
									character = STG_vasiliy_wheatin
									ruling_only = yes
								}
							}
							STG = {
								has_country_leader = {
									character = STG_sinister_serov
									ruling_only = yes
								}
							}
							STG = {
								has_country_leader = {
									character = STG_masha_norushkina
									ruling_only = yes
								}
							}
						}
					}
					
					country_event = {
						id = sugar_belle.2
						hours = 12
						random_hours = 6					
					}
				}
				else_if = {
					limit = {
						NOT = {
							has_country_flag = STL_friendly_STG
							has_country_flag = STL_not_friendly_STG
						}
						OR = {
							STG = {
								has_country_leader = {
									character = STG_caramelka_sladkaya
									ruling_only = yes
								}
							}
							STG = {
								has_country_leader = {
									character = STG_dark_wing
									ruling_only = yes
								}
							}
						}
					}
					
					country_event = {
						id = sugar_belle.1
						hours = 12
						random_hours = 6					
					}
				}
				# CHECKING EQS
				
				if = {
					limit = {
						EQS = {
							has_government = fascism
						}
						NOT = {
							has_country_flag = STL_eqs_not_friend
						}
					}
					
					country_event = {
						id = sugar_belle.3
						hours = 12
						random_hours = 6					
					}
				}
				else_if = {
					limit = {
						EQS = {
							exists = no
						}
						NOT = {
							has_country_flag = STL_eqs_not_friend
						}
					}
						
					country_event = {
						id = sugar_belle.4
						hours = 12
						random_hours = 6					
					}
				}
				# CHECKING CRY
				
				if = {
					limit = {
						CRY = {
							has_civil_war = yes
						}
						NOT = {
							has_country_flag = STL_civil_war_CRY
						}
					}
					
					country_event = {
						id = sugar_belle.5
						hours = 12
						random_hours = 6					
					}
				}

				
			}
			else_if = {
				limit = {
					has_country_flag = STL_ready_to_marry
					EQS = {
						has_war_with_major = no
						exists = yes
						has_government = democratic
					}
				}
				country_event = {
					id = starlight.23
					hours = 12
					random_hours = 6					
				}
			}
		#Checking STG Party Favor
			if = {
				limit = {
					has_country_leader = {
						character = STL_party_flavour
						ruling_only = yes
					}
				}
				if = { 
					limit = {  
						STG = { 
							OR = {
								has_country_flag = STG_flag_dark_wing_wins 
								has_country_flag = STG_flag_masha_wins 
								has_country_flag = STG_flag_evil_sinister_wins
							}
						}
					}
					country_event = { 
						id = party_favor.1
					}
				}
			}
			If = {
				limit = {
					has_global_flag = STG_STL_fighting
					has_country_flag = STL_military_unrest
				}
				country_event = feather_bangs.69
			}
		}
	}

	on_daily_STL = {
		effect = {
			# State Control - how many do we own, how many do we need to control
			IF = {
				limit = {
					has_global_flag = STG_STL_fighting
					NOT = { has_country_flag = STL_divisions_count_flag }
					STG = {
						num_divisions_in_states = {
							count > 24
							states = { 240 334 678 165 58 }
						}
					}
				}
				set_country_flag = STL_divisions_count_flag
				STL_escalate_the_war_effect = yes
			}
			IF = {
				limit = {
					has_global_flag = STG_STL_fighting
					NOT = { has_country_flag = STL_tank_divisions_flag }
					STG = {
						num_divisions_in_states = {
							count > 2
							states = { 240 334 678 165 58 }
							types = {
								light_armor medium_armor heavy_armor modern_armor super_heavy_armor
								light_tank_destroyer_brigade medium_tank_destroyer_brigade heavy_tank_destroyer_brigade modern_tank_destroyer_brigade super_heavy_tank_destroyer_brigade
								light_sp_anti_air_brigade medium_sp_anti_air_brigade heavy_sp_anti_air_brigade modern_sp_anti_air_brigade super_heavy_sp_anti_air_brigade
								light_sp_artillery_brigade medium_sp_artillery_brigade heavy_sp_artillery_brigade modern_sp_anti_air_brigade super_heavy_sp_artillery_brigade
							}
						}
					}
				}
				set_country_flag = STL_tank_divisions_flag
				STL_escalate_the_war_effect = yes
			}
			IF = {
				limit = {
					has_global_flag = STG_STL_fighting
					NOT = { has_country_flag = STL_planes_count_flag }
					STG = {
						num_planes_stationed_in_regions = {
							value > 299
				  			regions = { 65 }
				  		}
					}
				}
				LOG = "planes triggered"
				set_country_flag = STL_planes_count_flag
				STL_escalate_the_war_effect = yes
			}
			# trigger events (move to weekly if too performance heavy)
			IF = {
				limit = {
					has_global_flag = STG_STL_fighting
				}
				# Army grows in experience
				IF = {
					limit = {
						NOT = { has_country_flag = STL_army_experience_flag }
						check_variable = { STL.STL_war_escalation_level > 2 }
					}
					set_country_flag = STL_army_experience_flag
					country_event = {
						id = escalation_war.2
						hours = 6
						random_hours = 8
					}
				}
				# Army morale boosted
				IF = {
					limit = {
						NOT = { has_country_flag = STL_army_morale_flag }
						check_variable = { STL.STL_war_escalation_level > 4 }
					}
					set_country_flag = STL_army_morale_flag
					country_event = {
						id = escalation_war.3
						hours = 6
						random_hours = 8
					}
				}
				# International glory
				IF = {
					limit = {
						NOT = { has_country_flag = STL_international_glory_flag }
						check_variable = { STL.STL_war_escalation_level > 6 }
					}
					set_country_flag = STL_international_glory_flag
					country_event = {
						id = escalation_war.4
						hours = 6
						random_hours = 8
					}
				}
				# Partisans attack
				IF = {
					limit = {
						NOT = { has_country_flag = STL_partisans_attack_flag }
						check_variable = { STL.STL_war_escalation_level > 7 }
					}
					set_country_flag = STL_partisans_attack_flag
					country_event = {
						id = escalation_war.5
						hours = 6
						random_hours = 8
					}
				}
				# Survival against all odds
				IF = {
					limit = {
						NOT = { has_country_flag = STL_survival_flag }
						check_variable = { STL.STL_war_escalation_level > 8 }
					}
					set_country_flag = STL_survival_flag
					country_event = {
						id = escalation_war.6
						hours = 6
						random_hours = 8
					}
				}
				# The revolution eternal
				IF = {
					limit = {
						NOT = { has_country_flag = STL_revolution_eternal_flag }
						check_variable = { STL.STL_war_escalation_level > 9 }
					}
					set_country_flag = STL_revolution_eternal_flag
					country_event = {
						id = escalation_war.7
						hours = 6
						random_hours = 8
					}
				}
			}
		}
	}
}
