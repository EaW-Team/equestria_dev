on_actions = {
	on_declare_war = {
		effect = {
			if = {
				limit = {
					original_tag = OLE
					FROM = { original_tag = OLE }
				}
				if = {
					limit = {
						NOT = { tag = OLE }
						original_tag = OLE
					}
					if = {
						limit = {
							OR = {
								has_government = neutrality
								has_government = fascism
							}
						}
						set_cosmetic_tag = OLE_NEU
						else = {
							set_cosmetic_tag = OLE_DEM
						}
					}
				}
				FROM = {
					if = {
						limit = {
							NOT = { tag = OLE }
							original_tag = OLE
						}
						if = {
							limit = {
								OR = {
									has_government = neutrality
									has_government = fascism
								}
							}
							set_cosmetic_tag = OLE_NEU
							else = {
								set_cosmetic_tag = OLE_DEM
							}
						}
					}
				}
				if = {
					limit = {
						tag = OLE
						OR = {
							has_government = neutrality
							has_government = fascism
						}
					}
					set_cosmetic_tag = OLE_NEU
				}
			}
		}
	}

	on_civil_war_end = {
		effect = {
			if = {
				limit = { original_tag = OLE }
				set_capital = { state = 76 remember_old_capital = no }
				drop_cosmetic_tag = yes
			}
		}
	}

	on_civil_war_end_before_annexation = {
		effect = {
			if = {
				limit = { original_tag = OLE }
				FROM = {
					every_character = {
						limit = {
							is_unit_leader = yes
							is_country_leader = no
						}
						set_nationality = SOL
					}
				}
			}
		}
	}
}
