on_actions = {
	on_monthly_ROA = {
		effect = {
			if = {
				limit = {
					has_country_flag = resistance_mechanic_active
				}
				for_each_scope_loop = {
					array = ROA.owned_states
					if = {
						limit = {
							is_in_array = {
								array = ROA.recently_integrated_states
								value = THIS.id
							}
						}
						if = {
							limit = {
								check_variable = { months_since_integration > 4 } # 6 months
							}
							if = {
								limit = {
									is_in_array = {
										array = ROA.eaw_successfully_integrated_states
										value = THIS
									}
								}
								remove_from_array = { ROA.eaw_successfully_integrated_states = THIS }
							}
							clear_variable = months_since_integration
							remove_dynamic_modifier = {
								modifier = ROA_resistance_state_2_dm
							}
						}
						else = {
							add_to_variable = { months_since_integration = 1 }
						}
					}
					if = {
						limit = {
							NOT = { has_dynamic_modifier = ROA_resistance_state_1_dm }
							NOT = {
								is_in_array = {
									array = ROA.core_states
									value = THIS
								}
							}
						}
						add_dynamic_modifier = {
							modifier = ROA_resistance_state_1_dm
						}
					}
				}
			}
			resistance_dm_calculations = yes
		}
	}
	on_capitulation = {
		effect = {
			if = {
				limit = {
					ROOT = {
						original_tag = GFS
					}
					FROM = {
						has_country_leader = {
    						character = ROA_kaiser_augustus
    						ruling_only = yes
						}
					}
				}
				FROM = {
					country_event = {
						id = roam.6
						days = 5
					}
				}
			}
		}
	}
}
