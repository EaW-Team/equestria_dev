on_actions = {
	on_monthly_ROA = {
		effect = {
			if = {
				limit = {
					check_variable = { ROOT.recently_integrated_states^num > 0 }
					has_country_flag = resistance_mechanic_active
				}
				for_each_scope_loop = {
					array = ROOT.recently_integrated_states
					if = {
						limit = {
							check_variable = { months_since_integration > 4 } # 6 months
						}
						if = {
							limit = {
								is_in_array = {
									array = ROOT.eaw_successfully_integrated_states
									value = THIS
								}
							}
							remove_from_array = { ROOT.eaw_successfully_integrated_states = THIS }
						}
						remove_from_array = { ROOT.recently_integrated_states = THIS }
						clear_variable = months_since_integration
					}
					else = {
						add_to_variable = { months_since_integration = 1 }
					}
				}
				resistance_dm_calculations = yes
			}
			# Stuff for testing
			log = "[GetDateText]: ROA_TESTING: Amount of Civs [?num_of_civilian_factories]"
			log = "[GetDateText]: ROA_TESTING: Amount of Mils [?num_of_military_factories]"
			log = "[GetDateText]: ROA_TESTING: Armies [?num_armies]"
			log = "[GetDateText]: ROA_TESTING: Deployed Manpower [?deployed_army_manpower_k]k"
			log = "[GetDateText]: ROA_TESTING: Manpower in queue [?amount_manpower_in_deployment_queue]"
			log = "[GetDateText]: ROA_TESTING: Infantry Equipment [?num_equipment@infantry_equipment]"
			log = "[GetDateText]: ROA_TESTING: Artillery Equipment [?num_equipment@artillery_equipment]"
			log = "[GetDateText]: ROA_TESTING: Infantry Equipment in army [?num_equipment_in_armies@infantry_equipment]"
			log = "[GetDateText]: ROA_TESTING: Artillery Equipment in army [?num_equipment_in_armies@artillery_equipment]"
		}
	}
	on_capitulation = {
		effect = {
			if = {
				limit = {
					ROOT = {
						original_tag = GFS
					}
					FROM = {
						has_country_leader = {
    						character = ROA_kaiser_augustus
    						ruling_only = yes
						}
					}
				}
				FROM = {
					country_event = {
						id = roam.6
						days = 5
					}
				}
			}
			if = {
				limit = {
					OR = {
						AND = {
							ROOT = { original_tag = ZOO }
							NOT = {
								country_exists = KXN
								ROA = { has_war_with = KXN }
							}
						}
						AND = {
							ROOT = { original_tag = KXN }
							NOT = {
								country_exists = ZOO
								ROA = { has_war_with = ZOO }
							}
						}
					}
					FROM = {
						has_country_flag = uprising_mechanic_active
						has_country_leader = {
    						character = ROA_kaiser_augustus
    						ruling_only = yes
						}
					}
				}
				FROM = {
					country_event = {
						id = roam_lansin_uprising.9 # Lansin loses support due to Augustus's coolnest
						days = 5
					}
				}
			}
		}
	}
}
