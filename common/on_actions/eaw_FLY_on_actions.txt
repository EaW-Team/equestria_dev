on_actions = {

	on_capitulation = {
		effect = {
			
			if = {
				limit = {
					tag = FLY
				}
				#FLY = {
				#	every_character = { 
				#		set_nationality = SOL
				#	}
				#}
				every_state = {
					limit = {
						is_core_of = FLY
					}
					remove_core_of = FLY
				}
				if = {
					limit = {
						FROM = { 
							is_EQS_SOL = yes
						}
					}
					FROM = { country_event = solarempire.24 }
				}
			}
			else_if = {
				limit = {
					tag = FLY
					ROOT = {
						OR = {
							tag = NLR
							is_EQS_SOL = yes
						}
					}
				}
				transfer_navy = { target = FROM }
			}
		}
	}
	
	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					ROOT = {
						tag = FLY
					}
					FROM = {
						OR = {
							original_tag = EQS
							original_tag = NLR
							original_tag = CHN
							original_tag = OLE
							original_tag = PLB
						}
					}
					FROM.FROM = {
						NOT = {
							has_state_flag = FLY_state_captured
						}
						OR = {
							AND = {
								OR = {
									is_owned_by = EQS
									is_owned_by = NLR
								}
								OR = {
									is_core_of = EQS
									is_core_of = NLR
								}
							}
						}
					}
				}
				if = {
					limit = {
						ROOT = {
							NOT = {
								has_country_flag = FLY_skip_occupation_event
							}
						}
					}
					if = {
						limit = {
							FROM = {
								OR = {
									original_tag = CHN
									original_tag = OLE
									original_tag = PLB
								}
							}
						}
						FROM.FROM = {
							save_event_target_as = FLY_liberation_target
							save_event_target_as = FLY_liberation_target_changelings ### saving this for later to account for changeling invasion during the civil war
						}
						ROOT = {
							country_event = {
								id = fillydelphia_utility.1
							}
						}
					}
					else_if = {
						limit = {
							FROM = {
								OR = {
									original_tag = EQS
									original_tag = NLR
								}
							}
						}
						FROM.FROM = {
							save_event_target_as = FLY_liberation_target
						}
						ROOT = {
							country_event = {
								id = fillydelphia_utility.1
							}
						}
					}
				}
				else = {
					FROM.FROM = {
						save_event_target_as = FLY_liberation_target
					}
					ROOT = {
						country_event = {
							id = fillydelphia_utility.2
						}
					}
				}
			}
		}
	}
	
	on_unit_leader_created = {
		effect = {
			if = {
				limit = {
					OR = {
						is_army_leader = yes
						is_navy_leader = yes
					}
					var:creator = {
						tag = FLY
						OR = {
							has_completed_focus = FLY_guardians_harmony_cw
							has_completed_focus = FLY_guardians_harmony
						}
					}
				}
				if = {
					limit = {
						is_army_leader = yes
					}
					add_trait = {
						trait = guardian_of_harmony_army
					}
				}
				else_if = {
					limit = {
						is_navy_leader = yes
					}
					add_trait = {
						trait = guardian_of_harmony_navy
					}
				}
			}
		}
	}
	
	on_unit_leader_promote_from_ranks_green = {
		effect = {
			if = {
				limit = {
					OWNER = {
						tag = FLY
						OR = {
							has_completed_focus = FLY_guardians_harmony_cw
							has_completed_focus = FLY_guardians_harmony
						}
					}
				}
				if = {
					limit = {
						is_army_leader = yes
					}
					add_trait = {
						trait = guardian_of_harmony_army
					}
				}
			}
		}
	}
}
