scripted_gui = {
	eaw_SLC_gui_imamu_vision = {
		context_type = decision_category
		window_name = "SLC_gui_dual_imamu_vision_cat"

		properties = {
			progress_bar = {
				frame = SLC_imamu_vision_progress
			}
		}
	}

	@SLC_CBC_STEP_PER_LEVEL = 20.000 # 100/(Number of level - 1)

	eaw_SLC_gui_cbc = {
		context_type = decision_category
		window_name = "SLC_gui_commie_building_communism_cat"

		dirty = SLC_cbc_update_gui

		triggers = {
			authoritarianism_progress_bar_visible = {
				set_temp_variable = { autho_bar = SLC_cbc_authoritarianism_level }
				multiply_temp_variable = { autho_bar = @SLC_CBC_STEP_PER_LEVEL }
				round_temp_variable = autho_bar
				add_to_temp_variable = { autho_bar = 1 }
			}
			authoritarianism_state_visible = {
				set_temp_variable = { autho_icon = SLC_cbc_authoritarianism_level }
				add_to_temp_variable = { autho_icon = 1 }
			}

			praxial_leaning_progress_bar_visible = {
				set_temp_variable = { praxial_bar = SLC_cbc_praxial_leaning_level }
				multiply_temp_variable = { praxial_bar = @SLC_CBC_STEP_PER_LEVEL }
				round_temp_variable = praxial_bar
				add_to_temp_variable = { praxial_bar = 1 }
			}
			praxial_leaning_state_visible = {
				set_temp_variable = { praxial_icon = SLC_cbc_praxial_leaning_level }
				add_to_temp_variable = { praxial_icon = 1 }
			}

			fervour_frame_visible = {
				if = {
					limit = { has_country_flag = SLC_cbc_fervour_activated }
					set_temp_variable = { fervour_frame_frame = 2 }
				} else = {
					set_temp_variable = { fervour_frame_frame = 1 }
				}
			}
			fervour_text_visible = {
				has_country_flag = SLC_cbc_fervour_activated
			}
		}

		properties = {
			authoritarianism_progress_bar = {
				frame = autho_bar
			}
			authoritarianism_state = {
				frame = autho_icon
			}
			praxial_leaning_progress_bar = {
				frame = praxial_bar
			}
			praxial_leaning_state = {
				frame = praxial_icon
			}
			fervour_frame = {
				frame = fervour_frame_frame
			}
		}
	}
}
