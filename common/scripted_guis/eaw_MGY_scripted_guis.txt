scripted_gui = {
	MGY_GranaryOfZebrica_ui = {
		window_name = "MGY_GranaryOfZebrica_ui_window"
		context_type = decision_category
		dirty = MGY.MGY_GranaryOfZebrica_var

		triggers = {
			MGY_GranaryOfZebrica_ui_ToggleProduction_click_enabled = {
				always = yes
			}
			MGY_GranaryOfZebrica_ui_ToggleMarket_click_enabled = {
				always = yes
			}
		}

		effects = {
			MGY_GranaryOfZebrica_ui_ToggleProduction_click = {
				if = {
					limit = {
						check_variable = {
							MGY_ProductionDecision_var = 1
						}
					}
					set_variable = {
						MGY_ProductionDecision_var = 2
					}
				}
				else = {
					set_variable = {
						MGY_ProductionDecision_var = 1
					}
				}
				MGY_refreshGranaryScriptedUi = yes
			}
			MGY_GranaryOfZebrica_ui_ToggleMarket_click = {
				if = {
					limit = {
						check_variable = {
							MGY_MarketDecision_var = 1
						}
					}
					set_variable = {
						MGY_MarketDecision_var = 2
					}
				}
				else = {
					set_variable = {
						MGY_MarketDecision_var = 1
					}
				}
				MGY_refreshGranaryScriptedUi = yes
			}
		}

		properties = {
			MGY_GranaryOfZebrica_ui_ToggleProduction = {
				frame = MGY_ProductionDecision_var
			}
			MGY_GranaryOfZebrica_ui_ToggleMarket = {
				frame = MGY_MarketDecision_var
			}
		}
	}
}
