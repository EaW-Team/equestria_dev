		dirty = EaW_NVA_socialist_revolutionary_circus_var

		effects = {
			party_1_icon_click = {
				NVA_set_dummy_array = yes
				if = {
					limit = {
						check_variable = { party_select_var = 1 }
					}
					set_variable = { party_select_var = 0 }
				}
				else = {
					set_variable = { party_select_var = 1 }
				}
				add_to_variable = { EaW_NVA_socialist_revolutionary_circus_var = 1 }
				log = "party 1 was clicked"
			}
			party_2_icon_click = {
				NVA_set_dummy_array = yes
				if = {
					limit = {
						check_variable = { party_select_var = 2 }
					}
					set_variable = { party_select_var = 0 }
				}
				else = {
					set_variable = { party_select_var = 2 }
				}
				add_to_variable = { EaW_NVA_socialist_revolutionary_circus_var = 1 }
				log = "party 2 was clicked"
			}

					set_variable = { party_select_var = 0 }
				}

					set_variable = { party_select_var = 3 }
				}
				add_to_variable = { EaW_NVA_socialist_revolutionary_circus_var = 1 }
				log = "party 3 was clicked"

				NVA_set_dummy_array = yes

					limit = {
						check_variable = { party_select_var = 4 }
					}
					set_variable = { party_select_var = 0 }

					set_variable = { party_select_var = 4 }
				}
				add_to_variable = { EaW_NVA_socialist_revolutionary_circus_var = 1 }
				log = "party 4 was clicked"
			}
			special_gui_button_click = {
				add_to_variable = { EaW_NVA_socialist_revolutionary_circus_var = 1 }
				if = {
					limit = {
						has_country_flag = NVA_show_grand_circus_gui
					}
					clr_country_flag = NVA_show_grand_circus_gui
				}
				else = {
					set_country_flag = NVA_show_grand_circus_gui
				}
				if = {
					limit = {
						NOT = {
							has_country_flag = NVA_grand_circus_bop_var_initialised
						}
					}
					set_country_flag = NVA_grand_circus_bop_var_initialised
					set_variable = { NVA_socialist_grand_circus_bop_internal = 0 }
					set_variable = { NVA_socialist_grand_circus_bop_external = 0 }
					NVA_recalc_socialist_bop_positions = yes
				}
				log = "special gui button clicked"
			}
		}
