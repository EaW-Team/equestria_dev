scripted_gui = {
	HAU_naval_raiding_gui = {
		context_type = decision_category
		window_name = HAU_naval_raiding_gui_window
		dirty = HAU_naval_raiding_gui_dirty
		effects = {
			HAU_naval_raiding_route_1_click = {
				set_variable = { HAU.HAU_naval_raiding_route_selected_var = 1 }
				HAU_naval_raiding_probability_checkout_effect = yes
				add_to_variable = { HAU_naval_raiding_gui_dirty = 1}
			}
			HAU_naval_raiding_route_2_click = {
				set_variable = { HAU.HAU_naval_raiding_route_selected_var = 2 }
				HAU_naval_raiding_probability_checkout_effect = yes
				add_to_variable = { HAU_naval_raiding_gui_dirty = 1}
			}
			HAU_naval_raiding_route_3_click = {
				set_variable = { HAU.HAU_naval_raiding_route_selected_var = 3 }
				HAU_naval_raiding_probability_checkout_effect = yes
				add_to_variable = { HAU_naval_raiding_gui_dirty = 1}
			}
			HAU_naval_raiding_route_4_click = {
				set_variable = { HAU.HAU_naval_raiding_route_selected_var = 4 }
				HAU_naval_raiding_probability_checkout_effect = yes
				add_to_variable = { HAU_naval_raiding_gui_dirty = 1}
			}
			HAU_naval_raiding_route_5_click = {
				set_variable = { HAU.HAU_naval_raiding_route_selected_var = 5 }
				HAU_naval_raiding_probability_checkout_effect = yes
				add_to_variable = { HAU_naval_raiding_gui_dirty = 1}
			}
		}
	}
	HAU_naval_raiding_popup_gui = {
		context_type = player_context
		window_name = HAU_naval_raiding_popup_gui_window
		parent_window_token = decision_tab
		dirty = HAU_naval_raiding_gui_dirty
		visible = {
			has_variable = HAU.HAU_naval_raiding_route_selected_var
		}
		triggers = {
			HAU_naval_raiding_popup_gui_button_click_enabled = {
				NOT = {
					has_variable = HAU.HAU_naval_raiding_route_launched_var
				}
			}
		}
		effects = {
			HAU_naval_raiding_popup_gui_close_click = {
				clear_variable = HAU.HAU_naval_raiding_route_selected_var
				add_to_variable = { HAU_naval_raiding_gui_dirty = 1 }
			}
			HAU_naval_raiding_popup_gui_button_click = {
				if = {
					limit = {
						check_variable = { HAU.HAU_naval_raiding_route_selected_var = 1 }
					}
					clear_variable = HAU.HAU_naval_raiding_route_selected_var
					set_variable = { HAU.HAU_naval_raiding_route_launched_var = 1 }
				}
				else_if = {
					limit = {
						check_variable = { HAU.HAU_naval_raiding_route_selected_var = 2 }
					}
					clear_variable = HAU.HAU_naval_raiding_route_selected_var
					set_variable = { HAU.HAU_naval_raiding_route_launched_var = 2 }
				}
				else_if = {
					limit = {
						check_variable = { HAU.HAU_naval_raiding_route_selected_var = 3 }
					}
					clear_variable = HAU.HAU_naval_raiding_route_selected_var
					set_variable = { HAU.HAU_naval_raiding_route_launched_var = 3 }
				}
				else_if = {
					limit = {
						check_variable = { HAU.HAU_naval_raiding_route_selected_var = 4 }
					}
					clear_variable = HAU.HAU_naval_raiding_route_selected_var
					set_variable = { HAU.HAU_naval_raiding_route_launched_var = 4 }
				}
				else_if = {
					limit = {
						check_variable = { HAU.HAU_naval_raiding_route_selected_var = 5 }
					}
					clear_variable = HAU.HAU_naval_raiding_route_selected_var
					set_variable = { HAU.HAU_naval_raiding_route_launched_var = 5 }
				}
				add_to_variable = { HAU_naval_raiding_gui_dirty = 1 }
				activate_mission = HAU_naval_raid_timer
			}
		}
	}
}
