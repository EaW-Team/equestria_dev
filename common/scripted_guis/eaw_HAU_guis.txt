scripted_gui = {
	HAU_naval_raiding_gui = {
		context_type = decision_category
		window_name = HAU_naval_raiding_gui_window
		dirty = HAU_naval_raiding_gui_dirty
		triggers = {
			HAU_naval_raiding_popup_gui_name_textbox_visible = {
				OR = {
					has_country_flag = HAU_naval_raiding_route_1_selected
					has_country_flag = HAU_naval_raiding_route_2_selected
				}
			}
			HAU_naval_raiding_popup_gui_info_textbox_visible = {
				OR = {
					has_country_flag = HAU_naval_raiding_route_1_selected
					has_country_flag = HAU_naval_raiding_route_2_selected
				}
			}
			HAU_naval_raiding_popup_gui_button_visible = {
				OR = {
					has_country_flag = HAU_naval_raiding_route_1_selected
					has_country_flag = HAU_naval_raiding_route_2_selected
				}
			}
			HAU_naval_raiding_popup_gui_close_visible = {
				OR = {
					has_country_flag = HAU_naval_raiding_route_1_selected
					has_country_flag = HAU_naval_raiding_route_2_selected
				}
			}
			HAU_naval_raiding_popup_gui_bg_visible = {
				OR = {
					has_country_flag = HAU_naval_raiding_route_1_selected
					has_country_flag = HAU_naval_raiding_route_2_selected
				}
			}
		}
		effects = {
			HAU_naval_raiding_route_1_click = {
				set_country_flag = HAU_naval_raiding_popup_open
				set_country_flag = HAU_naval_raiding_route_1_selected
				clr_country_flag = HAU_naval_raiding_route_2_selected
				set_variable = { HAU_naval_raiding_popup_gui_bg_x = 140 }
				set_variable = { HAU_naval_raiding_popup_gui_bg_y = 340 }
				HAU_naval_raiding_probability_checkout_effect = yes
				add_to_variable = { HAU_naval_raiding_gui_dirty = 1}
			}
			HAU_naval_raiding_route_2_click = {
				set_country_flag = HAU_naval_raiding_popup_open
				clr_country_flag = HAU_naval_raiding_route_1_selected
				set_country_flag = HAU_naval_raiding_route_2_selected
				set_variable = { HAU_naval_raiding_popup_gui_bg_x = 120 }
				set_variable = { HAU_naval_raiding_popup_gui_bg_y = 0 }
				HAU_naval_raiding_probability_checkout_effect = yes
				add_to_variable = { HAU_naval_raiding_gui_dirty = 1}
			}
			HAU_naval_raiding_popup_gui_close_click = {
				clr_country_flag = HAU_naval_raiding_route_1_selected
				clr_country_flag = HAU_naval_raiding_route_2_selected
				add_to_variable = { HAU_naval_raiding_gui_dirty = 1 }
			}
			HAU_naval_raiding_popup_gui_button_click = {
				if = {
					limit = {
						has_country_flag = HAU_naval_raiding_route_1_selected
					}
					clr_country_flag = HAU_naval_raiding_route_1_selected
					set_country_flag = HAU_naval_raiding_route_1_launched
				}
				else_if = {
					limit = {
						has_country_flag = HAU_naval_raiding_route_2_selected
					}
					clr_country_flag = HAU_naval_raiding_route_2_selected
					set_country_flag = HAU_naval_raiding_route_2_launched
				}
				add_to_variable = { HAU_naval_raiding_gui_dirty = 1 }
				activate_mission = HAU_naval_raid_timer
			}
		}
		properties = {
			HAU_naval_raiding_popup_gui_name_textbox = {
				x = HAU_naval_raiding_popup_nametext_x
				y = HAU_naval_raiding_popup_nametext_y
			}
			HAU_naval_raiding_popup_gui_info_textbox = {
				x = HAU_naval_raiding_popup_infotext_x
				y = HAU_naval_raiding_popup_infotext_y
			}
			HAU_naval_raiding_popup_gui_button = {
				x = HAU_naval_raiding_popup_gui_button_x
				y = HAU_naval_raiding_popup_gui_button_y
			}
			HAU_naval_raiding_popup_gui_close = {
				x = HAU_naval_raiding_popup_gui_close_x
				y = HAU_naval_raiding_popup_gui_close_y
			}
			HAU_naval_raiding_popup_gui_bg = {
				x = HAU_naval_raiding_popup_gui_bg_x
				y = HAU_naval_raiding_popup_gui_bg_y
			}
		}
	}
}
