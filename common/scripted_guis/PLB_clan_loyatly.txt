scripted_gui = {

	PLB_clan_loyalty_button = {
		context_type = player_context
		
		window_name = "PLB_loyalty_button"
		parent_window_token = politics_tab 
		
		visible = {
            tag = PLB
		}
		
		effects = {
			PLB_shield_click = {
				set_temp_variable = { show = 1 }
				subtract_from_temp_variable = { show = show_PLB_loyalty }
				set_variable = { show_PLB_loyalty = show }
			}
		}
	}
	
	PLB_clan_loyalty_screen = {
		context_type = player_context
		
		window_name = "PLB_loyalty_screen"
		
		visible = {
			check_variable = { show_PLB_loyalty > 0.0 }
		}
		
		effects = {
			polar_close_button_click = { 
				set_variable = { 
					show_PLB_loyalty = 0
				}
			}
		}
	}
}
