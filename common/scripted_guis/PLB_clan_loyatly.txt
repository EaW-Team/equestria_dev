scripted_gui = {

	PLB_clan_loyalty_button = {
		context_type = player_context
		
		window_name = "PLB_loyalty_button"
		parent_window_token = politics_tab 
		
		visible = {
            tag = PLB
		}
		
		effects = {
			PLB_shield_click = {
				set_temp_variable = { show = 1 }
				subtract_from_temp_variable = { show = show_PLB_loyalty }
				set_variable = { show_PLB_loyalty = show }
			}
		}
	}
	
	PLB_clan_loyalty_screen = {
		context_type = player_context
		
		window_name = "PLB_loyalty_screen"
		
		visible = {
			check_variable = { show_PLB_loyalty > 0.0 }
		}
		
		effects = {
			polar_close_button_click = { 
				set_variable = { 
					show_PLB_loyalty = 0
				}
			}
			
			Hunter_Button_1_click = {
				if = {
					limit = {
						has_political_power > 99
					}
					
					add_to_variable = {
						hunter_loyalty = 20
					}
					
					add_political_power = -100
				}
			}
			
			Hunter_Button_2_click = {
				if = {
					limit = {
						check_variable = { 
							hunter_loyalty > 19
						}
					}
					
					add_to_variable = {
						hunter_loyalty = -20
					}
					
					add_timed_idea = {
						idea = PLB_festival
						days = 100
					}
				}
			}
			
			Hunter_Button_3_click = {
				if = {
					limit = {
						check_variable = { 
							hunter_loyalty > 59
						}
					}
					
					add_to_variable = {
						hunter_loyalty = -60
					}
					
					add_timed_idea = {
						idea = PLB_knowledge_from_elders
						days = 100
					}
				}
			}
		}
		
		triggers = {
			Loyalty_Symbol_Hunter_1_visible = {
				check_variable = { hunter_loyalty > 60 }
			}
			Loyalty_Symbol_Hunter_2_visible = {
				check_variable = { hunter_loyalty < 40 }
			}
			Loyalty_Symbol_Hunter_3_visible = {
				check_variable = { hunter_loyalty > 40 }
				check_variable = { hunter_loyalty < 60 }
			}
			
			Loyalty_Symbol_Warrior_1_visible = {
				check_variable = { warrior_loyalty > 60 }
			}
			Loyalty_Symbol_Warrior_2_visible = {
				check_variable = { warrior_loyalty < 40 }
			}
			Loyalty_Symbol_Warrior_3_visible = {
				check_variable = { warrior_loyalty > 40 }
				check_variable = { warrior_loyalty < 60 }
			}
			
			Loyalty_Symbol_Diplo_1_visible = {
				check_variable = { diplo_loyalty > 60 }
			}
			Loyalty_Symbol_Diplo_2_visible = {
				check_variable = { diplo_loyalty < 40 }
			}
			Loyalty_Symbol_Diplo_3_visible = {
				check_variable = { diplo_loyalty > 40 }
				check_variable = { diplo_loyalty < 60 }
			}
			
			Loyalty_Symbol_Industry_1_visible = {
				check_variable = { industry_loyalty > 60 }
			}
			Loyalty_Symbol_Industry_2_visible = {
				check_variable = { industry_loyalty < 40 }
			}
			Loyalty_Symbol_Industry_3_visible = {
				check_variable = { industry_loyalty > 40 }
				check_variable = { industry_loyalty < 60 }
			}
		}
	}
}
