scripted_gui = {

	MTA_artifacts_tab_gui = {
		context_type = player_context
		parent_window_token = top_bar
		window_name = "MTA_artifacts_tab_gui_container"

		visible = {
			has_country_flag = MTA_opened_cave_flag
		}

		triggers = {
			button_alert_visible = {
				NOT = {
					has_country_flag = MTA_artifacts_arrows_hide
				}
			}
		}

		effects = {
			button_open_click = {
				if = {
					limit = {
						has_country_flag = MTA_artifacts_tab_open
					}
					clr_country_flag = MTA_artifacts_tab_open
				}
				else = {
					set_country_flag = MTA_artifacts_tab_open
				}
				set_country_flag = MTA_artifacts_arrows_hide
			}
		}
	}

	MTA_artifact_list = {
		context_type = player_context
		window_name = "MTA_artifact_list_window"

		visible = {
			has_country_flag = MTA_artifacts_tab_open
		}
		
		properties = {
			selected_icon = {
				image = "[GetMTAArtifactIconSelected]"
			}
            icon = {
                image = "[GetMTAArtifactIconTHIS]"
            }
        }

		dynamic_lists = {
			MTA_artifacts_list_grid = {
				array = MTA_owned_artifacts
				change_scope = yes

				entry_container = MTA_artifact_entry
			}
		}

		effects = {
			button_close_click = {
				clr_country_flag = MTA_artifacts_tab_open
			}
			icon_click = {
				set_variable = { ROOT.MTA_artifact_selected = THIS }
			}
		}
	}
	
	MTA_artifact_document = {
		context_type = player_context
		window_name = "MTA_artifact_document_container"

		visible = {
			has_country_flag = MTA_artifacts_tab_open
		}

		triggers = {
			image_click_enabled = {
				always = no
			}
		}

		properties = {
			image = {
				image = "[GetMTAArtifactIconSelected]"
			}
		}
	}
}
