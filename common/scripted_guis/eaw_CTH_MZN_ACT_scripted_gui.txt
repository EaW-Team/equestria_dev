scripted_gui = {
	MZN_ACT_GUI = {
		window_name = MZN_ACT_GUI
		context_type = player_context

		visible = {
			has_country_flag = MZN_ACT_GUI_OPEN
		}
		triggers = {
			MZN_sigature_monzano_1_visible = {
				has_country_flag = MZN_SIGNATURE_SHOW
				set_temp_variable = { MZN_sigature_monzano_1_frame = 0 } # 1 for looping, 0 for no looping
				multiply_temp_variable = { MZN_sigature_monzano_1_frame = 256 }
				add_to_temp_variable = { MZN_sigature_monzano_1_frame = 180 } # FPS
				multiply_temp_variable = { MZN_sigature_monzano_1_frame = 64 }
				add_to_temp_variable = { MZN_sigature_monzano_1_frame = 19 } # End frame
				multiply_temp_variable = { MZN_sigature_monzano_1_frame = 64 }
				add_to_temp_variable = { MZN_sigature_monzano_1_frame = 1 } # Start frame
			}

			MZN_sigature_monzano_1_click_enabled = {
				has_country_flag = MZN_SIGNATURE_SHOW
			}
		}

		effects = {
			MZN_ACT_SIGNHERE_click = {
			if = {
				limit = {
					not = {
						has_country_flag = MZN_SIGNATURE_SHOW
					}
				}
				set_country_flag = MZN_SIGNATURE_SHOW
				scoped_sound_effect = "HIE_journal_o"
			}

			else = {
				if = {
					limit = {
						has_country_flag = MZN_resign_letter_flag
					}
					country_event = {
						id = colthage_monzano_successor.5
						days = 1
					}
				}
				if = {
					limit = {
						has_country_flag = MZN_consitution_amendment_lock
					}
					clr_country_flag = MZN_consitution_amendment_lock
				}
				clr_country_flag = MZN_ACT_GUI_OPEN
				clr_country_flag = MZN_SIGNATURE_SHOW
				MZN_ACT_CLEAR_ALL_ACT = yes
				MZN_ACT_CLEAR_EFFECT = yes
			}	
			}
		}

		properties = {
			MZN_sigature_monzano_1 = {
				frame = MZN_sigature_monzano_1_frame
			}
		}
	}
}
