scripted_gui = {
	eaw_peace_capitulation_gui = {
		context_type = player_context
		window_name = eaw_peace_capitulation_parent
		dirty = winning_country^num
		visible = {
			check_variable = { winning_country^num > 0 }
		}
		dynamic_lists = {
			capitulation_gridbox = {
				array = winning_country
				entry_container = eaw_peace_capitulation_gui
				index = peace_gui_index
			}
		}
		effects = {
			close_button_click = {
				remove_from_array = { array = losing_country index = peace_gui_index }
				remove_from_array = { array = winning_country index = peace_gui_index }
			}
		}
		properties = {
			winner_flag = { image = "[?winning_country^peace_gui_index.GetFlag]" }
			winner_flag_fade = { image = "[?winning_country^peace_gui_index.GetFlag]" }
			loser_flag = { image = "[?losing_country^peace_gui_index.GetFlag]" }
			loser_flag_fade = { image = "[?losing_country^peace_gui_index.GetFlag]" }
		}
	}
}
