				}

                set_temp_variable = { temp = avg_offensive_combat_status }
                check_variable = { temp > 0.45 }
                check_variable = { ai_random > temp }

                add = 2
            }
        }

        cost = 0.0
        duration = 168
		cooldown = 480

			var = TBK_AF_FP

        }
