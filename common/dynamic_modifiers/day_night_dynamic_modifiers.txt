
#Simulates an eternal night penalty. Applies any night bonuses towards removing the penalty.
EaW_night_override_modifier = {
	enable = {
		set_temp_variable = { 
			temp_land_night_attack = {
				value = 0.5
				subtract = modifier@land_night_attack
			}
		}
		set_temp_variable = { 
			temp_army_attack_factor = {
				value = temp_land_night_attack
				multiply = -1
			}
		}
		
		set_temp_variable = { 
			temp_carrier_night_traffic = {
				value = 1.0
				subtract = modifier@carrier_night_traffic
			}
		}
		set_temp_variable = { 
			temp_army_attack_factor = {
				value = temp_carrier_night_traffic
				multiply = -1
			}
		}
		
		set_temp_variable = { 
			temp_air_night_penalty = {
				value = -0.25
				add = modifier@air_night_penalty
			}
		}
		
		set_temp_variable = { 
			temp_air_strategic_bomber_night_penalty = {
				value = -0.5
				add = modifier@air_strategic_bomber_night_penalty
			}
		}
		set_temp_variable = { 
			temp_air_strategic_bomber_bombing_factor = {
				value = temp_air_strategic_bomber_night_penalty
				multiply = -1
			}
		}
		
	}
	#icon = GFX_modifiers_sabotaged_resource
	
	land_night_attack = temp_land_night_attack
	carrier_night_traffic = temp_carrier_night_traffic
	air_strategic_bomber_night_penalty = temp_air_strategic_bomber_night_penalty
	air_night_penalty = temp_air_night_penalty
	
	army_attack_factor = temp_army_attack_factor
	carrier_traffic= temp_carrier_traffic
	air_strategic_bomber_bombing_factor = temp_air_strategic_bomber_bombing_factor
}
