
#Simulates an eternal night penalty. Applies any night bonuses towards removing the penalty.
EaW_night_override_modifier = {
	enable = {
		set_temp_variable = { temp_land_night_attack = 0.5 }
		subtract_from_temp_variable = { temp_land_night_attack = modifier@land_night_attack }
		set_temp_variable = { temp_army_attack_factor = temp_land_night_attack }
		multiply_temp_variable = { temp_army_attack_factor = -1 }
		
		set_temp_variable = { temp_carrier_night_traffic = 1.0 }
		subtract_from_temp_variable = { temp_carrier_night_traffic = modifier@carrier_night_traffic }
		set_temp_variable = { temp_carrier_traffic = temp_carrier_night_traffic }
		multiply_temp_variable = { temp_carrier_traffic = -1 }
		
		set_temp_variable = { temp_air_night_penalty = 0.25 }
		subtract_from_temp_variable = { temp_air_night_penalty = modifier@air_night_penalty }
		
		set_temp_variable = { temp_air_strategic_bomber_night_penalty = -0.5 }
		add_to_temp_variable = { temp_air_strategic_bomber_night_penalty = modifier@air_strategic_bomber_night_penalty }
		set_temp_variable = { temp_air_carrier_night_penalty_reduction_factor = temp_air_strategic_bomber_night_penalty }
		multiply_temp_variable = { temp_air_carrier_night_penalty_reduction_factor = -1 }
		
	}
	icon = GFX_modifiers_sabotaged_resource
	
	land_night_attack = temp_land_night_attack
	carrier_night_traffic = temp_carrier_night_traffic
	air_strategic_bomber_night_penalty = temp_air_strategic_bomber_night_penalty
	air_night_penalty = temp_air_night_penalty
	
	army_attack_factor = temp_army_attack_factor
	carrier_traffic= temp_carrier_traffic
	air_strategic_bomber_bombing_factor = temp_air_strategic_bomber_bombing_factor
}
