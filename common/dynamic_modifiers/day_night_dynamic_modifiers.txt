
#Cancels out any night penalty and night bonus
EaW_day_override_modifier = {
	enable = {
		set_temp_variable = { temp_land_night_attack = 0.5 }
		subtract_from_temp_variable = { temp_land_night_attack = modifier@land_night_attack }
		
		set_temp_variable = { temp_carrier_night_traffic = 1.0 }
		subtract_from_temp_variable = { temp_carrier_night_traffic = modifier@carrier_night_traffic }
		
		set_temp_variable = { temp_air_night_penalty = 0.5 }
		subtract_from_temp_variable = { temp_air_night_penalty = modifier@air_night_penalty }
		
		set_temp_variable = { temp_air_strategic_bomber_night_penalty = -0.5 }
		add_to_temp_variable = { temp_air_strategic_bomber_night_penalty = modifier@air_strategic_bomber_night_penalty }
		
		set_temp_variable = { temp_air_carrier_night_penalty_reduction_factor = 0.5 }
		subtract_from_temp_variable = { temp_air_carrier_night_penalty_reduction_factor = modifier@air_carrier_night_penalty_reduction_factor }
	}
	icon = GFX_modifiers_sabotaged_resource
	
	land_night_attack = temp_land_night_attack
	#naval_night_attack = 0.3
	#night_spotting_chance = 0.3
	carrier_night_traffic = temp_carrier_night_traffic
	air_strategic_bomber_night_penalty = temp_air_strategic_bomber_night_penalty
	air_carrier_night_penalty_reduction_factor = temp_air_carrier_night_penalty_reduction_factor
}
