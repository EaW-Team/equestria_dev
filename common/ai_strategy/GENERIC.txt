	enable = {
		NOT = { tag = CHN }
		CHN = {
			has_government = fascism
		}
		NOT = { is_subject_of = CHN }
		NOT = { is_in_faction_with = CHN }

	abort = {
		OR = {
			is_subject_of = CHN

					NOT = {	has_government = fascism }
				}
			}
