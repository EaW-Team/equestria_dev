	enable = {
		NOT = { tag = CHN }
		CHN = {
			has_government = fascism
		}
		NOT = { 
			is_subject_of = CHN
			is_in_faction_with = CHN
		}
	}

	abort = {
		OR = {
			is_subject_of = CHN
			is_in_faction_with = CHN
			CHN = {
				NOT = {	has_government = fascism }
			}
		}
	}

	ai_strategy = {
		type = alliance
		id = "CHN"
		value = -500

		type = befriend
