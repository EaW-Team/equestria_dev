CHN_no_one_loves_them = {
	enable = {
		NOT = { tag = CHN }
		CHN = {
			has_government = fascism
		}
		NOT = { is_subject_of = CHN }
		NOT = { is_in_faction_with = CHN }
	}
	
	abort = {
		OR = {
			is_subject_of = CHN
			is_in_faction_with = CHN
			CHN = {
				OR = {
					NOT = {	has_government = fascism }
				}
			}
		}
	}
	
	ai_strategy = {
		type = alliance
		id = "CHN"
		value = -500
	}
	ai_strategy = {
		type = befriend
		id = "CHN"
		value = -50
	}
}
