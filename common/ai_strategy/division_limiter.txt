division_limiter = {
	allowed = {
		NOT = {
			tag = DED
			tag = EWI
		}
	}
	enable = {
		is_exempt_from_division_limit = no
		set_temp_variable = {
			upper_limit = num_of_civilian_factories
		}
		multiply_temp_variable = {
			upper_limit = 0.75
		}
		add_to_temp_variable = {
			upper_limit = num_of_military_factories
		}
		if = {
			limit = {
				manpower_per_military_factory < 10000
			}
			multiply_temp_variable = {
				upper_limit = 0.9
			}
		}
		if = {
			limit = {
				manpower_per_military_factory < 7500
			}
			multiply_temp_variable = {
				upper_limit = 0.8
			}
		}
		else_if = {
			limit = {
				manpower_per_military_factory < 5000
			}
			multiply_temp_variable = {
				upper_limit = 0.7
			}
		}
		else_if = {
			limit = {
				manpower_per_military_factory < 2500
			}
			multiply_temp_variable = {
				upper_limit = 0.6
			}
		}
		else_if = {
			limit = {
				manpower_per_military_factory < 2000
			}
			multiply_temp_variable = {
				upper_limit = 0.5
			}
		}
		else_if = {
			limit = {
				manpower_per_military_factory < 1500
			}
			multiply_temp_variable = {
				upper_limit = 0.4
			}
		}
		else_if = {
			limit = {
				manpower_per_military_factory < 1000
			}
			multiply_temp_variable = {
				upper_limit = 0.3
			}
		}
		else_if = {
			limit = {
				manpower_per_military_factory < 750
			}
			multiply_temp_variable = {
				upper_limit = 0.1
			}
		}
		else_if = {
			limit = {
				manpower_per_military_factory < 500
			}
			multiply_temp_variable = {
				upper_limit = 0.01
			}
		}
		if = {
			limit = {
				OR = {
					has_war = yes
					check_variable = {
						potential_and_current_enemies^num > 0
					}
				}
			}
			multiply_temp_variable = {
				upper_limit = 1.15
			}
		}
		else_if = {
			limit = {
				threat > 0.71
			}
			multiply_temp_variable = {
				upper_limit = 1.05
			}
		}
		if = {
			limit = {
				is_major = yes
			}
			multiply_temp_variable = {
				upper_limit = 1.10
			}
		}
		check_variable = {
			num_divisions > upper_limit
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = ai_wanted_divisions_factor
		value = -200
	}


}
