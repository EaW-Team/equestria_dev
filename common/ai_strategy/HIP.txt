HIP_stop_equestria_from_falling = {
	allowed = {
		tag = HIP
	}
	enable = {
		has_war = no
		num_divisions > 10 # we must have some divisions ourselves

		OR = {
			has_government = democratic
			has_country_flag = HIP_coalition_government
			AND = {
				has_completed_focus = HIP_rejoin_friends
				has_government = neutrality
			}
		}
		NOT = { has_war_with = EQS }
		EQS = {
			surrender_progress > 0.00
			has_capitulated = no
			has_government = democratic
			any_enemy_country = {
				OR = {
					has_government = fascism
					has_government = communism
				}
			}
			all_enemy_country = {
				NOT = { is_in_faction_with = ROOT }
			}
		}
	}

	abort = {
		OR = {
			has_war = yes
			AND = {
				NOT = { has_government = democratic }
				NOT = { has_country_flag = HIP_coalition_government }
				NOT = {
					AND = {
						has_completed_focus = HIP_rejoin_friends
						has_government = neutrality
					}
				}
			}
			has_war_with = EQS
			EQS = { has_war = no }
			# keep supporting EQS even if they flip to Solar Empire.
			# finish the fight!
			EQS = { any_enemy_country = { is_in_faction_with = ROOT } }
		}
	}

	ai_strategy = {
		type = ignore
		id = "EQS"
		value = -500
	}

	ai_strategy = {
		type = support
		id = "EQS"
		value = 500
	}

	ai_strategy = {
		type = befriend
		id = "EQS"
		value = 25
	}

	ai_strategy = {
		type = protect
		id = "EQS"
		value = 150
	}

	ai_strategy = {
		type = send_lend_lease_desire
		id = "EQS"
		value = 500
	}

	ai_strategy = {
		type = send_volunteers_desire
		id = "EQS"
		value = 500
	}
}

# This will prompt HIP to try and support countries who are fighting against extremist elements.
HIP_protect_against_extremism = { # Reverse strategy
	reversed = yes

	enable_reverse = {
		HIP = {
			has_government = democratic
			has_war = no
		}
	}

	enable = {
		NOT = { tag = HIP }
		NOT = { tag = event_target:griffonian_empire }
		NOT = { has_government = communism }
		NOT = { has_government = fascism }
		AND = {
			num_of_civilian_factories > 10
			is_subject = no
		}
		any_enemy_country = {
			NOT = { tag = HIP }
			OR = {
				has_government = communism
				has_government = fascism
				tag = event_target:griffonian_empire
			}
		}
	}

	abort = {
		OR = {
			has_war = no
			OR = {
				has_government = communism
				has_government = fascism
			}
			HIP = {
				OR = {
					NOT = {	has_government = democratic }
					has_war = yes
				}
			}
		}
	}

	ai_strategy = {
		type = ignore
		id = "HIP"
		value = -100
	}

	ai_strategy = {
		type = support
		id = "HIP"
		value = 150
	}

	ai_strategy = {
		type = send_lend_lease_desire
		id = "HIP"
		value = 50
	}

	ai_strategy = {
		type = send_volunteers_desire
		id = "HIP"
		value = 50
	}
}
