
HIP_stop_equestria_from_falling = {
	enable = {
		tag = HIP
		has_war = no
		num_divisions > 30 # we must have some divisions ourselves

		NOT = { has_government = fascism }
		NOT = { has_government = communism }
		NOT = { is_in_faction_with = CHN }
		NOT = { has_war_with = EQS }
		CHN = { 
			has_war_with = EQS
			has_government = fascism
		}
		EQS = {
			surrender_progress > 0.05
			has_capitulated = no
		}
	}

	abort = {
		OR = {
			has_war = yes
			has_war_with = EQS
			NOT = { CHN = { has_war_with = EQS } }
			is_in_faction_with = CHN
			has_government = fascism
			EQS = { has_war = no }
			EQS = { surrender_progress < 0.01 }
		}
	}

	ai_strategy = {
		type = support
		id = "EQS"
		value = 100
	}

	ai_strategy = {
		type = role_ratio
		id = infantry
		value = -75
	}

	ai_strategy = {
		type = template_prio
		id = light_armor
		value = -25
	}

	ai_strategy = {
		type = template_prio
		id = medium_armor
		value = -25
	}

	ai_strategy = {
		type = template_prio
		id = heavy_armor
		value = -25
	}
}

# This will prompt HIP to try and support countries who are fighting against extremist elements.
HIP_protect_against_extremism = { # Reverse strategy
	reversed = yes

	enable_reverse = {
		HIP = {
			has_government = democratic
		}
	}

	enable = {
		NOT = { tag = HIP }
		NOT = { tag = event_target:griffonian_empire }
		NOT = { has_government = communism }
		NOT = { has_government = fascism }
		AND = {
			num_of_civilian_factories > 10
			is_subject = no
		}
		any_of_scopes = {
			array = enemies
			NOT = { tag = HIP }
			OR = {
				has_government = communism
				has_government = fascism
				tag = event_target:griffonian_empire
			}
		}
	}

	abort = {
		OR = {
			has_war = no
			OR = {
				has_government = communism
				has_government = fascism
			}
			HIP = {
				OR = {
					NOT = {	has_government = democratic }
					has_war_with = ROOT
				}
			}
		}
	}

	ai_strategy = {
		type = support
		id = "HIP"
		value = 100
	}
}
