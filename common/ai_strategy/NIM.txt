		NOT = { tag = NIM }
		NIM = { 
			is_river_fed_member = yes
			NOT = { has_government = fascism }
		}
		NOT = { has_government = democratic }
		OR = {
			tag = LNS
			tag = HLQ
			tag = BAR
			tag = ZES
			tag = OLV
			original_tag = GRY

				original_tag = FIR
				has_government = fascism
			}
			AND = {
				any_neighbor_country = {
					is_river_fed_member = yes
					OR = {
						ROOT = { is_justifying_wargoal_against = PREV }
