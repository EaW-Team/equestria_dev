EQS_historical = {
	name = "Equestria historical plan"
	desc = ""

	allowed = {
		original_tag = EQS

	}

	enable = {

		OR = {
			AND = {
				is_historical_focus_on = yes
				has_game_rule = {
					rule = EQS_ai_behavior
					option = DEFAULT
				}
			}
			has_country_flag = EQS_AI_DEMOCRATIC_HISTORICAL 
			has_game_rule = {
				rule = EQS_ai_behavior
				option = DEMOCRATIC_HISTORICAL
			}
		}
	}
	abort = {
		OR = {
			has_war = yes
			CHN = {
				OR = {
					is_justifying_wargoal_against = EQS
					has_wargoal_against = EQS
					has_war_with = EQS
					AND = {
						CRY = { is_subject_of = EQS }
						OR = {
							has_wargoal_against = CRY
							is_justifying_wargoal_against = CRY
							has_war_with = CRY
						}
					}
				}
			NOT = {
				focus_progress = {
				    focus = demand_equestria
				    progress > 0
					}
				}
			}
		}
	}
	
	ai_national_focuses = {
		Friendship_Games
		examine_the_situation
		Ministry_of_defense
		worrisome_changelings
		unicorn_academy_f
		arms_potential
		need_to_reinovate
		mil_industry_fillydelphia
		luna_nova_academy
		Spitfire_research
		Fund_public_schools
		CTL_center_of_research
		Magcial_studies
		magical_training
		Matilda_research
		Extreme_war_comittee
		Hire_Generals
		Advisors
		Extreme_measures
		Forget_magic_of_friendship
		War_Plan
		Remove_Harmonic_Stagnation
		Suppress_bureaucracy
	}
	focus_factors = {

	}

	research = {
		ai_basic_start = 75.0
		ai_continued_start = 60.0
		land_doctrine = 50.0
		light_fighter = 40.0
	}

	ideas = {
		twilight_sparkle = 50.0
		cocopommel = 40.0
		rainbow_factory = 30.0
		flim = 20.0
		sweetie_drops = 0
	}

			factor = 1.0
		}
	}
