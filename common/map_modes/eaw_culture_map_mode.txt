scripted_map_modes = {
	eaw_culture_map_mode = {
		#Day/night override stuff
		top = {
			type = country
			color = {
				if = {
					limit = {
						FROM = {
							tag = ZZZ
						}
					}
					if = {
						limit = {
							has_global_flag = EaW_night_override
						}
						set_temp_variable = { red   = 0.006 }
					}
					else_if = {
						limit = {
							has_global_flag = EaW_day_override
						}
						set_temp_variable = { red   = 0.008 }
					}
					else = {
						set_temp_variable = { red   = 0 }
					}
					set_temp_variable = { green = 0 }
					set_temp_variable = { blue  = 0 }
					set_temp_variable = { alpha = 1 }
				}
				else = {
					set_temp_variable = { red   = 0 }
					set_temp_variable = { green = 0 }
					set_temp_variable = { blue  = 0 }
					set_temp_variable = { alpha = 0 }
				}
			}
		}
		bottom = {
			type = state
			color = {
				FROM = {
					set_temp_variable = { red   = global.map_culture_color_red^culture_map_mode_value }
					set_temp_variable = { green = global.map_culture_color_green^culture_map_mode_value }
					set_temp_variable = { blue  = global.map_culture_color_blue^culture_map_mode_value }
				}
				set_temp_variable = { alpha = 0.7 }
			}
		}
		far_text = none
		near_text = state
	}
}
