non_capitulated_majors = {
	input = game:all_countries
	operators = {
		limit = {
			is_major = yes
			has_capitulated = no
		}
	}
	name = COLLECTION_NON_CAPITULATED_MAJORS
}

non_capitulated_majors_with_smaller_navy_than_HIP = {
	input = collection:non_capitulated_majors
	operators = {
		limit = {
			naval_strength_comparison = {
				other = HIP
				ratio < 1.0
				sub_unit_def_weights = { # carriers REALLY scare us, and we don't really care about small ships
					carrier = 1.45
					battleship = 1.15
					battle_cruiser = 1
					heavy_cruiser = 0.85
					light_cruiser = 0.7
					destroyer = 0.55
					submarine = 0.3
				}
			}
		}
	}
	name = COLLECTION_NON_CAPITULATED_MAJORS_WITH_SMALLER_NAVY_THAN_HIP
}
