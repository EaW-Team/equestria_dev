
# River Coalition related
riverlands_controllable_states = {
	input = constant:state_groups.riverlands_states
	name = COLLECTION_RIVERLANDS_STATES

}

riverlands_controlled_states = {
	input = collection:riverlands_controllable_states
	name = COLLECTION_CONTROLLED_RIVERLANDS_STATES
	operators = {
		limit = {
			is_controlled_by_ROOT_or_ally = yes 
		}
	}
}

greater_equestrian_cores = {
	input = collection:states_in_europe
	name = COLLECTION_GREATER_EQUESTRIAN_CORES
	operators = {
		limit = {
			OR = {
				is_core_of = EQS
				is_core_of = CRY
				is_in_array = {
					state_original_cores = EQS
				}
				is_in_array = {
					state_original_cores = CRY
				}
			}
		}
	}
}

former_herzland_territories = {
	input = collection:states_in_asia
	name = COLLECTION_FORMER_HERZLAND_STATES

				is_state_main_culture_evi_valley = yes

				is_state_main_culture_southerner = yes

	}
}

non_capitulated_majors = {
	input = game:all_countries
	operators = {
		limit = {
			is_major = yes
			has_capitulated = no
		}
	}
	name = COLLECTION_NON_CAPITULATED_MAJORS
}

non_capitulated_majors_with_smaller_navy_than_HIP = {
	input = collection:non_capitulated_majors
	operators = {
		limit = {
			naval_strength_comparison = {
				other = HIP
				ratio < 1.0
				sub_unit_def_weights = { # carriers REALLY scare us, and we don't really care about small ships
					carrier = 1.45
					battleship = 1.15
					battle_cruiser = 1
					heavy_cruiser = 0.85
					light_cruiser = 0.7
					destroyer = 0.55
					submarine = 0.3
				}
			}
		}
	}
	name = COLLECTION_NON_CAPITULATED_MAJORS_WITH_SMALLER_NAVY_THAN_HIP
}
