autonomy_state = {
	id = autonomy_integrated_puppet

	is_puppet = yes

	min_freedom_level = 0.0

	rule = {
		can_not_declare_war = yes
		can_decline_call_to_war = no
		units_deployed_to_overlord = yes
	}

	modifier = {
		autonomy_manpower_share = 1.0
		can_master_build_for_us = 1
		extra_trade_to_overlord_factor = 1.0
		overlord_trade_cost_factor = -0.9
		cic_to_overlord_factor = 0.25
		mic_to_overlord_factor = 0.75

		research_sharing_per_country_bonus_factor = -0.3
	}

	ai_subject_wants_higher = {
		factor = 1.0
	}

	ai_overlord_wants_lower = {
		factor = 0.0
	}

	ai_overlord_wants_garrison = {
		always = yes
	}

	allowed = {
		has_dlc = "Together for Victory"

		if = {
			limit = { has_dlc = "Death or Dishonor" }
			overlord = { NOT = { has_government = fascism } }
		}
		OR = {
			AND = {
				is_griffon_race = no
				is_griffon_continent_country = no
				overlord = {
					is_griffon_race = no
					is_griffon_continent_country = no
				}
			}
			has_country_flag = be_regular_puppet
			overlord = {
				has_country_flag = have_regular_puppets
			}
		}
	}

	can_take_level = {
		if = {
			limit = {
				original_tag = CRY
				overlord = {
						original_tag = SOL
				}
			}
			NOT = { has_country_flag = SOL_CRY_force_protectorate }
		}
	}

	can_lose_level = {
		if = {
			limit = {
				original_tag = CRY
				overlord = {
					OR = {
						original_tag = EQS
						original_tag = SOL
					}
				}
			}
			always = no
		}
	}
}
