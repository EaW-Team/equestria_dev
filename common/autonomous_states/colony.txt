autonomy_state = {
	id = autonomy_colony

	is_puppet = yes

	min_freedom_level = 0.50

	manpower_influence = 0.7
	
	rule = {
		can_not_declare_war = yes
		can_decline_call_to_war = no
		can_be_spymaster = no
		contributes_operatives = no
		can_create_collaboration_government = no
	}

	modifier = {
		autonomy_manpower_share = 0.7

		extra_trade_to_overlord_factor = 0.5
		overlord_trade_cost_factor = -0.5
		can_master_build_for_us = 1
		license_subject_master_purchase_cost = -0.5
		research_sharing_per_country_bonus_factor = -0.3
	}

	ai_subject_wants_higher = {
		factor = 1.0
	}

	ai_overlord_wants_lower = {
		factor = 1.0
	}

	ai_overlord_wants_garrison = {
		always = no
	}

	allowed = {
		has_dlc = "Together for Victory"

		autonomy_states_default = yes

		if = {
			limit = {
				has_dlc = "Together for Victory"
				tag = FAT
			}
			NOT = { has_idea = FAT_griffonian_alliance_idea }
		}
	}

	can_take_level = {
		if = {
			limit = {
				original_tag = CRY
				overlord = { tag = event_target:equestria }
			}
			NOT = { has_completed_focus = CRY_subordination }
		}
		if = {
			limit = {
				original_tag = CRY
			}
			NOT = {
				has_country_flag = CRY_solar_governorship
				has_country_flag = CRY_lunar_governorship
			}
		}
		has_civil_war = no
	}


	can_lose_level = {
		overlord = {
			has_civil_war = no
		}
	}
}
