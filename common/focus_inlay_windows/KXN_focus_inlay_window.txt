KXN_DOCTRINARCH_Focus_Inlay = {
	window_name = KXN_DOCTRINARCH_Focus_Window
	internal = no
	visible = {
		NOT = { has_global_flag = KXN_The_Shattering_Happened }
	}
	scripted_buttons = {
        KXN_Complete_Focus_DOCTRINARCH_1A = {
			available = {
				has_completed_focus = KXN_Focus_DOCTRINARCH_1
				NOT = { has_completed_focus = KXN_Focus_DOCTRINARCH_1A }
				focus_progress = { focus = KXN_Focus_DOCTRINARCH_1A progress < 0 }
				custom_trigger_tooltip = {
					tooltip = KXN_Focus_DOCTRINARCH_Ongoing_Flag_TT
					NOT = { has_active_mission = KXN_Complete_Focus_DOCTRINARCH_Mission }
				}
			}
			click_effect = {
				if = {
					limit = {
						has_active_mission = KXN_Complete_Focus_DOCTRINARCH_Mission
						NOT = { has_completed_focus = KXN_Focus_DOCTRINARCH_1A }
						has_shine_effect_on_focus = KXN_Focus_DOCTRINARCH_1A
					}
					custom_effect_tooltip = KXN_Complete_Focus_DOCTRINARCH_Mission_Ongoing_TT
				}
				else_if = {
					limit = {
						NOT = { has_completed_focus = KXN_Focus_DOCTRINARCH_1A }
					}
					set_temp_variable = { GetKXNDoctrinarchTier = 1 }
					set_temp_variable = { KXN_Complete_Focus_DOCTRINARCH_Mission_Var = 1 }
					set_variable = { KXN_Complete_Focus_DOCTRINARCH_Mission_Var = 1 }
					set_variable = { KXN_Complete_Focus_DOCTRINARCH_Mission_Time = KXN_DOCTRINARCH_Title_2_Cost }
					set_temp_variable = { KXN_Complete_Focus_DOCTRINARCH_Mission_Time = KXN_DOCTRINARCH_Title_2_Cost }
					activate_mission = KXN_Complete_Focus_DOCTRINARCH_Mission
					custom_effect_tooltip = KXN_Complete_Focus_DOCTRINARCH_Mission_TT
					hidden_effect = { activate_shine_on_focus = KXN_Focus_DOCTRINARCH_1A }
				}
			}
        }
        KXN_Complete_Focus_DOCTRINARCH_1B = {

					tooltip = KXN_Focus_DOCTRINARCH_Ongoing_Flag_TT
					NOT = { has_active_mission = KXN_Complete_Focus_DOCTRINARCH_Mission }
				}
			}
			click_effect = {
				if = {
					limit = {
						has_active_mission = KXN_Complete_Focus_DOCTRINARCH_Mission
						NOT = { has_completed_focus = KXN_Focus_DOCTRINARCH_1B }
						has_shine_effect_on_focus = KXN_Focus_DOCTRINARCH_1B
					}
					custom_effect_tooltip = KXN_Complete_Focus_DOCTRINARCH_Mission_Ongoing_TT
				}
				else_if = {
					limit = {
						NOT = { has_completed_focus = KXN_Focus_DOCTRINARCH_1B }
					}
					set_temp_variable = { GetKXNDoctrinarchTier = 2 }
					set_temp_variable = { KXN_Complete_Focus_DOCTRINARCH_Mission_Var = 2 }
					set_variable = { KXN_Complete_Focus_DOCTRINARCH_Mission_Var = 2 }
					set_variable = { KXN_Complete_Focus_DOCTRINARCH_Mission_Time = KXN_DOCTRINARCH_Title_3_Cost }
					set_temp_variable = { KXN_Complete_Focus_DOCTRINARCH_Mission_Time = KXN_DOCTRINARCH_Title_3_Cost }

        KXN_Complete_Focus_DOCTRINARCH_1C = {
			available = {
				has_completed_focus = KXN_Focus_DOCTRINARCH_1B
				NOT = { has_completed_focus = KXN_Focus_DOCTRINARCH_1C }
				focus_progress = { focus = KXN_Focus_DOCTRINARCH_1C progress < 0 }

					tooltip = KXN_Focus_DOCTRINARCH_Ongoing_Flag_TT

				}
				has_character = KXN_Montane_Singe
				KXN_Montane_Singe_Alternate = {
					is_country_leader = yes
				}
			}
			click_effect = {
				if = {
