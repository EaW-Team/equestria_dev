DEP_vartai_is_friendly = {
	custom_trigger_tooltip = {
		tooltip = DEP_vartai_is_friendly
		OR = {
			AND = {
				LNS = {
					OR = {
						exists = no
						is_subject_of = HLQ
						is_in_faction_with = HLQ
					}
				}
				LSM = {
					OR = {
						exists = no
						is_subject_of = HLQ
						is_in_faction_with = HLQ
					}
				}
				LSC = {
					OR = {
						exists = no
						is_subject_of = HLQ
						is_in_faction_with = HLQ
					}
				}
				HLQ = {
					exists = yes
					OR = {
						AND = {
							OR = {
								has_government = democratic
								has_government = neutrality
							}
							OR = {
								has_completed_focus = HLQ_duchy
								has_completed_focus = HLQ_the_ponies
								has_completed_focus = HLQ_republic
								has_completed_focus = HLQ_the_bund
								has_completed_focus = HLQ_prince_elias
							}
						}
						AND = {
							has_government = communism
							has_completed_focus = HLQ_the_revolution_comes_for_all
						}
					}
				}
			}
			AND = {
				HLQ = {
					OR = {
						exists = no
						is_subject_of = LNS
						is_in_faction_with = LNS
						is_subject_of = LSM
						is_in_faction_with = LSM
						is_subject_of = LSC
						is_in_faction_with = LSC
					}
				}
				LSC = {
					exists = yes
					OR = {
						AND = {
							has_completed_focus = LSC_agrarian_revolution
							has_country_leader = {
								character = LSC_starry_night
								ruling_only = yes
							}
						}
						AND = {
							has_completed_focus = LSC_national_restoration
							has_country_leader = {
								character = LSC_flawless_justice
								ruling_only = yes
							}
						}
						AND = {
							has_completed_focus = LSC_divine_revolution
							has_country_leader = {
								character = LSC_abba_povner
								ruling_only = yes
							}
						}
					}
				}
			}
		}
	}
}
