MER_supremacy_plurality = {
    check_variable = { 
        var = MER_temp_supremacy_supp
        value =  MER_temp_communal_supp
        compare = greater_than_or_equals
    }
    check_variable = { 
        var = MER_temp_supremacy_supp
        value =  MER_temp_harmony_supp
        compare = greater_than_or_equals
    }
}

MER_communal_plurality = {
    check_variable = {  MER_temp_communal_supp > MER_temp_supremacy_supp }
    check_variable = { 
        var = MER_temp_communal_supp
        value =  MER_temp_harmony_supp
        compare = greater_than_or_equals
    }
}
