    check_variable = { 
        var = MER_temp_supremacy_supp

        compare = greater_than_or_equals
    }
    check_variable = { 
        var = MER_temp_supremacy_supp
        value =  MER_temp_harmony_supp
        compare = greater_than_or_equals
    }
}

        compare = greater_than_or_equals
    }
}
