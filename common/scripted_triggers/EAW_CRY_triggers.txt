
# Nations that are eligible to have and be interacted with through the joint Crystal Empire focus tree
CRY_is_potential_CRY_friend = {
	OR = {
		original_tag = CRY
		original_tag = EQS
		original_tag = STG
	}
}
CRY_joint_trigger = {
	hidden_trigger = {
		CRY_is_potential_CRY_friend = yes
	}
	OR = {
		original_tag = CRY
		is_in_faction_with = CRY
		is_subject_of = CRY
		has_subject = CRY
	}
}
CRY_monsoc_is_more_stalliongradian = {
	custom_trigger_tooltip = {
		tooltip = CRY_more_STG_influence
		check_variable = {
			STG.stg_crossroads_of_destiny_influence > EQS.eqs_crossroads_of_destiny_influence
		}
	}
}
CRY_monsoc_is_more_equestrian = {
	custom_trigger_tooltip = {
		tooltip = CRY_more_EQS_influence
		check_variable = {
			EQS.eqs_crossroads_of_destiny_influence > STG.stg_crossroads_of_destiny_influence
		}
	}
}
