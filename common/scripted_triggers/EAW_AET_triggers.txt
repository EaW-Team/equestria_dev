AET_HLDV_cost_trigger = {
	if = {
		limit = {
			check_variable = { var = j value = 3 compare = greater_than_or_equals }
			check_variable = { var = j value = 8 compare = less_than_or_equals }
		}
		set_temp_variable = { temp_pp_cost = 75 }
		set_temp_variable = { temp_pop_cost = 100 }
		any_controlled_state = {
			state_population > 100
		}
	}
	else_if = {
		limit = {
			check_variable = { j = 1 }
		}
		set_temp_variable = { temp_pp_cost = 25 }
	}
	else_if = {
		limit = {
			check_variable = { j = 2 }
		}
		set_temp_variable = { temp_pp_cost = 50 }
	}
	else_if = {
		limit = {
			check_variable = { j = 9 }
		}
		set_temp_variable = { temp_pp_cost = 60 }
	}
	else_if = {
		limit = {
			check_variable = { j = 10 }
		}
		set_temp_variable = { temp_pp_cost = 10 }
	}
	else_if = {
		limit = {
			check_variable = { j = 11 }
		}
		set_temp_variable = { temp_pp_cost = 60 }
	}
	else_if = {
		limit = {
			check_variable = { j = 12 }
		}
		set_temp_variable = { temp_pp_cost = 10 }
	}
	has_political_power > temp_pp_cost
}
