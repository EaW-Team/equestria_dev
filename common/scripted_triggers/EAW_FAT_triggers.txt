#assumed to be JER, but this should account for other possible Aquileian unifiers
FAT_aquileia_is_pangriffonian = {
	custom_trigger_tooltip = {
		tooltip = FAT_aquileia_is_pangriffonian_tt
		JED = {
			OR = {
				has_country_leader = { ruling_only = yes character = JER_theodore_verany }
			}
		}
	}
}

FAT_aquileia_is_amicable = {
	custom_trigger_tooltip = {
		tooltip = FAT_aquileia_is_amicable_tt
		JED = {
			OR = {
				#JER
				has_country_leader = { ruling_only = yes character = JER_cecile_gaudreau }
				has_country_leader = { ruling_only = yes character = JER_leonard_rodier }
				has_country_leader = { ruling_only = yes character = JER_grand_cru }
				#HEL
				has_country_leader = { ruling_only = yes character = HEL_alexandra_descieux }
				has_country_leader = { ruling_only = yes character = HEL_jaume_de_conard }
				has_country_leader = { ruling_only = yes character = HEL_even_baseder_council }
				#FLO
				has_country_leader = { ruling_only = yes character = FLO_anna_de_indigo }
			}
		}
	}
}

FAW_has_restored_monarchy = {
	FAW = {
		OR = {
			has_completed_focus = FST_the_unthinkable_option
			has_cosmetic_tag = FAT_Kingdom
			has_cosmetic_tag = FAW_UK
		}
	}
}
