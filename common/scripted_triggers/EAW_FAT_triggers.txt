			}
		}
	}
}

FAT_no_equestria = {
	custom_trigger_tooltip = {
		tooltip = FAT_no_equestria_tt
		OR = {
			has_country_flag = FAT_focus_debug_eqs_nonexistent
			EQ1 = {
				NOT = { original_tag = EQS }
				NOT = { original_tag = NLR }
			}
		}
	}
}

FAT_equestria_not_harmonic = {
	custom_trigger_tooltip = {
		tooltip = FAT_equestria_not_harmonic_tt
		OR = {
			has_country_flag = FAT_focus_debug_eqs_nonharmonic
			EQ1 = {
				OR = {
					AND = {
						original_tag = EQS
						is_subject = no
						has_global_flag = daybreaker_spawned
					}
					original_tag = NLR
				}
			}
		}
	}
}

FAT_equestria_stands_harmonic = {
	custom_trigger_tooltip = {
		tooltip = FAT_equestria_stands_harmonic_tt

		NOT = { has_country_flag = FAT_focus_debug_eqs_nonexistent }

			has_country_flag = FAT_focus_debug_eqs_harmonic

			}
		}
	}
}
