is_friend_or_potential_friend_of_EQC = {
	OR = {
		is_subject_of = EQC
		is_subject = no
	}
	OR = {
		is_in_faction_with = EQC
		is_in_faction = no
	}
	NOT = { has_war_with = EQC }
	has_capitulated = no
	exists = yes
}

is_alone_or_in_faction_at_peace_with_EQC = {
	OR = {
		is_in_faction = no
		is_in_faction_with = EQC
		AND = {
			is_in_faction = yes
			NOT = {
				is_in_faction_with = EQC
			}
		}
	}
	NOT = {
		any_other_country = {
			is_in_faction_with = PREV
			has_war_with = EQC
		}
	}
}
