BAR_expedition_can_be_launched = {
	AND = {
		OR = {
			check_variable = {
				bar_expedition_members^0 = 2
			}
			check_variable = {
				bar_expedition_members^1 = 2
			}
			check_variable = {
				bar_expedition_members^2 = 2
			}
			check_variable = {
				bar_expedition_members^3 = 2
			}
			check_variable = {
				bar_expedition_members^4 = 2
			}
		}
		has_variable = bar_expedition_target
		has_equipment = {
			infantry_equipment > bar_expedition_additional_equipment
		}		
		OR = {
			check_variable = {
				bar_expedition_additional_equipment_trucks = 0
			}
			has_equipment = {
				motorized_equipment > bar_expedition_additional_equipment_trucks
			}
		}		
		OR = {
			check_variable = {
				bar_expedition_additional_equipment_planes = 0
			}
			has_equipment = {
				motorized_equipment > bar_expedition_additional_equipment_planes
			}
		}
		NOT = {
			has_country_flag = bar_expedition_launched
		}
	}
}

bar_activities_can_be_launched = {
	AND = {
		OR = {
			check_variable = {
				bar_friendship_activity_members^0 = 2
			}
			check_variable = {
				bar_friendship_activity_members^1 = 2
			}
			check_variable = {
				bar_friendship_activity_members^2 = 2
			}
		}
		NOT = {
			check_variable = {
				bar_friendship_activity_type = 0
			}
		}
		NOT = {
			has_country_flag = bar_ongoing_activity
		}
		has_political_power > bar_friendship_activity_activities_cost^bar_friendship_activity_type
	}
}
