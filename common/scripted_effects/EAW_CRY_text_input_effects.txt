eaw_CRY_open_text_input = {
    if = {
        limit = { NOT = { has_variable = eaw_CRY_text_input_open } }
        clear_array = chars
        clear_variable = current_char
        set_variable = { eaw_CRY_text_input_open = 1 }
        if = {
            limit = { NOT = { has_variable = ROOT.rename_language} }
            set_variable = { ROOT.rename_language = 1 }
        }
    }
}

eaw_CRY_copy_char_array = {
    clear_array = cry_oc_name_chars
    for_each_loop = {
        array = chars
        value = v
        index = i
        add_to_array = {
            cry_oc_name_chars = v
        }
    }
}
