		limit = {

	}

	custom_effect_tooltip = BLT_boltrus_gain_trait_ruthless_tt
	hidden_effect = {
		set_variable = { BLT_king_boltrus_trait_ruthless = 2 }
		BLT_boltrus_lose_trait_gallant = yes
		if = {
			limit = {
				BLT_king_boltrus = { has_trait = bleeding_heart }
			}
			remove_trait = {
				character = BLT_king_boltrus
				trait = bleeding_heart
			}
		}
		if = {
			limit = { BLT_boltrus_outward_level_4 = yes }
			BLT_change_outward_stance_to_2 = yes
		}
		else_if = {
			limit = { BLT_boltrus_outward_level_3 = yes }
			BLT_change_outward_stance_to_1 = yes
		}
		else = { BLT_change_outward_stance_to_0 = yes }

BLT_boltrus_gain_motivation_1 = {
	if = {
		limit = {
			check_variable = { BLT_king_boltrus_inward_value < 5 }
			check_variable = { BLT_king_boltrus_inward_value > 0 }
		}
		custom_effect_tooltip = BLT_boltrus_lose_motivation_1_tt
		add_to_variable = { BLT_king_boltrus_inward_value = 1 }
	}
	else_if = {#sigma mode; nothing happens
		limit = {
			check_variable = { BLT_king_boltrus_inward_value = -1 }
		}
	}
	else_if = {#blackpilled; nothing happens
		limit = {
			check_variable = { BLT_king_boltrus_inward_value = 0 }
		}
	}
	else = {#motivation is 5
		custom_effect_tooltip = BLT_boltrus_lose_motivation_1_tt
		add_stability = 0.05
		add_war_support = 0.05
	}
}

BLT_boltrus_lose_motivation_1 = {
	if = {
		limit = {
			check_variable = { BLT_king_boltrus_inward_value > 1 }
		}
		custom_effect_tooltip = BLT_boltrus_lose_motivation_1_tt
		add_to_variable = { BLT_king_boltrus_inward_value = -1 }
	}
	else_if = {#sigma mode; nothing happens
		limit = {
			check_variable = { BLT_king_boltrus_inward_value = -1 }
		}
	}
	else_if = {#blackpilled; nothing happens
		limit = {
			check_variable = { BLT_king_boltrus_inward_value = 0 }
		}
	}
	else = {#motivation is 1
		custom_effect_tooltip = BLT_boltrus_lose_motivation_1_tt
		add_stability = -0.05
		add_war_support = -0.05
	}
}
