BLT_king_boltrus_tab_init_zero = {
	hidden_effect = {
		if = {
			limit = {
				NOT = { has_country_flag = BLT_king_boltrus_tab_unlocked }
			}
			set_country_flag = BLT_king_boltrus_tab_unlocked
		}
		set_variable = { BLT_king_boltrus_outward_value = 0 }
		set_variable = { BLT_king_boltrus_inward_value = 0 }
		set_variable = { BLT_king_boltrus_renown_value = 0 }
		#TRAITS
		set_variable = { BLT_king_boltrus_trait_explorer = 1 }
		set_variable = { BLT_king_boltrus_trait_ruthless = 1 }
		set_variable = { BLT_king_boltrus_trait_conniving = 1 }
		set_variable = { BLT_king_boltrus_trait_gallant = 1 }
		set_variable = { BLT_king_boltrus_trait_patient = 1 }
		set_variable = { BLT_king_boltrus_trait_austere = 1 }
		set_variable = { BLT_king_boltrus_trait_diligent = 1 }
		set_variable = { BLT_king_boltrus_trait_hardened = 1 }
		set_variable = { BLT_king_boltrus_trait_brave = 1 }
		set_variable = { BLT_king_boltrus_trait_tactical = 1 }
		set_variable = { BLT_king_boltrus_trait_erudite = 1 }
		set_variable = { BLT_king_boltrus_trait_persuasive = 1 }
	}
}

BLT_king_boltrus_tab_init = {
	custom_effect_tooltip = BLT_king_boltrus_tab_unlocked_tt
	hidden_effect = {
		if = {
			limit = {
				NOT = { has_country_flag = BLT_king_boltrus_tab_unlocked }
			}
			set_country_flag = BLT_king_boltrus_tab_unlocked
		}
		set_variable = { BLT_king_boltrus_outward_value = 3 }
		set_variable = { BLT_king_boltrus_inward_value = 4 }
		set_variable = { BLT_king_boltrus_renown_value = 0 }
		#TRAITS
		set_variable = { BLT_king_boltrus_trait_explorer = 2 }
		set_variable = { BLT_king_boltrus_trait_ruthless = 1 }
		set_variable = { BLT_king_boltrus_trait_conniving = 1 }
		set_variable = { BLT_king_boltrus_trait_gallant = 1 }
		set_variable = { BLT_king_boltrus_trait_patient = 1 }
		set_variable = { BLT_king_boltrus_trait_austere = 1 }
		set_variable = { BLT_king_boltrus_trait_diligent = 1 }
		set_variable = { BLT_king_boltrus_trait_hardened = 1 }
		set_variable = { BLT_king_boltrus_trait_brave = 1 }
		set_variable = { BLT_king_boltrus_trait_tactical = 1 }
		set_variable = { BLT_king_boltrus_trait_erudite = 1 }
		set_variable = { BLT_king_boltrus_trait_persuasive = 1 }
	}
}

BLT_update_trait_gui = {
	add_to_variable = { BLT_traits_gui_update_value = 1 }
}

BLT_boltrus_gain_trait_tactical = {
	custom_effect_tooltip = BLT_boltrus_gain_trait_tactical_tt
	hidden_effect = {
		set_variable = { BLT_king_boltrus_trait_tactical = 2 }
		BLT_update_trait_gui = yes

		if = {
			limit = {
				NOT = {
					BLT_king_boltrus = { has_trait = brilliant_strategist }
				}
			}
			BLT_king_boltrus = {
				add_trait = { trait = brilliant_strategist }
			}
		}
	}
}
