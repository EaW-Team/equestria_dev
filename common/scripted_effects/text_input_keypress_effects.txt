keypress_init = {
    if = {
        limit = { NOT = { has_variable = current_char } }
        set_variable = { current_char = -1 }
        resize_array = {
            array = chars
            size = 32
        }
    }
    add_to_variable = { current_char = 1 }
}

get_keypress_backspace = {
    set_variable = { chars^current_char = 0 }
    subtract_from_variable = { current_char = 1 }
    clamp_variable = {
        var = current_char
        min = 0
        max = 32
    }
}
