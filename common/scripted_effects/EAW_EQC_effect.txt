EQC_update_parliament_chart = {
	set_temp_variable = { temp_frame_var = 100 }
	set_temp_variable = { temp_prev_popularity = 0 }
	
	for_each_loop = {
		array = EQC_party_token_array
		value = v
		index = i
		
		subtract_from_temp_variable = { temp_frame_var = temp_prev_popularity }
		set_variable = { EQC_party_popularity_frame@var:v = temp_frame_var }
		set_temp_variable = { temp_prev_popularity = EQC_party_popularity@var:v }
	}
	
	add_to_variable = { EQC_parliament_dirty = 0.01 }
}

EQC_calculate_total_relations = {
	for_each_loop = {
		array = EQC_party_token_array
		value = v
		index = i
		
		set_variable = { EQC_party_opinion@var:v = 0 }
		for_each_loop = {
			array = EQC_party_opinion_breakdown@var:v
			value = w
			index = j
			
			add_to_variable = { EQC_party_opinion@var:v = w }
		}
	}
	
	add_to_variable = { EQC_parliament_dirty = 0.01 }
}

#Use variable temp_target to select the party to break coalition with.
EQC_break_coalition_with_temp_target = {		
	if = {
		limit = {
			is_in_array = {
				array = EQC_coalition_partners
				value = temp_target
			}
		}
		for_each_loop = {
			array = EQC_party_opinion_loc@var:temp_target
			value = v
			index = i
			break = b
				
			if = {
				limit = {
					check_variable = { v = token:in_coalition_modifier }
				}
				remove_from_array = { 
					array = EQC_party_opinion_breakdown@var:temp_target
					index = i 
				}
				remove_from_array = { 
					array = EQC_party_opinion_loc@var:temp_target
					index = i
				}
				set_temp_variable = { b = 1 }
			}
		}
		
		add_to_array = { EQC_party_opinion_breakdown@var:temp_target = -25 }
		add_to_array = { EQC_party_opinion_loc@var:temp_target = token:broke_coalition_modifier }
		EQC_calculate_total_relations = yes
		
		set_variable = { EQC_party_coalition_btn_frame@var:temp_target = 1 }
		
		remove_from_array = { EQC_coalition_partners = temp_target }
		EQC_calculate_coalition_size = yes
		EQC_calculate_coalition_size_projected = yes
	}
}

#Set temp_amount for amount
#Set temp_target for the target
EQC_add_rep_for_party = {
	add_to_variable = { EQC_party_popularity@var:temp_target = temp_amount }
	
	set_temp_variable = { total_value_check = 100 }
	subtract_from_temp_variable = { total_value_check = EQC_party_popularity@var:temp_target }
	
	set_temp_variable = { total = 0 }
	for_each_loop = {
		array = EQC_party_token_array
		index = i
		value = v
		
		if = {
			limit = {
				NOT = { check_variable = { v = temp_target } }
			}
			add_to_temp_variable = { total = EQC_party_popularity@var:v }
		}
	}
	
	divide_temp_variable = { total = total_value_check }
	
	for_each_loop = {
		array = EQC_party_token_array
		index = i
		value = v
		
		if = {
			limit = {
				NOT = { check_variable = { v = temp_target } }
			}
			divide_variable = { EQC_party_popularity@var:v = total }
		}
	}
	
	EQC_calculate_coalition_size = yes
	EQC_update_parliament_chart = yes
}

EQC_hold_election = {
	# if MARESOC or Haygleians are powerful than shenanigans
	if = {
		limit = {
			OR = {
				check_variable = {
					EQC_party_projected_popularity@token:maresoc_party > 20
				}
				check_variable = {
					EQC_party_projected_popularity@token:duke_haygle_party > 20
				}
			}
		}
		set_temp_variable = {
			EQC_maresoc_coup_level = EQC_party_projected_popularity@token:maresoc_party
		}
		subtract_from_temp_variable = {
			EQC_maresoc_coup_level = EQC_party_projected_popularity@token:duke_haygle_party
		}
		if = {
			limit = {
				check_variable = {
					EQC_maresoc_coup_level > 0
				}
			}	
			# maresoc path
			country_event = mareland.29
		}
		else = {
			# haygleians path
			# their popularity is doubled
			set_temp_variable = { temp_amount = EQC_party_projected_popularity@token:duke_haygle_party }
			set_temp_variable = { temp_target = token:duke_haygle_party }
			EQC_add_support_for_party = yes
			for_each_loop = {
				array = EQC_party_token_array
				value = v
				index = i
				
				set_variable = { EQC_party_popularity@var:v = EQC_party_projected_popularity@var:v }
			}
			country_event = mareland_new.5
		}
	}
	else = { # otherwise elections are free and fair
		for_each_loop = {
			array = EQC_party_token_array
			value = v
			index = i
			
			set_variable = { EQC_party_popularity@var:v = EQC_party_projected_popularity@var:v }
		}
		# choosing who you play as
		if = {
			limit = {
				# if the Haygleians didn't make the 20% threshold they will not be able to secure absolute power
				has_completed_focus = EQC_the_government_of_national_salvation
			}
			EQC_become_picket = yes # even if the NMPP got more votes Picket is a very loud voice
		}
		else_if = {
			limit = {
				has_completed_focus = EQC_no_deals_with_extremists
				check_variable = {
					EQC_party_popularity@token:kingfisher_party > 50
				}
			}
			EQC_become_kingfisher = yes
		}
		else_if = {
			limit = {
				has_completed_focus = EQC_no_deals_with_extremists
			}
			EQC_become_jet_set = yes # you will probably already be him but this is being careful
		}
		else_if = {
			limit = {
				has_completed_focus = EQC_unite_the_left
				
				set_temp_variable = {
					EQC_blueberry_level = EQC_party_popularity@token:blueberry_frost_party
				}
				subtract_from_temp_variable = {
					EQC_blueberry_level = EQC_party_popularity@token:lufty_star_party
				}
				check_variable = {
					EQC_blueberry_level < 0
				}
			}
			EQC_become_lufty_star = yes
		}
		else = {
			EQC_become_blueberry = yes
		}
	}
