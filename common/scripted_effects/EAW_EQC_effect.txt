EQC_initialize_parliament = {
	
	set_country_flag = EQC_parliament_active
	
	#Main token array
	add_to_array = { EQC_party_token_array = token:jet_set_party }
	add_to_array = { EQC_party_token_array = token:lufty_star_party }
	add_to_array = { EQC_party_token_array = token:blueberry_frost_party }
	add_to_array = { EQC_party_token_array = token:maresoc_party }
	add_to_array = { EQC_party_token_array = token:kingfisher_party }
	add_to_array = { EQC_party_token_array = token:duke_haygle_party }
	add_to_array = { EQC_party_token_array = token:picket_fence_party }
	
	set_variable = { EQC_party_popularity@token:jet_set_party = 28 }
	set_variable = { EQC_party_popularity@token:lufty_star_party = 18 }
	set_variable = { EQC_party_popularity@token:blueberry_frost_party = 11 }
	set_variable = { EQC_party_popularity@token:maresoc_party = 6 }
	set_variable = { EQC_party_popularity@token:kingfisher_party = 19 }
	set_variable = { EQC_party_popularity@token:duke_haygle_party = 9 }
	set_variable = { EQC_party_popularity@token:picket_fence_party = 9 }
	
	for_each_loop = {
		array = EQC_party_token_array
		value = v
		index = i
		
		#Some frame variables for the gui
		set_variable = { EQC_party_coalition_btn_frame@var:v = 1 }
		set_variable = { EQC_party_improve_relations_frame@var:v = 1 }
	}
	
	add_dynamic_modifier = { modifier = EQC_parliament_effects }
	
	EQC_update_parliament_chart = yes
}

EQC_update_parliament_chart = {
	set_temp_variable = { temp_frame_var = 100 }
	
	for_each_loop = {
		array = EQC_party_token_array
		value = v
		index = i
		
		subtract_from_temp_variable = { temp_frame_var = temp_prev_popularity }
		set_variable = { EQC_party_popularity_frame@var:v = temp_frame_var }
		set_temp_variable = { temp_prev_popularity = EQC_party_popularity@var:v }
	}
	
	add_to_variable = { EQC_parliament_dirty = 0.01 }
}

EQC_calculate_total_relations = {
	for_each_loop = {
		array = EQC_party_token_array
		value = v
		index = i
		
		set_variable = { EQC_party_opinion@var:v = 0 }
		for_each_loop = {
			array = EQC_party_opinion_breakdown@var:v
			value = w
			index = j
			
			add_to_variable = { EQC_party_opinion@var:v = w }
		}
	}
	
	add_to_variable = { EQC_parliament_dirty = 0.01 }
}

EQC_break_current_coalition = {				
	for_each_loop = {
		array = EQC_party_opinion_loc@var:EQC_coalition_partner
		value = v
		index = i
		break = b
			
		if = {
			limit = {
				check_variable = { v = token:in_coalition_modifier }
			}
			remove_from_array = { 
				array = EQC_party_opinion_breakdown@var:EQC_coalition_partner
				index = i 
			}
			remove_from_array = { 
				array = EQC_party_opinion_loc@var:EQC_coalition_partner
				index = i
			}
			set_temp_variable = { b = 1 }
		}
	}
	
	add_to_array = { EQC_party_opinion_breakdown@var:EQC_coalition_partner = -25 }
	add_to_array = { EQC_party_opinion_loc@var:EQC_coalition_partner = token:broke_coalition_modifier }
	EQC_calculate_total_relations = yes
	
	set_variable = { EQC_party_coalition_btn_frame@var:EQC_coalition_partner = 1 }
	
	clear_variable = EQC_coalition_partner
}
