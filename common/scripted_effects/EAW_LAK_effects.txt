LAK_factions_events_effect = {
	hidden_effect = {
		set_temp_variable = {
			temp_threat = { # 0-4
				value = LAK_moderate_tier
				add = LAK_radical_tier
				subtract = 2
			}
		}
		random_list = {
			40 = { # Chance of Radicals doing their move
				random_list = {
					10 = { country_event = rumare_factions.1 modifier = { factor = 0 has_variable = rumare_event_fired^1 } }
					10 = { country_event = rumare_factions.2 modifier = { factor = 0 has_variable = rumare_event_fired^2 } }
					10 = { country_event = rumare_factions.3 modifier = { factor = 0 has_variable = rumare_event_fired^3 } modifier = { factor = 0 check_variable = { temp_threat < 2 } }}
					10 = { country_event = rumare_factions.4 modifier = { factor = 0 has_variable = rumare_event_fired^4 } modifier = { factor = 0 check_variable = { temp_threat < 2 } }}
					10 = { country_event = rumare_factions.5 modifier = { factor = 0 has_variable = rumare_event_fired^5 } modifier = { factor = 0 check_variable = { temp_threat < 3 } } }
				}
			}
			40 = { # Chance of Moderates doing their move
				random_list = {
					10 = { country_event = rumare_factions.6 modifier = { factor = 0 has_variable = rumare_event_fired^6 } }
					10 = { country_event = rumare_factions.7 modifier = { factor = 0 has_variable = rumare_event_fired^7 } }
					10 = { country_event = rumare_factions.8 modifier = { factor = 0 has_variable = rumare_event_fired^8 } modifier = { factor = 0 check_variable = { temp_threat < 2 } } }
					10 = { country_event = rumare_factions.9 modifier = { factor = 0 has_variable = rumare_event_fired^9 } modifier = { factor = 0 check_variable = { temp_threat < 2 } } }
					10 = { country_event = rumare_factions.10 modifier = { factor = 0 has_variable = rumare_event_fired^10 } modifier = { factor = 0 check_variable = { temp_threat < 3 } } }
				}
			}
			20 = {} # Nothing Happens
		}
		set_temp_variable = {
			temp_days = {
				value = 180
				divide = {
					value = LAK_moderate_tier
					add = LAK_radical_tier
				}
			}
		}
		country_event = {
			id = rumare_factions.0
			days = temp_days # 30-90 days
		}
	}
}
LAK_update_GUI = {
	clamp_variable = { var = LAK_moderate_var min = 0 max = 100 }
	clamp_variable = { var = LAK_radical_var min = 0 max = 100 }
	set_variable = {
		LAK_moderate_tier = {
			value = 1
			if = {
				limit = {
					value = LAK_moderate_var greater_than_or_equals = 75
				}
				add = 2
			}
			else_if = {
				limit = {
					value = LAK_moderate_var greater_than_or_equals = 24
				}
				add = 1
			}
		}
		LAK_radical_tier = {
			value = 1
			if = {
				limit = {
					value = LAK_radical_var greater_than_or_equals = 75
				}
				add = 2
			}
			else_if = {
				limit = {
					value = LAK_radical_var greater_than_or_equals = 24
				}
				add = 1
			}
		}
	}
	add_to_variable = { LAK_GUI_dirty = 1 }
}

		limit = {

		}
		# TODO

		clr_country_flag = LAK_integration_GUI_visible_flag
		clear_variable = LAK_integration_progress_var
	}
	add_to_variable = { LAK_GUI_dirty = 1 }
}
