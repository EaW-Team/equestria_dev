LAK_update_GUI = {
	clamp_variable = { var = LAK_moderate_var min = 0 max = 100 }
	clamp_variable = { var = LAK_radical_var min = 0 max = 100 }
	set_variable = {
		LAK_moderate_tier = {
			value = 1
			if = {
				limit = {
					value = LAK_moderate_var greater_than_or_equals = 75
				}
				add = 2
			}
			else_if = {
				limit = {
					value = LAK_moderate_var greater_than_or_equals = 24
				}
				add = 1
			}
		}
		LAK_radical_tier = {
			value = 1
			if = {
				limit = {
					value = LAK_radical_var greater_than_or_equals = 75
				}
				add = 2
			}
			else_if = {
				limit = {
					value = LAK_radical_var greater_than_or_equals = 24
				}
				add = 1
			}
		}
	}
	add_to_variable = { LAK_GUI_dirty = 1 }
}
