IST_recalculate_coordinates = {
	set_temp_variable = { x_coordinate_temp_var = actual_x_position }
	set_temp_variable = { y_coordinate_temp_var = actual_y_position }

	# Calculate X
	
	add_to_temp_variable = { x_coordinate_temp_var = y_coordinate_temp_var }
	multiply_temp_variable = { x_coordinate_temp_var = 0.707 }
	subtract_from_temp_variable = { x_coordinate_temp_var = 104 }
	set_variable = { x_coordinate_var = x_coordinate_temp_var }

	set_temp_variable = { x_coordinate_temp_var = actual_x_position }

	# Calculate Y

	subtract_from_temp_variable = { x_coordinate_temp_var = y_coordinate_temp_var }
	multiply_temp_variable = { x_coordinate_temp_var = 0.707 }
	add_to_temp_variable = { x_coordinate_temp_var = 246 }
	set_variable = { y_coordinate_var = x_coordinate_temp_var }
}
