			}

	}
	else_if = {
		limit = {
			power_balance_value = {
				id = GRF_civil_war_bop
				value < 0.75
			}
		}
		set_temp_variable = { civ_war_size = 0.35 }
	}
	else = {
		set_temp_variable = { civ_war_size = 0.3 }
		set_country_flag = GRF_super_army_buff

	}
	if = {
		limit = {
			has_country_flag = GRF_ulla_helped
		}
		add_to_temp_variable = { civ_war_size = 0.05 }
		add_to_variable = { temp_repubs_spawned_troop_nb = 3 }
	}
	clr_country_flag = GRF_have_bop_flag
	remove_power_balance = {
		id = GRF_civil_war_bop
	}

	set_temp_variable = { air_civ_war_size = civ_war_size }
	add_to_temp_variable = { air_civ_war_size = -0.1 } # auths should get more air
		
	start_civil_war = {
		ideology = communism
		keep_unit_leaders_trigger = {
			has_character_flag = GRF_authoritarian_character
			NOT = { has_character_flag = GRF_republican_character }
		}
		size = var:civ_war_size
		air_ratio = var:air_civ_war_size
		navy_ratio = 0
		capital = 271
		states = all
		states_filter = {
			NOT = {
				is_in_array = {
					array = temp_state_auths_state_array
					value = id
				}
			}
		}
		set_country_flag = GRF_republicans_flag
		add_popularity = {
			ideology = democratic
			popularity = 0.45
		}

		# Troop spawning
		GRF_civil_war_repub_templates = yes
		set_country_flag = GRF_refuse_junta

		GRF_repub_buffs_from_STG = yes


		every_character = {
			limit = {
				has_character_flag = GRF_authoritarian_character
			}
			set_nationality = PREV
		}
		add_ideas = { GRF_civil_war_stage_0 }	
		set_cosmetic_tag = GRF_REPUB	
		load_focus_tree = {
			tree = Griffin_focus
			keep_completed = yes
		}
	}
	set_country_flag = GRF_authoritarians_flag

	every_character = {
		limit = {
			has_character_flag = GRF_republican_character
		}
		set_nationality = GRA
	}

	GRF_civil_war_auth_templates = yes
	hidden_effect = { # moved out of CW effect for sanity
		GRB = {
			GRF_civil_war_auth_spawning = yes
		}
		GRA = {
			GRF_civil_war_repub_spawning = yes
		}
	}
	add_ideas = {	
		GRF_fight_for_freedom
		GRF_communist_militias
		GRF_civil_war_stage_0
	}
	if = {
		limit = {
			has_country_flag = GRF_super_army_buff
		}

	}
	set_cosmetic_tag = GRF_AUTH
	GRF_auth_nerfs_from_STG = yes
	GRA = {
