### Example of usage:
### set_temp_variable = { temp_amount = 5 } -- number to add to the clan
### set_temp_variable = { temp_party = WYV } -- clan we add influence to (influence of other clans will decrease automatically by following effect)
### influence_clan = yes -- executes effects and updates GUI

influence_clan = { # sets new influence of the chosen clan and updates other clans' influences
	set_temp_variable = { old_size = 100 }
	subtract_from_temp_variable = { old_size = GAR_clans_influence_array^temp_party }
	if = {
		limit = {
			has_country_flag = GAR_equal_your_odds_flag
		}
		divide_temp_variable = { temp_amount = 2 }
	}
	add_to_variable = { 
		tooltip = influence_clan_tt
		GAR_clans_influence_array^temp_party = temp_amount 
	}
	clamp_variable = {
		var = GAR_clans_influence_array^temp_party
		max = 100
		min = 0
	}
	round_variable = GAR_clans_influence_array^temp_party
	set_temp_variable = { new_size = 100 }
	subtract_from_temp_variable = { new_size = GAR_clans_influence_array^temp_party }
	divide_temp_variable = { new_size = old_size }
	for_each_loop = {
		array = GAR_clans_influence_array
		index = i
		if = {
			limit = {
				check_variable = { 
					var = i
					value = temp_party
					compare = not_equals
				}
			}
			multiply_variable = { GAR_clans_influence_array^i = new_size }
			round_variable = GAR_clans_influence_array^i
		}
		add_to_temp_variable = { curent_size = GAR_clans_influence_array^i }
	}
	subtract_from_temp_variable = { curent_size = 100 }
	multiply_temp_variable = { curent_size = -1 }
	add_to_variable = { GAR_clans_influence_array^temp_party = curent_size }
	find_highest_in_array = {
		array = GAR_clans_influence_array
		index = temp_largest_party
	}
	set_variable = { GAR_current_strongest_clan = temp_largest_party }
	update_clans_gui = yes
}
approval_clan = {
	add_to_variable = { 
		GAR_clans_approval_array^temp_party = temp_amount
