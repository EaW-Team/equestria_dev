ZUG_gulu_calculate_faction_disunity = {
	set_variable = {
		var = ZUG_faction_disunity
		value = 0
	}
	if = {
		limit = {
			has_completed_focus = ZUG_expand_traditional_rights
			check_variable = {
				ZUG_gulu_influence < 30
			}
		}
		add_to_variable = {
			var = ZUG_faction_disunity
			value = 1
		}
	}
	if = {
		limit = {
			has_completed_focus = ZUG_royal_estates
			check_variable = {
				ZUG_gulu_influence < 50
			}
		}
		add_to_variable = {
			var = ZUG_faction_disunity
			value = 1
		}
	}
	if = {
		limit = {
			check_variable = {
				ZUG_gulu_influence < 80
			}
		}
		add_to_variable = {
			var = ZUG_faction_disunity
			value = 1
		}
	}
}
