HAU_naval_raiding_probability_trigger_effect = {
	set_temp_variable = { temp_probability_1 = HAU.HAU_naval_raiding_probability_var }
	CONTROLLER = {
		if = {
			limit = {
				has_idea = modern_society
			}
			set_temp_variable = { temp_add = -5 }
			set_variable = { PREV.HAU_society_probability = -5 }
		}
		else_if = {
			limit = {
				OR = {
					has_idea = outdated_industry
					has_idea = outdated_industry2
				}
			}
			set_temp_variable = { temp_add = 5 }
			set_variable = { PREV.HAU_society_probability = 5 }
		}
		else_if = {
			limit = {
				OR = {
					has_idea = pre_industrial_society
					has_idea = pre_industrial_society2
					has_idea = pre_industrial_society3
				}
			}
			set_temp_variable = { temp_add = 7.5 }
			set_variable = { PREV.HAU_society_probability = 7.5 }
		}
		else_if = {
			limit = {
				OR = {
					has_idea = agrarian_society
					has_idea = agrarian_society2
					has_idea = agrarian_society3
					has_idea = agrarian_society4
					has_idea = agrarian_society5
					has_idea = agrarian_society6
				}
			}
			set_temp_variable = { temp_add = 10 }
			set_variable = { PREV.HAU_society_probability = 10 }
		}
		#
		if = {
			limit = {
				has_idea = limited_exports
			}
			add_to_temp_variable = { temp_add = 5 }
			set_variable = { PREV.HAU_trade_probability = 5 }
		}
		else_if = {
			limit = {
				has_idea = export_focus
			}
			add_to_temp_variable = { temp_add = 7.5 }
			set_variable = { PREV.HAU_trade_probability = 7.5 }
		}
		else_if = {
			limit = {
				has_idea = free_trade
			}
			add_to_temp_variable = { temp_add = 10 }
			set_variable = { PREV.HAU_trade_probability = 10 }
		}
		else = {
			add_to_temp_variable = { temp_add = 0 }
			set_variable = { PREV.HAU_trade_probability = 0 }
		}
		#
		if = {
			limit = {
				has_navy_size = { size < 45 }
			}
			add_to_temp_variable = { temp_add = 5 }
			set_variable = { PREV.HAU_navy_probability = 5 }
		}
		else_if = {
			limit = {
				has_navy_size = { size < 30 }
			}
			add_to_temp_variable = { temp_add = 7.5 }
			set_variable = { PREV.HAU_navy_probability = 7.5 }
		}
		else_if = {
			limit = {
				has_navy_size = { size < 15 }
			}
			add_to_temp_variable = { temp_add = 10 }
			set_variable = { PREV.HAU_navy_probability = 10 }
		}
	}
	add_to_temp_variable = { temp_add = route_temp_probability }
	add_to_temp_variable = { temp_probability_1 = temp_add }
	clamp_temp_variable = {
		var = temp_probability_1
		min = 0
		max = 100
	}
}
