HAU_naval_raiding_effect = {
	if = {
		limit = {
			has_country_flag = HAU_naval_raiding_route_1_selected
		}
		# Manehattan #
		set_temp_variable = { temp_probability_1 = HAU_naval_raiding_probability_var }
		75 = {
			HAU_naval_raiding_probability_trigger_effect = yes
		}
		set_temp_variable = { temp_probability_1a = temp_probability_1 }
		set_temp_variable = { temp_probability_2a = temp_probability_2 }
		# Rila #
		set_temp_variable = { temp_probability_1 = HAU_naval_raiding_probability_var }
		435 = {
			HAU_naval_raiding_probability_trigger_effect = yes
		}
		set_temp_variable = { temp_probability_1b = temp_probability_1 }
		set_temp_variable = { temp_probability_2b = temp_probability_2 }
		random_list = {
			50 = { # Manehattan
				random_list = {
					0 = { # Raid Succesful
						modifier = {
							add = temp_probability_1a
						}
						country_event = haukland_raiding.1
					}
					0 = { # Raid Failed
						modifier = {
							add = temp_probability_2a
						}
						country_event = haukland_raiding.2
					}
				}
			}
			50 = { # Rila
				random_list = {
					0 = { # Raid Succesful
						modifier = {
							add = temp_probability_1b
						}
						country_event = haukland_raiding.1
					}
					0 = { # Raid Failed
						modifier = {
							add = temp_probability_2b
						}
						country_event = haukland_raiding.2
					}
				}
			}
		}
	}
}
HAU_naval_raiding_probability_trigger_effect = {
	CONTROLLER = {
		if = {
			limit = {
				OR = {
					has_idea = outdated_industry
					has_idea = outdated_industry2
				}
			}
			add_to_temp_variable = { temp_probability_1 = 2.5 } # +5%
		}
		else_if = {
			limit = {
				OR = {
					has_idea = pre_industrial_society
					has_idea = pre_industrial_society2
					has_idea = pre_industrial_society3
				}
			}
			add_to_temp_variable = { temp_probability_1 = 5 } # +10%
		}
		else_if = {
			limit = {
				OR = {
					has_idea = agrarian_society
					has_idea = agrarian_society2
					has_idea = agrarian_society3
					has_idea = agrarian_society4
					has_idea = agrarian_society5
					has_idea = agrarian_society6
				}
			}
			add_to_temp_variable = { temp_probability_1 = 7.5 } # +15%
		}
	}
	set_temp_variable = { temp_probability_2 = 100 }
	subtract_from_temp_variable = { temp_probability_2 = temp_probability_1 }
}
