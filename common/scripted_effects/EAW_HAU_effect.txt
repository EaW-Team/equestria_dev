HAU_naval_raiding_probability_trigger_effect = {
	CONTROLLER = {
		if = {
			limit = {
				OR = {
					has_idea = outdated_industry
					has_idea = outdated_industry2
				}
			}
			add_to_temp_variable = { temp_probability_1 = 2.5 } # +5%
		}
		else_if = {
			limit = {
				OR = {
					has_idea = pre_industrial_society
					has_idea = pre_industrial_society2
					has_idea = pre_industrial_society3
				}
			}
			add_to_temp_variable = { temp_probability_1 = 5 } # +10%
		}
		else_if = {
			limit = {
				OR = {
					has_idea = agrarian_society
					has_idea = agrarian_society2
					has_idea = agrarian_society3
					has_idea = agrarian_society4
					has_idea = agrarian_society5
					has_idea = agrarian_society6
				}
			}
			add_to_temp_variable = { temp_probability_1 = 7.5 } # +15%
		}
	}
	set_temp_variable = { temp_probability_2 = 100 }
	subtract_from_temp_variable = { temp_probability_2 = temp_probability_1 }
}
