FAT_mobilize_law_by_1 = {
	if = {
		limit = { has_idea = disarmed_nation }
		add_ideas = volunteer_only
	}
	else_if = {
		limit = { has_idea = volunteer_only }
		add_ideas = limited_conscription
	}
	else_if = {
		limit = { has_idea = limited_conscription }
		add_ideas = extensive_conscription
	}
	else_if = {
		limit = { has_idea = extensive_conscription }
		add_ideas = service_by_requirement
	}
	else_if = {
		limit = { has_idea = service_by_requirement }
		add_ideas = all_adults_serve
	}
	else_if = {
		limit = { has_idea = all_adults_serve }
		add_ideas = scraping_the_barrel
	}
	else_if = {
		limit = { has_idea = scraping_the_barrel }
		add_manpower = 15000
	}
}

FAT_create_dynamic_LCT = {
	create_dynamic_country = {
		original_tag = LCT
		PREV = {
			every_controlled_state = {
				limit = {
					is_in_array = { state_original_cores = LCT }
				}
				PREV.PREV = {
					transfer_state = PREV
					add_state_core = PREV
				}
			}
			release_puppet = PREV
		}
		load_focus_tree = { tree = generic_griffon_focus }
		set_capital = {
			state = 387
			remember_old_capital = no
		}
		generate_character = {
			token_base = FAT_LCT_council
			name = FAT_LCT_council
		}
		set_party_name = {
			ideology = fascism
			long_name = LCT_FATASS_F_LONG
			name = LCT_FATASS_F
		}
		set_party_name = {
			ideology = communism
			long_name = LCT_FATASS_C_LONG
			name = LCT_FATASS_C
		}
		set_party_name = {
			ideology = neutrality
			long_name = LCT_FATASS_N_LONG
			name = LCT_FATASS_N
		}
		set_party_name = {
			ideology = democratic
			long_name = LCT_FATASS_D_LONG
			name = LCT_FATASS_D
		}
		if = {
			limit = { has_character = FAT_LCT_council }
			set_portraits = {
				character = FAT_LCT_council
				civilian = {
					large = "gfx/leaders/FAT/Council.tga"
				}
			}
			add_country_leader_role = {
				character = FAT_LCT_council
				country_leader = {
					ideology = stratocracy
					expire = "1965.1.1.1"
					id=-1
				}
				promote_leader = yes
			}
			add_country_leader_role = {
				character = FAT_LCT_council
				country_leader = {
					ideology = khakistocracy
					expire = "1965.1.1.1"
					id=-1
				}
				promote_leader = yes
			}
			add_country_leader_role = {
				character = FAT_LCT_council
				country_leader = {
					ideology = liberalism
					expire = "1965.1.1.1"
					id=-1
				}
				promote_leader = yes
			}
			add_country_leader_role = {
				character = FAT_LCT_council
				country_leader = {
					ideology = democratic_socialism
					expire = "1965.1.1.1"
					id=-1
				}
				promote_leader = yes
			}
		}
		add_ideas = {
			pony_race
			modern_society
			no_lack_of_scientists

# Safe incrementation for Reconstruction progress that does not exceed 100
# Always use this instead of adding to variable directly
# set_temp_variable = { reconstruction_progress_add_value = [VALUE] }
FAT_increment_reconstruction_progress = {
	set_temp_variable = { reconstruction_value_check = FAT_reconstruction_progress_value }
	add_to_temp_variable = { reconstruction_value_check = reconstruction_progress_add_value }
	if = {
		limit = {
			check_variable = { reconstruction_value_check > 99 }
		}
		set_variable = {
			FAT_reconstruction_progress_value = 100
			tooltip = FAT_reconstruction_progress_value_max_tt
		}
	}
	else = {
		add_to_variable = {
			FAT_reconstruction_progress_value = reconstruction_progress_add_value
			tooltip = FAT_reconstruction_progress_value_tt
		}
	}
	FAT_update_reconstruction_gui = yes
}

# Same as above but for Sectarianism
# set_temp_variable = { sectarianism_add_value = [VALUE] }
FAT_increment_sectarianism = {
	effect_tooltip = {
		add_to_variable = {
			FAT_sectarianism_value = sectarianism_add_value
			tooltip = FAT_sectarianism_value_tt
		}
	}
	set_temp_variable = { sectarianism_value_check = FAT_sectarianism_value }
	add_to_temp_variable = { sectarianism_value_check = sectarianism_add_value }
	if = {
		limit = {
			check_variable = { sectarianism_value_check > @FAT_SECTARIANISM_CAP }
		}
		set_variable = { FAT_sectarianism_value = @FAT_SECTARIANISM_CAP }
	}
	else_if = {
		limit = {
			check_variable = { sectarianism_value_check < @FAT_SECTARIANISM_CAP_MIN }
		}
		set_variable = { FAT_sectarianism_value = @FAT_SECTARIANISM_CAP_MIN }
	}
	else = {
		add_to_variable = { FAT_sectarianism_value = sectarianism_add_value }
	}
	FAT_update_reconstruction_gui = yes
}
