FAT_create_republican_pact = {
	if = {
		limit = { has_government = fascism }
		create_faction_from_template = faction_template_republican_pact_supremacist
	}
	else = { create_faction_from_template = faction_template_republican_pact_moderate }
}

FAT_create_dynamic_NTR = {
	create_dynamic_country = {
		original_tag = NTR
		set_variable = { collaboration_formed_by = PREV }
		set_temp_variable = { new_country = this }
		PREV = {
			every_controlled_state = {
				limit = { occupied_country_tag = NTR }
				var:new_country = { transfer_state = PREV }
			}
			puppet = new_country
		}
		load_focus_tree = { tree = generic_griffon_focus }
		set_capital = {
			state = 562
			remember_old_capital = no
		}
		generate_character = {
			token_base = FAT_NTR_council
			name = FAT_NTR_council
		}
		if = {
			limit = { has_character = FAT_NTR_council }
			set_portraits = {
				character = FAT_NTR_council
				civilian = {
					large = "gfx/leaders/FAT/Council.tga"
				}
			}
			add_country_leader_role = {
				character = FAT_NTR_council
				country_leader = {
					ideology = stratocracy
					expire = "1965.1.1.1"
					id=-1
				}
				promote_leader = yes
			}
			add_country_leader_role = {
				character = FAT_NTR_council
				country_leader = {
					ideology = khakistocracy
					expire = "1965.1.1.1"
					id=-1
				}
				promote_leader = yes
			}
			add_country_leader_role = {
				character = FAT_NTR_council
				country_leader = {
					ideology = liberalism
					expire = "1965.1.1.1"
					id=-1
				}
				promote_leader = yes
			}
			add_country_leader_role = {
				character = FAT_NTR_council
				country_leader = {
					ideology = democratic_socialism
					expire = "1965.1.1.1"
					id=-1
				}
				promote_leader = yes
			}
		}
		set_party_name = {
			ideology = fascism
			long_name = NTR_FATASS_LONG
			name = NTR_FATASS
		}
		set_party_name = {
			ideology = communism
			long_name = NTR_FATASS_LONG
			name = NTR_FATASS
		}
		set_party_name = {
			ideology = neutrality
			long_name = NTR_FATASS_LONG
			name = NTR_FATASS
		}
		set_party_name = {
			ideology = democratic
			long_name = NTR_FATASS_LONG
			name = NTR_FATASS
		}
		add_ideas = {
			pony_race
			poverty5
			outdated_industry
			lack_of_scientists2
			better_science_base
			civilian_economy
		}
		FAT = { NTR = { transfer_technology = yes } }
		set_AI_strategies_on_every_possible_country_delayed = yes
	}
}

FAT_invade_herzland = {
	if = {
		limit = {
			382 = {
				owner = {
					NOT = { tag = ROOT }
					tag = GRD
				}
			}
		}
		declare_war_on = { type = annex_everything target = GRD }
	}
	else = { unlock_decision_tooltip = FAT_demand_griffenheim }
	if = {
		limit = {
			FEA = { exists_outside_sphere_of_influence = yes }
		}
		declare_war_on = { type = annex_everything target = FEA }
	}
	if = {
		limit = {
			STW = { exists_outside_sphere_of_influence = yes }
		}
		declare_war_on = { type = annex_everything target = STW }
	}
	if = {
		limit = {
			PYT = { exists_outside_sphere_of_influence = yes }
		}
		declare_war_on = { type = annex_everything target = PYT }
	}
	if = {
		limit = {
			BRZ = { exists_outside_sphere_of_influence = yes }
		}
		declare_war_on = { type = annex_everything target = BRZ }
	}
	if = {
		limit = {
			YAL = { exists_outside_sphere_of_influence = yes }
		}
		declare_war_on = { type = annex_everything target = YAL }
	}
	if = {
		limit = {
			ANG = {
				exists_outside_sphere_of_influence = yes
				has_country_leader = { ruling_only = yes character = ANG_Gerhard_Bluhm }
			}
		}
		ANG = { country_event = griffonRepublic.275 }
	}
	else_if = {
		limit = {
			ANG = { exists_outside_sphere_of_influence = yes }
		}
		declare_war_on = { type = annex_everything target = ANG }
	}
	if = {
		limit = {
			ROU = {
				exists_outside_sphere_of_influence = yes
				has_government = democratic
			}
		}
		ROU = { country_event = griffonRepublic.275 }
	}
	else_if = {
		limit = {
			ROU = { exists_outside_sphere_of_influence = yes }
		}
		declare_war_on = { type = annex_everything target = ROU }
	}
	if = {
		limit = {
			KAT = {
				exists_outside_sphere_of_influence = no
				has_government = democratic
			}
		}
		KAT = { country_event = griffonRepublic.275 }
	}
	else_if = {
		limit = {
			KAT = { exists_outside_sphere_of_influence = yes }
		}
		declare_war_on = { type = annex_everything target = KAT }
	}
}
