#Console command for setting day/night overrides. Takes an integer arguement.
#0 - clear overrides
#1 - night override
#2 - day override
#Example usage: setDayNightOverride 1
d_setDayNightOverride = {
	if = {
		limit = {
			check_variable = { args^num < 1 }
		}
		log = "RPlease enter an arguement!!"
	}
	else = {
		if = {
			limit = {
				check_variable = { args^0 = 0 }
			}
			EaW_clear_day_night_overrides = yes
			log = "Cleared override"
		}
		else_if = {
			limit = {
				check_variable = { args^0 = 1 }
			}
			EaW_night_override = yes
			log = "Set override to Ynight!"
		}
		else_if = {
			limit = {
				check_variable = { args^0 = 2 }
			}
			EaW_day_override = yes
			log = "Set override to Yday!"
		}
		else = {
			log = "RInvalid arguement!!"
		}
	}
}

EaW_clear_day_night_overrides = {
	if = {
		limit = {
			has_global_flag = EaW_dn_override_enabled
		}
		if = {
			limit = {
				has_global_flag = EaW_night_override
			}
			hidden_effect = {
				every_possible_country = {
					limit = {
						NOT = { tag = ZZZ }
					}
					night_remove_opinion_modifiers = yes
				}
				clr_global_flag = EaW_night_override
			}
			custom_effect_tooltip = EaW_eternal_night_end
		}
		if = {
			limit = {
				has_global_flag = EaW_day_override
			}
			hidden_effect = {
				every_possible_country = {
					limit = {
						NOT = { tag = ZZZ }
					}
					day_remove_opinion_modifiers = yes
				}
				clr_global_flag = EaW_day_override
			}
			custom_effect_tooltip = EaW_eternal_day_end
		}
		clr_global_flag = EaW_dn_override_enabled
		hidden_effect = {
			ZZZ = {
				drop_cosmetic_tag = yes
				set_politics = {
					ruling_party = neutrality
					elections_allowed = no
				}
			}
		}
	}
}

EaW_night_override = {
	hidden_effect = {
		EaW_clear_day_night_overrides = yes
		set_global_flag = EaW_night_override
		set_global_flag = EaW_dn_override_enabled
		every_possible_country = {
			limit = {
				NOT = { tag = ZZZ }
			}
			night_add_opinion_modifiers = yes
		}
		ZZZ = {
			set_cosmetic_tag = ZZZ_night_override
			set_politics = {
				ruling_party = democratic
				elections_allowed = no
			}
		}
	}
	custom_effect_tooltip = EaW_eternal_night
}

EaW_day_override = {
	hidden_effect = {
		EaW_clear_day_night_overrides = yes
		set_global_flag = EaW_day_override
		set_global_flag = EaW_dn_override_enabled
		every_possible_country = {
			limit = {
				NOT = { tag = ZZZ }
			}
			day_add_opinion_modifiers = yes
		}
		force_update_dynamic_modifier = yes
		ZZZ = {
			set_cosmetic_tag = ZZZ_day_override
			set_politics = {
				ruling_party = communism
				elections_allowed = no
			}
		}
	}
	custom_effect_tooltip = EaW_eternal_day
}
