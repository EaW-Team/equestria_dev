YAL_karl_boost_anarchist_effect = {
	if = {
		limit = {
			has_idea = YAL_anarchist_support_4
		}
		swap_ideas = {
			remove_idea = YAL_anarchist_support_4
			add_idea = YAL_anarchist_support_5
		}
	}
	else_if = {
		limit = {
			has_idea = YAL_anarchist_support_3
		}
		swap_ideas = {
			remove_idea = YAL_anarchist_support_3
			add_idea = YAL_anarchist_support_4
		}
	}
	else_if = {
		limit = {
			has_idea = YAL_anarchist_support_2
		}
		swap_ideas = {
			remove_idea = YAL_anarchist_support_2
			add_idea = YAL_anarchist_support_3
		}
	}
	else_if = {
		limit = {
			has_idea = YAL_anarchist_support_1
		}
		swap_ideas = {
			remove_idea = YAL_anarchist_support_1
			add_idea = YAL_anarchist_support_2
		}
	}
}

YAL_reduce_economic_devastation_effect = {
	# Does nothing if ideas are already removed
	if = {
		limit = {
			has_idea = YAL_economic_downturn
		}
		swap_ideas = {
			remove_idea = YAL_economic_downturn
			add_idea = YAL_sluggish_economy
		}
	}
	else_if = {
		limit = {
			has_idea = YAL_sluggish_economy
		}
		swap_ideas = {
			remove_idea = YAL_sluggish_economy
			add_idea = YAL_sluggish_economy_2
		}
	}
	else_if = {
		limit = {
			has_idea = YAL_sluggish_economy_2
		}
		swap_ideas = {
			remove_idea = YAL_sluggish_economy_2
			add_idea = YAL_lingering_economic_woes
		}
	}
	else_if = {
		limit = {
			has_idea = YAL_lingering_economic_woes
		}
		remove_ideas = YAL_lingering_economic_woes
	}
}
