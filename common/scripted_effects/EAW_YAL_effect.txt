	}
}

YAL_update_university_network = {
	set_variable = {
		YAL_university_network_pop = {
			value = 0.02
			multiply = YAL_subject_counter
		}
	}
	set_variable = {
		YAL_university_network_pp = {
			value = 0.05
			multiply = YAL_subject_counter
		}
	}
	if = {
		limit = {
			has_completed_focus = YAL_the_reconstruction_mutual_fund
		}
		set_variable = {
			YAL_university_network_civs = {
				value = 1
				multiply = YAL_subject_counter
			}
		}
	}
	else = {
		set_variable = { YAL_university_network_civs = 0 }
	}
	force_update_dynamic_modifier = yes
}

YAL_centrist_puppet_effect = {
	if = {
		limit = {
			YAL = { has_country_flag = YAL_is_democratic_path }
		}
		ROOT = {
			set_autonomy = {
				target = PREV
				autonomy_state = autonomy_supervised_state
			}
		}
		set_politics = {
			ruling_party = democratic
		}

		set_popularities = {
			democratic = 70
			neutrality = 10
			fascism = 10
			communism = 10
		}
	}
	else = {
		ROOT = {
			set_autonomy = {
				target = PREV
				autonomy_state = autonomy_research_associate
			}
		}
	}

	load_focus_tree = { tree = generic_griffon_focus keep_completed = no }					
	reserve_dynamic_country = yes
}
