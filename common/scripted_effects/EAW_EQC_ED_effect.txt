artifacts_apply_mods_pb = {
	set_temp_variable = { new_pb = 0 }
	for_each_loop = {
		array = artifacts_acquired_power 
		value = n

		add_to_temp_variable = { new_pb = n }
	}
	set_variable = { power_pb_var = new_pb }
	set_temp_variable = { pb_mod = 1 }
	add_to_temp_variable = { pb_mod = modifier@artifact_strength_factor }
	multiply_temp_variable = { new_pb = pb_mod }
	set_variable = { art_debug_mod = new_pb }
	add_to_variable = { zzz_update_arts = 0.01 }
}

remove_art_power_effect = {
	
	for_each_loop = {
		array = ROOT.artifacts_affect_token
		value = fel_v 
		index = fel_i
		break = token_ptr

		if = {
			limit = {
				check_variable = { fel_v = remove_idea_pointer }
			}
			set_variable = { current_idea_index = fel_i }
			set_temp_variable = { token_ptr = 1 }
		}
		
		remove_from_array = { array = ROOT.artifacts_affect_token index = current_idea_index }
		remove_from_array = { array = ROOT.artifacts_affect_value index = current_idea_index }
	}
	
}

thaumaturgy_roll = {
	
	set_temp_variable = { success_chance = 3.5 }

	if = {
		limit = {
			has_mastery = { amount = 500 track = operations }
		}
		add_to_temp_variable = { success_chance = 0.6 }
	}
	else_if = {
		limit = {
			has_mastery = { amount = 300 track = operations }
		}
		add_to_temp_variable = { success_chance = 0.4 }
	}
	else_if = {
		limit = {
			has_mastery = { amount = 100 track = operations }
		}
		add_to_temp_variable = { success_chance = 0.2 }
	}
	
	set_temp_variable = { aa_effect = FROM.building_level@anti_air_building }
	divide_temp_variable = { aa_effect = 10 } ### 1 AA gives -0.1 
	subtract_from_temp_variable = { success_chance = aa_effect }
	
	if = {
		limit = {
			FROM.owner = { has_tech = future_radar_5 }
		}
		subtract_from_temp_variable = { success_chance = 0.3 }
	}
	else_if = {
		limit = {
			FROM.owner = { has_tech = future_radar_4 }
		}
		subtract_from_temp_variable = { success_chance = 0.25 }
	}
	else_if = {
		limit = {
			FROM.owner = { has_tech = future_radar_3 }
		}
		subtract_from_temp_variable = { success_chance = 0.2 }
	}
	else_if = {
		limit = {
			FROM.owner = { has_tech = future_radar_2 }
		}
		subtract_from_temp_variable = { success_chance = 0.15 }
	}
	else_if = {
		limit = {
			FROM.owner = { has_tech = future_radar }
		}
		subtract_from_temp_variable = { success_chance = 0.1 }
	}
	
	if = {
		limit = {
			ROOT = { is_special_project_completed = sp:sp_pegasi_void_suit }
		}
		add_to_temp_variable = { success_chance = 0.25 }
	}
	if = {
		limit = {
			ROOT = { is_special_project_completed = sp:sp_magical_imperceptibility_studies } ### LaR Only 
		}
		add_to_temp_variable = { success_chance = 0.15 }
	}
	
	if = {
		limit = {
			ROOT = { is_special_project_completed = sp:sp_air_advanced_jet_engine } 
		}
		add_to_temp_variable = { success_chance = 0.3 }
	}
	else_if = {
		limit = {
			ROOT = { is_special_project_completed = sp:sp_air_axial_jet_engine } 
		}
		add_to_temp_variable = { success_chance = 0.2 }
	}
	else_if = {
		limit = {
			ROOT = { is_special_project_completed = sp:sp_air_jet_engine } 
		}
		add_to_temp_variable = { success_chance = 0.1 }
	}	
	
	if = {
		limit = {
			NOT = { has_dlc = "La Resistance" }
		}
		set_temp_variable = { dec_st = modifier@decryption }
		FROM = { set_temp_variable = { ROOT.enc_st = owner:modifier@encryption } }
		
		subtract_from_temp_variable = { dec_st = enc_st }
		if = {
			limit = {
				check_variable = { var = dec_st value = 0 compare = greater_than_or_equals }
			}
			add_to_temp_variable = { success_chance = 0.2 }
		}
	}
	else = {
		
		set_temp_variable = { int_level = ROOT.air_intel@FROM.owner }
		
		divide_temp_variable = { int_level = 400 } #### 0 to 1 -> 0 to 0.25
		add_to_temp_variable = { success_chance = int_level }
	}
	
	if = {
		limit = {
			FROM = { is_owned_by = EQC }
		}
		add_to_temp_variable = { success_chance = -1.5 }
	}
	
	set_variable = { roll_result = success_chance }
}

ED_conquering_equestria_effect = {
	# The result is written into the temp variable effective_magic_balance
	# Formula is arctan( (EQS magic - EQS magic) / 5000 )
	ED_calculate_effective_magic_balance = yes 

	EQC = { set_variable = { current_magic_balance = effective_magic_balance } }
	EQS = { 
		set_variable = { current_magic_balance = effective_magic_balance }
		multiply_variable = { current_magic_balance = -1 }
	}

	hidden_effect = {
		EQC = { # Make sure we know the scope
			set_variable = { mission_days = 365 } # 1 year
			set_temp_variable = { mission_days_mod = effective_magic_balance }
			multiply_temp_variable = { mission_days_mod = 165 }
			add_to_variable = { mission_days = mission_days_mod }
			log = "Days for Equestria conquest mission: [?mission_days]"
			
			activate_mission = EQS_ED_the_invasion_of_equestria_mission
			add_dynamic_modifier = { modifier = ED_magic_balance_effect }
			ED_calculate_magic_balance_modifier = yes
			force_update_dynamic_modifier = yes
		}
		EQS = {
			activate_mission = EQS_ED_the_invasion_of_equestria_mission
			# The effect of this is the negative to the EQC one due to the variable being multiplied by -1 for EQS
			add_dynamic_modifier = { modifier = ED_magic_balance_effect } 
			ED_calculate_magic_balance_modifier = yes
			force_update_dynamic_modifier = yes
		}
	}
}

ED_calculate_effective_magic_balance = {
	set_temp_variable = { magic_power_difference = EQC.art_debug_mod } 
	subtract_from_temp_variable = { magic_power_difference = EQS.art_debug_mod } 
	divide_temp_variable = { magic_power_difference = 5000 } 
	set_temp_variable = {
		sign = 1
	}
	if = {
		limit = {
			check_variable = { magic_power_difference < 0 }
		}
		set_temp_variable = { sign = -1 }
	}
	set_temp_variable = { arctan_argument = 1 }
	if = {
		limit = {
			check_variable = { magic_power_difference > 1 }
		}
		set_country_flag = inverse_active
		divide_temp_variable = { arctan_argument = magic_power_difference }
	}
	else = {
		set_temp_variable = { arctan_argument = magic_power_difference }
	}
	ED_calculate_arctan = yes

	set_temp_variable = { effective_magic_balance = arctan_result }
	if = {
		limit = {
			has_country_flag = inverse_active
		}
		set_temp_variable = { effective_magic_balance = 1.571 } # pi/2
		multiply_temp_variable = { effective_magic_balance = sign }
		subtract_from_temp_variable = { effective_magic_balance = arctan_result }
	}
	log = "Effective Magic Balance: [?effective_magic_balance]"

	clr_country_flag = inverse_active
}

ED_calculate_arctan = {
	# x
	set_temp_variable = { arctan_result = arctan_argument }

	# x^3 / 3
	set_temp_variable = { cubic = arctan_argument }
	multiply_temp_variable = { cubic = arctan_argument }
	multiply_temp_variable = { cubic = arctan_argument }
	divide_temp_variable = { cubic = 3 }

	# x^5 / 5
	set_temp_variable = { quintic = arctan_argument }
	multiply_temp_variable = { quintic = arctan_argument }
	multiply_temp_variable = { quintic = arctan_argument }
	multiply_temp_variable = { quintic = arctan_argument }
	multiply_temp_variable = { quintic = arctan_argument }
	divide_temp_variable = { quintic = 5 }

	# x^7 / 7
	set_temp_variable = { septic = arctan_argument }
	multiply_temp_variable = { septic = arctan_argument }
	multiply_temp_variable = { septic = arctan_argument }
	multiply_temp_variable = { septic = arctan_argument }
	multiply_temp_variable = { septic = arctan_argument }
	multiply_temp_variable = { septic = arctan_argument }
	multiply_temp_variable = { septic = arctan_argument }
	divide_temp_variable = { septic = 7 }

	# x^9 / 9
	set_temp_variable = { nonic = arctan_argument }
	multiply_temp_variable = { nonic = arctan_argument }
	multiply_temp_variable = { nonic = arctan_argument }
	multiply_temp_variable = { nonic = arctan_argument }
	multiply_temp_variable = { nonic = arctan_argument }
	multiply_temp_variable = { nonic = arctan_argument }
	multiply_temp_variable = { nonic = arctan_argument }
	multiply_temp_variable = { nonic = arctan_argument }
	multiply_temp_variable = { nonic = arctan_argument }
	divide_temp_variable = { nonic = 9 }

	# x - x^3/3 + x^5/5 - x^7/7 + x^9/9
	# This is the taylor series of the arctan to ninth order
	# Yes, I know that this isn't optimal
	# We should also use arctan(x) = pi/4 - arctan( (1-x)/(1+x) ) for values close to 1
	# But I'm lazy
	subtract_from_temp_variable = { arctan_result = cubic }
	add_to_temp_variable = { arctan_result = quintic }
	subtract_from_temp_variable = { arctan_result = septic }
	add_to_temp_variable = { arctan_result = nonic }
}

ED_decay_magical_balance_effect = {
	if = {
		limit = {
			check_variable = {
				current_magic_balance < 0
			}
		}
		add_to_variable = {
			current_magic_balance = 0.05
		}
		if = {
			limit = {
				current_magic_balance > -0.01
			}
			remove_dynamic_modifier = {
				modifier = ED_magic_balance_effect
			}
		}
	}
	else = {
		subtract_from_variable = {
			current_magic_balance = 0.05
		}
		if = {
			limit = {
				current_magic_balance < 0.01
			}
			remove_dynamic_modifier = {
				modifier = ED_magic_balance_effect
			}
		}
	}
}
