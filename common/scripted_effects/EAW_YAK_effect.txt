YAK_hegemony_popularity = {
	if = {
		limit = {
			has_government = communism
		}
		add_popularity = { ideology = communism popularity = 0.1 }
	}
	else_if = {
		limit = {
			has_government = neutrality
		}
		add_popularity = { ideology = neutrality popularity = 0.1 }
	}
	else_if = {
		limit = {
			has_government = democratic
		}
		add_popularity = { ideology = democratic popularity = 0.1 }
	}
	else = {
		add_popularity = { ideology = fascism popularity = 0.1 }
	}
}

YAK_clan_juyilkhai_effect_calculation = {
	if = {
		limit = {
			check_variable = {
				global.yak_juyilkhai_clan_opinion_var = 3
			}
			check_variable = {
				global.yak_juyilkhai_prev_var = 2
			}
		}
		every_owned_state = {
			limit = {
				has_state_flag = YAK_juyilkhai_control
			}
			add_state_modifier = {
				modifier = {
					local_manpower = 0.10
					
					monthly_population = -0.10
				}
			}
		}
	}
	else_if = {
		limit = {
			check_variable = {
				global.yak_juyilkhai_clan_opinion_var = 2
			}
			check_variable = {
				global.yak_juyilkhai_prev_var = 3
			}
		}
		add_state_modifier = {
			modifier = {
				local_manpower = -0.10
				
				monthly_population = 0.10
			}
		}		
	}
}
