af_init_radom_events = {
	scramble_random_seed = yes
	
	add_to_array = { diarchy_random_events = 1 }
	add_to_array = { diarchy_random_events = 2 }
	add_to_array = { diarchy_random_events = 3 }
	add_to_array = { diarchy_random_events = 5 }
	add_to_array = { diarchy_random_events = 6 }
	add_to_array = { diarchy_random_events = 7 }
	add_to_array = { diarchy_random_events = 8 }
}

af_fire_random_event = {
	if = {
		limit = {
			check_variable = { diarchy_random_events^num > 0 }
		}
		set_temp_variable_to_random = {
			var = random_index
			min = 0
			max = diarchy_random_events^num
			integer = yes
		}
		
		meta_effect = {
			text = {
				country_event = evildiarchy_random.[ID]
			}
			ID = "[?diarchy_random_events^random_index]"
		}
		
		remove_from_array = {
			array = diarchy_random_events
			index = random_index
		}
	}
}
