
FLY_photos_starting_location = {
	add_to_array = { FLY_applejack_position_x = 0 } # Starting position
	add_to_array = { FLY_applejack_position_y = 0 }

	add_to_array = { FLY_applejack_position_y = 200 }

	add_to_array = { FLY_fluttershy_position_x = 20 } # Starting position
	add_to_array = { FLY_fluttershy_position_y = 0 }

	add_to_array = { FLY_pinkie_position_x = -20 } # Starting position
	add_to_array = { FLY_pinkie_position_y = 0 }

	add_to_array = { FLY_rarity_position_x = 0 } # Starting position
	add_to_array = { FLY_rarity_position_y = 100 }

	add_to_array = { FLY_rainbow_position_x = 0 } # Starting position
	add_to_array = { FLY_rainbow_position_y = -100 }

	add_to_array = { FLY_twilight_position_x = 0 } # Starting position
	add_to_array = { FLY_twilight_position_y = 80 }

	set_variable = { FLY_applejack_position_index = 0 }
	set_variable = { FLY_fluttershy_position_index = 0 }
	set_variable = { FLY_pinkie_position_index = 0 }
	set_variable = { FLY_rarity_position_index = 0 }
	set_variable = { FLY_twilight_position_index = 0 }

	add_to_variable = { FLY_m6_map_update_true = 1 }
}
