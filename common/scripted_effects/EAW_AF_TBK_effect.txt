EAW_af_TBK_recalc_daily = {

	###1 is base speed
	set_variable = { THIS.TBK_AF_state_view_jadd_progress_daily = 0.2 }
	if = {
		limit = {
			NOT = {
				is_owned_by = TBK
			}
		}
		multiply_variable = { THIS.TBK_AF_state_view_jadd_progress_daily = 0.5 }
	}
	multiply_variable = {
		THIS.TBK_AF_state_view_jadd_progress_daily = {
			#TBK_AF_state_view_jadd_progress_daily_coef
			value = 1 
			add = THIS.TBK_AF_state_view_jadd_progress_daily_coef_bonus
			add = TBK.TBK_AF_state_view_jadd_progress_daily_coef_bonus_global
		}
	}
	
	###GUI Reload 
	add_to_variable = { THIS.TBK_AF_state_view_gui_reloading = 1 }

}

EAW_af_DEL_recalc_daily = {

	###1 is base speed
	set_variable = {
		THIS.TBK_AF_state_view_jadd_progress_daily = {
			value = 0.2
			multiply = {
				#TBK_AF_state_view_jadd_progress_daily_coef
				value = 1
				add = THIS.TBK_AF_state_view_jadd_progress_daily_coef_bonus
				add = DEL.TBK_AF_state_view_jadd_progress_daily_coef_bonus_global
			}
			multiply = -1
		}
	}
	
	###GUI Reload 
	add_to_variable = { THIS.TBK_AF_state_view_gui_reloading = 1 }

}

EAW_af_DEL_pop_into_mp = {

	if = {
		limit = {
			check_variable = {
				DEL_state_pop_save = 0
			}
		}
		set_variable = {
			DEL_state_pop_save = state_population
		}
	}

	if = {
		limit = {
			check_variable = {
				DEL_state_pop_save < state_population
			}
		}
		set_temp_variable = {
			DEL_state_pop_save_temp = {
				value = state_population
				subtract = DEL_state_pop_save
			}
		}
		owner = {
			add_manpower = DEL_state_pop_save_temp
		}
		multiply_temp_variable = {
			DEL_state_pop_save_temp = -1
		}
		add_manpower = DEL_state_pop_save_temp
	}

	if = {
		limit = {
			check_variable = {
				state_population > 0
			}
		}
		set_variable = {
			DEL_state_pop = state_population_k
		}
		if = {
			limit = {
				has_dynamic_modifier = {
					modifier = TBK_AF_jadd_evil_lvl3_state_modifier
				}
			}
			multiply_variable = {
				DEL_state_pop = 2.5
			}
		}
		if = {
			limit = {
				has_dynamic_modifier = {
					modifier = TBK_AF_jadd_evil_lvl4_state_modifier
				}
			}
			multiply_variable = {
				DEL_state_pop = 5
			}
		}
		if = {
			limit = {
				has_dynamic_modifier = {
					modifier = TBK_AF_jadd_evil_lvl5_state_modifier
				}
			}
			multiply_variable = {
				DEL_state_pop = 10
			}
		}
		if = {
			limit = {
				is_core_of = DEL
			}
				add_to_variable = {
					DEL_state_pop = 100
				}
		}
		add_to_variable = {
			DEL_state_pop = 1
		}
		owner = {
			add_manpower = DEL_state_pop
		}
		multiply_variable = {
			DEL_state_pop = -1
		}
		add_manpower = DEL_state_pop
		if = {
			limit = {
				check_variable = {
					DEL_state_pop_save > state_population
				}
			}
			set_variable = {
				DEL_state_pop_save = state_population
			}
		}
	}

}
