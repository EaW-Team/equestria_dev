FLO_reroll_ai_loan_desire = {
	every_other_country = {
		limit = {
			is_griffon_continent_griffon_country_tag = yes
			NOT = {
				original_tag = FLO
				original_tag = KAT
				original_tag = BAN
				original_tag = MNC
			}
		}
		set_variable_to_random = {
			var = flowenan_loan_desire_var
			min = 0
			max = 4
			integer = no
		}
		# Formula: x^4/4 - 1

		multiply_variable = { flowenan_loan_desire_var = flowenan_loan_desire_var } # Square
		multiply_variable = { flowenan_loan_desire_var = flowenan_loan_desire_var } # Square again
		divide_variable = { flowenan_loan_desire_var = 4 }
		subtract_from_variable = { flowenan_loan_desire_var = 1 }

		log = "FLO load desire rolled: [THIS.GetName]: [?flowenan_loan_desire_var]"
	}
}
