		}

		modulo_variable = { ZOO_railway_tiles_stored_rounded = 1 }

		add_to_temp_variable = { temp_2 = 1 }
		for_loop_effect = {
			start = start_index
			end = ZOO_railway_tiles_stored_rounded
			value = temp

				text = {
					build_railway = {

			}
		}
	}
}
