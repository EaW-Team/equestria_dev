ZOO_weekly_railroad_construction = {
	if = {
		limit = {
			check_variable = {
				var = ZOO_weekly_railroad_construction
				value = 1
				compare = greater_than_or_equals
			}
		}
		set_variable = { ZOO_railway_tiles_stored_rounded = ZOO_railway_tiles_stored }
		modulo_variable = { ZOO_railway_tiles_stored_rounded = 1 }
		for_loop_effect = {
			start = 0
			end = ZOO_railway_tiles_stored_rounded
			value = temp
			set_variable = { ZOO_railway_tiles_construction_1 = ZOO_railway_tiles_construction^temp }
			set_temp_variable = { temp_2 = temp }
			add_to_temp_variable = {
				temp_2 = 1
			}
			set_variable = { ZOO_railway_tiles_construction_2 = ZOO_railway_tiles_construction^temp_2 }
			meta_effect = {
				text = {
					build_railway = {
						level = 1
						build_only_on_allied = yes
						path = { [FIRST_TILE] [SECOND_TILE] }				
					}
				}
				
				FIRST_TILE = "[?ZOO_railway_tiles_construction_1]]"
				SECOND_TILE = "[?ZOO_railway_tiles_construction_2]]"
			}
		}
	}
}
