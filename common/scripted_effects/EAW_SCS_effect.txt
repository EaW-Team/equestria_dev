SCS_reduce_organizational_difficulties = {
	if = {
		limit = { has_idea = SCS_organizational_difficulties }
		swap_ideas = { add_idea = SCS_organizational_difficulties2 remove_idea = SCS_organizational_difficulties }
	}
	else_if = {
		limit = { has_idea = SCS_organizational_difficulties2 }
		swap_ideas = { add_idea = SCS_organizational_difficulties3 remove_idea = SCS_organizational_difficulties2 }
	}
	else_if = {
		limit = { has_idea = SCS_organizational_difficulties3 }
		remove_ideas = SCS_organizational_difficulties3
	}
}

SCS_LSP_setup_faction = {
	if = {
		limit = {
			
			SCS = {
				is_ai = no
			}
			LSP = {
				is_ai = yes
			}
		}
		SCS = {
			create_faction = BUF_ROC_faction
			add_to_faction = LSP
			LSP = {
				set_major = yes
			}
		}
	}
	else = {
		LSP = {
			create_faction = BUF_ROC_faction
			add_to_faction = SCS
			SCS = {
				set_major = yes
			}
		}
	}
}
