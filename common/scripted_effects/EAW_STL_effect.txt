STL_enable_firing_advisors = {
	STL_night_glider = {
		set_can_be_fired_in_advisor_role = { slot = air_chief value = yes } 
	}
	STL_party_flavour = {
		set_can_be_fired_in_advisor_role = { slot = political_advisor value = yes } 
	}
	STL_feather_bangs = {
		set_can_be_fired_in_advisor_role = { slot = political_advisor value = yes } 
	}
	STL_sugar_belle = {
		set_can_be_fired_in_advisor_role = { slot = political_advisor value = yes } 	
	}
}
STL_fire_everyone = {
	if = {
		limit = {
			STL_feather_bangs = {
				is_political_advisor = yes
			}
		}
		deactivate_advisor = STL_feather_bangs
	}
	if = {
		limit = {
			STL_party_flavour = {
				is_political_advisor = yes
			}
		}
		deactivate_advisor = STL_party_flavour
	}
	if = {
		limit = {
			STL_sugar_belle = {
				is_political_advisor = yes
			}
		}
		deactivate_advisor = STL_sugar_belle
	}
	if = {
		limit = {
			STL_background_pony_1 = {
				is_political_advisor = yes
			}
		}
		deactivate_advisor = STL_background_pony_1
	}
	if = {
		limit = {

				is_political_advisor = yes
			}
		}

			}
		}
		deactivate_advisor = STL_background_pony_3
	}

}

STL_set_up_STG_war_defensive = {
	set_global_flag = STG_STL_fighting
	STL = {
		STL_escalate_the_war_effect = yes
		set_country_flag = STG_attacked_STL
	}
	news_event = {
		id = ourtown_news.7
		days = 1
	}
}


STL_escalate_the_war_effect = {
	custom_effect_tooltip = ETH_war_escalation_inc_tt
	add_to_variable = { STL.STL_war_escalation_level = 1 }
}

STL_upgrade_settlers_effect = {
	if = {
		limit = {
			has_idea = STL_settlers_idea_1
		}
		swap_ideas = {
			remove_idea = STL_settlers_idea_1
			add_idea = STL_settlers_idea_2
		}
	}
	else_if = {
		limit = {
			has_idea = STL_settlers_idea_2
		}
		swap_ideas = {
			remove_idea = STL_settlers_idea_2
			add_idea = STL_settlers_idea_3
		}
	}
	else_if = {
		limit = {
			has_idea = STL_settlers_idea_3
		}
		swap_ideas = {
			remove_idea = STL_settlers_idea_3
			add_idea = STL_settlers_idea_4
		}
	}
	else_if = {
		limit = {
			has_idea = STL_settlers_idea_4
		}
		swap_ideas = {
			remove_idea = STL_settlers_idea_4
			add_idea = STL_settlers_idea_5
		}
	}
	else_if = {
		limit = {
			has_idea = STL_settlers_idea_5
		}
		swap_ideas = {
			remove_idea = STL_settlers_idea_5
			add_idea = STL_settlers_idea_6
		}
	}
}
