	STL_night_glider = {
		set_can_be_fired_in_advisor_role = { slot = air_chief }
	}
	STL_party_flavour = {
		set_can_be_fired_in_advisor_role = { slot = political_advisor }
	}
	STL_feather_bangs = {
		set_can_be_fired_in_advisor_role = { slot = political_advisor }		
	}
	STL_sugar_belle = {
		set_can_be_fired_in_advisor_role = { slot = political_advisor }		
	}
}

STL_set_up_STG_war_offensive = {
	STL = {
		if = {
			limit = {
				has_government = democratic
			}
			set_global_flag = STG_STL_fighting
		}
		else_if = {
			limit = { 
				has_government = neutrality
			}
			set_global_flag = STG_STL_fighting
		}
		else = {	
			STG = {
				add_ideas = STL_STG_easily_capitulate
			}
		}
	}
}
STL_set_up_STG_war_defensive = {

	STL = {
		if = {
			limit = {
				has_government = democratic
			}
			set_global_flag = STG_STL_fighting
			STL_escalate_the_war_effect = yes
			STL_escalate_the_war_effect = yes
			set_country_flag = STG_attacked_STL
		}
		else_if = {
			limit = { 
				has_government = neutrality
			}
			set_global_flag = STG_STL_fighting
			STL_escalate_the_war_effect = yes
			STL_escalate_the_war_effect = yes
			set_country_flag = STG_attacked_STL
		}
		else = {
			STG = {
				add_ideas = STL_STG_easily_capitulate_alt
			}
		}
	}
}


STL_escalate_the_war_effect = {
	custom_effect_tooltip = ETH_war_escalation_inc_tt
	add_to_variable = { STL.STL_war_escalation_level = 1 }
}
