MSH_Influences_re_render = {
	MSH = {
		set_variable = { MSH_piety = 0 }
		set_variable = { MSH_purism = 0 }
		set_variable = { MSH_reformism = 0 }
		set_temp_variable = { population_total = 0 }
		for_each_scope_loop = { # can't trust max_manpower_k
			array = MSH_Influences_States
			
			if = {
				limit = {
					is_owned_by = MSH
				}
				add_to_temp_variable = { population_total = state_population_k }
			}
		}
		set_variable = { population_total_permanent = population_total }
		set_temp_variable = { population_total_check = 0 }
		for_each_scope_loop = {
			array = MSH_Influences_States
			
			if = {
				limit = {
					is_owned_by = MSH
				}
				set_variable = { population_factor = state_population_k }
				divide_variable = { population_factor = population_total }
			}
		}
		
		resize_temp_array = { temp_influence_average = 3 }
		
		for_each_loop = {
			array = MSH_Influences_States
			
			var:v = {
				if = {
					limit = {
						is_owned_by = MSH
					}
					
					## Set local progressbar variables to be called for in local view

					set_variable = { MSH_local_display = 100 }
					subtract_from_variable = { MSH_local_display = MSH_Influences^2 }
					
					if = {
						limit = {
							check_variable = { MSH_Influences^0 > MSH_Influences^1 }
							check_variable = { MSH_Influences^0 > MSH_Influences^2 }
						}
						set_variable = { MSH_Influences_Frame = 2 }
					}
					else_if = {
						limit = {
							check_variable = { MSH_Influences^1 > MSH_Influences^0 }
							check_variable = { MSH_Influences^1 > MSH_Influences^2 }
						}
						set_variable = { MSH_Influences_Frame = 3 }
					}
					else_if = {
						limit = {
							check_variable = { MSH_Influences^2 > MSH_Influences^0 }
							check_variable = { MSH_Influences^2 > MSH_Influences^1 }
						}
						set_variable = { MSH_Influences_Frame = 4 }
					}
					else = {
						set_variable = { MSH_Influences_Frame = 1 }
					}
					
					# Piety
					set_temp_variable = { local_piety = MSH_Influences^0 }
					multiply_temp_variable = { local_piety = population_factor }
					set_variable = { MSH_local_piety_contribution = local_piety }

					# Purism
					set_temp_variable = { local_purism = MSH_Influences^1 }
					multiply_temp_variable = { local_purism = population_factor }
					set_variable = { MSH_local_purism_contribution = local_purism }

					# Reformism
					set_temp_variable = { local_reformism = MSH_Influences^2 }
					multiply_temp_variable = { local_reformism = population_factor }
					set_variable = { MSH_local_reformism_contribution = local_reformism }
					
					for_each_loop = {
						array = temp_influence_average
						index = j
						value = k
						
						set_temp_variable = { temp_influence_average_calc = MSH_Influences^j }
						multiply_temp_variable = { temp_influence_average_calc = population_factor }
						add_to_temp_variable = { temp_influence_average^j = temp_influence_average_calc }
					}
					
					multiply_variable = { population_factor = 100 }
				}
			}
		}
		
		clear_array = MSH_Influences_National
		set_temp_variable = { total_average_check = 100 }
		set_temp_variable = { total_average_check_v = 0 }
		for_each_loop = {
			array = temp_influence_average
			add_to_array = { MSH_Influences_National = temp_influence_average^i }
			add_to_temp_variable = { total_average_check_v = v }
		}
		subtract_from_temp_variable = { total_average_check = total_average_check_v }
		divide_temp_variable = { total_average_check = 3 }
		for_each_loop = {
			array = MSH_Influences_National
			add_to_variable = { MSH_Influences_National^i = total_average_check }
		}

		set_variable = { MSH_piety = MSH_Influences_National^0 }
		set_variable = { MSH_purism = MSH_Influences_National^1 }
		set_variable = { MSH_reformism = MSH_Influences_National^2 }
		
		# Compute curved bar frame
		set_variable = { MSH_piety_frame = MSH_piety }
		multiply_variable = { MSH_piety_frame = 1.2 }
		add_to_variable = { MSH_piety_frame = 210 }

		multiply_variable = { MSH_reformism_frame = -1.2 }
		add_to_variable = { MSH_reformism_frame = 150 }
		
		MSH_Influences_re_position_num = yes
		
		# Dynmod apply
		for_each_loop = {
			array = MSH_Influences_States
			
			var:v = {
				if = {
					limit = {
						is_owned_by = MSH
					}
					
					if = {
						limit = {
							has_state_flag = MSH_scholarch_legalist
						}
						add_dynamic_modifier = { modifier = MSH_scholarch_legalist }
						clr_state_flag = MSH_scholarch_legalist
					}

							has_state_flag = MSH_scholarch_purist
						}
						add_dynamic_modifier = { modifier = MSH_scholarch_purist }
						clr_state_flag = MSH_scholarch_purist
					}
					else_if = {
						limit = {
							has_state_flag = MSH_scholarch_reformist
						}
						add_dynamic_modifier = { modifier = MSH_scholarch_reformist }
						clr_state_flag = MSH_scholarch_reformist
					}
		
				}
			}
		}

### MASSIVE INFLUENCE CODE INCOMING ###

## t_index values:
# 0 - Legalists
# 1 - Purists
# 2 - Reformists

## How to use:

# 1000 = {
#	 set_temp_variable = { t_index = 1 }
#	 set_temp_variable = { t_change = 10 }
#	 MSH_change_support = yes
# }

## This will change 1000's purist influence in the by 10

MSH_change_support = {
	set_temp_variable = { v_check = MSH_Influences^t_index }
	add_to_temp_variable = { v_check = t_change }

	if = {
		limit = {
			check_variable = { v_check < 0 }
		}
		subtract_from_temp_variable = { t_change = v_check }
	}
	else_if = {
		limit = {
			check_variable = { v_check > 100 }
		}
		set_temp_variable = { n_100 = v_check }
		subtract_from_temp_variable = { n_100 = 100 }
		subtract_from_temp_variable = { t_change = n_100 }
	}
	
	if = {
		limit = {
			NOT = {
				check_variable = { t_change = 0 }
			}
		}
		custom_effect_tooltip = MSH_change_support_CET
		add_to_variable = { MSH_Influences^t_index = t_change }
		set_temp_variable = { t_change_divide = t_change }
		
		for_each_loop = {
			array = MSH_Influences
			if = {
				limit = {
					check_variable = { v = 0 }
					NOT = { check_variable = { i = t_index } }
				}
				add_to_temp_array = { has_0 = i }
			}
		}
		
		set_temp_variable = { divide_factor = 2 }
		subtract_from_temp_variable = { divide_factor = has_0^num }
		divide_temp_variable = { t_change_divide = divide_factor }
		
		set_temp_variable = { excess = 0 }
		for_each_loop = {
			array = MSH_Influences
			value = inf_v
			index = inf_i

			if = {
				limit = {
					NOT = {
						check_variable = { inf_i = t_index }
						is_in_array = { array = has_0 value = inf_i }
					}
				}
				subtract_from_variable = { MSH_Influences^inf_i = t_change_divide }
			}
			if = {
				limit = { check_variable = { MSH_Influences^inf_i < 0 } }
				add_to_temp_variable = { excess = MSH_Influences^inf_i }
				set_variable = { MSH_Influences^inf_i = 0 }
				add_to_temp_array = { zero_value = inf_i }
			}
		}
		for_each_loop = {
			array = MSH_Influences
			value = inf_v
			index = inf_i
			if = {
				limit = {
					NOT = {
						is_in_array = { array = zero_value value = inf_i }
						check_variable = { inf_i = t_index }
					}
				}
				add_to_variable = { MSH_Influences^inf_i = excess }
			}
		}
	}
	
	if = {
		limit = {
			NOT = {
				check_variable = { manually_re_render_gui = 1 }
			}
		}
		MSH_Influences_re_render = yes
	}
	
	clear_temp_array = has_0
	clear_temp_array = zero_value
}

	MSH_change_support = yes
