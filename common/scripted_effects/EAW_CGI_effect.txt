CGI_end_bop_and_apply_idea = {
	if = {
		limit = {
			has_power_balance = {
				id = CGI_crossroads_bop
			}
		}
		
		if = {
			limit = {
				power_balance_value = {
					id = CGI_crossroads_bop
					value < -0.5
				}
			}
			add_ideas = CGI_powerbalance_end_ascending_clans
			set_country_flag = CGI_moderate_khanate
		}
		else_if = {
			limit = { 
				power_balance_value = {
					id = CGI_crossroads_bop
					value < -0.25
				}
			}
			add_ideas = CGI_powerbalance_end_powerful_clans
			set_country_flag = CGI_moderate_khanate
		}
		else_if = {
			limit = {
				power_balance_value = {
					id = CGI_crossroads_bop
					value < 0.25
				}
			}
			add_ideas = CGI_powerbalance_end_balanced_government
			set_country_flag = CGI_moderate_khanate
		}
		else_if = {
			limit = {
				power_balance_value = {
					id = CGI_crossroads_bop
					value < 0.5
				}
			}
			add_ideas = CGI_powerbalance_end_powerful_khanate
			set_country_flag = CGI_moderate_khanate
		}
		else_if = {
			limit = {
				power_balance_value = {
					id = CGI_crossroads_bop
					value < 0.75
				}
			}
			add_ideas = CGI_powerbalance_end_ascending_khanate
			set_country_flag = CGI_moderate_khanate
		}
		else = {
			add_ideas = CGI_powerbalance_end_dominant_khanate
		}

		remove_power_balance = {
			id = CGI_crossroads_bop
		}
	}
}

CGI_modify_colthaginian_resistance = {
	# First step: Moke sure we never overshoot anywhere
	set_temp_variable = { consumer_goods_compvar = CGI_colthaginian_resistance_consumer_goods }
	add_to_temp_variable = { consumer_goods_compvar = CGI_CR_consumer_goods_val }
	if = {
		limit = { check_variable = { consumer_goods_compvar < 0 } }
		subtract_from_temp_variable = { CGI_CR_consumer_goods_val = consumer_goods_compvar }
	}
	set_temp_variable = { construction_speed_compvar = CGI_colthaginian_resistance_construction_speed }
	add_to_temp_variable = { construction_speed_compvar = CGI_CR_construction_speed_val }
	if = {
		limit = { check_variable = { construction_speed_compvar > 0 } }
		subtract_from_temp_variable = { CGI_CR_construction_speed_val = construction_speed_compvar }
	}
	set_temp_variable = { factory_output_compvar = CGI_colthaginian_resistance_factory_output }
	add_to_temp_variable = { factory_output_compvar = CGI_CR_factory_output_val }
	if = {
		limit = { check_variable = { factory_output_compvar > 0 } }
		subtract_from_temp_variable = { CGI_CR_factory_output_val = factory_output_compvar }
	}
	set_temp_variable = { dockyard_output_compvar = CGI_colthaginian_resistance_dockyard_output }
	add_to_temp_variable = { dockyard_output_compvar = CGI_CR_dockyard_output_val }
	if = {
		limit = { check_variable = { dockyard_output_compvar > 0 } }
		subtract_from_temp_variable = { CGI_CR_dockyard_output_val = dockyard_output_compvar }
	}
	set_temp_variable = { political_power_compvar = CGI_colthaginian_resistance_political_power }
	add_to_temp_variable = { political_power_compvar = CGI_CR_political_power_val }
	if = {
		limit = { check_variable = { political_power_compvar > 0 } }
		subtract_from_temp_variable = { CGI_CR_political_power_val = political_power_compvar }
	}
	set_temp_variable = { stability_compvar = CGI_colthaginian_resistance_stability }
	add_to_temp_variable = { stability_compvar = CGI_CR_stability_val }
	if = {
		limit = { check_variable = { stability_compvar > 0 } }
		subtract_from_temp_variable = { CGI_CR_stability_val = stability_compvar }
	}
	set_temp_variable = { war_support_compvar = CGI_colthaginian_resistance_war_support }
	add_to_temp_variable = { war_support_compvar = CGI_CR_war_support_val }
	if = {
		limit = { check_variable = { war_support_compvar > 0 } }
		subtract_from_temp_variable = { CGI_CR_war_support_val = war_support_compvar }
	}
	set_temp_variable = { garrison_damage_compvar = CGI_colthaginian_resistance_garrison_damage }
	add_to_temp_variable = { garrison_damage_compvar = CGI_CR_garrison_damage_val }
	if = {
		limit = { check_variable = { garrison_damage_compvar < 0 } }
		subtract_from_temp_variable = { CGI_CR_garrison_damage_val = garrison_damage_compvar }
	}

	# Second step: Apply the verified effects
	if = {
		limit = {
			has_dynamic_modifier = {
				modifier = CGI_colthaginian_resistance_modifier
			}
			# Check if we actually modify stuff (this might dynamically not 
			# be the case anymore in case the resistance effect got removed elsewhere)
			OR = {
				NOT = { check_variable = { CGI_CR_consumer_goods_val = 0 } }
				NOT = { check_variable = { CGI_CR_construction_speed_val = 0 } }
				NOT = { check_variable = { CGI_CR_factory_output_val = 0 } }
				NOT = { check_variable = { CGI_CR_dockyard_output_val = 0 } }
				NOT = { check_variable = { CGI_CR_political_power_val = 0 } }
				NOT = { check_variable = { CGI_CR_stability_val = 0 } }
				NOT = { check_variable = { CGI_CR_war_support_val = 0 } }
				NOT = { check_variable = { CGI_CR_garrison_damage_val = 0 } }
			}
		}
		custom_effect_tooltip = CGI_decrease_CTH_res
		if = {
			limit = {
				NOT = { check_variable = { CGI_CR_consumer_goods_val = 0 } }
			}
			add_to_variable = { CGI_colthaginian_resistance_consumer_goods = CGI_CR_consumer_goods_val }
			custom_effect_tooltip = CGI_CR_consumer
		}
		if = {
			limit = {
				NOT = { check_variable = { CGI_CR_construction_speed_val = 0 } }
			}
			add_to_variable = { CGI_colthaginian_resistance_construction_speed = CGI_CR_construction_speed_val }
			custom_effect_tooltip = CGI_CR_construction
		}
		if = {
			limit = {
				NOT = { check_variable = { CGI_CR_factory_output_val = 0 } }
			}
			add_to_variable = { CGI_colthaginian_resistance_factory_output = CGI_CR_factory_output_val }
			custom_effect_tooltip = CGI_CR_factory
		}
		if = {
			limit = {
				NOT = { check_variable = { CGI_CR_dockyard_output_val = 0 } }
			}
			add_to_variable = { CGI_colthaginian_resistance_dockyard_output = CGI_CR_dockyard_output_val }
			custom_effect_tooltip = CGI_CR_dockyard
		}
		if = {
			limit = {
				NOT = { check_variable = { CGI_CR_political_power_val = 0 } }
			}
			add_to_variable = { CGI_colthaginian_resistance_political_power = CGI_CR_political_power_val }
			custom_effect_tooltip = CGI_CR_pp
		}
		if = {
			limit = {
				NOT = { check_variable = { CGI_CR_stability_val = 0 } }
			}
			add_to_variable = { CGI_colthaginian_resistance_stability = CGI_CR_stability_val }
			custom_effect_tooltip = CGI_CR_stability
		}
		if = {
			limit = {
				NOT = { check_variable = { CGI_CR_war_support_val = 0 } }
			}
			add_to_variable = { CGI_colthaginian_resistance_war_support = CGI_CR_war_support_val }
			custom_effect_tooltip = CGI_CR_war_support
		}
		if = {
			limit = {
				NOT = { check_variable = { CGI_CR_garrison_damage_val = 0 } }
			}
			add_to_variable = { CGI_colthaginian_resistance_garrison_damage = CGI_CR_garrison_damage_val }
			custom_effect_tooltip = CGI_CR_garrison_damage
		}
		force_update_dynamic_modifier = yes
	}
}

# Made a scripted effect as it appears in multiple focus trees
CGI_end_of_diamond_road_effect = {
	# Get some industry
	one_random_industrial_complex = yes
	effect_tooltip = {
		custom_effect_tooltip = {
			localization_key = CGI_if_diamond_has_strength_points
			STRENGTH = "20"
		}
		one_random_industrial_complex = yes
		custom_effect_tooltip = {
			localization_key = CGI_if_diamond_has_strength_points
			STRENGTH = "40"
		}
		one_random_industrial_complex = yes
		if = {
			limit = {
				check_variable = {
					CGI_diamond_road_mil_cost < 1000
				}
			}
			custom_effect_tooltip = {
				localization_key = CGI_if_diamond_has_strength_points
				STRENGTH = "25"
			}
			one_random_arms_factory = yes
			custom_effect_tooltip = {
				localization_key = CGI_if_diamond_has_strength_points
				STRENGTH = "50"
			}
			one_random_arms_factory = yes
		}
	}
	hidden_effect = {
		if = {
			limit = {
				check_variable = { CGI_diamond_road_strength > 19.9 }
			}
			one_random_industrial_complex = yes
		}
		if = {
			limit = {
				check_variable = { CGI_diamond_road_strength > 39.9 }
			}
			one_random_industrial_complex = yes
		}
		if = {
			limit = {
				check_variable = {
					CGI_diamond_road_mil_cost < 1000
				}
			}
			if = {
				limit = {
					check_variable = { CGI_diamond_road_strength > 24.9 }
				}
				one_random_arms_factory = yes
			}
			if = {
				limit = {
					check_variable = { CGI_diamond_road_strength > 49.9 }
				}
				one_random_arms_factory = yes
			}
		}
	}

	# Get some resources (only steel though, resources don't grow on trees)
	set_temp_variable = {
		steel_to_get = CGI_diamond_road_strength
	}
	divide_temp_variable = {
		steel_to_get = CGI_diamond_road_steel_cost
	}
	round_temp_variable = steel_to_get
	add_resource = {
		type = steel
		amount = steel_to_get
		state = 883 # Koltshgar
	}

	# Get 25 times the weekly equipment
	set_temp_variable = {
		guns_to_get = CGI_diamond_road_strength
	}
	divide_temp_variable = {
		guns_to_get = CGI_diamond_road_gun_cost
	}
	multiply_temp_variable = {
		guns_to_get = 25
	}
	add_equipment_to_stockpile = {
		type = infantry_equipment
		amount = guns_to_get
	}
	if = {
		limit = {
			check_variable = {
				CGI_diamond_road_arty_cost < 1000
			}
		}
		set_temp_variable = {
			arty_to_get = CGI_diamond_road_strength
		}
		divide_temp_variable = {
			arty_to_get = CGI_diamond_road_arty_cost
		}
		multiply_temp_variable = {
			arty_to_get = 25
		}
		add_equipment_to_stockpile = {
			type = artillery_equipment
			amount = arty_to_get
		}
	}
	if = {
		limit = {
			check_variable = {
				CGI_diamond_road_support_cost < 1000
			}
		}
		set_temp_variable = {
			support_to_get = CGI_diamond_road_strength
		}
		divide_temp_variable = {
			support_to_get = CGI_diamond_road_support_cost
		}
		multiply_temp_variable = {
			support_to_get = 25
		}
		add_equipment_to_stockpile = {
			type = artillery_equipment
			amount = support_to_get
		}
	}

	# Diamond Road over
	remove_dynamic_modifier = {
		modifier = CGI_diamond_road_modifier
	}
	clr_country_flag = CGI_diamond_road_active
	custom_effect_tooltip = CGI_disables_diamond_road
	add_ideas = CGI_diamond_road_external # Leftover
}
