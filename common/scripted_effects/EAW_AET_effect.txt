change_AET_style_meter = {
	if = {
		limit = {
			check_variable = {
				var = AET_style_count
				value = 100
				compare = greater_than
			}
		}
		subtract_from_variable = {
			AET_style_count = 100
		}
		level_up_AET_style_meter = yes
	}
	else_if = {
		limit = {
			check_variable = {
				var = AET_style_count
				value = 0
				compare = less_than
			}
		}
		add_to_variable = {
			AET_style_count = 100
		}
		level_down_AET_style_meter = yes
	}
}

level_up_AET_style_meter = {
	add_to_variable = { AET_style_count_div = 1 }
}

level_down_AET_style_meter = {
	subtract_from_variable = { AET_style_count_div = 1 }
}
