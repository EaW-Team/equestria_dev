AET_book_update_page_gridbox = {
	# f(n) = 2n - 1
	set_variable = { AET_book_left_stratagem_id = AET_book_current_page }
	multiply_variable = { AET_book_left_stratagem_id = 2 }
	subtract_from_variable = { AET_book_left_stratagem_id = 1 }

	# f(n) = 2n
	set_variable = { AET_book_right_stratagem_id = AET_book_current_page }
	multiply_variable = { AET_book_right_stratagem_id = 2 }
	
	clear_array = AET_HLDV_stratagem_left_display
	clear_array = AET_HLDV_stratagem_right_display
	if = {
		limit = {
			NOT = {
				check_variable = { AET_HLDV_stratagem_list^AET_book_left_stratagem_id = 0 }
			}
		}
		add_to_array = { AET_HLDV_stratagem_left_display = AET_HLDV_stratagem_list^AET_book_left_stratagem_id }
	}
	if = {
		limit = {
			NOT = {
				check_variable = { AET_HLDV_stratagem_list^AET_book_right_stratagem_id = 0 }
			}
		}
		add_to_array = { AET_HLDV_stratagem_right_display = AET_HLDV_stratagem_list^AET_book_right_stratagem_id }
	}
}

AET_HLDV_evaluate_sequence = {
	for_each_loop = {
		array = AET_HLDV_stratagem_list
		index = effect_i
		value = effect_j
		break = b

		if = {
			limit = {
				all_of = {
					array = AET_HLDV_stratagem_code_list_@var:effect_j
					index = effect_k
					value = effect_l

					check_variable = { effect_l = AET_HLDV_current_arrows^effect_k }
				}
				check_variable = { var = AET_HLDV_current_arrows^num value = AET_HLDV_stratagem_code_list_@var:effect_j^num compare = less_than_or_equals }
			}
			# I'm going to hell
			if = {
				limit = {
					meta_trigger = {
						text = {
							AET_HLDV_trigger_[INPUT] = yes
						}
						INPUT = "[?effect_j]"
					}
				}
				meta_effect = {
					text = {
						AET_HLDV_effect_[INPUT] = yes
					}
					INPUT = "[?effect_j]"
				}
				scoped_sound_effect = squish_sound
				clear_array = AET_HLDV_current_arrows
				clear_variable = AET_HLDV_last_arrow_direction
				AET_HLDV_highlight = yes

				set_temp_variable = { b = 1 }
			}
		}
	}
	add_to_variable = { AET_HLDV_dirty = 0.001 }
}

# When you explicitly want two spells to be next to each other, add a dummy entry to skip an odd page
AET_HLDV_check_for_odd = {
	if = {
		limit = {
			set_temp_variable = { temp_count = AET_HLDV_stratagem_list^num }
			subtract_from_temp_variable = { temp_count = 1 }
			modulo_temp_variable = { temp_count = 2 }
			check_variable = { temp_count = 1 }
		}
		add_to_array = { AET_HLDV_stratagem_list = 0 }
	}
}

# args: `temp_spell_id`, `temp_stratagem_sequence`
AET_HLDV_add_to_spell_list = {
	if = {
		limit = {
			NOT = {
				has_country_flag = AET_HLDV_active
			}
		}
		AET_HLDV_start = yes
	}
	add_to_array = { AET_HLDV_stratagem_list = temp_spell_id }
	while_loop_effect = {
		limit = { check_variable = { temp_stratagem_sequence > 0 } }
		set_temp_variable = { temp_current = temp_stratagem_sequence }
		modulo_temp_variable = { temp_current = 10 }
		log = "temp_current: [?temp_current] is inputted"
		meta_effect = {
			text = {
				add_to_array = { array = AET_HLDV_stratagem_code_list_[INPUT] value = temp_current index = 0 }
			}
			INPUT = "[?temp_spell_id]"
		}

		divide_temp_variable = { temp_stratagem_sequence = 10 }
		subtract_from_temp_variable = { temp_stratagem_sequence = 0.5 }
		round_temp_variable = temp_stratagem_sequence
	}

	add_to_variable = { AET_HLDV_dirty = 0.001 }
}

AET_HLDV_trimmel_daily_pulse = {
	for_each_loop = {
		array = AET_HLDV_trimmel_timer_array
		if = {
			limit = {
				check_variable = { AET_HLDV_trimmel_timer_array^i > 0 }
				check_variable = { AET_HLDV_trimmel_timer_array_status^i = 1 }
				log = "First check, i: [?i]"
				log = "AET_HLDV_trimmel_timer_array^i: [?AET_HLDV_trimmel_timer_array^i]"
				log = "AET_HLDV_trimmel_timer_array_status^i: [?AET_HLDV_trimmel_timer_array_status^i]"
			}
			subtract_from_variable = { AET_HLDV_trimmel_timer_array^i = 1 }

			# rescale for frames = raw * ( 100 / max )
			set_temp_variable = { temp_timer_calc = AET_HLDV_trimmel_timer_array^i }
			multiply_temp_variable = { temp_timer_calc = 100 }
			divide_temp_variable = { temp_timer_calc = AET_HLDV_trimmel_timer_array_max^i }

			set_variable = { AET_HLDV_trimmel_timer_array_frame_dial^i = temp_timer_calc }
			set_variable = { AET_HLDV_trimmel_timer_array_frame_circle^i = 100 }
			subtract_from_variable = { AET_HLDV_trimmel_timer_array_frame_circle^i = temp_timer_calc }
		}
		else_if = {
			limit = {
				check_variable = { AET_HLDV_trimmel_timer_array_status^i = 1 }
				log = "Second check, i: [?i]"
				log = "AET_HLDV_trimmel_timer_array_status^i: [?AET_HLDV_trimmel_timer_array_status^i]"
			}
			# ongoing effect is attached to the init stratagem
			set_temp_variable = { temp_stratagem_id = i }
			multiply_temp_variable = { temp_stratagem_id = 2 }
			subtract_from_temp_variable = { temp_stratagem_id = 1 }
			meta_effect = {
				text = {
					AET_HLDV_effect_[INPUT]_ongoing = yes
				}
				INPUT = "[?AET_HLDV_stratagem_list^temp_stratagem_id]"
			}
			set_variable = { AET_HLDV_trimmel_timer_array^i = AET_HLDV_trimmel_timer_array_max^i }
			set_variable = { AET_HLDV_trimmel_timer_array_frame_circle^i = 0 }
			set_variable = { AET_HLDV_trimmel_timer_array_frame_dial^i = 100 }
		}
	}
	add_to_variable = { AET_HLDV_dirty = 0.00001 }
}
