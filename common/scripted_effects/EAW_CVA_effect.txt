CVA_to_the_strongest = {
	set_temp_variable = { strongest_griffonian_nation = 0 }
	set_temp_variable = { strongest_nation = 0 }
	
	add_to_temp_array = { states_to_check = 558 }
	add_to_temp_array = { states_to_check = 578 }
	add_to_temp_array = { states_to_check = 413 }
	add_to_temp_array = { states_to_check = 680 }
	
	for_each_scope_loop = {
		array = states_to_check
		every_neighbor_state = {
			limit = {
				owner = {
					NOT = {
						tag = ROOT
						is_subject_of = ROOT
						has_idea = detached_country
						has_war_with = ROOT
					}
				}
			}
			owner = {
				set_temp_variable = { country_to_check = THIS }
				if = {
					limit = {
						is_subject = yes 
					}
					overlord = {
						set_temp_variable = { country_to_check = THIS }
					}
				}
			}
			var:country_to_check = {
				if = {
					limit = {
						is_griffon_continent_country = yes
						OR = {
							check_variable = { strongest_griffonian_nation = 0 }
							check_variable = { num_battalions > strongest_griffonian_nation:num_battalions }
						}
					}
					set_temp_variable = { strongest_griffonian_nation = THIS }
				}
				if = {
					limit = {
						OR = {
							check_variable = { strongest_nation = 0 }
							check_variable = { num_battalions > strongest_nation:num_battalions }
						}
					}
					set_temp_variable = { strongest_nation = THIS }
				}
			}
		}
	}
}
