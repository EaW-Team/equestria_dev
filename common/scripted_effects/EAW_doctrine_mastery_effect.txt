# Usage:
# 	set_temp_variable = { temp_track = 3 } -- sets track, 1-8: infantry -> support -> armor -> operations -> [future doctrines in the same direction starting from 5]
# 	set_temp_variable = { temp_value = 50 } -- amount of mastery you're adding. If doctrine isn't selected, gives an equivalent percentage doctrine cost reduction. In this case, 50%
#	land_mastery_doctrine_effect = yes

land_mastery_doctrine_effect = {
	if = {
		limit = {
			has_any_grand_doctrine = land 
			meta_trigger = {
				text = {
					has_subdoctrine_in_track = [TRACK]
				}
				TRACK = "[land_mastery_doctrine_GetTrack]"
			}
		}
		meta_effect = {
			text = {
				add_mastery = {
					amount = [NUMBER]
					track = [TRACK]
				}
			}
			NUMBER = "[?temp_value]"
			TRACK = "[land_mastery_doctrine_GetTrack]"
		}
	}
	else = {
		divide_temp_variable = { temp_value = 100 }
		meta_effect = {
			text = {
				add_doctrine_cost_reduction = {
					cost_reduction = [NUMBER]
					name = land_doc_bonus
					uses = 1
					category = land_doctrine
				}
			}
			NUMBER = "[?temp_value]"
		}
	}
}
