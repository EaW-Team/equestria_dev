land_infantry_mastery_doctrine_25 = {		#for former 25% doctrine cost reduction
	if = {
		limit = { 
			has_any_grand_doctrine = land 
			has_subdoctrine_in_track = infantry
		}
		add_mastery = {
			amount = 25
			folder = land
			track = infantry
		}
	}
	else = {
		add_doctrine_cost_reduction = {
			cost_reduction = 0.25
			name = doctrine_bonus
			uses = 1
			category = land_doctrine
		}
	}
}
