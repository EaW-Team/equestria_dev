land_infantry_mastery_doctrine_25 = {		#for former 25% doctrine cost reduction
	if = {
		limit = { 
			has_any_grand_doctrine = land 
			has_subdoctrine_in_track = infantry
		}
		add_mastery = {
			amount = 25
			folder = land
			track = infantry
		}
	}
	else = {
		add_doctrine_cost_reduction = {
			cost_reduction = 0.25
			name = doctrine_bonus
			uses = 1
			category = land_doctrine
		}
	}
}

# Usage:
# 	set_temp_variable = { temp_track = 3 } -- sets track, 1-8: infantry -> support -> armor -> operations -> [future doctrines in the same direction starting from 5]
# 	set_temp_variable = { temp_value = 50 } -- amount of mastery you're adding. If doctrine isn't selected, gives 50% doctrine cost reduction
#	land_mastery_doctrine_effect = yes

land_mastery_doctrine_effect = {
	if = {
		limit = {
			has_any_grand_doctrine = land 
			meta_trigger = {
				text = {
					has_subdoctrine_in_track = [TRACK]
				}
				TRACK = "[land_mastery_doctrine_GetTrack]"
			}
		}
		meta_effect = {
			text = {
				add_mastery = {
					amount = [NUMBER]
					folder = land
					track = [TRACK]
				}
			}
			NUMBER = "[?temp_value]"
			TRACK = "[land_mastery_doctrine_GetTrack]"
		}
	}
	else = {
		add_doctrine_cost_reduction = {
			cost_reduction = 0.5
			name = doctrine_bonus
			uses = 1
			category = land_doctrine
		}
			######################################################################################################################
