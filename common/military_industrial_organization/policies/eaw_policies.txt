		owner = { original_tag = STG }
	}

		has_mio_size > 4 
	}
	
	visible = {
		owner = {
			has_completed_focus = STG_invest_crystals_to_research_labs
		}
	}
	
	organization_modifier = {
		military_industrial_organization_research_bonus = 0.33
		custom_modifier_tooltip = STG_policy_5_consgood_30_cryst
	}
	production_bonus = {
		same_as_mio = {
			production_resource_need_factor = -0.1
		}
	}

		hidden_effect = {
			owner = {
				add_ideas = STG_polic_use_of_crystals

			}
		}
	}
	
	on_remove = {
		hidden_effect = {
			owner = {
				remove_ideas = STG_polic_use_of_crystals
			}
		}
	}	
}
