# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

###BOLTRUS VARIABLES###
defined_text = {
	name = BLTGetBoltrusOutwardValue
	text = {
		trigger = {
			check_variable = { BLT_king_boltrus_outward_value = 4 }
		}
		localization_key = BLTGetBoltrusOutwardValue_4
	}
	text = {
		trigger = {
			check_variable = { BLT_king_boltrus_outward_value = 3 }
		}
		localization_key = BLTGetBoltrusOutwardValue_3
	}
	text = {
		trigger = {
			check_variable = { BLT_king_boltrus_outward_value = 2 }
		}
		localization_key = BLTGetBoltrusOutwardValue_2
	}
	text = {
		trigger = {
			check_variable = { BLT_king_boltrus_outward_value = 1 }
		}
		localization_key = BLTGetBoltrusOutwardValue_1
	}
	text = {
		trigger = {
			check_variable = { BLT_king_boltrus_outward_value = 0 }
		}
		localization_key = BLTGetBoltrusOutwardValue_0
	}
}
