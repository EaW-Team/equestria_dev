# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes


### All scripted loc is basically just for the tooltips
defined_text = { 
	name = LOC_Hunter
	text = {
		trigger = {
			has_idea = PLB_LOY_Hunter_bad
		}
		localization_key = PLB_LOC_Hunter_bad
	}
	text = {
		trigger = {
			has_idea = PLB_LOY_Hunter_bad_half
		}
		localization_key = PLB_LOC_Hunter_bad_half
	}
	text = {
		trigger = {
			has_idea = PLB_LOY_Hunter_good
		}
		localization_key = PLB_LOC_Hunter_good
	}
	text = {
		trigger = {
			has_idea = PLB_LOY_Hunter_good_half
		}
		localization_key = PLB_LOC_Hunter_good_half
	}
	
	text = {
		trigger = {
		}
		localization_key = PLB_LOC_Hunter_neutral
	}
}
