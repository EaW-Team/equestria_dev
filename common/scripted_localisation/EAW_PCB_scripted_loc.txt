defined_text = {
    name = PCB_corporate_unity_LOC
    text = {
        trigger = {
            check_variable = { PCB_corp_unity > 79 } #If unity is 80 or higher
        }
        localization_key = PCB_corporate_unity_very_good_LOC
    }
    text = {
        trigger = {
            check_variable = { PCB_corp_unity > 59 } #If unity is 60 or higher
        }
        localization_key = PCB_corporate_unity_good_LOC
    }
    text = {
        trigger = {
            check_variable = { PCB_corp_unity > 39 } #If unity is 40 or higher
        }
        localization_key = PCB_corporate_unity_mid_LOC
    }
    text = {
        trigger = { 
            check_variable = { PCB_corp_unity > 19 } #If unity is 20 or higher
        }
        localization_key = PCB_corporate_unity_bad_LOC
    }
    text = {
        localization_key = PCB_corporate_unity_very_bad_LOC #Base case (unity is less than 20)
    }
}

			OR = {

    }

        localization_key = PCB_estrella_gone_LOC
    }
}

##### Divider Civil War #####

defined_text = {
    name = PCB_divider_civil_war
    text = {
        trigger = {
			OR = {
				has_variable = PCB_estrella_resistance_var
				has_completed_focus = PCB_ECW_1
				NOT = {
					AND = {
						check_variable = { PCB_ECW_supply_units = 0 }
						has_completed_focus = PCB_ECW_EQS_2
					}
				}
				has_country_flag = PCB_debug_CIVILWARNOW
			}
        }
        localization_key = PCB_seperator_yes_LOC
    }
	text = {
        localization_key = PCB_seperator_no_LOC
    }
}
