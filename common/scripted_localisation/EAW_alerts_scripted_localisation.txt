defined_text = {
	name = EAW_alert_imminent_declaration_of_war_recursive_scriptedloc
	## Continued entry
	# Check for index not being the last entry
	text = {
		trigger = {
			set_temp_variable = { size = eaw_alert_list_of_imminent_declaration^num }
			subtract_from_temp_variable = { size = 1 }
			check_variable = { current_index < size }
			set_temp_variable = { current_tag = eaw_alert_list_of_imminent_declaration^current_index }
			add_to_temp_variable = { current_index = 1 }
		}
		localization_key = eaw_alert_imminent_declaration_of_war_entry_continual
	}
	## Last entry
	text = {
		trigger = {
			check_variable = { current_index = size }
			set_temp_variable = { current_tag = eaw_alert_list_of_imminent_declaration^current_index }
		}
		localization_key = eaw_alert_imminent_declaration_of_war_entry_last
	}
}
