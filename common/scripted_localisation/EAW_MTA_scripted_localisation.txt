defined_text = {
	name = GetMTAArtifactThreat
	text = {
		trigger = {
			set_temp_variable = { artifact_scope = MTA_artifact_selected?1 }
			always = no
		}
	}

	text = {
		trigger = {
			var:artifact_scope = {
				NOT = {
					has_state_flag = MTA_figured_out_threat_flag
				}
			}
		}
		localization_key = GetMTAArtifactThreat_Unknown_tt
	}
	text = {
		trigger = { 
			var:artifact_scope = {
				check_variable = { MTA_artifact_threat_level = 5 }
			}
		}
		localization_key = GetMTAArtifactThreat_Highest_tt
	}
	text = {
		trigger = { 
			var:artifact_scope = {
				check_variable = { MTA_artifact_threat_level = 4 }
			}
		}
		localization_key = GetMTAArtifactThreat_High_tt
	}
	text = {
		trigger = { 
			var:artifact_scope = {
				check_variable = { MTA_artifact_threat_level = 3 }
			}
		}
		localization_key = GetMTAArtifactThreat_Average_tt
	}
	text = {
		trigger = { 
			var:artifact_scope = {
				check_variable = { MTA_artifact_threat_level = 2 }
			}
		}
		localization_key = GetMTAArtifactThreat_Low_tt
	}
	text = {
		trigger = { 
			var:artifact_scope = {
				check_variable = { MTA_artifact_threat_level = 1 }
			}
		}
		localization_key = GetMTAArtifactThreat_Lowest_tt
	}
	text = {
		localization_key = GetMTAArtifactThreat_Unknown_tt
	}
}

defined_text = {
	name = GetMTAArtifactStatus
	text = {
		trigger = {
			set_temp_variable = { artifact_scope = MTA_artifact_selected?1 }
			always = no
		}
	}

	text = {
		trigger = {
			var:artifact_scope = {
				has_state_flag = MTA_sealed_artifact_flag

		localization_key = GetMTAArtifactStatus_Sealed
	}
	text = {
		trigger = {
			var:artifact_scope = {

			}
		}
		localization_key = GetMTAArtifactStatus_Active
	}
	text = {
		localization_key = GetMTAArtifactStatus_NotActive
	}
}
