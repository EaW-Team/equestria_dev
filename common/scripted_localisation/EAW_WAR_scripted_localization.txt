defined_text = {
	name = WAR_get_satellite_tag_value
	text = {
		trigger = {
			check_variable = { tag_id:WAR_satellite_value > -1 }
			check_variable = { tag_id:WAR_satellite_value < 1 }
		}
		localization_key = WAR_satellite_lowest_tt
	}
	text = {
		trigger = {
			check_variable = { tag_id:WAR_satellite_value > 0.9 }
			check_variable = { tag_id:WAR_satellite_value < 2 }
		}
		localization_key = WAR_satellite_low_tt
	}
	text = {
		trigger = {
			check_variable = { tag_id:WAR_satellite_value > 1.9 }
			check_variable = { tag_id:WAR_satellite_value < 3 }
		}
		localization_key = WAR_satellite_middle_tt
	}
	text = {
		trigger = {
			check_variable = { tag_id:WAR_satellite_value > 2.9 }
			check_variable = { tag_id:WAR_satellite_value < 4 }
		}
		localization_key = WAR_satellite_high_tt
	}
	text = {
		trigger = {
			check_variable = { tag_id:WAR_satellite_value > 3.9 }
			check_variable = { tag_id:WAR_satellite_value < 5 }
		}
		localization_key = WAR_satellite_highest_tt
	}
}
