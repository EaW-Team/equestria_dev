## Matrotanzas ##

defined_text = {
    name = PCB_1194_control_gc
    text = {
		trigger = {
			1194 = {
				has_dynamic_modifier = {
					modifier = PCB_modifier_nacro_state
				}
			}
        }
        localization_key = PCB_state_cartel_control
    }
    text = {
		trigger = {
			1194 = {
				NOT = {
					has_dynamic_modifier = {
						modifier = PCB_modifier_nacro_state
					}
				}
			}
        }
        localization_key = PCB_state_government_control
    }
}
