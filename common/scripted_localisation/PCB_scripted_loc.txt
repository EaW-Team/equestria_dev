    name = PCB_approval_rating_LOC
    text = {
        trigger = {
            check_variable = { PCB_leader_popularity > 79 } #If approval is 80 or higher
        }
        localization_key = PCB_approval_very_good_LOC
    }
    text = {
        trigger = {
            check_variable = { PCB_leader_popularity > 59 } #If approval is 60 or higher
        }
        localization_key = PCB_approval_good_LOC
    }
    text = {
        trigger = {
            check_variable = { PCB_leader_popularity > 39 } #If approval is 40 or higher
        }
        localization_key = PCB_approval_mid_LOC
    }
    text = {
        trigger = { 
            check_variable = { PCB_leader_popularity > 19 } #If approval is 20 or higher
        }
        localization_key = PCB_approval_bad_LOC
    }
    text = {
        localization_key = PCB_approval_very_bad_LOC #Base case (approval is less than 20)
    }
}
