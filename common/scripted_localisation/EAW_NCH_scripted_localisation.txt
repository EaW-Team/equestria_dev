# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = {
	name = GetAdressingFirtreeFocusName
	text = {
		trigger = {
			NOT = { FIR = { has_government = fascism } }

		}
		localization_key = NCH_adressing_firtree_historical
	}
	text = {
		trigger = {
			OR = {
				FIR = { has_government = fascism }

						NOT = { FIR = { is_subject_of = NCH } }
					}
				}
			}
		}
		localization_key = NCH_adressing_firtree_fascist
	}
}
