defined_text = {
	name = PopupIntegrationRecursiveScriptedLoc
	## Continued entry
	# Check for index not being the last entry
	text = {
		trigger = {
			set_temp_variable = { size = eaw_successfully_integrated_states^num }
			subtract_from_temp_variable = { size = 1 }
			check_variable = { current_index < size }
			set_temp_variable = { current_state = eaw_successfully_integrated_states^current_index }
			add_to_temp_variable = { current_index = 1 }
		}
		localization_key = eaw_diplo_popup_window_entry_continual
	}
	## Last entry
	text = {
		trigger = {
			check_variable = { current_index = size }
			set_temp_variable = { current_state = eaw_successfully_integrated_states^current_index }
		}
		localization_key = eaw_diplo_popup_window_entry_last
	}
}
